74 void build_masked_KKT(const StaticVector<double, N_vars>& u_k) {
75 KKT_Matrix.set_zero();
76 KKT_rhs.set_zero();
77
78 // --------------------------------------------------------
79 // Step 1 : 좌상단 블록 P (Hessian) 및 변수에 대한 우변 잔차
80 // --------------------------------------------------------
81 // KKT 행렬의 (0, 0) 위치에 P 행렬 삽입
82 KKT_Matrix.insert_block(0, 0, P);
83
84 // 탐색 방향 p에 대한 우변(RHS)은 현재 위치에서의 목적 함수 기울기의 음수: - (P * u_k + q)
85 StaticVector<double, N_vars> grad = (P * u_k) + q;
86 grad = grad * static_cast<double>(-1.0);
87 KKT_rhs.insert_block(0, 0, grad);
88
89 // --------------------------------------------------------
90 // Step 2 : 등식 제약 블록 (항상 활성화 상태이므로 무조건 KKT 행렬에 포함)
91 // --------------------------------------------------------
92 if constexpr (N_eq > 0) {
93 // KKT 행렬의 좌하단 및 우상단 블록에 A_eq 및 A_eq^T 삽입
94 KKT_Matrix.insert_transposed_block(0, N_vars, A_eq); // 우상단: A_eq^T
95 KKT_Matrix.insert_block(N_vars, 0, A_eq); // 좌하단: A_eq
96
97 // 등식 제약에 대한 우변(RHS) 잔차 : b_eq - A_eq * u_k
98 StaticVector<double, N_eq> eq_res = b_eq - (A_eq * u_k);
99 KKT_rhs.insert_block(N_vars, 0, eq_res);
100 }
101
102 // --------------------------------------------------------
103 // Step 3 : 부등식 제약 블록 (Working Set 배열에 따른 런타임 마스킹)
104 // --------------------------------------------------------
105 if constexpr (N_ineq > 0) {
106 const size_t ineq_offset = N_vars + N_eq; // 부등식 제약 조건이 삽입될 시작 인덱스
107 for (size_t i = 0; i < N_ineq; ++i) {
108 if (working_set[i]) {
109 // [활성화 됨] : 현재 해가 부등식 경계에 도달해 있으므로 등식 제약처럼 시스템에 삽입
110 for (size_t j = 0; j < N_vars; ++j) {
111 // 좌하단 (A_ineq) 및 우상단 (A_ineq^T) 삽입
112 KKT_Matrix(static_cast<int>(ineq_offset + i), static_cast<int>(j)) =
113 A_ineq(static_cast<int>(i), static_cast<int>(j));
114 KKT_Matrix(static_cast<int>(j), static_cast<int>(ineq_offset + i)) =
115 A_ineq(static_cast<int>(i), static_cast<int>(j));
116 }
117
118 // 활성화된 제약의 우변(RHS) 잔차 : b_ineq - A_ineq * u_k
119 // p를 이동하여 이 잔차를 정확히 0으로 만들고자 함.
120 double ineq_val = 0;
121 for (size_t j = 0; j < N_vars; ++j) {
122 ineq_val += A_ineq(static_cast<int>(i), static_cast<int>(j)) *
123 u_k(static_cast<int>(j));
124 }
125 KKT_rhs(static_cast<int>(ineq_offset + i)) =
126 b_ineq(static_cast<int>(i)) - ineq_val;
127 } else {
128 // [비활성화 됨] : 제약 조건 내부 영역에 있어 여유가 있는 상태
129 // KKT 행렬의 크기를 동적으로 줄이지 않고 대각 원소를 1.0으로, 우변을 0.0으로 두어
130 // 해당 라그랑주 승수가 0이 되도록 유도하고 행렬이 가역행렬(Invertible)이 되도록 보장함.
131 KKT_Matrix(static_cast<int>(ineq_offset + i),
132 static_cast<int>(ineq_offset + i)) = 1.0;
133 KKT_rhs(static_cast<int>(ineq_offset + i)) = 0.0;
134 }
135 }
136 }
137 }

147 bool solve(StaticVector<double, N_vars>& u, int max_iter = 100) {
148 for (int iter = 0; iter < max_iter; ++iter) {
149 // 1. 현재 Working Set과 변수 u를 기반으로 KKT 시스템 조립
150 build_masked_KKT(u);
151
152 // 2. KKT 행렬을 풀기 위해 LDLT 분해(Decomposition) 시도
153 // Active-Set 방법에서 KKT 행렬은 Indefinite 형태(양수 및 음수 고윳값 혼재)를 띠므로,
154 // Cholesky 대신 LDLT 분해 등 더 강력한 솔버가 필요합니다.
155 if (!KKT_Matrix.LDLT_decompose()) {
156 // 특이 행렬(Singular Matrix) 발생. 주로 제약 조건들이 선형 종속이거나
157 // P 행렬이 양의 정부호가 아닐 때 발생합니다.
158 return false;
159 }
160
161 // 3. 탐색 방향 p 및 라그랑주 승수 λ 계산
162 KKT_solution = KKT_Matrix.LDLT_solve(KKT_rhs);
163
164 // KKT_solution 벡터의 상단 N_vars 만큼이 실제 공간에서의 탐색 방향 p에 해당합니다.
165 StaticVector<double, N_vars> p = KKT_solution.template extract_block<N_vars, 1>(0, 0);
166
167 // 탐색 방향 벡터 p의 크기(L-infinity norm) 계산
168 double p_norm = 0.0;
169 for (size_t i = 0; i < N_vars; ++i) {
170 p_norm = std::max(p_norm, std::abs(p(static_cast<int>(i))));
171 }
172
173 // -------------------------------------------------------------
174 // Branch A : p == 0 (현재 활성화된 제약 조건들의 부분 공간 내에서 최적점에 도달)
175 // -------------------------------------------------------------
176 // 탐색 방향이 거의 0이라는 것은 현재의 Working Set 하에서는 더 이상 갈 곳이 없음을 의미합니다.
177 if (p_norm < 1e-6) {
178 double min_lambda = 0.0;
179 int drop_idx = -1;
180 const size_t ineq_offset = N_vars + N_eq;
181
182 // KKT 최적성 조건 중 쌍대 실행 가능성(Dual Feasibility) 검사:
183 // 부등식 제약(g_i(x) <= 0)에 대한 라그랑주 승수 λ_i 는 반드시 0 이상이어야 합니다.
184 // 만약 λ_i < 0 이라면, 해당 제약 조건을 경계에서 해제(Drop)했을 때
185 // 목적 함수 값을 더 낮출 수 있는 방향으로 진행할 수 있다는 뜻입니다.
186 for (size_t i = 0; i < N_ineq; ++i) {
187 if (working_set[i]) {
188 double lambda_i = KKT_solution(static_cast<int>(ineq_offset + i));
189 // 가장 음수 값이 큰(즉, 해제했을 때 가장 이득이 큰) 제약 조건을 찾습니다.
190 if (lambda_i < min_lambda) {
191 min_lambda = lambda_i;
192 drop_idx = static_cast<int>(i);
193 }
194 }
195 }
196
197 if (drop_idx == -1) {
198 // 음수인 라그랑주 승수가 없다면 (모든 λ_i >= 0),
199 // 현재 해가 전역적인 최적점(Global Optimum)이며 KKT 조건을 완벽히 충족한 것입니다.
200 return true;
201 } else {
202 // 가장 '나쁜' 영향을 주는 제약 조건 하나를 Working Set에서 제거(Drop)하여
203 // 다음 반복에서 해당 제약 경계면 안쪽으로 파고들어갈 수 있게 합니다.
204 working_set[drop_idx] = false;
205 }
206 }
207 // -------------------------------------------------------------
208 // Branch B : p != 0 (탐색 방향 p를 따라 더 나은 목적 함수 값을 향해 이동 가능)
209 // -------------------------------------------------------------
210 else {
211 double alpha = 1.0; // 기본 스텝 사이즈 (최대 이동 거리)
212 int blocking_idx = -1; // 이동 중에 부딪히게 될 새로운 제약 조건의 인덱스
213
214 // p 방향으로 이동했을 때의 각 부등식 제약 조건의 변화량(a_p)과 현재 제약 조건 값(a_u)
215 StaticVector<double, N_ineq> a_p = A_ineq * p;
216 StaticVector<double, N_ineq> a_u = A_ineq * u;
217
218 for (size_t i = 0; i < N_ineq; ++i) {
219 // 비활성화된 제약 중, 탐색 방향 p가 제약 경계면을 향해 가고 있을 때만 (a_p_i > 0) 검사합니다.
220 // a_p_i <= 0 인 경우는 제약 조건 안쪽(더 안전한 방향)으로 가고 있으므로 충돌할 일이 없습니다.
221 if (!working_set[i] && a_p(static_cast<int>(i)) > 1e-8) {
222 // 현재 위치에서 경계면까지 도달하는 데 필요한 이동 비율(거리) 계산
223 // (A_ineq * (u + alpha * p) = b_ineq) 를 alpha에 대해 푼 결과입니다.
224 double alpha_i = (b_ineq(static_cast<int>(i)) - a_u(static_cast<int>(i))) / a_p(static_cast<int>(i));
225
226 // 현재까지 찾은 가장 짧은 거리보다 짧다면 갱신 (가장 먼저 부딪히는 경계면 찾기)
227 if (alpha_i < alpha) {
228 alpha = alpha_i;
229 blocking_idx = static_cast<int>(i);
230 }
231 }
232 }
233
234 // 상태 변수 u를 방향 p를 따라 스텝 사이즈 alpha 만큼 이동시킵니다.
235 for (size_t i = 0; i < N_vars; ++i) {
236 u(static_cast<int>(i)) = u(static_cast<int>(i)) + (p(static_cast<int>(i)) * alpha);
237 }
238
239 // 스텝 사이즈가 1.0 미만으로 잘렸다는 것은 이동 중에 새로운 부등식 경계면과 부딪혔음을 의미합니다.
240 // 부딪힌 제약 조건을 Working Set에 추가(Add)하여,
241 // 다음 반복부터는 이 제약을 넘어서 밖으로 나가지 못하도록 등식 제약처럼 고정시킵니다.
242 if (blocking_idx != -1) {
243 working_set[blocking_idx] = true;
244 }
245 }
246 }
247 // 반복 횟수가 max_iter에 도달했다면 수렴에 실패한 것입니다.
248 // 흔한 실패 원인으로는 제약 조건이 상충(Infeasible)하여 Cycling(Add/Drop 반복 무한루프)이 발생한 경우가 있습니다.
249 return false;
250 }
251};
