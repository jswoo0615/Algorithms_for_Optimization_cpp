46 void build_masked_KKT(const StaticVector<double, N_vars>& u_k) {
47 KKT_Matrix.set_zero();
48 KKT_rhs.set_zero();
49
50 // --------------------------------------------------------
51 // Step 1 : 좌상단 블록 P (Hessian) 및 비용 함수 Gradient 잔차
52 // --------------------------------------------------------
53 KKT_Matrix.insert_block(0, 0, P);
54
55 // 탐색 방향 p에 대한 RHS : - (P * u_k + q)
56 StaticVector<double, N_vars> grad = (P * u_k) + q;
57 grad = grad * static_cast<double>(-1.0);
58 KKT_rhs.insert_block(0, 0, grad);
59
60 // --------------------------------------------------------
61 // Step 2 : 등식 제약 블록 (항상 활성화 상태로 매핑)
62 // --------------------------------------------------------
63 if constexpr (N_eq > 0) {
64 KKT_Matrix.insert_transposed_block(0, N_vars, A_eq);
65 KKT_Matrix.insert_block(N_vars, 0, A_eq);
66
67 // RHS 잔차 : b_eq - A_eq * u_k
68 StaticVector<double, N_eq> eq_res = b_eq - (A_eq * u_k);
69 KKT_rhs.insert_block(N_vars, 0, eq_res);
70 }
71
72 // --------------------------------------------------------
73 // Step 3 : 부등식 제약 블록 (Working Set 배열에 따른 런타임 마스킹)
74 // --------------------------------------------------------
75 if constexpr (N_ineq > 0) {
76 const size_t ineq_offset = N_vars + N_eq;
77 for (size_t i = 0; i < N_ineq; ++i) {
78 if (working_set[i]) {
79 // 활성화 됨 : 실제 A_ineq의 해당 행을 시스템에 삽입
80 for (size_t j = 0; j < N_vars; ++j) {
81 KKT_Matrix(static_cast<int>(ineq_offset + i), static_cast<int>(j)) = A_ineq(static_cast<int>(i), static_cast<int>(j));
82 KKT_Matrix(static_cast<int>(j), static_cast<int>(ineq_offset + i)) = A_ineq(static_cast<int>(i), static_cast<int>(j));
83 }
84
85 // RHS 잔차 : 활성화된 제약의 거리 b_ineq - A_ineq * u_k
86 double ineq_val = 0;
87 for (size_t j = 0; j < N_vars; ++j) {
88 ineq_val += A_ineq(static_cast<int>(i), static_cast<int>(j)) * u_k(static_cast<int>(j));
89 }
90 KKT_rhs(static_cast<int>(ineq_offset + i)) = b_ineq(static_cast<int>(i)) - ineq_val;
91 } else {
92 // 비활성화 됨 : 역행렬 특이성 (Singular)을 방지하기 위해 대각 원소 1.0 삽입
93 KKT_Matrix(static_cast<int>(ineq_offset + i), static_cast<int>(ineq_offset + i)) = 1.0;
94 KKT_rhs(static_cast<int>(ineq_offset + i)) = 0.0;
95 }
96 }
97 }
98 }

106 bool solve(StaticVector<double, N_vars>& u, int max_iter = 100) {
107 for (int iter = 0; iter < max_iter; ++iter) {
108 // 1. 현재 Working Set과 상태 u를 기반으로 KKT 시스템 조립
109 build_masked_KKT(u);
110
111 if (!KKT_Matrix.LDLT_decompose()) {
112 // 특이 행렬 발생 (제약 조건의 선형 종속 등) -> Fallback 전략 필요
113 return false;
114 }
115
116 KKT_solution = KKT_Matrix.LDLT_solve(KKT_rhs);
117
118 // 탐색 방향 p 추출
119 StaticVector<double, N_vars> p = KKT_solution.template extract_block<N_vars, 1>(0, 0);
120
121 // p의 무한 노름 (Infinity Norm) 계산
122 double p_norm = 0.0;
123 for (size_t i = 0; i < N_vars; ++i) {
124 p_norm = std::max(p_norm, std::abs(p(i)));
125 }
126
127 // -------------------------------------------------------------
128 // Branch A : p == 0 (현재 Working Set 내에서 최적점에 도달)
129 // -------------------------------------------------------------
130 if (p_norm < 1e-6) {
131 double min_lambda = 0.0;
132 int drop_idx = -1;
133 const size_t ineq_offset = N_vars + N_eq;
134
135 // KKT 조건의 Dual Feasibility 검사 (활성화된 부등식의 λ >= 0 인가?)
136 for (size_t i = 0; i < N_ineq; ++i) {
137 if (working_set[i]) {
138 double lambda_i = KKT_solution(static_cast<int>(ineq_offset + i));
139 // 가장 음수인 라그랑주 승수를 찾음
140 if (lambda_i < min_lambda) {
141 min_lambda = lambda_i;
142 drop_idx = static_cast<int>(i);
143 }
144 }
145 }
146
147 if (drop_idx == -1) {
148 // 모든 활성 부등식 제약의 λ >= 0 -> KKT 최적성 조건 완벽 충족
149 return true;
150 } else {
151 // 해를 영역 안쪽으로 끌어당기는 '나쁜' 제약 조건 해제 (Drop)
152 working_set[drop_idx] = false;
153 }
154 }
155 // -------------------------------------------------------------
156 // Branch B : p != 0 (아직 더 나은 비용 함수 값을 향해 이동 가능)
157 // -------------------------------------------------------------
158 else {
159 double alpha = 1.0;
160 int blocking_idx = -1;
161
162 StaticVector<double, N_ineq> a_p = A_ineq * p;
163 StaticVector<double, N_ineq> a_u = A_ineq * u;
164
165 for (size_t i = 0; i < N_ineq; ++i) {
166 // 비활성화된 제약 중, 이동 방향 p가 제약 경계면을 향해 갈 때만 (a_p > 0) 검사
167 if (!working_set[i] && a_p(i) > 1e-8) {
168 double alpha_i = (b_ineq(i) - a_u(i)) / a_p(i);
169
170 // 가장 먼저 부딪히는 경계면 찾기
171 if (alpha_i < alpha) {
172 alpha = alpha_i;
173 blocking_idx = static_cast<int>(i);
174 }
175 }
176 }
177
178 // 상태 변수 업데이트
179 u = u + (p * alpha);
180
181 // 스텝이 가로막혔다면 해당 제약을 Working Set에 추가 (Add)
182 if (blocking_idx != -1) {
183 working_set[blocking_idx] = true;
184 }
185 }
186 }
187 // Iteration Limit 초과 (Cycling 발생 혹은 스텝 폭발)
188 return false;
189 }
