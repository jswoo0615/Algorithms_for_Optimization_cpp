73 template <size_t N, typename Func>
74 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
75 double gamma = 0.9, double epsilon = 1e-8,
76 size_t max_iter = 15000,
77 double tol = 1e-4,
78 bool verbose = false) noexcept {
79 // 변수의 차원 N은 최소 1차원 이상이어야 합니다.
80 static_assert(N > 0, "Dimension N must be greater than 0");
81
82 // 최적화 시작 시간을 기록 (나노초 단위 측정을 위함)
83 auto start_clock = std::chrono::high_resolution_clock::now();
84
85 // 캐시 라인 정렬(64바이트)을 통해 SIMD 벡터 연산 및 캐시 히트 성능을 극대화한 변수 배열
86 // 선언
87 alignas(64) std::array<double, N> x = x_init;
88
89 // s: 과거 기울기(Gradient) 제곱의 지수 이동 평균을 저장하는 배열 (E[g^2])
90 // 분모에 위치하여 각 차원별 스텝 크기를 스케일링하는 역할을 합니다.
91 alignas(64) std::array<double, N> s = {0.0};
92
93 // u: 과거 변수 위치 변화량(Δx) 제곱의 지수 이동 평균을 저장하는 배열 (E[Δx^2])
94 // 분자에 위치하며, 글로벌 학습률을 대체하여 파라미터 업데이트 시 단위(Unit)의 일관성을
95 // 맞추는 핵심 요소입니다.
96 alignas(64) std::array<double, N> u = {0.0};
97
98 // 수렴 판정 시 제곱근(sqrt) 연산을 피하기 위해 허용 오차를 미리 제곱해 둡니다.
99 const double tol_sq = tol * tol;
100 double f_x = 0.0;
101 size_t iter = 0;
102
103 // 상세 출력(verbose) 모드가 활성화된 경우 시작 메시지 출력
104 if (verbose) {
105 std::cout << "========================================================\n";
106 std::cout << " 📐 AdaDelta Optimizer Started (gamma=" << gamma << ")\n";
107 std::cout << "========================================================\n";
108 }
109
110 // 최적화 메인 반복 루프 시작
111 for (iter = 1; iter <= max_iter; ++iter) {
112 // 현재 스텝에서의 기울기(Gradient) 벡터를 저장할 배열을 0으로 초기화 및 정렬
113 alignas(64) std::array<double, N> g = {0.0};
114
115 // 1. 순방향/역방향 자동 미분(Auto Differentiation) 호출
116 // 현재 파라미터 위치 `x`에서의 목적 함수 값(`f_x`)과 기울기 벡터(`g`)를 계산합니다.
117 // 정적 배열을 활용한 AutoDiff 기능으로 동적 메모리 할당 없이 O(1)의 안정성을
118 // 보장합니다.
119 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
120
121 // 현재 위치에서 기울기 벡터의 제곱합(L2-Norm의 제곱)을 계산합니다.
122 double g_norm_sq = 0.0;
123#pragma omp simd // 컴파일러에게 이 루프를 SIMD(Single Instruction Multiple Data) 병렬 명령어로
124 // 처리하도록 지시
125 for (size_t i = 0; i < N; ++i) {
126 // std::fma(a, b, c) = (a * b) + c
127 // 단일 사이클 하드웨어 연산으로 처리하여 정밀도를 높이고 부동소수점 반올림 오차를
128 // 줄입니다.
129 g_norm_sq = std::fma(g[i], g[i], g_norm_sq);
130 }
131
132 // 2. 수렴 판정 (Convergence Check)
133 // 기울기 벡터의 크기(의 제곱)가 지정된 허용 오차 제곱보다 작아지면 더 이상
134 // 업데이트할 필요가 없는 극소점(수렴)에 도달했다고 판단하고 루프를 탈출합니다.
135 if (g_norm_sq < tol_sq) {
136 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
137 break;
138 }
139
140 // 3. AdaDelta 핵심 파라미터 업데이트 로직
141#pragma omp simd // 벡터화(Vectorization) 강제 지시어
142 for (size_t i = 0; i < N; ++i) {
143 // [Step 3-1] 기울기 제곱의 지수 이동 평균 (E[g^2]_t) 갱신
144 // 수식: E[g^2]_t = gamma * E[g^2]_{t-1} + (1 - gamma) * g_t^2
145 // 이전까지의 누적 정보 s[i]에 gamma를 곱하여 서서히 잊게 만들고, 현재 기울기의 제곱
146 // 비중을 (1-gamma)만큼 반영합니다.
147 s[i] = std::fma(gamma, s[i], (1.0 - gamma) * g[i] * g[i]);
148
149 // [Step 3-2] 업데이트 변위 산출을 위한 RMS (Root Mean Square) 계산
150 // u[i]와 s[i]에 epsilon을 더하여 0으로 나누어지는 것을 방지하고 RMS를 구합니다.
151 // rms_u: 과거 변위량(Δx)들의 RMS 값
152 // rms_s: 과거 기울기(g)들의 RMS 값
153 double rms_u = std::sqrt(u[i] + epsilon);
154 double rms_s = std::sqrt(s[i] + epsilon);
155
156 // [Step 3-3] 이번 반복 스텝에서의 변위(Δx) 계산
157 // 수식: Δx_t = - (RMS[Δx]_{t-1} / RMS[g]_t) * g_t
158 // AdaDelta의 가장 큰 특징인 '파라미터별 맞춤형 스텝 폭' 결정 부분입니다.
159 // 사용자가 지정하는 Learning Rate가 없으며 (rms_u / rms_s) 비율 자체가 그 역할을
160 // 대신합니다.
161 double delta_x = -(rms_u / rms_s) * g[i];
162
163 // [Step 3-4] 변위(Δx) 제곱의 지수 이동 평균 (E[Δx^2]_t) 갱신
164 // 수식: E[Δx^2]_t = gamma * E[Δx^2]_{t-1} + (1 - gamma) * Δx_t^2
165 // 계산된 현재 스텝의 변위(delta_x)를 바탕으로, 다음 반복에서 사용할 E[Δx^2] 값을
166 // 업데이트합니다.
167 u[i] = std::fma(gamma, u[i], (1.0 - gamma) * delta_x * delta_x);
168
169 // [Step 3-5] 실제 파라미터(x) 업데이트 적용
170 // 현재 파라미터에 계산된 변위량(delta_x)을 더하여 위치를 갱신합니다.
171 x[i] += delta_x;
172 }
173
174 // 진행 상태 주기적 로깅 (1000번 반복마다 또는 첫 번째 반복 시 출력)
175 if (verbose && (iter % 1000 == 0 || iter == 1)) {
176 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed
177 << std::setprecision(6) << f_x << " | ||g||: " << std::sqrt(g_norm_sq)
178 << "\n";
179 }
180 }
181
182 // 최적화 루프 종료 후 종료 시간 기록 및 소요 시간(나노초) 계산
183 auto end_clock = std::chrono::high_resolution_clock::now();
184 auto duration =
185 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
186
187 // 최적화 최종 결과 및 파라미터 상태 출력
188 if (verbose) {
189 std::cout << "========================================================\n";
190 std::cout << " 🏁 Final Optimal Point: [" << x[0];
191 if constexpr (N > 1) std::cout << ", " << x[1];
192 if constexpr (N > 2) std::cout << ", ...";
193 std::cout << "]\n========================================================\n";
194 }
195
196 // 최종 최적 해, 마지막 목적 함수 값, 실제 소요된 반복 횟수, 경과 시간을 구조체로 묶어 반환
197 return {x, f_x, iter, duration.count()};
198 }
