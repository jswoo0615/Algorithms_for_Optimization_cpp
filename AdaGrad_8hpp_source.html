19 // ==============================================================================
20 // Algorithm 5.4 : AdaGrad
21 // 차원별로 과거 기울기의 제곱을 누적하여 맞춤형 학습률 제공
22 // ==============================================================================
23 template <size_t N, typename Func>
24 [[nodiscard]] static std::array<double, N> optimize(Func f, std::array<double, N> x, double alpha = 0.1,
25 double epsilon = 1e-8, size_t max_iter = 15000, double tol=1e-4,
26 bool verbose = false) {
27 // 0차원 배열 주입을 컴파일 타임에 원천 차단
28 static_assert(N > 0, "Dimension N must be greater than 0.");
29 // 과거 기울기 제곱 누적합 (동적 할당 없는 스택 메모리)
30 // 제어 이론 관례에 따라 Velocity(v)와 구분하기 위해 G 사용
31 std::array<double, N> G = {0.0};
32
33 if (verbose) {
34 std::cout << "========================================================\n";
35 std::cout << " 🎯 AdaGrad Optimizer Started (alpha=" << alpha << ", eps=" << epsilon << ")\n";
36 std::cout << "========================================================\n";
37 }
38
39 for (size_t iter = 1; iter <= max_iter; ++iter) {
40 double f_x = 0.0;
41 std::array<double, N> g = {0.0};
42
43 // 1. O(1) 할당 초고속 Auto Diff 계산
44 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
45
46 // 2. 기울기 l2-Norm 계산 및 조기 종료 검증
47 double g_norm_sq = 0.0;
48 for (size_t i = 0; i < N; ++i) {
49 g_norm_sq += g[i] * g[i];
50 }
51
52 const double g_norm = std::sqrt(g_norm_sq);
53 if (g_norm < epsilon) {
54 if (verbose) {
55 std::cout << "✅ Convergence achieved at iteration " << iter << " with gradient norm " << g_norm << ".\n";
56 }
57 break;
58 }
59
60 // 3. AdaGrad 파라미터 업데이트
61 for (size_t i = 0; i < N; ++i) {
62 G[i] += g[i] * g[i]; // 과거 기울기 제곱 누적
63 // [수치적 안정성] epsilon을 루트 밖에서 다하여 발산 (Divergence) 방지
64 // [실시간성] -= In-place 연산자로 캐시 히트율 극대화
65 x[i] -= (alpha / (std::sqrt(G[i]) + epsilon)) * g[i];
66 }
67
68 // 런타임 분기 예측 (Branch Prediction) 최적화를 위해 % 연산을 최소화
69 if (verbose && (iter % 1000 == 0)) {
70 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): "
71 << std::fixed << std::setprecision(6) << f_x
72 << " | ||g||: " << g_norm << "\n";
73 }
74 }
75 if (verbose) {
76 std::cout << "========================================================\n";
77 std::cout << " 🏁 Final Optimal Point: [" << x[0];
78 // C++17 constexpr if를 활용하여 N차원에 따른 안전한 출력 보장
79 if constexpr (N > 1) std::cout << ", " << x[1];
80 if constexpr (N > 2) std::cout << ", ...";
81 std::cout << "]\n========================================================\n";
82 }
83 return x;
84 }
