21 [[nodiscard]] static std::array<double, N> optimize(Func f, std::array<double, N> x,

24 size_t max_iter = 15000, double tol = 1e-4,
25 bool verbose = false) {
26 static_assert(N > 0, "Dimension N must be greater than 0");
27 std::array<double, N> m = {0.0}; // 1차 모멘트 초기화
28 std::array<double, N> v = {0.0}; // 2차 모멘트 초기화
29
30 // [성능 최적화] 매 이터레이션마다 sqrt를 호출하기 않기 위함
31 const double tol_sq = tol * tol;
32
33 double beta1_t = beta1;
34 double beta2_t = beta2;
35 if (verbose) {
36 std::cout << "========================================================\n";
37 std::cout << " 👑 Adam Optimizer Started (alpha=" << alpha << ")\n";
38 std::cout << "========================================================\n";
39 }
40 for (size_t iter = 1; iter <= max_iter; ++iter) {
41 double f_x = 0.0;
42 std::array<double, N> g = {0.0};
43 AutoDiff::value_and_gradient(f, x, f_x, g);
44
45 double g_norm_sq = 0.0;
46 for (size_t i = 0; i < N; ++i) {
47 g_norm_sq += g[i] * g[i];
48 }
49
50 if (g_norm_sq < tol_sq) {
51 if (verbose) {
52 std::cout << "✅ Convergence achieved at iteration " << iter
53 << " (f(x) = " << f_x << ", ||g||^2 = " << g_norm_sq << ")\n";
54 }
55 break;
56 }
57
58 // [성능 최적화] 차원 (N) 루프 진입 전, 공통 분모 (Bias correction)를 미리 계산
59 const double one_minus_beta1_t = 1.0 - beta1_t;
60 const double one_minus_beta2_t = 1.0 - beta2_t;
61
62 for (size_t i = 0; i < N; ++i) {
63 // 1. 모멘텀 업데이트
64 m[i] = beta1 * m[i] + (1.0 - beta1) * g[i];
65 v[i] = beta2 * v[i] + (1.0 - beta2) * g[i] * g[i];
66
67 // 2. 편향 보정 (수만 번의 std::pow를 단 2번의 나눗셈을 압축)
68 const double m_hat = m[i] / one_minus_beta1_t;
69 const double v_hat = v[i] / one_minus_beta2_t;
70
71 // 3. 파라미터 업데이트 (In-place)
72 x[i] -= (alpha / (std::sqrt(v_hat) + epsilon)) * m_hat;
73 }
74
75 // 다음 이터레이션을 위해 누적 곱 업데이트 (O(1) 연산)
76 beta1_t *= beta1;
77 beta2_t *= beta2;
78
79 if (verbose && (iter % 1000 == 0)) {
80 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed
81 << std::setprecision(6) << f_x << " | ||g||: " << std::sqrt(g_norm_sq)
82 << "\n";
83 }
84 }
85 if (verbose) {
86 std::cout << "========================================================\n";
87 std::cout << " 🏁 Final Optimal Point: [" << x[0];
88 if constexpr (N > 1) std::cout << ", " << x[1];
89 if constexpr (N > 2) std::cout << ", ...";
90 std::cout << "]\n========================================================\n";
91 }
92 return x;
93 }
