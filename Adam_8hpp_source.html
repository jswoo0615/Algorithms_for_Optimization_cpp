55 template <size_t N, typename Func>
56 [[nodiscard]] static std::array<double, N> optimize(Func f, std::array<double, N> x,
57 double alpha = 0.1, double beta1 = 0.9,
58 double beta2 = 0.999, double epsilon = 1e-8,
59 size_t max_iter = 15000, double tol = 1e-4,
60 bool verbose = false) {
61 // 0차원 배열 주입을 컴파일 타임에 원천 차단
62 static_assert(N > 0, "Dimension N must be greater than 0");
63
64 std::array<double, N> m = {0.0}; // 1차 모멘트 (기울기의 지수 이동 평균) 초기화
65 std::array<double, N> v = {0.0}; // 2차 모멘트 (기울기 제곱의 지수 이동 평균) 초기화
66
67 // [성능 최적화] 매 이터레이션(루프)마다 g_norm 계산 시 std::sqrt()를 호출하는 것을 막기
68 // 위해 허용 오차(tol)를 미리 제곱해 둡니다.
69 const double tol_sq = tol * tol;
70
71 // 편향 보정에 사용될 beta1^t 와 beta2^t 의 누적값
72 // std::pow(beta1, iter) 를 매번 호출하는 대신, 매 반복마다 변수에 beta1, beta2를
73 // 곱해줍니다.
74 double beta1_t = beta1;
75 double beta2_t = beta2;
76
77 if (verbose) {
78 std::cout << "========================================================\n";
79 std::cout << " 👑 Adam Optimizer Started (alpha=" << alpha << ")\n";
80 std::cout << "========================================================\n";
81 }
82
83 // 최적화 반복 루프
84 for (size_t iter = 1; iter <= max_iter; ++iter) {
85 double f_x = 0.0;
86 std::array<double, N> g = {0.0};
87
88 // 1. Auto Diff 호출 (O(1) 정적 메모리 환경에서 기울기 계산)
89 AutoDiff::value_and_gradient(f, x, f_x, g);
90
91 // 2. 기울기 제곱합(Norm^2) 계산 및 수렴 판정
92 double g_norm_sq = 0.0;
93 for (size_t i = 0; i < N; ++i) {
94 g_norm_sq += g[i] * g[i];
95 }
96
97 if (g_norm_sq < tol_sq) {
98 if (verbose) {
99 std::cout << "✅ Convergence achieved at iteration " << iter
100 << " (f(x) = " << f_x << ", ||g||^2 = " << g_norm_sq << ")\n";
101 }
102 break;
103 }
104
105 // [성능 최적화] 차원(N) 루프 진입 전, 모든 차원에 공통으로 적용되는
106 // 편향 보정(Bias Correction) 분모를 단 한 번만 미리 계산합니다.
107 const double one_minus_beta1_t = 1.0 - beta1_t;
108 const double one_minus_beta2_t = 1.0 - beta2_t;
109
110 // 3. 파라미터 업데이트 (차원별 순회)
111 for (size_t i = 0; i < N; ++i) {
112 // Step 3-1: 1차 모멘텀과 2차 모멘텀 업데이트
113 // m_t = beta1 * m_{t-1} + (1 - beta1) * g_t
114 m[i] = beta1 * m[i] + (1.0 - beta1) * g[i];
115 // v_t = beta2 * v_{t-1} + (1 - beta2) * g_t^2
116 v[i] = beta2 * v[i] + (1.0 - beta2) * g[i] * g[i];
117
118 // Step 3-2: 편향 보정 (Bias Correction)
119 // 초기화 시점(t=0)에 m과 v가 0으로 초기화되어 있으므로, 초기 몇 번의 반복 동안은
120 // 0에 가깝게 편향됩니다. 이를 방지하기 위해 (1 - beta^t) 로 나누어 스케일을
121 // 원복시킵니다.
122 const double m_hat = m[i] / one_minus_beta1_t;
123 const double v_hat = v[i] / one_minus_beta2_t;
124
125 // Step 3-3: 파라미터 업데이트 (In-place 연산으로 캐시 히트율 증가)
126 // x_t = x_{t-1} - alpha * m_hat / (sqrt(v_hat) + epsilon)
127 x[i] -= (alpha / (std::sqrt(v_hat) + epsilon)) * m_hat;
128 }
129
130 // 4. 다음 이터레이션을 위해 누적 곱 업데이트 (O(1) 연산)
131 // 비싼 std::pow 연산을 피하는 핵심 메커니즘입니다.
132 beta1_t *= beta1;
133 beta2_t *= beta2;
134
135 // 진행 상태 로깅 (분기 예측 최적화를 위해 1000번에 1번만 출력)
136 if (verbose && (iter % 1000 == 0)) {
137 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed
138 << std::setprecision(6) << f_x << " | ||g||: " << std::sqrt(g_norm_sq)
139 << "\n";
140 }
141 }
142
143 if (verbose) {
144 std::cout << "========================================================\n";
145 std::cout << " 🏁 Final Optimal Point: [" << x[0];
146 // C++17 constexpr if를 활용하여 차원 수에 맞춰 컴파일 타임에 분기
147 if constexpr (N > 1) std::cout << ", " << x[1];
148 if constexpr (N > 2) std::cout << ", ...";
149 std::cout << "]\n========================================================\n";
150 }
151
152 // 최종적으로 업데이트된 최적해 배열 반환
153 return x;
154 }
