91 template <size_t N, typename Func>
92 static std::array<std::array<double, N>, N> hessian(Func f, const std::array<double, N>& x) {
93 std::array<std::array<double, N>, N> H;
94 const double eps = 1e-5; // 중앙 차분 미소 간격
95
96 for (size_t i = 0; i < N; ++i) {
97 auto x_plus = x;
98 x_plus[i] += eps;
99 auto x_minus = x;
100 x_minus[i] -= eps;
101
102 auto g_plus = gradient<N>(f, x_plus);
103 auto g_minus = gradient<N>(f, x_minus);
104
105 for (size_t j = 0; j < N; ++j) {
106 H[i][j] = (g_plus[j] - g_minus[j]) / (2.0 * eps);
107 }
108 }
109
110 // 헤시안 행렬은 대칭행렬이므로, 수치 오차 보정을 위해 대칭화
111 for (size_t i = 0; i < N; ++i) {
112 for (size_t j = 0; j < N; ++j) {
113 double sym = (H[i][j] + H[j][i]) / 2.0;
114 H[i][j] = sym;
115 H[j][i] = sym;
116 }
117 }
118 return H;
119 }
