64
65 // N개의 입력 변수 각각에 대해 편미분 수행
66 for (size_t i = 0; i < N; ++i) {
67 StaticVector<Dual<double>, N> x_dual;
68 for (size_t j = 0; j < N; ++j) {
69 // i번째 변수에 대해서만 미분값(d, Dual Number의 쌍대 부분)을 1.0으로 부여하여
70 // "Seed"를 주입합니다. 이는 수식 상 편미분 ∂f / ∂x_i 를 구하기 위한 설정입니다. j
71 // != i 인 다른 변수들의 미분값은 0.0으로 두어 상수로 취급합니다.
72 x_dual(static_cast<int>(j)) =
73 Dual<double>(x(static_cast<int>(j)), (i == j) ? 1.0 : 0.0);
74 }
75
76 // 이원수를 인자로 함수를 평가하면 반환되는 결과 역시 이원수(Dual Number)가 됩니다.
77 // 반환된 이원수의 'd' 부분이 바로 우리가 찾고자 하는 i번째 변수에 대한 편미분 값입니다.
78 Dual<double> res = f(x_dual);
79 g(static_cast<int>(i)) = res.d;
80 }

102 StaticVector<double, N>& g_out) {

104
105 for (size_t i = 0; i < N; ++i) {
106 StaticVector<Dual<double>, N> x_dual;
107 for (size_t j = 0; j < N; ++j) {
108 // i번째 변수에만 편미분 Seed(1.0)를 주입
109 x_dual(static_cast<int>(j)) =
110 Dual<double>(x(static_cast<int>(j)), (i == j) ? 1.0 : 0.0);
111 }
112 Dual<double> res = f(x_dual);
113
114 // 첫 번째 변수에 대한 순회(i == 0)에서 함수의 스칼라 결과값(실수부 v)을 추출합니다.
115 // 모든 반복에서 res.v는 동일한 f(x) 값을 가지므로 한 번만 저장하면 됩니다.
116 if (i == 0) {
117 v_out = res.v;
118 }
119 // 쌍대부(d)는 i번째 방향으로의 방향 도함수(편미분 값)입니다.
120 g_out(static_cast<int>(i)) = res.d;
121 }
122 }

141 template <size_t M, size_t N, typename Func>
142 [[nodiscard]] static StaticMatrix<double, M, N> jacobian(Func f,
143 const StaticVector<double, N>& x) {
144 StaticMatrix<double, M, N> J;
145 J.set_zero();
146
147 // 입력 차원 N번의 평가를 통해 전체 야코비안 행렬을 구축합니다. (Forward-Mode AD의 특징)
148 // 입력 변수 개수가 출력 함수 개수보다 현저히 많을 경우(M << N),
149 // Reverse-Mode AD보다 계산 비용이 더 클 수 있으나, 소규모 최적화 문제에서는 구현이 단순하고
150 // 매우 효율적입니다.
151 for (size_t i = 0; i < N; ++i) {
152 StaticVector<Dual<double>, N> x_dual;
153 for (size_t j = 0; j < N; ++j) {
154 // i번째 입력 변수에 대해 Seed(1.0)를 주입
155 x_dual(static_cast<int>(j)) =
156 Dual<double>(x(static_cast<int>(j)), (i == j) ? 1.0 : 0.0);
157 }
158
159 // 함수 f는 입력으로 전달된 이원수 벡터를 이용해 연산을 수행하고,
160 // 출력 차원 M개 만큼의 이원수 벡터를 반환해야 합니다.
161 StaticVector<Dual<double>, M> res_vec = f(x_dual);
162
163 // 반환된 각각의 이원수에서 'd'(쌍대부)를 추출하면,
164 // 이는 각 출력 함수 f_k 가 i번째 입력 변수 x_i 의 미세한 변화에 어떻게 반응하는지를
165 // 의미합니다. 따라서 각 출력 차원 k에 대하여, i번째 열에 편미분 값을 채워 넣습니다.
166 for (size_t k = 0; k < M; ++k) {
167 J(static_cast<int>(k), static_cast<int>(i)) = res_vec(static_cast<int>(k)).d;
168 }
169 }
170 return J;
171 }
