77 std::array<std::array<double, N>, N> H{};
78 constexpr double eps = 1e-5;
79 constexpr double inv_two_eps = 1.0 / (2.0 * eps);
80
81 for (size_t i = 0; i < N; ++i) {
82 auto x_plus = x;
83 x_plus[i] += eps;
84 auto x_minus = x;
85 x_minus[i] -= eps;
86
87 const auto g_plus = gradient<N>(f, x_plus);
88 const auto g_minus = gradient<N>(f, x_minus);
89
90 for (size_t j = 0; j < N; ++j) {
91 H[i][j] = (g_plus[j] - g_minus[j]) * inv_two_eps;
92 }
93 }
94
95 for (size_t i = 0; i < N; ++i) {
96 for (size_t j = 0; j < N; ++j) {
97 const double sym = (H[i][j] + H[j][i]) * 0.5;
98 H[i][j] = sym;
99 H[j][i] = sym;
100 }
101 }
102 return H;
103 }
