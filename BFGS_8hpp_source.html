103 template <size_t N>
104 [[nodiscard]] static constexpr std::array<double, N> mat_vec_mul(
105 const std::array<std::array<double, N>, N>& M, const std::array<double, N>& v) noexcept {
106 std::array<double, N> res = {0.0};
107 // 행렬의 각 행(row)에 대해 반복 계산
108 for (size_t i = 0; i < N; ++i) {
109 double sum = 0.0;
110// 각 행의 열(column) 요소들과 벡터 요소를 내적합니다. SIMD 최적화 적용.
111#pragma omp simd
112 for (size_t j = 0; j < N; ++j) {
113 sum = std::fma(M[i][j], v[j], sum);
114 }
115 res[i] = sum;
116 }
117 return res;
118 }

122 // 완전히 금지합니다.

146 template <size_t N, typename Func>
147 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
148 double tol = 1e-6,
149 size_t max_iter = 200) noexcept {
150 // 성능 분석 및 최적화 수행 시간을 측정하기 위한 고해상도 타이머 시작
151 auto start_clock = std::chrono::high_resolution_clock::now();
152
153 std::array<double, N> x = x_init; // 현재 추정 위치 변수 (초기값으로 세팅)
154 const double tol_sq = tol * tol; // 매 반복마다 발생하는 sqrt 연산을 방지하고자 허용 오차를
155 // 제곱 형태로 변환하여 비교
156
157 // V : 역헤시안 근사 행렬 (Inverse Hessian Approximation Matrix)
158 // 메모리를 64바이트(일반적인 CPU L1 캐시 라인 크기) 단위로 정렬(alignas)하여 메모리 접근
159 // 병목 현상을 줄이고, 캐시 히트율(Cache Hit Rate) 및 SIMD 벡터 연산 효율성을 극대화합니다.
160 alignas(64) std::array<std::array<double, N>, N> V = {0.0};
161
162 // 초기화 단계:
163 // 알고리즘 초기 단계에서는 목적 함수의 곡률(2차 미분)에 대한 정보가 전혀 없으므로,
164 // 역헤시안 근사 행렬 V를 항등 행렬(Identity Matrix, 대각 원소가 1인 행렬)로 초기화합니다.
165 // 이는 첫 번째 반복 시 탐색 방향이 순수 경사 하강법(Steepest Gradient Descent, p = -g)과
166 // 완전히 동일해지도록 만들어 초기 안정성을 확보하기 위함입니다.
167 for (size_t i = 0; i < N; ++i) {
168 V[i][i] = 1.0;
169 }
170
171 double f_val = 0.0; // 현재 위치 x에서의 목적 함수 평가값 저장
172 std::array<double, N> g = {0.0}; // 현재 위치 x에서의 1차 기울기(Gradient) 벡터 저장
173
174 // 최적화 루프 진입 전, 초기 위치에서의 목적 함수 값과 1차 미분(Gradient) 값을 자동
175 // 미분(AutoDiff) 기법으로 동시 계산
176 AutoDiff::value_and_gradient<N>(f, x, f_val, g);
177
178 size_t iter = 0;
179 // ==========================================
180 // 메인 최적화 반복(Iteration) 루프 시작
181 // ==========================================
182 for (iter = 1; iter <= max_iter; ++iter) {
183 // ------------------------------------------
184 // 1. 수렴 검사 (Convergence Check)
185 // ------------------------------------------
186 // 현재 기울기 벡터의 제곱합(L2-norm의 제곱)을 계산
187 double g_norm_sq = dot<N>(g, g);
188 // 기울기가 사실상 0에 수렴했다면(tol_sq 미만), 현재 위치가 평고점(극솟값 또는 안장점)에
189 // 도달한 것으로 간주하고 루프 조기 탈출
190 if (g_norm_sq < tol_sq) {
191 break;
192 }
193
194 // ------------------------------------------
195 // 2. 탐색 방향(Search Direction, p) 결정
196 // ------------------------------------------
197 // 순수 뉴턴법의 탐색 방향 공식은 p = -H^{-1} * g 입니다.
198 // BFGS에서는 H^{-1} 대신 역헤시안의 근사 행렬인 V를 사용하므로 p = -V * g 가 됩니다.
199 // 우선 행렬 V와 기울기 벡터 g를 곱합니다.
200 std::array<double, N> p = mat_vec_mul<N>(V, g);
201
202 // 구해진 결과의 부호를 반전시켜, 목적 함수값이 가장 가파르게 감소하는 하강 방향(Descent
203 // Direction)을 취합니다.
204#pragma omp simd
205 for (size_t i = 0; i < N; ++i) {
206 p[i] = -p[i];
207 }
208
209 // ------------------------------------------
210 // 3. 선 탐색 (Line Search)
211 // ------------------------------------------
212 // 찾은 탐색 방향 p를 따라 "얼마만큼의 크기(Step Size, alpha)"로 이동해야
213 // 함수값이 충분히 감소하는지(Armijo Condition 만족 등)를 결정합니다.
214 // 뉴턴 기반 기법은 스케일이 자체적으로 맞춰져 있으므로, 이론적으로 최적의 이동 거리
215 // alpha는 1.0에 가깝습니다. 따라서 기본 스텝 사이즈 1.0에서 시작하여 백트래킹(조건을
216 // 만족할 때까지 0.5씩 곱하여 스텝을 줄임)을 수행합니다. (파라미터: 초기값 1.0, 감소율
217 // 0.5, c1 파라미터 1e-4, wolfe 조건 강제 여부 false)
218 double alpha =
219 BacktrackingLineSearch::search<N>(f, x, p, f_val, g, 1.0, 0.5, 1e-4, false);
220
221 // ------------------------------------------
222 // 4. 새로운 위치(x_new) 산출 및 함수 재평가
223 // ------------------------------------------
224 std::array<double, N> x_new = {0.0};
225 // 결정된 이동 거리 alpha와 방향 p를 바탕으로 다음 위치를 업데이트 (x_new = x + alpha *
226 // p)
227#pragma omp simd
228 for (size_t i = 0; i < N; ++i) {
229 x_new[i] = std::fma(alpha, p[i], x[i]);
230 }
231
232 // 새롭게 이동한 위치(x_new)에서의 목적 함수 값(f_new)과 새로운 기울기 벡터(g_new)를
233 // 계산
234 double f_new = 0.0;
235 std::array<double, N> g_new = {0.0};
236 AutoDiff::value_and_gradient<N>(f, x_new, f_new, g_new);
237
238 // ------------------------------------------
239 // 5. 역헤시안 근사 행렬(V) 업데이트를 위한 벡터 변화량 계산
240 // ------------------------------------------
241 std::array<double, N> s = {0.0}; // 위치 변화 벡터 (Step vector): s = x_new - x
242 std::array<double, N> y = {0.0}; // 기울기 변화 벡터 (Yield vector): y = g_new - g
243#pragma omp simd
244 for (size_t i = 0; i < N; ++i) {
245 s[i] = x_new[i] - x[i];
246 y[i] = g_new[i] - g[i];
247 }
248
249 // ------------------------------------------
250 // 6. 곡률 조건 (Curvature Condition) 검사 및 V 행렬 갱신
251 // ------------------------------------------
252 // 두 벡터의 내적 (y^T * s) 연산
253 double y_dot_s = dot<N>(y, s);
254
255 // [곡률 조건(Curvature Condition)]
256 // BFGS 알고리즘이 안정적으로 동작하여 최적해로 수렴하기 위해서는
257 // 근사 행렬 V가 항상 양의 정부호(Positive Definite, 모든 고윳값이 양수) 성질을 유지해야
258 // 합니다. 이를 수학적으로 보장하기 위한 필요충분조건(Secant Equation 제약 조건)이 바로
259 // y^T * s > 0 입니다. 컴퓨터의 부동소수점 수치적 오차를 고려하여 단순 0이 아닌
260 // 1e-10(아주 작은 양수)보다 큰지 엄격하게 검사합니다.
261 if (y_dot_s > 1e-10) {
262 // 조건을 만족하면 Sherman-Morrison-Woodbury 공식을 전개한 형태의 역헤시안
263 // 업데이트(Rank-2 Update)를 수행합니다. 정석적인 BFGS 역헤시안 업데이트 공식: V_new
264 // = (I - rho * s * y^T) * V * (I - rho * y * s^T) + rho * s * s^T 이를 실제
265 // 구현에서 행렬 곱셈 연산량을 줄이기 위해 수학적으로 전개하여 최적화된 공식: V_new
266 // = V - rho*(V*y*s^T + s*y^T*V) + rho*(rho*y^T*V*y + 1)*s*s^T
267
268 const double rho = 1.0 / y_dot_s; // 곡률 스케일링 팩터 (rho)
269 auto Vy = mat_vec_mul<N>(V, y); // 중간 벡터 행렬 곱 연산: V * y
270 const double yVy = dot<N>(y, Vy); // 스칼라 값 연산: y^T * V * y
271
272 // 모든 업데이트 항에 공통으로 곱해지는 스칼라 계수 항 계산: rho * (1 + rho * (y^T *
273 // V * y))
274 const double scalar_term = rho * (1.0 + rho * yVy);
275
276 // 행렬 V 업데이트 로직
277 // 근사 행렬 V는 본질적으로 대칭 행렬(Symmetric Matrix)입니다.
278 // 따라서 N^2 번의 연산을 모두 수행하지 않고, 상삼각(Upper Triangular) 부분만 직접
279 // 계산한 뒤 하삼각 부분은 대칭 복사(Symmetric Copy)하여 연산 속도를 크게
280 // 향상시킵니다.
281 for (size_t i = 0; i < N; ++i) {
282 for (size_t j = i; j < N; ++j) {
283 // 공식에 따른 각 행렬 원소(i, j)별 업데이트 계산
284 double updated_val = V[i][j] + scalar_term * s[i] * s[j] -
285 rho * (Vy[i] * s[j] + s[i] * Vy[j]);
286
287 V[i][j] = updated_val;
288 // 대각 원소가 아닌 경우(상삼각 부분인 경우), 대칭되는 하삼각 위치(j, i)에도
289 // 동일한 값 할당
290 if (i != j) {
291 V[j][i] = updated_val;
292 }
293 }
294 }
295 } else if (iter > 1) {
296 // [리스타트 기법 (Restart Strategy)]
297 // 선 탐색이 부정확하거나 최적화하는 목적 함수의 비선형성 및 굴곡이 너무 강해서,
298 // 수치적 이유로 곡률 조건(y^T * s > 0)을 만족하지 못하는 경우가 발생할 수 있습니다.
299 // 이 상태로 강제로 업데이트를 진행하면 V 행렬의 양의 정부호 성질이 깨져(Not
300 // Positive Definite) 탐색 방향이 망가집니다. 이를 방지하기 위해, 이런 이상
301 // 상황에서는 V 행렬에 누적된 정보가 오염되었다고 판단하고 행렬을 초기 상태인 단위
302 // 행렬(Identity Matrix)로 완전히 초기화하여 안전한 경사 하강법 방향부터 다시
303 // 시작(Restart)합니다.
304 for (size_t i = 0; i < N; ++i) {
305 for (size_t j = 0; j < N; ++j) {
306 V[i][j] = (i == j) ? 1.0 : 0.0;
307 }
308 }
309 }
310
311 // ------------------------------------------
312 // 7. 상태 갱신 (State Update)
313 // ------------------------------------------
314 // 다음 반복(Iteration)을 위해 현재 상태 정보를 새롭게 갱신된 정보로 치환합니다.
315 x = x_new;
316 f_val = f_new;
317 g = g_new;
318 }
319
320 // 메인 최적화 루프 완전 종료 후 소요 시간 측정 종료
321 auto end_clock = std::chrono::high_resolution_clock::now();
322 // 측정된 두 시간차를 나노초(ns) 단위로 변환
323 auto duration =
324 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
325
326 // 도출된 최적해 벡터, 그 때의 목적 함수 값, 최종 수행된 반복 횟수, 경과 시간(나노초)을
327 // 구조체로 묶어 반환
328 return {x, f_val, iter, duration.count()};
329 }
