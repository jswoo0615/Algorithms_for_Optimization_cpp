49 template <size_t N>
50 [[nodiscard]] static constexpr double dot(const std::array<double, N>& a,
51 const std::array<double, N>& b) noexcept {
52 double sum = 0.0;
53#pragma omp simd
54 for (size_t i = 0; i < N; ++i) {
55 sum = std::fma(a[i], b[i], sum); // a[i] * b[i] + sum
56 }

64 template <size_t N>
65 [[nodiscard]] static constexpr std::array<double, N> mat_vec_mul(
66 const std::array<std::array<double, N>, N>& M, const std::array<double, N>& v) noexcept {
67 std::array<double, N> res = {0.0};
68 for (size_t i = 0; i < N; ++i) {
69 double sum = 0.0;
70#pragma omp simd
71 for (size_t j = 0; j < N; ++j) {
72 sum = std::fma(M[i][j], v[j], sum);
73 }
74 res[i] = sum;
75 }
76 return res;

94 template <size_t N, typename Func>
95 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
96 double tol = 1e-6,
97 size_t max_iter = 200) noexcept {
98 auto start_clock = std::chrono::high_resolution_clock::now();
99
100 std::array<double, N> x = x_init;
101 const double tol_sq = tol * tol; // 제곱 형태의 톨러런스 (루트 연산 방지용)
102
103 // V : 역헤시안 근사 행렬 (Inverse Hessian Approximation Matrix)
104 // 메모리 정렬(alignas 64)을 통해 캐시 히트율 및 SIMD 연산 효율을 높입니다.
105 alignas(64) std::array<std::array<double, N>, N> V = {0.0};
106
107 // 초기 역헤시안 근사 행렬을 단위 행렬(Identity Matrix)로 설정합니다.
108 // 이는 초기에는 경사 하강법(Gradient Descent) 방향으로 시작함을 의미합니다.
109 for (size_t i = 0; i < N; ++i) {
110 V[i][i] = 1.0;
111 }
112
113 double f_val = 0.0;
114 std::array<double, N> g = {0.0};
115
116 // 초기 위치에서의 목적 함수 값과 기울기(Gradient) 평가
117 AutoDiff::value_and_gradient<N>(f, x, f_val, g);
118
119 size_t iter = 0;
120 // 메인 최적화 루프
121 for (iter = 1; iter <= max_iter; ++iter) {
122 // 1. 수렴 검사 (기울기의 제곱합이 허용치보다 작으면 종료)
123 double g_norm_sq = dot<N>(g, g);
124 if (g_norm_sq < tol_sq) {
125 break; // 평고점(최적해) 도달
126 }
127
128 // 2. 탐색 방향(Search Direction, p) 계산
129 // p = -V * g (역헤시안 근사 행렬과 기울기를 곱하여 뉴턴 방향 근사)
130 std::array<double, N> p = mat_vec_mul<N>(V, g);
131#pragma omp simd
132 for (size_t i = 0; i < N; ++i) {
133 p[i] = -p[i]; // 방향을 음수(하강 방향)로 반전
134 }
135
136 // 3. 선 탐색 (Line Search)
137 // 결정된 탐색 방향 p를 따라 얼마나 이동할지(alpha)를 결정합니다.
138 // BFGS는 근사된 뉴턴 스텝을 따르므로 기본 스텝 사이즈 1.0부터 백트래킹을 시작합니다.
139 double alpha =
140 BacktrackingLineSearch::search<N>(f, x, p, f_val, g, 1.0, 0.5, 1e-4, false);
141
142 // 4. 새로운 위치(x_new) 산출
143 std::array<double, N> x_new = {0.0};
144#pragma omp simd
145 for (size_t i = 0; i < N; ++i) {
146 x_new[i] = std::fma(alpha, p[i], x[i]); // x_new = x + alpha * p
147 }
148
149 // 새로운 위치에서의 함수 값과 기울기 평가
150 double f_new = 0.0;
151 std::array<double, N> g_new = {0.0};
152 AutoDiff::value_and_gradient<N>(f, x_new, f_new, g_new);
153
154 // 5. 역헤시안 근사 행렬(V) 업데이트를 위한 준비
155 std::array<double, N> s = {0.0}; // 변위 벡터: s = x_new - x
156 std::array<double, N> y = {0.0}; // 기울기 변화 벡터: y = g_new - g
157#pragma omp simd
158 for (size_t i = 0; i < N; ++i) {
159 s[i] = x_new[i] - x[i];
160 y[i] = g_new[i] - g[i];
161 }
162
163 // 6. 곡률 조건 (Curvature Condition) 검사
164 double y_dot_s = dot<N>(y, s);
165
166 // y^T * s > 0 을 만족해야만 V 행렬이 양의 정부호(Positive Definite) 성질을 유지할 수 있습니다.
167 if (y_dot_s > 1e-10) {
168 // Sherman-Morrison 공식을 전개한 역헤시안 업데이트 (Rank-2 Update)
169 // 수학적 공식: V_new = V - rho*(V*y*s^T + s*y^T*V) + rho*(rho*y^T*V*y + 1)*s*s^T
170
171 const double rho = 1.0 / y_dot_s;
172 auto Vy = mat_vec_mul<N>(V, y); // V * y
173 const double yVy = dot<N>(y, Vy); // y^T * V * y
174
175 // 공통 스칼라 항 계산: rho * (1 + rho * y^T * V * y)
176 const double scalar_term = rho * (1.0 + rho * yVy);
177
178 // 행렬 V 업데이트 (대칭 행렬이므로 상삼각 부분만 계산하고 대칭 복사)
179 for (size_t i = 0; i < N; ++i) {
180 for (size_t j = i; j < N; ++j) {
181 double updated_val = V[i][j]
182 + scalar_term * s[i] * s[j]
183 - rho * (Vy[i] * s[j] + s[i] * Vy[j]);
184
185 V[i][j] = updated_val;
186 if (i != j) {
187 V[j][i] = updated_val; // 대칭성 유지
188 }
189 }
190 }
191 } else if (iter > 1) {
192 // 수치적 이유로 곡률 조건을 만족하지 못한 경우 (y^T * s <= 0)
193 // 행렬이 무너지는 것을 막기 위해 V를 안전한 초기 상태(단위 행렬)로 리셋합니다 (Restart).
194 for (size_t i = 0; i < N; ++i) {
195 for (size_t j = 0; j < N; ++j) {
196 V[i][j] = (i == j) ? 1.0 : 0.0;
197 }
198 }
199 }
200
201 // 7. 상태 갱신 (다음 이터레이션을 위함)
202 x = x_new;
203 f_val = f_new;
204 g = g_new;
205 }
206
207 auto end_clock = std::chrono::high_resolution_clock::now();
208 auto duration =
209 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
210
211 // 결과 반환
212 return {x, f_val, iter, duration.count()};
213 }
