45 const std::array<double, N>& d,
46 double alpha) noexcept {
47 std::array<double, N> pt = {0.0};
48// 컴파일러에게 벡터화(SIMD)를 지시하여 하드웨어 병렬 처리를 극대화합니다.
49#pragma omp simd
50 for (size_t i = 0; i < N; ++i) {
51 // std::fma(a, b, c) = a * b + c : 부동소수점 오차를 줄이고 속도를 높이는 FMA 연산
52 pt[i] = std::fma(alpha, d[i], x[i]); // x[i] + (alpha * d[i])
53 }
54 return pt;
55 }

75 template <size_t N, typename Func>
76 [[nodiscard]] static double search(Func f, const std::array<double, N>& x,
77 const std::array<double, N>& d, double f_x,
78 const std::array<double, N>& grad_x, double alpha = 1.0,
79 double p = 0.5, double c = 1e-4,
80 bool verbose = false) noexcept {
81 // 1. 방향 도함수 (Directional Derivative) 계산 : ∇f(x)^T * d
82 // 탐색 방향 d로 이동할 때, 시작점에서 함수값이 감소하는 초기 기울기를 의미합니다.
83 double dir_deriv = 0.0;
84#pragma omp simd
85 for (size_t i = 0; i < N; ++i) {
86 dir_deriv = std::fma(grad_x[i], d[i], dir_deriv);
87 }
88
89 // 방향 도함수가 0 이상이라는 것은 현재 방향(d)이 목적 함수 값을 줄이는 '하강 방향(Descent
90 // Direction)'이 아님을 의미합니다.
91 if (dir_deriv >= 0.0 && verbose) {
92 std::cout << " [Warning] Not a descent direction! dir_deriv: " << dir_deriv << "\n";
93 }
94
95 size_t iter = 0;
96
97 // 2. Armijo 조건을 만족할 때까지 반복하며 alpha를 줄여나갑니다.
98 while (true) {
99 iter++;
100
101 // 현재 스텝 사이즈(alpha)만큼 이동한 새로운 좌표(x_new) 산출
102 auto x_new = ray_point<N>(x, d, alpha);
103
104 // 새로운 좌표에서의 목적 함수 값(f_new) 평가
105 double f_new = AutoDiff::value<N>(f, x_new);
106
107 // 3. 목표 감소치(Target Value) 계산 (Armijo 충분 감소 조건)
108 // 수식: f(x + alpha * d) <= f(x) + c * alpha * (∇f(x)^T * d)
109 // 단순히 함수값이 감소하는 것(f_new < f_x)만으로는 무한 루프에 빠질 수 있으므로,
110 // 1차 테일러 전개 기반의 예측 감소량에 상수 c를 곱한 만큼은 '최소한' 감소해야 한다고
111 // 엄격하게 제한합니다.
112 double target_val = f_x + c * alpha * dir_deriv;
113
114 if (verbose) {
115 std::cout << " ↳ [Backtrack Iter " << std::setw(2) << iter
116 << "] alpha: " << std::fixed << std::setprecision(6) << alpha
117 << " | f_new: " << f_new << " | Target: " << target_val << "\n";
118 }
119
120 // Armijo 조건 검사 : 함수 값이 목표치보다 작거나 같으면 승인(Accept)하고 스텝 사이즈
121 // 확정
122 if (f_new <= target_val) {
123 if (verbose) std::cout << " ↳ [Accepted] Armijo condition satisfied!\n";
124 return alpha;
125 }
126
127 // 조건을 만족하지 못했으므로 스텝 사이즈를 p 비율(예: 절반)만큼 줄여서 다시 시도
128 alpha *= p;
129
130 // 4. 무한 루프 방지를 위한 Failsafe (수치적으로 너무 작은 alpha는 의미가 없음)
131 if (alpha < 1e-10) {
132 if (verbose) std::cout << " ↳ [Failsafe] Alpha reached minimum limit.\n";
133 return alpha;
134 }
135 }
136 }
