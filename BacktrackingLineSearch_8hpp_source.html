41 [[nodiscard]] static constexpr std::array<double, N> ray_point(const std::array<double, N>& x,
42 const std::array<double, N>& d,
43 double alpha) noexcept {
44 std::array<double, N> pt = {0.0};
45// 컴파일러에게 벡터화(SIMD)를 지시하여 하드웨어 병렬 처리를 극대화합니다.
46#pragma omp simd
47 for (size_t i = 0; i < N; ++i) {
48 // std::fma(a, b, c) = a * b + c : 부동소수점 오차를 줄이고 속도를 높이는 FMA 연산
49 pt[i] = std::fma(alpha, d[i], x[i]); // x[i] + (alpha * d[i])
50 }
51 return pt;
52 }

71 template <size_t N, typename Func>
72 [[nodiscard]] static double search(Func f, const std::array<double, N>& x,
73 const std::array<double, N>& d, double f_x,
74 const std::array<double, N>& grad_x, double alpha = 1.0,
75 double p = 0.5, double c = 1e-4,
76 bool verbose = false) noexcept {
77 // 1. 방향 도함수 (Directional Derivative) 계산 : ∇f(x)^T * d
78 // 탐색 방향 d로 이동할 때, 시작점에서 함수값이 감소하는 초기 기울기를 의미합니다.
79 double dir_deriv = 0.0;
80#pragma omp simd
81 for (size_t i = 0; i < N; ++i) {
82 dir_deriv = std::fma(grad_x[i], d[i], dir_deriv);
83 }
84
85 // 방향 도함수가 0 이상이라는 것은 현재 방향(d)이 목적 함수 값을 줄이는 '하강 방향(Descent Direction)'이 아님을 의미합니다.
86 if (dir_deriv >= 0.0 && verbose) {
87 std::cout << " [Warning] Not a descent direction! dir_deriv: " << dir_deriv << "\n";
88 }
89
90 size_t iter = 0;
91
92 // 2. Armijo 조건을 만족할 때까지 반복하며 alpha를 줄여나갑니다.
93 while (true) {
94 iter++;
95
96 // 현재 스텝 사이즈(alpha)만큼 이동한 새로운 좌표(x_new) 산출
97 auto x_new = ray_point<N>(x, d, alpha);
98
99 // 새로운 좌표에서의 목적 함수 값(f_new) 평가
100 double f_new = AutoDiff::value<N>(f, x_new);
101
102 // 3. 목표 감소치(Target Value) 계산 (Armijo 충분 감소 조건)
103 // 수식: f(x + alpha * d) <= f(x) + c * alpha * (∇f(x)^T * d)
104 // 단순히 함수값이 감소하는 것(f_new < f_x)만으로는 무한 루프에 빠질 수 있으므로,
105 // 1차 테일러 전개 기반의 예측 감소량에 상수 c를 곱한 만큼은 '최소한' 감소해야 한다고 엄격하게 제한합니다.
106 double target_val = f_x + c * alpha * dir_deriv;
107
108 if (verbose) {
109 std::cout << " ↳ [Backtrack Iter " << std::setw(2) << iter
110 << "] alpha: " << std::fixed << std::setprecision(6) << alpha
111 << " | f_new: " << f_new << " | Target: " << target_val << "\n";
112 }
113
114 // Armijo 조건 검사 : 함수 값이 목표치보다 작거나 같으면 승인(Accept)하고 스텝 사이즈 확정
115 if (f_new <= target_val) {
116 if (verbose) std::cout << " ↳ [Accepted] Armijo condition satisfied!\n";
117 return alpha;
118 }
119
120 // 조건을 만족하지 못했으므로 스텝 사이즈를 p 비율(예: 절반)만큼 줄여서 다시 시도
121 alpha *= p;
122
123 // 4. 무한 루프 방지를 위한 Failsafe (수치적으로 너무 작은 alpha는 의미가 없음)
124 if (alpha < 1e-10) {
125 if (verbose) std::cout << " ↳ [Failsafe] Alpha reached minimum limit.\n";
126 return alpha;
127 }
128 }
129 }
