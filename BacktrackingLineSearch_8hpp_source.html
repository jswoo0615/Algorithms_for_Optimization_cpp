21 #pragma omd simd
22 for (size_t i = 0; i < N; ++i) {
23 pt[i] = std::fma(alpha, d[i], x[i]); // x[i] + (alpha * d[i])

29 template <size_t N, typename Func>
30 [[nodiscard]] static double search(Func f, const std::array<double, N>& x,
31 const std::array<double, N>& d, double f_x,
32 const std::array<double, N>& grad_x, double alpha = 1.0,
33 double p = 0.5, double c = 1e-4, bool verbose = false) noexcept {
34 // 방향 도함수 (Directional Derivative) 계산 : ∇f(x)^T * d
35 double dir_deriv = 0.0;
36 #pragma omp simd
37 for (size_t i = 0; i < N; ++i) {
38 dir_deriv = std::fma(grad_x[i], d[i], dir_deriv);
39 }
40
41 if (dir_deriv >= 0.0 && verbose) {
42 std::cout << " [Warning] Not a descent direction! dir_deriv: " << dir_deriv << "\n";
43 }
44
45 size_t iter = 0;
46 while (true) {
47 iter++;
48 auto x_new = ray_point<N>(x, d, alpha);
49 double f_new = AutoDiff::value<N>(f, x_new);
50
51 // 목표 감소치 계산 (출발점 함수값 + c * alpha * dir_deriv)
52 double target_val = f_x + c * alpha * dir_deriv;
53
54 if (verbose) {
55 std::cout << " ↳ [Backtrack Iter " << std::setw(2) << iter
56 << "] alpha: " << std::fixed << std::setprecision(6) << alpha
57 << " | f_new: " << f_new << " | Target: " << target_val << "\n";
58 }
59
60 // Armijo 조건 검사
61 if (f_new <= target_val) {
62 if (verbose) std::cout << " ↳ [Accepted] Armijo condition satisfied!\n";
63 return alpha;
64 }
65
66 alpha *= p;
67
68 if (alpha < 1e-10) {
69 if (verbose) std::cout << " ↳ [Failsafe] Alpha reached minimum limit.\n";
70 return alpha;
71 }
72 }
73 }
