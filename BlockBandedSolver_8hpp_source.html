33 bool backward_pass() {
34 // 1. 종단 조건 (Terminal Condition) 초기화
35 StaticMatrix<double, Nx, Nx> P_next = P_N;
36 StaticVector<double, Nx> p_next = p_N_vec;
37
38 // 2. 미래 (Np-1)부터 현재 (0)로 역행
39 for (int k = static_cast<int>(Np) - 1; k >= 0; --k) {
40 auto& s = stages[k];
41
42 // H_xx = Q_k + A_k^T * P_{k+1} * A_k
43 StaticMatrix<double, Nx, Nx> H_xx = s.Q + s.A.quadratic_multiply(P_next);
44
45 // H_uu = R_k + B_k^T * P_{k+1} * B_k
46 StaticMatrix<double, Nu, Nu> H_uu = s.R + s.B.quadratic_multiply(P_next);
47
48 // H_ux = B_k^T * P_{k+1} * A_k
49 StaticMatrix<double, Nx, Nx> P_A = P_next * s.A;
50 StaticMatrix<double, Nu, Nx> H_ux = s.B.transpose() * P_A;
51
52 // 벡터 항 계산 (Gradient)
53 // g_u = r_k + B_k^T * (P_{k+1} * d_k + p_{k+1})
54 StaticVector<double, Nx> P_d_p = (P_next * s.d) + p_next;
55 StaticVector<double, Nu> g_u = s.r + (s.B.transpose() * P_d_p);
56
57 // g_x = q_k + A_k^T * (P_{k+1} * d_k + p_{k+1})
58 StaticVector<double, Nx> g_x = s.q + (s.A.transpose() * P_d_p);
59
60 // H_uu 분해 (Positive Definite 검증)
61 if (!H_uu.LDLT_decompose()) {
62 // 수치적 불안정성 방어 (Damping)
63 for (size_t i = 0; i < Nu; ++i) {
64 H_uu(i, i) += 1e-4;
65 }
66 if (!H_uu.LDLT_decompose()) {
67 return false;
68 }
69 }
70
71 // 피드백 게인 K_k = -H_uu^{-1} * H_ux
72 StaticMatrix<double, Nu, Nx> neg_H_ux = H_ux * -1.0;
73 s.K = H_uu.template solve_multiple<Nx>(neg_H_ux);
74
75 // 피드포워드 게인 k_ff = -H_uu^{-1} * g_u
76 StaticVector<double, Nu> neg_g_u;
77 for (size_t i = 0; i < Nu; ++i) {
78 neg_g_u(i) = -g_u(i);
79 }
80 s.k_ff = H_uu.LDLT_solve(neg_g_u);
81
82 // Value Function 갱신 (P_k, p_k)
83 // P_k = H_xx + K_k^T * H_uu * K_k + K_k^T * H_ux + H_ux^T * K_k
84 // 구조적 단순화 : P_k = H_xx + H_ux^T * K_k
85 s.P = H_xx + (H_ux.transpose() * s.K);
86
87 // p_k = g_x + H_ux^T * k_ff
88 s.p_vec = g_x + (H_ux.transpose() * s.k_ff);
89
90 // 다음 스텝 (뒤로 이동)을 위해 갱신
91 P_next = s.P;
92 p_next = s.p_vec;
93 }
94 return true;
95 }

102 void forward_pass(const StaticVector<double, Nx>& dx_0) {
103 dx[0] = dx_0; // 초기 상태 오차
104 for (size_t k = 0; k < Np; ++k) {
105 auto& s = stages[k];
106
107 // u_k = K_k * x_k + k_ff
108 du[k] = (s.K * dx[k]) + s.k_ff;
109
110 // x_{k+1} = A_k * x_k + B_k * u_k + d_k
111 dx[k + 1] = (s.A * dx[k]) + (s.B * du[k]) + s.d;
112 }
113 }

118 bool solve(const StaticVector<double, Nx>& dx_0) {
119 if (!backward_pass()) {
120 return false; // H_uu 행렬 특이성 발생 (Infeasible)

122 forward_pass(dx_0);
123 return true;
