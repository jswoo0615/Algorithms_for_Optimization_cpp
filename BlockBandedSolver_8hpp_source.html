78 bool backward_pass() {
79 // 1. 종단 조건 (Terminal Condition)을 초기 가치 함수의 파라미터로 설정 (V_N(x) = Terminal
80 // Cost)
81 StaticMatrix<double, Nx, Nx> P_next = P_N;
82 StaticVector<double, Nx> p_next = p_N_vec;
83
84 // 2. 미래 (Np-1)부터 현재 (0)로 역행하며 동적 계획법(Dynamic Programming) 수행
85 for (int k = static_cast<int>(Np) - 1; k >= 0; --k) {
86 auto& s = stages[k];
87
88 // [Step 2-1] Q 함수 (State-Action Value Function) 근사를 위한 블록 텐서(Hessian 블록)
89 // 계산 H_xx = Q_k + A_k^T * P_{k+1} * A_k (상태에 대한 2차 미분)
90 StaticMatrix<double, Nx, Nx> H_xx = s.Q + s.A.quadratic_multiply(P_next);
91
92 // H_uu = R_k + B_k^T * P_{k+1} * B_k (제어 입력에 대한 2차 미분)
93 StaticMatrix<double, Nu, Nu> H_uu = s.R + s.B.quadratic_multiply(P_next);
94
95 // H_ux = B_k^T * P_{k+1} * A_k (상태와 제어 입력 간의 교차 2차 미분)
96 StaticMatrix<double, Nx, Nx> P_A = P_next * s.A;
97 StaticMatrix<double, Nu, Nx> H_ux = s.B.transpose() * P_A;
98
99 // [Step 2-2] Q 함수 (State-Action Value Function) 근사를 위한 1차 Gradient 벡터 계산
100 // 보조 벡터: P_{k+1} * d_k + p_{k+1}
101 StaticVector<double, Nx> P_d_p = (P_next * s.d) + p_next;
102
103 // g_u = r_k + B_k^T * (P_{k+1} * d_k + p_{k+1}) (제어 입력에 대한 1차 미분)
104 StaticVector<double, Nu> g_u = s.r + (s.B.transpose() * P_d_p);
105
106 // g_x = q_k + A_k^T * (P_{k+1} * d_k + p_{k+1}) (상태에 대한 1차 미분)
107 StaticVector<double, Nx> g_x = s.q + (s.A.transpose() * P_d_p);
108
109 // [Step 2-3] H_uu의 역행렬 연산을 위한 LDLT 분해 (H_uu는 Positive Definite여야 함)
110 if (!H_uu.LDLT_decompose()) {
111 // H_uu가 양의 정부호(Positive Definite)가 아니어 역행렬을 구할 수 없는 경우 (수치적
112 // 불안정성 발생) 대각 원소에 작은 양수(1e-4)를 더해주는 Damping(Tikhonov
113 // Regularization)을 적용하여 방어합니다.
114 for (size_t i = 0; i < Nu; ++i) {
115 H_uu(i, i) += 1e-4;
116 }
117 // Damping 적용 후 다시 분해 시도. 그래도 실패하면 문제를 풀 수 없으므로 false 반환.
118 if (!H_uu.LDLT_decompose()) {
119 return false;
120 }
121 }
122
123 // [Step 2-4] 최적 제어 이득(Gain) 계산
124 // 피드백 게인: K_k = -H_uu^{-1} * H_ux
125 // (H_ux 각 열에 대해 H_uu의 역행렬을 곱하는 solve_multiple 연산 수행)
126 StaticMatrix<double, Nu, Nx> neg_H_ux = H_ux * -1.0;
127 s.K = H_uu.template solve_multiple<Nx>(neg_H_ux);
128
129 // 피드포워드 게인: k_ff_k = -H_uu^{-1} * g_u
130 StaticVector<double, Nu> neg_g_u;
131 for (size_t i = 0; i < Nu; ++i) {
132 neg_g_u(i) = -g_u(i);
133 }
134 s.k_ff = H_uu.LDLT_solve(neg_g_u);
135
136 // [Step 2-5] 가치 함수(Value Function) 갱신 (P_k, p_k 계산)
137 // 원래 식: P_k = H_xx + K_k^T * H_uu * K_k + K_k^T * H_ux + H_ux^T * K_k
138 // 위 식은 K_k = -H_uu^{-1} * H_ux 를 대입하면 구조적으로 아래와 같이 단순화됩니다:
139 // P_k = H_xx + H_ux^T * K_k
140 s.P = H_xx + (H_ux.transpose() * s.K);
141
142 // p_k = g_x + H_ux^T * k_ff_k
143 s.p_vec = g_x + (H_ux.transpose() * s.k_ff);
144
145 // 다음 시간 스텝(더 과거의 시간, k-1)의 계산을 위해 P_next와 p_next를 갱신합니다.
146 P_next = s.P;
147 p_next = s.p_vec;
148 }
149 return true;
150 }

161 void forward_pass(const StaticVector<double, Nx>& dx_0) {
162 dx[0] = dx_0; // 시점 t = 0 의 초기 상태 세팅
163
164 // 현재 (0)부터 미래 (Np-1)까지 순행하며 궤적 생성
165 for (size_t k = 0; k < Np; ++k) {
166 auto& s = stages[k];
167
168 // 최적 제어 입력 산출: u_k = K_k * x_k + k_ff_k
169 du[k] = (s.K * dx[k]) + s.k_ff;
170
171 // 다음 상태 예측 (선형화된 시스템 동역학 방정식 적용):
172 // x_{k+1} = A_k * x_k + B_k * u_k + d_k
173 dx[k + 1] = (s.A * dx[k]) + (s.B * du[k]) + s.d;
174 }
175 }

187 bool solve(const StaticVector<double, Nx>& dx_0) {
188 // 1. 역방향 탐색을 통해 최적 제어 이득(K, k_ff) 계산
189 if (!backward_pass()) {
190 // H_uu 행렬이 특이행렬(Singular)이 되거나 Damping으로도 해결 불가능한 경우 (Infeasible)
191 return false;
192 }
193
194 // 2. 도출된 제어 이득을 바탕으로 초기 상태 dx_0부터 순방향으로 제어 입력과 상태 궤적
195 // 시뮬레이션
196 forward_pass(dx_0);
197
198 return true;
199 }
