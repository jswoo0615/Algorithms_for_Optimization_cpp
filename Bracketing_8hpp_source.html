40template <typename Func>
41inline Range bracket_minimum(Func f, double x, double s = 0.01, double k = 2.0) {
42 double a = x;
43 double ya = f(a);
44 double b = a + s;
45 double yb = f(b);
46
47 // [Step 1] 탐색 방향 결정
48 // 현재 스텝 방향(오른쪽)으로 이동했을 때 함수값이 커진다면(yb > ya),
49 // 이는 하강 방향(Descent Direction)이 아니라는 의미입니다.
50 // 따라서 탐색 방향을 반대(왼쪽)로 바꾸기 위해 a와 b를 스왑하고 스텝(s)의 부호를 뒤집습니다.
51 if (yb > ya) {
52 std::swap(a, b);
53 std::swap(ya, yb);
54 s = -s;
55 }
56
57 // [Step 2] 하강 방향을 따라 구간 확장
58 while (true) {
59 // 이전 지점(b)에서 확장된 스텝(s)만큼 더 이동한 새로운 지점(c)를 평가합니다.
60 double c = b + s;
61 double yc = f(c);
62
63 // 함수값이 이전 지점(yb)보다 커진다면(yc > yb),
64 // 하강하던 추세가 꺾이고 다시 상승한다는 의미이므로 구간(a, c) 사이에 극솟값이 존재하게
65 // 됩니다.
66 if (yc > yb) {
67 // 구간의 양 끝점이 항상 (작은 값, 큰 값) 순서가 되도록 정렬하여 반환합니다.
68 return {std::min(a, c), std::max(a, c)};
69 }
70
71 // 탐색 지점들을 한 칸씩 앞으로 당깁니다. (a <- b, b <- c)
72 a = b;
73 ya = yb;
74 b = c;
75 yb = yc;
76
77 // 다음 탐색을 위해 스텝 크기를 k배(보통 2배)로 확장하여 탐색 속도를 가속합니다.
78 s *= k;
79 }
80}

96template <typename Func>
97inline double golden_section_search(Func f, Range r, int n) {
98 // 황금비(Golden Ratio): (sqrt(5) - 1) / 2 ≈ 0.61803398...
99 const double phi = (std::sqrt(5.0) - 1.0) / 2.0;
100 double a = r.a;
101 double b = r.b;
102
103 // 구간 내에서 황금비를 기준으로 두 개의 내분점(x1, x2)을 생성
104 double x1 = b - phi * (b - a);
105 double x2 = a + phi * (b - a);
106 double y1 = f(x1);
107 double y2 = f(x2);
108
109 for (int i = 0; i < n; ++i) {
110 // [구간 축소 로직]
111 if (y1 < y2) {
112 // x1 지점의 함수값이 더 작다면, 최솟값은 무조건 [a, x2] 사이에 존재합니다.
113 // 따라서 오른쪽 경계(b)를 x2로 축소합니다.
114 b = x2;
115 x2 = x1; // 이전의 x1은 새로운 구간의 오른쪽 내분점(x2)이 됩니다.
116 y2 = y1; // 함수 평가 재사용 (성능 이점)
117 x1 = b - phi * (b - a); // 새로운 왼쪽 내분점 계산
118 y1 = f(x1);
119 } else {
120 // x2 지점의 함수값이 더 작거나 같다면, 최솟값은 무조건 [x1, b] 사이에 존재합니다.
121 // 따라서 왼쪽 경계(a)를 x1로 축소합니다.
122 a = x1;
123 x1 = x2; // 이전의 x2는 새로운 구간의 왼쪽 내분점(x1)이 됩니다.
124 y1 = y2; // 함수 평가 재사용 (성능 이점)
125 x2 = a + phi * (b - a); // 새로운 오른쪽 내분점 계산
126 y2 = f(x2);
127 }
128 }
129 // 탐색이 완료된 후 최종적으로 축소된 구간의 중앙값을 최적점으로 반환합니다.
130 return (a + b) / 2.0;
131}

147template <typename FuncDual>
148inline double bisection_method(FuncDual f, Range range, int n) {
149 double a = range.a;
150 double b = range.b;
151
152 for (int i = 0; i < n; ++i) {
153 // 현재 탐색 구간의 정중앙(mid) 지점을 계산합니다.
154 double mid = (a + b) / 2.0;
155
156 // 중앙 지점(mid)에서 미분값을 평가하기 위해 이원수(Dual Number)를 생성합니다.
157 // 실수부(Real)에는 위치 'mid'를, 쌍대부(Dual)에는 미분 계수 시딩(Seeding)을 위한 '1.0'을
158 // 넣습니다.
159 auto res = f(Dual<double>(mid, 1.0));
160
161 // [구간 축소 로직]
162 // res.d 는 이원수 연산을 통해 자동으로 계산된 현재 위치(mid)에서의 1차 미분값(기울기,
163 // f'(mid))입니다.
164 if (res.d > 0) {
165 // 기울기가 양수(+)라는 것은 현재 지점 기준 오른쪽은 함수값이 증가한다는 뜻입니다.
166 // 극솟값은 함수값이 감소하는 왼쪽 구간 [a, mid]에 존재하므로 오른쪽 경계(b)를 당깁니다.
167 b = mid;
168 } else {
169 // 기울기가 음수(-)라는 것은 현재 지점 기준 왼쪽은 함수값이 증가(또는 감소 중)한다는
170 // 뜻입니다. 극솟값은 오른쪽 구간 [mid, b]에 존재하므로 왼쪽 경계(a)를 당깁니다.
171 a = mid;
172 }
173 }
174 // 탐색이 완료된 후 최종적으로 축소된 구간의 중앙값을 최적점으로 반환합니다.
175 return (a + b) / 2.0;
176}

204template <typename Func>
205inline double shubert_piyavskii(Func f, Range r, double l, double eps, int max_iter) {
206 // 초기화: 구간의 양 끝점(a, b)을 평가하여 톱날의 시작점으로 리스트에 저장합니다.
207 // X 좌표를 기준으로 항상 오름차순 정렬 상태를 유지해야 합니다.
208 std::vector<Pt> pts = {{r.a, f(r.a)}, {r.b, f(r.b)}};
209
210 // 지금까지 발견된 가장 작은 실제 함수값 추적
211 double min_f = std::min(pts[0].y, pts[1].y);
212
213 for (int i = 0; i < max_iter; ++i) {
214 int best_idx = 0;
215 double best_z = std::numeric_limits<double>::infinity();
216
217 // [Step 1] V자 톱날(하한선)의 교점 중 가장 깊게 파인(가장 작은) z 좌표 찾기
218 // pts 배열에 인접한 두 점 사이에서 만들어지는 V자 모양의 꼭짓점 높이(z)를 계산합니다.
219 for (size_t j = 0; j < pts.size() - 1; ++j) {
220 // 립시츠 상수를 이용한 하한선 교점의 Y 좌표(z) 수식
221 double z = (pts[j].y + pts[j + 1].y) / 2.0 - l * (pts[j + 1].x - pts[j].x) / 2.0;
222 if (z < best_z) {
223 best_z = z;
224 best_idx =
225 static_cast<int>(j); // 가장 유망한(낮은) 교점이 속한 구간의 왼쪽 인덱스 저장
226 }
227 }
228
229 // [Step 2] 가장 유망한 교점의 X 좌표를 계산하고 실제 함수 평가 수행
230 double new_x = (pts[best_idx].x + pts[best_idx + 1].x) / 2.0 -
231 (pts[best_idx + 1].y - pts[best_idx].y) / (2.0 * l);
232
233 // 새로 예측한 지점에서 실제 함수값을 평가합니다.
234 double new_y = f(new_x);
235
236 // 탐색된 함수값 중 최소값을 갱신합니다.
237 min_f = std::min(min_f, new_y);
238
239 // [Step 3] 새로운 점 삽입 및 배열 정렬 상태 유지
240 // best_idx와 best_idx + 1 사이에 새로운 점을 삽입하여 X 좌표 오름차순을 유지합니다.
241 pts.insert(pts.begin() + best_idx + 1, {new_x, new_y});
242
243 // [Step 4] 수렴 조건(Convergence Check) 확인
244 // 현재 발견된 최소 함수값(min_f)과 립시츠 하한선의 최소값(best_z) 사이의 갭(오차)이
245 // 허용 오차(eps)보다 작아졌다면 전역 최솟값을 충분히 찾았다고 판단하고 종료합니다.
246 if (std::abs(min_f - best_z) < eps) break;
247 }
248
249 // 최종적으로 발견된 가장 작은 함수값을 반환합니다.
250 return min_f;
251}
