27template <typename Func>
28inline Range bracket_minimum(Func f, double x, double s = 0.01, double k = 2.0) {
29 double a = x;
30 double ya = f(a);
31 double b = a + s;
32 double yb = f(b);
33
34 // 하강 방향이 아니면 방향 전환
35 if (yb > ya) {
36 std::swap(a, b);
37 std::swap(ya, yb);
38 s = -s;
39 }
40
41 while (true) {
42 double c = b + s;
43 double yc = f(c);
44
45 // 함수값이 다시 커지는 지점을 발견하면 구간 변환
46 if (yc > yb) {
47 return {std::min(a, c), std::max(a, c)};
48 }
49 a = b;
50 ya = yb;
51 b = c;
52 yb = yc;
53 s *= k; // 스텝 확장
54 }
55}

61template <typename Func>
62inline double golden_section_search(Func f, Range r, int n) {
63 const double phi = (std::sqrt(5.0) - 1.0) / 2.0; // 황금비 (~0.618)
64 double a = r.a;
65 double b = r.b;
66
67 double x1 = b - phi * (b - a);
68 double x2 = a + phi * (b - a);
69 double y1 = f(x1);
70 double y2 = f(x2);
71
72 for (int i = 0; i < n; ++i) {
73 if (y1 < y2) {
74 b = x2;
75 x2 = x1;
76 y2 = y1;
77 x1 = b - phi * (b - a);
78 y1 = f(x1);
79 } else {
80 a = x1;
81 x1 = x2;
82 y1 = y2;
83 x2 = a + phi * (b - a);
84 y2 = f(x2);
85 }
86 }
87 return (a + b) / 2.0;
88}

94template <typename FuncDual>
95inline double bisection_method(FuncDual f, Range range, int n) {
96 double a = range.a;
97 double b = range.b;
98
99 for (int i = 0; i < n; ++i) {
100 double mid = (a + b) / 2.0;
101
102 // 1.0을 시딩 (Seeding)하여 mid 지점에서 1차 미분값 획득
103 auto res = f(Dual<double>(mid, 1.0));
104
105 // 기울기가 양수 (+)면 오른쪽이 높으므로 왼쪽 구간 (a ~ mid) 선택
106 // 기울기가 음수 (-)면 왼쪽이 높으므로 오른쪽 구간 (mid ~ b) 선택
107 if (res.d > 0) {
108 b = mid;
109 } else {
110 a = mid;
111 }
112 }
113 return (a + b) / 2.0;
114}

125template <typename Func>
126inline double shubert_piyavskii(Func f, Range r, double l, double eps, int max_iter) {
127 // std::vector 오류 수정 및 리스트 초기화 문법 준수
128 std::vector<Pt> pts = {{r.a, f(r.a)}, {r.b, f(r.b)}};
129
130 double min_f = std::min(pts[0].y, pts[1].y);
131
132 for (int i = 0; i < max_iter; ++i) {
133 int best_idx = 0;
134 double best_z = std::numeric_limits<double>::infinity();
135
136 // 가장 깊게 파인 톱날 (하한선) 탐색
137 for (size_t j = 0; j < pts.size() - 1; ++j) {
138 double z = (pts[j].y + pts[j + 1].y) / 2.0 - l * (pts[j + 1].x - pts[j].x) / 2.0;
139 if (z < best_z) {
140 best_z = z;
141 best_idx = static_cast<int>(j);
142 }
143 }
144
145 // 새로운 탐색 지점 계산 및 평가
146 double new_x = (pts[best_idx].x + pts[best_idx + 1].x) / 2.0 -
147 (pts[best_idx + 1].y - pts[best_idx].y) / (2.0 * l);
148 double new_y = f(new_x);
149
150 min_f = std::min(min_f, new_y);
151
152 // 새로운 점 삽입 (x좌표 기분 정렬 유지)
153 pts.insert(pts.begin() + best_idx + 1, {new_x, new_y});
154
155 // 수렴 조건 판단
156 if (std::abs(min_f - best_z) < eps) break;
157 }
158 return min_f;
159}
