44 template <size_t N, typename Func>
45 static std::array<double, N> optimize(Func f, std::array<double, N> x, size_t max_iter = 1000,
46 double tol = 1e-4, bool verbose = false) {
47 if (verbose) {
48 std::cout << "========================================================\n";
49 std::cout << " ⚔️ Conjugate Gradient (Fletcher-Reeves) Started\n";
50 std::cout << "========================================================\n";
51 }
52
53 double f_x;
54 std::array<double, N> g;
55
56 // [초기화] AutoDiff를 통해 초기 위치 x에서의 함수값(f_x)과 기울기(g)를 계산합니다.
57 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
58
59 // 첫 번째 탐색 방향(d)은 순수 경사 하강법과 동일하게 가장 가파른 역방향(-g)으로 설정합니다.
60 std::array<double, N> d;
61 for (size_t i = 0; i < N; ++i) {
62 d[i] = -g[i];
63 }
64
65 for (size_t iter = 1; iter <= max_iter; ++iter) {
66 // [Step 1] 현재 기울기 벡터의 크기(Norm의 제곱) 계산 및 수렴 확인
67 double g_norm_sq = 0.0;
68 for (size_t i = 0; i < N; ++i) {
69 g_norm_sq += g[i] * g[i];
70 }
71
72 // 기울기 크기(루트 g_norm_sq)가 지정된 허용 오차(tol)보다 작으면 극솟값에 도달했다고 판단합니다.
73 if (std::sqrt(g_norm_sq) < tol) {
74 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
75 break;
76 }
77
78 // [Step 2] Exact Line Search를 통한 최적의 보폭(alpha) 탐색
79 // 사용자가 고정된 학습률(Learning Rate)을 지정하지 않고, 해당 방향(d)으로 가장 많이 감소할 수 있는
80 // 최적의 전진 거리(alpha)를 수학적으로 찾아냅니다.
81
82 // 2-1. 브래킷 부호 변경(Bracket Sign Change)을 통해 방향 미분값이 0을 가로지르는(감소하다가 증가하는) 확실한 구간을 잡습니다.
83 auto bracket = LineSearch::bracket_sign_change<N>(f, x, d, 0.0, 0.001, 2.0, false);
84
85 // 2-2. 확보된 구간(bracket.first ~ bracket.second) 내에서 이분법(Bisection)을 수행하여 정확한 보폭 alpha를 산출합니다.
86 double alpha =
87 LineSearch::bisection<N>(f, x, d, bracket.first, bracket.second, 1e-4, false);
88
89 // [Step 3] 실제 파라미터(위치 x) 업데이트
90 // 현재 위치에서 최적 보폭(alpha)만큼 탐색 방향(d)으로 전진합니다.
91 for (size_t i = 0; i < N; ++i) {
92 x[i] += alpha * d[i];
93 }
94
95 // [Step 4] 새로운 위치에서의 함수값과 기울기(g_new) 계산
96 std::array<double, N> g_new;
97 AutoDiff::value_and_gradient<N>(f, x, f_x, g_new);
98
99 // [Step 5] Fletcher-Reeves 공식을 이용한 공액 비율(beta) 계산
100 // 이전 방향(d)의 정보를 얼마나 섞어서 다음 방향을 결정할지 그 비율을 계산합니다.
101 double g_new_norm_sq = 0.0;
102 for (size_t i = 0; i < N; ++i) {
103 g_new_norm_sq += g_new[i] * g_new[i];
104 }
105
106 // Fletcher-Reeves 식: beta = ||g_new||^2 / ||g||^2
107 // 새로운 기울기의 크기와 이전 기울기의 크기의 비율을 사용합니다.
108 double beta = g_new_norm_sq / g_norm_sq;
109
110 // [Step 6] 새로운 공액 탐색 방향(Conjugate Direction, d) 설정
111 // 새로운 방향은 현재의 가장 가파른 하강 방향(-g_new)에 이전 방향(d)을 beta 비율만큼 섞어 만들어집니다.
112 // 이를 통해 이전까지 탐색해온 계곡의 축(Axis) 정보를 잃지 않고 나아갈 수 있습니다.
113 for (size_t i = 0; i < N; ++i) {
114 d[i] = -g_new[i] + beta * d[i];
115 }
116
117 // 다음 루프의 연산을 위해 새로운 기울기(g_new)를 현재 기울기(g)로 덮어씁니다.
118 g = g_new;
119
120 // 진행 상태 주기적 로깅 (10번 반복마다 또는 첫 번째 반복 시 출력)
121 if (verbose && (iter % 10 == 0 || iter == 1)) {
122 std::cout << "[Iter " << std::setw(3) << iter << "] f(x): " << std::fixed
123 << std::setprecision(6) << f_x << "\n";
124 }
125 }
126
127 if (verbose) {
128 std::cout << "========================================================\n";
129 std::cout << " 🏁 Final Optimal Point: [" << x[0];
130 if constexpr (N > 1) std::cout << ", " << x[1];
131 if constexpr (N > 2) std::cout << ", ...";
132 std::cout << "]\n========================================================\n";
133 }
134
135 return x;
136 }
