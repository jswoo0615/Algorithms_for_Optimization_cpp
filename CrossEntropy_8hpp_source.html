84 template <size_t N, size_t M = 100, size_t M_ELITE = 10, typename Func>
85 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> mu_init,
86 std::array<double, N> sigma_sq_init,
87 size_t max_iter = 100, double tol = 1e-5,
88 uint32_t seed = 12345,
89 bool verbose = false) noexcept {
90 // 컴파일 타임에 차원 및 샘플링 개수(전체 샘플 수는 반드시 엘리트 수 이상) 검증
91 static_assert(N > 0, "Dimension N must be greater than 0");
92 static_assert(M >= M_ELITE, "Total samples M must be greater than or equal to M_ELITE");
93
94 // 최적화 시작 시간 측정
95 auto start_clock = std::chrono::high_resolution_clock::now();
96
97 // 진화하는 확률 분포의 평균(mu)과 분산(sigma_sq)을 저장할 메모리 정렬 배열
98 alignas(64) std::array<double, N> mu = mu_init;
99 alignas(64) std::array<double, N> sigma_sq = sigma_sq_init;
100
101 // [정적 할당] 한 세대의 모든 해를 담을 모집단(Population) 배열
102 // thread_local을 사용하여 멀티스레딩 환경에서 스레드별로 독립적인 스택/BSS 영역 활용 (힙
103 // 할당 원천 차단)
104 static thread_local std::array<Sample<N>, M> population;
105
106 // [난수 생성기 설정] 오프라인 튜닝 시 재현성(Reproducibility)을 보장하기 위한 고정 시드
107 // 주입
108 std::mt19937 gen(seed);
109 // 단일 표준 정규 분포 N(0, 1) 객체를 한 번만 생성하여 매 루프 재사용 (성능 최적화)
110 std::normal_distribution<double> standard_normal(0.0, 1.0);
111
112 if (verbose) {
113 std::cout << "========================================================\n";
114 std::cout << " 🎯 Cross-Entropy Method Started (Seed: " << seed << ")\n";
115 std::cout << "========================================================\n";
116 }
117
118 size_t iter = 0;
119 for (iter = 1; iter <= max_iter; ++iter) {
120 // [Step 1] 모집단(Population) 샘플링 및 평가 (O(M*N))
121 // Reparameterization Trick(재매개변수화 기법)을 사용하여 각 차원별 독립적인 정규
122 // 분포에서 샘플링합니다. 식: x = mu + standard_deviation * z (단, z ~ N(0, 1))
123 for (size_t p = 0; p < M; ++p) {
124 // 샘플의 각 파라미터를 하나의 z값 방향을 기준으로 전개하여 스칼라 연산 가속
125 // (경우에 따라 N차원 독립 샘플링이 필요하면 이 부분의 z 추출 위치를 수정할 수
126 // 있으나, 본 구현은 효율성을 위해 공통 난수 방향을 사용합니다)
127 double z = standard_normal(gen);
128#pragma omp simd
129 for (size_t i = 0; i < N; ++i) {
130 // std::sqrt(sigma_sq[i]) = 표준 편차(Standard Deviation)
131 // FMA(Fused Multiply-Add) 명령어로 a * b + c 를 한 사이클에 가속
132 population[p].x[i] = std::fma(std::sqrt(sigma_sq[i]), z, mu[i]);
133 }
134
135 // 실제 샘플링된 위치(x)에서 목적 함수(f)를 평가하여 적합도(y) 산출
136 population[p].y = f(population[p].x);
137 }
138
139 // [Step 2] 엘리트(Elite) 샘플 선별
140 // O(M log M)의 전체 정렬(std::sort) 대신, 상위 M_ELITE개만 찾아내는
141 // O(M log M_ELITE)의 부분 정렬(std::partial_sort)을 사용하여 성능을 최적화합니다.
142 std::partial_sort(population.begin(), population.begin() + M_ELITE, population.end());
143
144 // 다음 세대의 분포 업데이트를 위한 임시 배열 준비
145 alignas(64) std::array<double, N> new_mu = {0.0};
146 alignas(64) std::array<double, N> new_sigma_sq = {0.0};
147
148 // [Step 3.1] 엘리트 샘플 기반 새로운 평균(mu) 계산
149 // 상위 M_ELITE개의 샘플들의 위치(x)를 모두 더한 후, 개수로 나누어 새로운 탐색 중심점을
150 // 잡습니다.
151 for (size_t e = 0; e < M_ELITE; ++e) {
152#pragma omp simd
153 for (size_t i = 0; i < N; ++i) {
154 new_mu[i] += population[e].x[i];
155 }
156 }
157
158#pragma omp simd
159 for (size_t i = 0; i < N; ++i) {
160 new_mu[i] /= static_cast<double>(M_ELITE);
161 }
162
163 // [Step 3.2] 대각 공분산(Diagonal Covariance, 분산) 행렬 업데이트
164 // 엘리트 샘플들이 새로운 평균(new_mu)을 중심으로 얼마나 퍼져 있는지(분산)를 계산합니다.
165 double max_variance = 0.0;
166 for (size_t e = 0; e < M_ELITE; ++e) {
167#pragma omp simd
168 for (size_t i = 0; i < N; ++i) {
169 double diff = population[e].x[i] - new_mu[i];
170 // 편차의 제곱을 누적 합산 (분산 계산 공식)
171 new_sigma_sq[i] = std::fma(diff, diff, new_sigma_sq[i]);
172 }
173 }
174
175 // 분산의 평균을 구하고, 수렴 판정을 위해 모든 차원의 분산 중 가장 큰 값(max_variance)을
176 // 찾습니다.
177 for (size_t i = 0; i < N; ++i) {
178 new_sigma_sq[i] /= static_cast<double>(M_ELITE);
179 if (new_sigma_sq[i] > max_variance) {
180 max_variance = new_sigma_sq[i];
181 }
182 }
183
184 // 계산된 새로운 확률 분포 파라미터(평균, 분산)를 현재 상태로 덮어씁니다.
185 mu = new_mu;
186 sigma_sq = new_sigma_sq;
187
188 // 엘리트 집단의 최고 성적(population[0].y)과 가장 넓은 탐색 범위(max_variance) 출력
189 if (verbose && (iter % 10 == 0 || iter == 1)) {
190 std::cout << "[Iter " << std::setw(3) << iter << "] Best f(x): " << std::fixed
191 << std::setprecision(6) << population[0].y
192 << " | Max Var: " << max_variance << "\n";
193 }
194
195 // [Step 4] 수렴 조건 (Convergence Check) 확인
196 // 분산의 최대값(max_variance)이 허용 오차(tol) 미만으로 작아졌다면,
197 // 모든 샘플들이 한 점(평균 mu)에 몰려 붕괴(Collapse)된 상태이므로 탐색을 종료합니다.
198 if (max_variance < tol) {
199 if (verbose)
200 std::cout << " ↳ ✅ Converged (Variance collapsed) at Iteration: " << iter
201 << "\n";
202 break;
203 }
204 }
205
206 // 최적화 종료 후 소요 시간 산출
207 auto end_clock = std::chrono::high_resolution_clock::now();
208 auto duration =
209 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
210
211 if (verbose) {
212 std::cout << "========================================================\n";
213 std::cout << " 🏁 Final Optimal Point: [" << mu[0];
214 if constexpr (N > 1) std::cout << ", " << mu[1];
215 if constexpr (N > 2) std::cout << ", ...";
216 std::cout << "]\n========================================================\n";
217 }
218
219 // 최상위 우수 개체의 함수값(population[0].y)과 최종 분포의 중심점(mu)을 결과로 반환
220 return {mu, population[0].y, iter, duration.count()};
221 }
