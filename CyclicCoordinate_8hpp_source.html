37struct GoldenSectionStrategy {
48 template <size_t N, typename Func>
49 [[nodiscard]] std::array<double, N> operator()(Func f, const std::array<double, N>& x,
50 const std::array<double, N>& d) const noexcept {
51 // 1. 최솟값이 존재하는 구간 [a, b] 탐색
52 // 초기 step size는 1e-2, 확장 비율은 2.0으로 설정하여 최솟값이 포함된 구간을 찾습니다.
53 auto [a, b] = LineSearch::bracket_minimum<N>(f, x, d, 1e-2, 2.0, false);
54
55 // 2. 황금 분할 탐색으로 최적의 이동 보폭 (alpha) 계산
56 // 앞서 찾은 구간 [a, b] 내에서 허용 오차 1e-5 내로 들어올 때까지 황금 분할 탐색을
57 // 수행합니다.
58 double alpha = LineSearch::golden_section_search<N>(f, x, d, a, b, 1e-5, false);
59
60 // 3. 최적 위치로 이동 (FMA 연산 적용)
61 // x_new = x + alpha * d 공식을 FMA(Fused Multiply-Add)를 사용하여 빠르고 정확하게
62 // 계산합니다. 메모리 정렬(alignas(64))과 OpenMP SIMD 지시어를 사용하여 벡터화 연산을
63 // 최적화합니다.
64 alignas(64) std::array<double, N> x_new = {0.0};
65#pragma omp simd
66 for (size_t i = 0; i < N; ++i) {
67 x_new[i] = std::fma(alpha, d[i], x[i]);
68 }
69 return x_new;
70 }
71};

140 }

146 const double diff = x[i] - x_prime[i]; // (현재 위치 - 시작 위치)

148 }

193 template <size_t N, typename Func, typename LineSearchFunc = GoldenSectionStrategy>
194 [[nodiscard]] static OptimizationResultND<N> optimize_accelerated(
195 Func f, std::array<double, N> x_init, LineSearchFunc line_search = GoldenSectionStrategy{},
196 double tol = 1e-5, size_t max_iter = 500, bool verbose = false) noexcept {
197 static_assert(N > 0, "Dimension N must be greater than 0");
198
199 // 타이머 시작
200 auto start_clock = std::chrono::high_resolution_clock::now();
201
202 alignas(64) std::array<double, N> x = x_init;
203 const double tol_sq = tol * tol;
204 size_t iter = 0;
205
206 for (iter = 1; iter <= max_iter; ++iter) {
207 // 사이클 시작 시점의 위치 기록
208 alignas(64) std::array<double, N> x_prime = x;
209
210 // 1단계: 각 축 방향별로 순환하며 선 탐색 수행 (기본 Cyclic Coordinate Search와 동일)
211 for (size_t i = 0; i < N; ++i) {
212 alignas(64) std::array<double, N> d = {0.0};
213 d[i] = 1.0;
214 x = line_search(f, x, d);
215 }
216
217 // 2단계: 가속 방향(패턴 방향, Pattern Direction) 계산
218 // 이전 사이클 시작점(x_prime)에서 현재 점(x)으로 향하는 벡터를 구합니다.
219 alignas(64) std::array<double, N> d_acc = {0.0};
220#pragma omp simd
221 for (size_t i = 0; i < N; ++i) {
222 d_acc[i] = x[i] - x_prime[i];
223 }
224
225 // 3단계: 가속 방향으로 추가적인 선 탐색 수행
226 // 전체 진행 방향으로 크게 점프(이동)할 수 있어 좁은 협곡 등에서 빠른 수렴을 돕습니다.
227 x = line_search(f, x, d_acc);
228
229 // 4단계: 수렴 여부 확인을 위해 이동 거리 계산
230 double delta_sq = 0.0;
231#pragma omp simd
232 for (size_t i = 0; i < N; ++i) {
233 const double diff =
234 x[i] - x_prime[i]; // 한 사이클(가속 스텝 포함) 전체의 이동 거리
235 delta_sq = std::fma(diff, diff, delta_sq);
236 }
237
238 // 수렴 조건을 만족하면 루프 탈출
239 if (delta_sq < tol_sq) {
240 break;
241 }
242 }
243
244 // 실행 시간 측정 종료 및 반환
245 auto end_clock = std::chrono::high_resolution_clock::now();
246 auto duration =
247 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
248
249 return {x, f(x), iter, duration.count()};
250 }
