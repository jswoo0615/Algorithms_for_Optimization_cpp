26struct GoldenSectionStrategy {
27 template <size_t N, typename Func>
28 [[nodiscard]] std::array<double, N> operator()(Func f, const std::array<double, N>& x,
29 const std::array<double, N>& d) const noexcept {
30 // 1. 최솟값이 존재하는 구간 [a, b] 탐색
31 auto [a, b] = LineSearch::bracket_minimum<N>(f, x, d, 1e-2, 2.0, false);
32
33 // 2. 황금 분할 탐색으로 최적의 이동 보폭 (alpha) 계산
34 double alpha = LineSearch::golden_section_search<N>(f, x, d, a, b, 1e-5, false);
35
36 // 3. 최적 위치로 이동 (FMA 적용)
37 alignas(64) std::array<double, N> x_new = {0.0};
38#pragma omp simd
39 for (size_t i = 0; i < N; ++i) {
40 x_new[i] = std::fma(alpha, d[i], x[i]);
41 }
42 return x_new;
43 }
44};
