64 size_t md = depths[0];
65 for (size_t i = 1; i < N; ++i) {
66 if (depths[i] < md) {

68 }

82 [[nodiscard]] static constexpr std::array<double, N> denormalize(
83 const std::array<double, N>& x, const std::array<double, N>& a,
84 const std::array<double, N>& b) noexcept {
85 std::array<double, N> real_x = {0.0};
86 // SIMD 지시어를 통해 벡터화 연산 수행
87#pragma omp simd
88 for (size_t i = 0; i < N; ++i) {
89 // real_x[i] = x[i] * (b[i] - a[i]) + a[i]
90 // FMA(Fused Multiply-Add)를 사용하여 부동소수점 오차를 줄이고 연산 속도 향상
91 real_x[i] = std::fma(x[i], b[i] - a[i], a[i]);
92 }
93 return real_x;
94 }

111 template <typename Func>
112 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> a,
113 std::array<double, N> b,
114 double epsilon = 1e-4,
115 size_t max_iter = 100,
116 bool verbose = false) noexcept {
117 static_assert(N > 0, "Dimension N must be greater than 0"); // 최소 1차원 이상 필요
118 auto start_clock = std::chrono::high_resolution_clock::now(); // 타이머 시작
119
120 // 정적 메모리 풀(Static Memory Pool)
121 // 동적 메모리 할당(new, malloc)으로 인한 파편화나 지연을 막기 위해 thread_local로 큰 배열을 스택/데이터 영역에 할당
122 static thread_local std::array<Interval, MAX_INTERVALS> pool;
123 for (size_t i = 0; i < MAX_INTERVALS; ++i) {
124 pool[i].active = false; // 풀 초기화
125 }
126 size_t pool_size = 0; // 현재 풀에 활성화된(분할된) 구간의 개수
127
128 // 1. 초기 하이퍼큐브 설정: 정규화된 공간의 정중앙 점(0.5, 0.5, ...)
129 std::array<double, N> c_init = {0.0};
130 for (size_t i = 0; i < N; ++i) {
131 c_init[i] = 0.5;
132 }
133
134 // 첫 번째 구간을 풀에 등록하고 중앙점 함수값을 평가
135 pool[0].c = c_init;
136 pool[0].y = f(denormalize(c_init, a, b));
137 for (size_t i = 0; i < N; ++i) {
138 pool[0].depths[i] = 0; // 아직 어느 차원으로도 분할되지 않음
139 }
140 pool[0].active = true;
141 pool_size = 1;
142
143 // 현재까지 발견된 가장 작은 함수값과 그 때의 좌표 기록
144 double y_best = pool[0].y;
145 alignas(64) std::array<double, N> c_best = pool[0].c;
146
147 if (verbose) {
148 std::cout << "========================================================\n";
149 std::cout << " 🌐 DIRECT (Divided Rectangles) Search Started \n";
150 std::cout << "========================================================\n";
151 }
152
153 // 잠재적 최적 구간의 인덱스들을 저장할 버퍼 (std::vector 대신 정적 배열 사용)
154 static thread_local std::array<size_t, MAX_INTERVALS> S;
155 size_t iter = 0;
156
157 // 메인 최적화 루프
158 for (iter = 1; iter <= max_iter; ++iter) {
159 size_t S_size = 0;
160 double min_y_val = y_best;
161
162 // 2. 잠재적 최적 구간 (Potentially Optimal Intervals) 탐색 및 식별
163 // 간략화된 구현: 현재 최고 기록(y_best)에 가까운(epsilon 내에 있는) 구간들을 후보로 선정합니다.
164 // (정석적인 DIRECT는 립시츠 상수 하한을 이용해 볼록 껍질(Convex Hull)을 구하여 식별합니다)
165 for (size_t i = 0; i < pool_size; ++i) {
166 if (pool[i].active && pool[i].y <= min_y_val + epsilon) {
167 S[S_size++] = i;
168 }
169 }
170
171 // 3. 식별된 잠재적 최적 구간들을 분할(Division)
172 for (size_t s_idx = 0; s_idx < S_size; ++s_idx) {
173 size_t idx = S[s_idx]; // 분할할 부모 구간의 인덱스
174
175 // 한 번의 분할로 차원 당 2개(좌/우)의 새 구간이 생기므로, 메모리 풀 용량을 사전 검사
176 if (pool_size + 2 * N > MAX_INTERVALS) {
177 break;
178 }
179
180 Interval& parent = pool[idx];
181 size_t d = parent.min_depth(); // 분할을 진행할 깊이 (가장 적게 분할된 차원부터 분할)
182
183 // 각 차원을 순회하며 해당 차원이 최소 분할 상태(d)라면 분할을 수행
184 for (size_t dim = 0; dim < N; ++dim) {
185 if (parent.depths[dim] == d) {
186 // 3-등분할 때 중앙점의 이동 거리 계산
187 // 깊이 d일 때 한 변의 길이는 3^(-d)에 비례하므로 중앙점 간의 간격은 3^(-(d+1))이 됨
188 double delta = std::pow(3.0, -static_cast<double>(d + 1));
189
190 // 3.1 왼쪽(Left) 구간 생성 및 평가
191 Interval left = parent;
192 left.c[dim] -= delta; // 왼쪽으로 중심점 이동
193 left.depths[dim] += 1; // 해당 차원 분할 횟수 증가
194 left.y = f(denormalize(left.c, a, b)); // 목적 함수 평가
195
196 // 3.2 오른쪽(Right) 구간 생성 및 평가
197 Interval right = parent;
198 right.c[dim] += delta; // 오른쪽으로 중심점 이동
199 right.depths[dim] += 1; // 해당 차원 분할 횟수 증가
200 right.y = f(denormalize(right.c, a, b)); // 목적 함수 평가
201
202 // 새 구간들을 풀에 추가
203 pool[pool_size++] = left;
204 pool[pool_size++] = right;
205
206 // 부모 구간도 해당 차원에 대해 분할되었으므로 깊이 증가
207 parent.depths[dim] += 1;
208
209 // 최고 기록(y_best) 갱신 여부 확인
210 if (left.y < y_best) {
211 y_best = left.y;
212 c_best = left.c;
213 }
214 if (right.y < y_best) {
215 y_best = right.y;
216 c_best = right.c;
217 }
218 }
219 }
220 }
221
222 // 상태 출력
223 if (verbose && (iter % 10 == 0 || iter == 1)) {
224 std::cout << "[Iter " << std::setw(3) << iter << "] Best f(x): " << std::fixed
225 << std::setprecision(6) << y_best << " | Active Intervals: " << pool_size
226 << "\n";
227 }
228
229 // 4. 종료 조건 검사: 메모리 풀이 가득 찼으면 조기 종료
230 if (pool_size >= MAX_INTERVALS - 2 * N) {
231 if (verbose) std::cout << " ↳ ⚠️ MAX_INTERVALS capacity reached!\n";
232 break;
233 }
234 }
235
236 // 5. 최종 결과 반환
237 // 최고 기록을 달성한 정규화 좌표를 실제 범위 좌표로 변환
238 std::array<double, N> final_x = denormalize(c_best, a, b);
239
240 auto end_clock = std::chrono::high_resolution_clock::now();
241 auto duration =
242 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
243
244 if (verbose) {
245 std::cout << "========================================================\n";
246 std::cout << " 🏁 Final Global Optimal Point: [" << final_x[0];
247 if constexpr (N > 1) std::cout << ", " << final_x[1];
248 if constexpr (N > 2) std::cout << ", ...";
249 std::cout << "]\n========================================================\n";
250 }
251
252 return {final_x, y_best, iter, duration.count()};
253 }
