29template <size_t N, size_t MAX_INTERVALS = 10000>
30class DIRECT {
31 private:
32 struct Interval {
33 alignas(64) std::array<double, N> c; // 중앙점 좌표 (정규화된 공간 [0, 1]^N)
34 double y; // 함수값 f(c)
35 std::array<size_t, N> depths; // 차원별 분할 횟수
36 bool active; // 메모리 풀 사용 여부

38 [[nodiscard]] size_t min_depth() const noexcept {
39 size_t md = depths[0];
40 for (size_t i = 1; i < N; ++i) {
41 if (depths[i] < md) {
42 md = depths[i];
43 }
44 }
45 return md;
46 }
47 };

49 [[nodiscard]] static constexpr std::array<double, N> denormalize(
50 const std::array<double, N>& x, const std::array<double, N>& a,
51 const std::array<double, N>& b) noexcept {
52 std::array<double, N> real_x = {0.0};
53#pragma omp simd
54 for (size_t i = 0; i < N; ++i) {
55 // x[i] * (b[i] - a[i]) + a[i]
56 real_x[i] = std::fma(x[i], b[i] - a[i], a[i]);
57 }
58 return real_x;
59 }

64 template <typename Func>
65 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> a,
66 std::array<double, N> b,
67 double epsilon = 1e-4,
68 size_t max_iter = 100,
69 bool verbose = false) noexcept {
70 static_assert(N > 0, "Dimension N must be greater than 0");
71 auto start_clock = std::chrono::high_resolution_clock::now();
72
73 // 정적 메모리 풀 (RT 환경 스택 오버플로우 방지 및 스레드 안정성 확보)
74 static thread_local std::array<Interval, MAX_INTERVALS> pool;
75 for (size_t i = 0; i < MAX_INTERVALS; ++i) {
76 pool[i].active = false;
77 }
78 size_t pool_size = 0;
79
80 std::array<double, N> c_init = {0.0};
81 for (size_t i = 0; i < N; ++i) {
82 c_init[i] = 0.5;
83 }
84 pool[0].c = c_init;
85 pool[0].y = f(denormalize(c_init, a, b));
86
87 for (size_t i = 0; i < N; ++i) {
88 pool[0].depths[i] = 0;
89 }
90
91 pool[0].active = true;
92 pool_size = 1;
93
94 double y_best = pool[0].y;
95 alignas(64) std::array<double, N> c_best = pool[0].c;
96
97 if (verbose) {
98 std::cout << "========================================================\n";
99 std::cout << " 🌐 DIRECT (Divided Rectangles) Search Started \n";
100 std::cout << "========================================================\n";
101 }
102
103 // 동적 할당 std::vector를 대체하는 정적 인덱스 버퍼
104 static thread_local std::array<size_t, MAX_INTERVALS> S;
105 size_t iter = 0;
106 for (iter = 1; iter <= max_iter; ++iter) {
107 size_t S_size = 0;
108 double min_y_val = y_best;
109
110 // 잠재적 최적 구간 (Potentially Optimal Intervals) 탐색
111 for (size_t i = 0; i < pool_size; ++i) {
112 if (pool[i].active && pool[i].y <= min_y_val + epsilon) {
113 S[S_size++] = i;
114 }
115 }
116
117 for (size_t s_idx = 0; s_idx < S_size; ++s_idx) {
118 size_t idx = S[s_idx];
119
120 if (pool_size + 2 * N > MAX_INTERVALS) {
121 break;
122 }
123
124 Interval& parent = pool[idx];
125 size_t d = parent.min_depth();
126
127 for (size_t dim = 0; dim < N; ++dim) {
128 if (parent.depths[dim] == d) {
129 double delta = std::pow(3.0, -static_cast<double>(d + 1));
130
131 Interval left = parent;
132 left.c[dim] -= delta;
133 left.depths[dim] += 1;
134 left.y = f(denormalize(left.c, a, b));
135
136 Interval right = parent;
137 right.c[dim] += delta;
138 right.depths[dim] += 1;
139 right.y = f(denormalize(right.c, a, b));
140
141 pool[pool_size++] = left;
142 pool[pool_size++] = right;
143 parent.depths[dim] += 1;
144
145 if (left.y < y_best) {
146 y_best = left.y;
147 c_best = left.c;
148 }
149 if (right.y < y_best) {
150 y_best = right.y;
151 c_best = right.c;
152 }
153 }
154 }
155 }
156 if (verbose && (iter % 10 == 0 || iter == 1)) {
157 std::cout << "[Iter " << std::setw(3) << iter << "] Best f(x): " << std::fixed
158 << std::setprecision(6) << y_best << " | Active Intervals: " << pool_size
159 << "\n";
160 }
161
162 if (pool_size >= MAX_INTERVALS - 2 * N) {
163 if (verbose) std::cout << " ↳ ⚠️ MAX_INTERVALS capacity reached!\n";
164 break;
165 }
166 }
167
168 std::array<double, N> final_x = denormalize(c_best, a, b);
169
170 auto end_clock = std::chrono::high_resolution_clock::now();
171 auto duration =
172 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
173
174 if (verbose) {
175 std::cout << "========================================================\n";
176 std::cout << " 🏁 Final Global Optimal Point: [" << final_x[0];
177 if constexpr (N > 1) std::cout << ", " << final_x[1];
178 if constexpr (N > 2) std::cout << ", ...";
179 std::cout << "]\n========================================================\n";
180 }
181
182 return {final_x, y_best, iter, duration.count()};
183 }
