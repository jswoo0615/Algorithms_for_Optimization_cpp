49template <size_t N, size_t MAX_INTERVALS = 10000>
50class DIRECT {
51 private:
55 struct Interval {
56 alignas(64) std::array<double, N> c;
58 double y;
59 std::array<size_t, N> depths;
60 bool active;

68 [[nodiscard]] size_t min_depth() const noexcept {
69 size_t md = depths[0];
70 for (size_t i = 1; i < N; ++i) {
71 if (depths[i] < md) {
72 md = depths[i];
73 }
74 }
75 return md;
76 }
77 };

87 [[nodiscard]] static constexpr std::array<double, N> denormalize(
88 const std::array<double, N>& x, const std::array<double, N>& a,
89 const std::array<double, N>& b) noexcept {
90 std::array<double, N> real_x = {0.0};
91 // SIMD 지시어를 통해 벡터화 연산 수행
92#pragma omp simd
93 for (size_t i = 0; i < N; ++i) {
94 // real_x[i] = x[i] * (b[i] - a[i]) + a[i]
95 // FMA(Fused Multiply-Add)를 사용하여 부동소수점 오차를 줄이고 연산 속도 향상
96 real_x[i] = std::fma(x[i], b[i] - a[i], a[i]);
97 }
98 return real_x;
99 }

117 template <typename Func>
118 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> a,
119 std::array<double, N> b,
120 double epsilon = 1e-4,
121 size_t max_iter = 100,
122 bool verbose = false) noexcept {
123 static_assert(N > 0, "Dimension N must be greater than 0"); // 최소 1차원 이상 필요
124 auto start_clock = std::chrono::high_resolution_clock::now(); // 타이머 시작
125
126 // 정적 메모리 풀(Static Memory Pool)
127 // 동적 메모리 할당(new, malloc)으로 인한 파편화나 지연을 막기 위해 thread_local로 큰 배열을
128 // 스택/데이터 영역에 할당
129 static thread_local std::array<Interval, MAX_INTERVALS> pool;
130 for (size_t i = 0; i < MAX_INTERVALS; ++i) {
131 pool[i].active = false; // 풀 초기화
132 }
133 size_t pool_size = 0; // 현재 풀에 활성화된(분할된) 구간의 개수
134
135 // 1. 초기 하이퍼큐브 설정: 정규화된 공간의 정중앙 점(0.5, 0.5, ...)
136 std::array<double, N> c_init = {0.0};
137 for (size_t i = 0; i < N; ++i) {
138 c_init[i] = 0.5;
139 }
140
141 // 첫 번째 구간을 풀에 등록하고 중앙점 함수값을 평가
142 pool[0].c = c_init;
143 pool[0].y = f(denormalize(c_init, a, b));
144 for (size_t i = 0; i < N; ++i) {
145 pool[0].depths[i] = 0; // 아직 어느 차원으로도 분할되지 않음
146 }
147 pool[0].active = true;
148 pool_size = 1;
149
150 // 현재까지 발견된 가장 작은 함수값과 그 때의 좌표 기록
151 double y_best = pool[0].y;
152 alignas(64) std::array<double, N> c_best = pool[0].c;
153
154 if (verbose) {
155 std::cout << "========================================================\n";
156 std::cout << " 🌐 DIRECT (Divided Rectangles) Search Started \n";
157 std::cout << "========================================================\n";
158 }
159
160 // 잠재적 최적 구간의 인덱스들을 저장할 버퍼 (std::vector 대신 정적 배열 사용)
161 static thread_local std::array<size_t, MAX_INTERVALS> S;
162 size_t iter = 0;
163
164 // 메인 최적화 루프
165 for (iter = 1; iter <= max_iter; ++iter) {
166 size_t S_size = 0;
167 double min_y_val = y_best;
168
169 // 2. 잠재적 최적 구간 (Potentially Optimal Intervals) 탐색 및 식별
170 // 간략화된 구현: 현재 최고 기록(y_best)에 가까운(epsilon 내에 있는) 구간들을 후보로
171 // 선정합니다. (정석적인 DIRECT는 립시츠 상수 하한을 이용해 볼록 껍질(Convex Hull)을
172 // 구하여 식별합니다)
173 for (size_t i = 0; i < pool_size; ++i) {
174 if (pool[i].active && pool[i].y <= min_y_val + epsilon) {
175 S[S_size++] = i;
176 }
177 }
178
179 // 3. 식별된 잠재적 최적 구간들을 분할(Division)
180 for (size_t s_idx = 0; s_idx < S_size; ++s_idx) {
181 size_t idx = S[s_idx]; // 분할할 부모 구간의 인덱스
182
183 // 한 번의 분할로 차원 당 2개(좌/우)의 새 구간이 생기므로, 메모리 풀 용량을 사전
184 // 검사
185 if (pool_size + 2 * N > MAX_INTERVALS) {
186 break;
187 }
188
189 Interval& parent = pool[idx];
190 size_t d =
191 parent.min_depth(); // 분할을 진행할 깊이 (가장 적게 분할된 차원부터 분할)
192
193 // 각 차원을 순회하며 해당 차원이 최소 분할 상태(d)라면 분할을 수행
194 for (size_t dim = 0; dim < N; ++dim) {
195 if (parent.depths[dim] == d) {
196 // 3-등분할 때 중앙점의 이동 거리 계산
197 // 깊이 d일 때 한 변의 길이는 3^(-d)에 비례하므로 중앙점 간의 간격은
198 // 3^(-(d+1))이 됨
199 double delta = std::pow(3.0, -static_cast<double>(d + 1));
200
201 // 3.1 왼쪽(Left) 구간 생성 및 평가
202 Interval left = parent;
203 left.c[dim] -= delta; // 왼쪽으로 중심점 이동
204 left.depths[dim] += 1; // 해당 차원 분할 횟수 증가
205 left.y = f(denormalize(left.c, a, b)); // 목적 함수 평가
206
207 // 3.2 오른쪽(Right) 구간 생성 및 평가
208 Interval right = parent;
209 right.c[dim] += delta; // 오른쪽으로 중심점 이동
210 right.depths[dim] += 1; // 해당 차원 분할 횟수 증가
211 right.y = f(denormalize(right.c, a, b)); // 목적 함수 평가
212
213 // 새 구간들을 풀에 추가
214 pool[pool_size++] = left;
215 pool[pool_size++] = right;
216
217 // 부모 구간도 해당 차원에 대해 분할되었으므로 깊이 증가
218 parent.depths[dim] += 1;
219
220 // 최고 기록(y_best) 갱신 여부 확인
221 if (left.y < y_best) {
222 y_best = left.y;
223 c_best = left.c;
224 }
225 if (right.y < y_best) {
226 y_best = right.y;
227 c_best = right.c;
228 }
229 }
230 }
231 }
232
233 // 상태 출력
234 if (verbose && (iter % 10 == 0 || iter == 1)) {
235 std::cout << "[Iter " << std::setw(3) << iter << "] Best f(x): " << std::fixed
236 << std::setprecision(6) << y_best << " | Active Intervals: " << pool_size
237 << "\n";
238 }
239
240 // 4. 종료 조건 검사: 메모리 풀이 가득 찼으면 조기 종료
241 if (pool_size >= MAX_INTERVALS - 2 * N) {
242 if (verbose) std::cout << " ↳ ⚠️ MAX_INTERVALS capacity reached!\n";
243 break;
244 }
245 }
246
247 // 5. 최종 결과 반환
248 // 최고 기록을 달성한 정규화 좌표를 실제 범위 좌표로 변환
249 std::array<double, N> final_x = denormalize(c_best, a, b);
250
251 auto end_clock = std::chrono::high_resolution_clock::now();
252 auto duration =
253 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
254
255 if (verbose) {
256 std::cout << "========================================================\n";
257 std::cout << " 🏁 Final Global Optimal Point: [" << final_x[0];
258 if constexpr (N > 1) std::cout << ", " << final_x[1];
259 if constexpr (N > 2) std::cout << ", ...";
260 std::cout << "]\n========================================================\n";
261 }
262
263 return {final_x, y_best, iter, duration.count()};
264 }
265};
