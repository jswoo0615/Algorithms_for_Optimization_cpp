168 [[nodiscard]] static constexpr DualVec make_variable(T value, size_t index) noexcept {
169 DualVec res(value);
170 if (index < N) {
171 res.g[index] = T(1); // 자기 자신에 대한 편미분은 1
172 }
173 return res;
174 }

177 [[nodiscard]] constexpr DualVec operator-() const noexcept {
178 DualVec res;
179 res.v = -v;
180 for (size_t i = 0; i < N; ++i) {
181 res.g[i] = -g[i];
182 }
183 return res;
184 }

187 [[nodiscard]] constexpr DualVec operator+(const DualVec& rhs) const noexcept {
188 DualVec res;
189 res.v = v + rhs.v;
190 for (size_t i = 0; i < N; ++i) {
191 res.g[i] = g[i] + rhs.g[i];
192 }
193 return res;
194 }
195
197 [[nodiscard]] constexpr DualVec operator-(const DualVec& rhs) const noexcept {
198 DualVec res;
199 res.v = v - rhs.v;
200 for (size_t i = 0; i < N; ++i) {
201 res.g[i] = g[i] - rhs.g[i];
202 }
203 return res;
204 }

207 [[nodiscard]] constexpr DualVec operator*(const DualVec& rhs) const noexcept {
208 DualVec res;
209 res.v = v * rhs.v;
210 for (size_t i = 0; i < N; ++i) {
211 res.g[i] = g[i] * rhs.v + v * rhs.g[i];

353#define DUAL_MATH_OVERLOAD(funcName, derivativeExpr) \
354 template <typename T> \
355 inline Dual<T> funcName(const Dual<T>& u) { \
356 using std::funcName; \
357 T res_v = funcName(u.v); \
358 return {res_v, (derivativeExpr) * u.d}; \
359 } \
360 template <typename T, size_t N> \

362 using std::funcName; \
363 T res_v = funcName(u.v); \
364 DualVec<T, N> res(res_v); \
365 T factor = (derivativeExpr); \
366 for (size_t i = 0; i < N; ++i) { \
367 res.g[i] = factor * u.g[i]; \
368 } \
369 return res; \
370 }

373DUAL_MATH_OVERLOAD(sin, std::cos(u.v)) // sin'(x) = cos(x)

379DUAL_MATH_OVERLOAD(atan, T(1.0) / (T(1.0) + u.v * u.v)) // atan'(x) = 1 / (1 + x^2)

433 using std::atan2;

481#define CSD_MATH_OVERLOAD(funcName, derivativeExpr) \
482 template <typename T> \
483 inline std::complex<T> funcName(const std::complex<T>& u) { \
484 using std::funcName; \
485 T r = u.real(); \
486 T i = u.imag(); \
487 return std::complex<T>(funcName(r), i * (derivativeExpr)); \
488 }
489CSD_MATH_OVERLOAD(sin, std::cos(r))
490CSD_MATH_OVERLOAD(cos, -std::sin(r))
491CSD_MATH_OVERLOAD(tan, T(1.0) + std::pow(std::tan(r), 2.0))
492CSD_MATH_OVERLOAD(atan, T(1.0) / (T(1.0) + r * r))
493CSD_MATH_OVERLOAD(exp, std::exp(r))
494CSD_MATH_OVERLOAD(log, T(1.0) / r)
495CSD_MATH_OVERLOAD(tanh, T(1.0) - std::pow(std::tanh(r), 2.0))
496
500template <typename T>
501inline std::complex<T> pow(const std::complex<T>& u, double n) {
502 using std::pow;
503 T r = u.real(), i = u.imag();
504 T res_v = pow(r, n);
505 T deriv = (r <= T(1e-16) && n < 1.0) ? T(1.0) : (n * pow(r, n - 1.0));
506 return std::complex<T>(res_v, i * deriv);
507}
508
512template <typename T>
513inline std::complex<T> atan2(const std::complex<T>& y, const std::complex<T>& x) {
514 T yr = y.real(), yi = y.imag();
515 T xr = x.real(), xi = x.imag();
516 T den = xr * xr + yr * yr;
517
518 if (den <= T(1e-16)) {
519 return std::complex<T>(std::atan2(yr, xr), T(0.0));
520 }
521 T real_part = std::atan2(yr, xr);
522 T imag_part = std::atan2(xr * yi - yr * xi) / den; // CSD 확장 규칙 적용
523
524 return std::complex<T>(real_part, imag_part);
525}
