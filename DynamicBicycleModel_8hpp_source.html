16struct DynamicBicycleModel {
17 // 차량 파라미터 (임시값, 실차 데이터로 교체 가능)
18 double m = 1500.0; // 질량 (kg)

20 double Lf = 1.2; // 무게중심 - 전륜 거리 (m)
21 double Lr = 1.6; // 무게중심 - 후륜 거리 (m)
22 double Cf = 100000.0; // 전륜 코너링 강성 (N/rad)
23 double Cr = 100000.0; // 후륜 코너링 강성 (N/rad)
24
25 // Functor 오버로딩 (T는 double 또는 DualVec이 들어옴)
26 template <typename T>
27 StaticVector<T, 6> operator()(const StaticVector<T, 6>& x, const StaticVector<T, 2>& u) const {
28 using namespace ad; // ad::sin, ad::cos, ad::atan2 등을 사용하기 위함
29
30 // State 추출
31 T psi = x(2);
32 T vx = x(3);
33 T vy = x(4);
34 T r = x(5);
35
36 // Control 추출
37 T delta = u(0);
38 T a = u(1);
39
40 // 특이점 방어 (속도가 0에 가까울 때 Slip angle 발산 방지)
41 double vx_val = get_value(vx); // T가 double 이든 DualVec이든 순수 값만 추출
42 T vx_safe = (vx_val >= 0.1) ? vx : ((vx_val <= -0.1) ? vx : T(0.1));
43
44 // 타이어 슬립각 (Slip Angle) 계산
45 T alpha_f = delta - atan2(vy + T(Lf) * r, vx_safe);
46 T alpha_r = -atan2(vy - T(Lr) * r, vx_safe);
47
48 // 선형 타이어 모델 기반 횡력 (Lateral Force)
49 T Fyf = T(Cf) * alpha_f;
50 T Fyr = T(Cr) * alpha_r;
51
52 // 상태 변화율 (State Derivatives) dx/dt = f(x, u)
53 StaticVector<T, 6> x_dot;
54
55 x_dot(0) = vx * cos(psi) - vy * sin(psi); // X_dot
56 x_dot(1) = vx * sin(psi) + vy * cos(psi); // Y_dot
57 x_dot(2) = r; // psi_dot
58 x_dot(3) = a + r * vy; // vx_dot
59 x_dot(4) = (Fyf * cos(delta) + Fyr) / T(m) - r * vx; // vy_dot
60 x_dot(5) = (T(Lf) * Fyf * cos(delta) - T(Lr) * Fyr) / T(Iz); // r_dot
61
62 return x_dot;
63 }
