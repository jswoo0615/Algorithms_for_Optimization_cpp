40 // 역행렬기 (추후 StaticMatrix 레이어 또는 별도 Math 모듈로 분리 권장)
41 StaticMatrix<double, Nx, Nx> invert(StaticMatrix<double, Nx, Nx> mat) {
42 StaticMatrix<double, Nx, Nx> inv;
43 inv.set_zero();
44 for (size_t i = 0; i < Nx; ++i) inv(i, i) = 1.0;
45
46 for (size_t i = 0; i < Nx; ++i) {
47 double pivot = mat(i, i);
48 if (std::abs(pivot) < 1e-12) pivot = (pivot < 0) ? -1e-12 : 1e-12;
49 for (size_t j = 0; j < Nx; ++j) {
50 mat(i, j) /= pivot;
51 inv(i, j) /= pivot;
52 }
53 for (size_t k = 0; k < Nx; ++k) {
54 if (k != i) {
55 double factor = mat(k, i);
56 for (size_t j = 0; j < Nx; ++j) {
57 mat(k, j) -= factor * mat(i, j);
58 inv(k, j) -= factor * inv(i, j);
59 }
60 }
61 }
62 }
63 return inv;
64 }

66 template <typename Model>
67 void predict(Model& model, const StaticVector<double, Nu>& u, double dt) {
68 // 1. 상태 예측 (RK4)
69 x_est = integrator::step_rk4<Nx, Nu>(model, x_est, u, dt);
70
71 // 2. 비선형 동역학 자코비안 F 추출 (AD Engine 적용)
72 StaticMatrix<double, Nx, Nx> F;
73 using ADVar = DualVec<double, Nx>;
74 StaticVector<ADVar, Nx> x_dual;
75 StaticVector<ADVar, Nu> u_dual;
76
77 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(x_est(i), i);
78 for (size_t i = 0; i < Nu; ++i) u_dual(i) = ADVar(u(i));
79
80 StaticVector<ADVar, Nx> x_next_dual =
81 integrator::step_rk4<Nx, Nu>(model, x_dual, u_dual, dt);
82
83 for (size_t i = 0; i < Nx; ++i) {
84 for (size_t j = 0; j < Nx; ++j) {
85 F(i, j) = x_next_dual(i).g[j];
86 }
87 }
88
89 // 3. 공분산 예측 (3중 for문 제거 및 완벽한 수식화)
90 P = (F * P * F.transpose()) + Q;
91 }

93 void update(const StaticVector<double, Nx>& z) {
94 // 1. S = P + R
95 StaticMatrix<double, Nx, Nx> S = P + R;
96
97 // 2. 칼만 게인 K = P * S^-1
98 StaticMatrix<double, Nx, Nx> K = P * invert(S);
99
100 // 3. 잔차 y = z - x_est (각도 wrap-around 처리 포함)
101 StaticVector<double, Nx> y = z - x_est;
102 while (y(2) > M_PI) y(2) -= 2.0 * M_PI;
103 while (y(2) < -M_PI) y(2) += 2.0 * M_PI;
104
105 // 4. 상태 업데이트 x_est = x_est + K * y
106 x_est = x_est + (K * y);
107
108 // 5. 공분산 업데이트 P = (I - K) * P
109 StaticMatrix<double, Nx, Nx> I;
110 I.set_zero();
111 for (size_t i = 0; i < Nx; ++i) I(i, i) = 1.0;
112
113 P = (I - K) * P;
114 }
