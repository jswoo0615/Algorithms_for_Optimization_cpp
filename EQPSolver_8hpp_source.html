43 // LDLT_decompose는 행렬을 인플레이스 (In-place)로 덮어쓰며 파괴합니다.
44 // 따라서 매 solve 호출 시마다 이전 데이터의 찌꺼기를 완전히 지워야 합니다.
45 KKT_Matrix.set_zero();
46 KKT_rhs.set_zero();
47
48 // -------------------------------------------------------
49 // Step 1 : KKT_Matrix 블록 조립
50 // [P A^T]
51 // [A 0 ]
52 // -------------------------------------------------------
53 KKT_Matrix.insert_block(0, 0, P);
54 KKT_Matrix.insert_transposed_block(0, N_vars, A); // Zero-copy Transpose
55 KKT_Matrix.insert_block(N_vars, 0, A);
56 // 우하단 N_cons x N_cons 크기의 '0' 블록은 set_zero()로 인해 이미 0입니다
57
58 // -------------------------------------------------------
59 // Step 2 : RHS 블록 조립
60 // [ -q ]
61 // [ b ]
62 // -------------------------------------------------------
63 StaticVector<double, N_vars> neg_q = q * static_cast<double>(-1.0);
64 KKT_rhs.insert_block(0, 0, neg_q);
65 KKT_rhs.insert_block(N_vars, 0, b);
66

68 // Step 3 : 최적화 엔진 가동 (LDLT 분해)
69 // -------------------------------------------------------
70 // 주의 : KKT 행렬의 우하단이 0이므로 전체 행렬은 엄밀히 양정치 (PD)가 아닙니다.
71 // LDLT가 중간에 터진다면 KKT의 0 위치에 Regularization (예 : 1e-8 * I)이 필요할 수
72 // 있습니다.
73 if (!KKT_Matrix.LDLT_decompose()) {
