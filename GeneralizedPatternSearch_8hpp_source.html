80 double alpha = 1.0, double epsilon = 1e-5,
81 double gamma = 0.5,
82 size_t max_iter = 10000,
83 bool verbose = false) noexcept {
84 // 컴파일 타임 검증: 차원은 0보다 커야 하며,
85 // 공간 전체를 양수로 스패닝(Positive Spanning)하려면 최소 N+1개의 방향 벡터가 필수입니다.

87 static_assert(M >= N + 1, "A positive spanning set must have at least N+1 directions");
88

90 auto start_clock = std::chrono::high_resolution_clock::now();
91
92 // 64바이트 캐시 라인에 맞게 정렬하여 메모리 접근 속도 및 SIMD 벡터화 효율을 극대화
93 alignas(64) std::array<double, N> x = x_init;
94 double y = f(x); // 초기 위치에서의 목적 함수 값
95 size_t iter = 0;
96
97 if (verbose) {
98 std::cout << "========================================================\n";

101 }
102

104 // 보폭(alpha)이 허용 오차(epsilon)보다 크고, 최대 반복 횟수에 도달하지 않을 때까지 반복
105 while (alpha > epsilon && iter < max_iter) {
106 iter++;
107 bool improved = false; // 이번 iteration에서 업데이트(성공)가 일어났는지 추적
108
109 // 주어진 방향 집합(M개)을 순서대로 탐색
110 for (size_t i = 0; i < M; ++i) {
111 alignas(64) std::array<double, N> x_prime = {0.0};
112
113 // 탐색 위치 계산: x_prime = x + alpha * D[i]
114 // SIMD 지시어와 FMA(Fused Multiply-Add) 연산을 사용하여 부동소수점 오차를 줄이고
115 // 병렬로 빠르게 연산
116#pragma omp simd
117 for (size_t j = 0; j < N; ++j) {
118 x_prime[j] = std::fma(alpha, D[i][j], x[j]);
119 }
120
121 // 새로운 위치에서의 목적 함수값 평가
122 double y_prime = f(x_prime);
123
124 // 1. 기회주의적 업데이트 (Opportunistic Update)
125 // 만약 단 하나라도 현재(y)보다 더 나은(목적 함수값이 작은) 위치를 발견하면,
126 // 나머지 방향은 더 이상 평가하지 않고 즉시 위치를 이동합니다.
127 if (y_prime < y) {
128 x = x_prime;
129 y = y_prime;
130 improved = true;
131
132 // 2. 동적 정렬 (Dynamic Ordering)
133 // 방금 성공한 방향(D[i])이 다음번 탐색에서도 또 성공할 확률이 높다는 가정 하에
134 // 이 방향 벡터를 탐색 우선순위 1위(배열의 맨 앞, index 0)로
135 // 재배치(Shift)합니다. 이는 불필요한 함수 평가를 획기적으로 줄여주는 휴리스틱
136 // 기법입니다.
137 if (i > 0) {
138 alignas(64) std::array<double, N> best_d = D[i];
139 // 성공한 방향을 빼내고, 그 앞의 요소들을 한 칸씩 뒤로 밉니다.
140 for (size_t k = i; k > 0; --k) {
141 D[k] = D[k - 1];
142 }
143 D[0] = best_d; // 성공한 방향을 배열 맨 앞에 삽입
144 }
145
146 // 기회주의적 탐색이므로 한 번 성공하면 현재 루프를 빠져나가고 다음 step으로
147 // 넘어감
148 break;
149 }
150 }
151
152 // 3. 패턴 탐색 실패 처리 (Step size 축소)
153 // M개의 방향을 모두 탐색했지만 어느 곳으로 가도 목적 함수값이 줄어들지 않은 경우,
154 // 현재 지점이 골짜기의 바닥 근처이거나 국소 최적점(Local Minimum)에 가까워졌다는
155 // 뜻입니다. 이때는 보폭(alpha)에 축소 비율(gamma, 주로 0.5)을 곱해 탐색 반경을 세밀하게
156 // 좁힙니다.
157 if (!improved) {
158 alpha *= gamma;
159 }
160
161 // 상태 출력
162 // 실패해서 보폭이 줄어들었을 때 또는 특정 주기(100번)마다 진행 상황을 알립니다.
163 if (verbose && (!improved || iter % 100 == 0)) {
164 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
165 << std::setprecision(6) << y << " | alpha: " << alpha << "\n";
166 }
167 }
168
169 // 최적화 루프 종료 및 소요 시간 계산
170 auto end_clock = std::chrono::high_resolution_clock::now();
171 auto duration =
172 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
173
174 if (verbose) {
175 std::cout << "========================================================\n";
176 std::cout << " 🏁 Final Optimal Point: [" << x[0];
177 if constexpr (N > 1) std::cout << ", " << x[1];
178 if constexpr (N > 2) std::cout << ", ...";
179 std::cout << "]\n========================================================\n";
180 }
181 return {x, y, iter, duration.count()};
182 }
