34 template <size_t N, size_t M, typename Func>
35 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
36 std::array<std::array<double, N>, M> D,
37 double alpha = 1.0, double epsilon = 1e-5,
38 double gamma = 0.5,
39 size_t max_iter = 10000,
40 bool verbose = false) noexcept {
41 static_assert(N > 0, "Dimension N must be greater than 0");
42 static_assert(M >= N + 1, "A positive spanning set must have at least N+1 directions");
43
44 auto start_clock = std::chrono::high_resolution_clock::now();
45
46 alignas(64) std::array<double, N> x = x_init;
47 double y = f(x);
48 size_t iter = 0;
49
50 if (verbose) {
51 std::cout << "========================================================\n";
52 std::cout << " 🕸️ Generalized Pattern Search Started (alpha0=" << alpha << ")\n";
53 std::cout << "========================================================\n";
54 }
55
56 while (alpha > epsilon && iter < max_iter) {
57 iter++;
58 bool improved = false;
59
60 for (size_t i = 0; i < M; ++i) {
61 alignas(64) std::array<double, N> x_prime = {0.0};
62
63// 탐색 위치 계산 (SIMD + FMA)
64#pragma omp simd
65 for (size_t j = 0; j < N; ++j) {
66 x_prime[j] = std::fma(alpha, D[i][j], x[j]);
67 }
68
69 double y_prime = f(x_prime);
70
71 // 기회주의적 (Opportunistic) 업데이트
72 if (y_prime < y) {
73 x = x_prime;
74 y = y_prime;
75 improved = true;
76
77 // 동적 정렬 (Dynamic Ordering) : 성공한 방향을 맨 앞으로 시프트
78 if (i > 0) {
79 alignas(64) std::array<double, N> best_d = D[i];
80 for (size_t k = i; k > 0; --k) {
81 D[k] = D[k - 1];
82 }
83 D[0] = best_d;
84 }
85 break;
86 }
87 }
88
89 // M개의 방향 모두 실패 시 보폭 축소
90 if (!improved) {
91 alpha *= gamma;
92 }
93 if (verbose && (!improved || iter % 100 == 0)) {
94 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
95 << std::setprecision(6) << y << " | alpha: " << alpha << "\n";
96 }
97 }
98
99 auto end_clock = std::chrono::high_resolution_clock::now();
100 auto duration =
101 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
102
103 if (verbose) {
104 std::cout << "========================================================\n";
105 std::cout << " 🏁 Final Optimal Point: [" << x[0];
106 if constexpr (N > 1) std::cout << ", " << x[1];
107 if constexpr (N > 2) std::cout << ", ...";
108 std::cout << "]\n========================================================\n";
109 }
110 return {x, y, iter, duration.count()};
111 }
