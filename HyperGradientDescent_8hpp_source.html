22 template <size_t N, typename Func>
23 [[nodiscard]] static constexpr std::array<double, N> optimize(
24 Func f, std::array<double, N> x, double alpha_init = 0.001, double mu = 1e-6,
25 double tol = 1e-5, size_t max_iter = 50000, bool verbose = false) noexcept {
26 static_assert(N > 0, "Dimension N must be greater than 0.");
27
28 double alpha = alpha_init;
29 std::array<double, N> g_prev = {0.0};
30 std::array<double, N> x_valid = x; // 발산 방어용 백업 상태
31
32 const double tol_sq = tol * tol;
33
34 // [안전성 보장] 수치적 발산을 막기 위한 학습률 범위
35 constexpr double MIN_ALPHA = 1e-8;
36 constexpr double MAX_ALPHA = 0.005; // Rosenbrock과 같은 급경사를 위해 보수적 상한 적용
37 constexpr double MAX_GRAD_NORM = 100.0; // Gradient Clipping 임계값
38
39 for (size_t iter = 1; iter <= max_iter; ++iter) {
40 double f_x = 0.0;
41 std::array<double, N> g = {0.0};
42
43 // 1. Auto Differentiation (O(1) 할당)
44 AutoDiff::value_and_gradient(f, x, f_x, g);
45
46 // [Fail-safe] 목적 함수값이 NaN 또는 Inf로 발산한 경우, 직전의 유효한 x 반환
47 if (std::isnan(f_x) || std::isinf(f_x)) {
48 if (verbose)
49 std::cout << " ⚠️ Warning: Numerical explosion detected. Rolling back.\n";
50 return x_valid;
51 }
52 x_valid = x; // 현재 상태가 유효하므로 백업
53
54 double g_norm_sq = 0.0;
55 double g_dot_gprev = 0.0;
56
57// 2. 기울기 노름 및 내적 동시 계산 (FMA 가속)
58#pragma omp simd
59 for (size_t i = 0; i < N; ++i) {
60 g_norm_sq = std::fma(g[i], g[i], g_norm_sq);
61 g_dot_gprev = std::fma(g[i], g_prev[i], g_dot_gprev);
62 }
63
64 // 3. 종료 조건 검사
65 if (g_norm_sq < tol_sq) {
66 break;
67 }
68
69 // [안전성 보장] Gradient Clipping: 기울기가 비정상적으로 클 때 스케일링
70 double g_norm = std::sqrt(g_norm_sq);
71 double clip_scale = (g_norm > MAX_GRAD_NORM) ? (MAX_GRAD_NORM / g_norm) : 1.0;
72
73 // 4. 하이퍼그레디언트 업데이트
74 alpha += mu * (g_dot_gprev * clip_scale * clip_scale);
75 alpha = std::clamp(alpha, MIN_ALPHA, MAX_ALPHA);
76
77// 5. 파라미터 업데이트 (In-place)
78#pragma omp simd
79 for (size_t i = 0; i < N; ++i) {
80 g[i] *= clip_scale; // 클리핑 적용
81 x[i] -= alpha * g[i];
82 g_prev[i] = g[i];
83 }
84
85 if (verbose && (iter % 5000 == 0)) {
86 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed
87 << std::setprecision(6) << f_x << " | ||g||: " << g_norm
88 << " | alpha: " << alpha << "\n";
89 }
90 }
91 return x_valid;
92 }
