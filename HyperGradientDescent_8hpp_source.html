55 double tol = 1e-5, size_t max_iter = 50000, bool verbose = false) noexcept {
56 static_assert(N > 0, "Dimension N must be greater than 0.");
57
58 double alpha = alpha_init; // 현재 학습률 (동적으로 업데이트됨)
59 std::array<double, N> g_prev = {0.0}; // 이전 단계의 기울기를 저장하는 배열
60 std::array<double, N> x_valid = x; // 발산 시 복구하기 위한 안전 백업(Fail-safe) 상태
61
62 const double tol_sq = tol * tol; // 루트 연산을 피하기 위한 제곱 허용 오차
63
64 // [안전성 보장] 수치적 발산을 막기 위한 학습률(alpha)의 범위 설정
65 constexpr double MIN_ALPHA = 1e-8; // 학습률이 너무 작아지는 것 방지
66 constexpr double MAX_ALPHA = 0.005; // Rosenbrock과 같은 급경사 함수에서 발산하는 것을 막기 위한 보수적 상한치
67 constexpr double MAX_GRAD_NORM = 100.0; // Gradient Clipping을 위한 기울기 크기 최대 임계값
68
69 for (size_t iter = 1; iter <= max_iter; ++iter) {
70 double f_x = 0.0;
71 std::array<double, N> g = {0.0};
72
73 // 1. Auto Differentiation (자동 미분)
74 // 목적 함수의 현재 위치(x)에서의 값(f_x)과 기울기(g)를 계산합니다. (메모리 재할당 없음 O(1))
75 AutoDiff::value_and_gradient(f, x, f_x, g);
76
77 // [Fail-safe] 목적 함수값이 NaN(Not a Number) 또는 Inf(Infinity)로 발산한 경우,
78 // 최적화를 중단하고 가장 마지막에 확인된 유효한 위치(x_valid)를 반환합니다.
79 if (std::isnan(f_x) || std::isinf(f_x)) {
80 if (verbose)
81 std::cout << " ⚠️ Warning: Numerical explosion detected. Rolling back.\n";
82 return x_valid;
83 }
84 // 현재 상태가 정상(유효)하므로 백업을 갱신합니다.
85 x_valid = x;
86
87 double g_norm_sq = 0.0; // 현재 기울기의 L2 Norm 제곱
88 double g_dot_gprev = 0.0; // 현재 기울기(g)와 이전 기울기(g_prev)의 내적
89
90 // 2. 기울기 크기(Norm) 및 내적(Dot product) 동시 계산
91 // SIMD 및 FMA(Fused Multiply-Add) 명령어를 통해 하드웨어 가속을 유도합니다.
92#pragma omp simd
93 for (size_t i = 0; i < N; ++i) {
94 g_norm_sq = std::fma(g[i], g[i], g_norm_sq); // g_norm_sq += g[i] * g[i]
95 g_dot_gprev = std::fma(g[i], g_prev[i], g_dot_gprev); // g_dot_gprev += g[i] * g_prev[i]
96 }
97
98 // 3. 수렴(종료) 조건 검사
99 // 기울기의 크기 제곱이 목표 허용 오차 제곱보다 작다면 (즉, 극소점에 충분히 가까워졌다면) 종료
100 if (g_norm_sq < tol_sq) {
101 break;
102 }
103
104 // [안전성 보장] Gradient Clipping (기울기 자르기)
105 // 기울기가 비정상적으로 커서 매개변수가 우주로 날아가는(발산하는) 것을 방지합니다.
106 double g_norm = std::sqrt(g_norm_sq);
107 // 기울기 크기가 최대 허용치(MAX_GRAD_NORM)를 넘으면 그 비율만큼 축소(Scale down)합니다.
108 double clip_scale = (g_norm > MAX_GRAD_NORM) ? (MAX_GRAD_NORM / g_norm) : 1.0;
109
110 // 4. 하이퍼그레디언트(학습률 alpha) 업데이트
111 // g_dot_gprev (이전 기울기와 현재 기울기의 내적)이 양수이면, 같은 방향으로 가고 있으므로 보폭(alpha) 증가.
112 // 음수이면, 계곡을 지그재그로 넘나들고(Overshooting) 있다는 뜻이므로 보폭(alpha) 감소.
113 // 이때 클리핑된 스케일을 제곱하여 보폭 업데이트에도 안전성을 반영합니다.
114 alpha += mu * (g_dot_gprev * clip_scale * clip_scale);
115
116 // 학습률이 설정된 안전 범위 [MIN_ALPHA, MAX_ALPHA]를 벗어나지 않도록 제한(Clamp)합니다.
117 alpha = std::clamp(alpha, MIN_ALPHA, MAX_ALPHA);
118
119 // 5. 파라미터 업데이트 (In-place)
120 // 새로운 위치 = 현재 위치 - 학습률 * 기울기
121#pragma omp simd
122 for (size_t i = 0; i < N; ++i) {
123 g[i] *= clip_scale; // 클리핑(비율 축소)을 실제 기울기 벡터에 적용
124 x[i] -= alpha * g[i]; // 경사하강법 스텝 이동
125 g_prev[i] = g[i]; // 다음 반복(Iteration)을 위해 현재 기울기를 저장
126 }
127
128 // 일정 주기(5000회)마다 진행 상태를 콘솔에 출력합니다.
129 if (verbose && (iter % 5000 == 0)) {
130 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed
131 << std::setprecision(6) << f_x << " | ||g||: " << g_norm
132 << " | alpha: " << alpha << "\n";
133 }
134 }
135
136 // 탐색이 끝난 후 유효한 최적해를 반환
137 return x_valid;
138 }
