71 // Dual 변수 (Lagrange Multipliers)
72 StaticVector<double, (N_eq > 0 ? N_eq : 1)> y;
73 y.set_zero();
74 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)>
75 z;
76 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)>
77 s;
78
79 // 부등식 제약조건에 대한 초기화
80 for (size_t i = 0; i < N_ineq; ++i) {
81 z(static_cast<int>(i)) = 1.0; // 엄격히 양수여야 함 (Strictly feasible)
82 // 슬랙 변수 초기화 (최소 1.0을 보장하여 로그 장벽 함수의 정의 구역 내에 있도록 함)
83 s(static_cast<int>(i)) = std::max(1.0, b_ineq(static_cast<int>(i)));
84 }

86 // KKT 시스템의 전체 크기: Primal 변수 + 등식 Dual 변수 + 부등식 Dual 변수

88
89 // Newton Step을 계산하기 위한 KKT 행렬(K)과 우변 벡터(rhs), 그리고 해를 담을 벡터(delta)
90 StaticMatrix<double, (KKT_size > 0 ? KKT_size : 1), (KKT_size > 0 ? KKT_size : 1)> K;
91 StaticVector<double, (KKT_size > 0 ? KKT_size : 1)> rhs;
92 StaticVector<double, (KKT_size > 0 ? KKT_size : 1)> delta;
93
94 // 메인 IPM 반복 루프
95 for (int iter = 0; iter < max_iter; ++iter) {
96 // 1. KKT 잔여물 (Residuals) 계산
97 // 1.1 라그랑지안(Lagrangian)의 그래디언트 잔여물 (r_L = P*x + q + A_eq^T*y +
98 // A_ineq^T*z)
99 StaticVector<double, N_vars> r_L = (P * x) + q;
100 if constexpr (N_eq > 0) r_L = r_L + (A_eq.transpose() * y);
101 if constexpr (N_ineq > 0) r_L = r_L + (A_ineq.transpose() * z);
102
103 // 1.2 등식 제약조건 잔여물 (r_eq = A_eq*x - b_eq)
104 StaticVector<double, (N_eq > 0 ? N_eq : 1)> r_eq;
105 if constexpr (N_eq > 0) r_eq = (A_eq * x) - b_eq;
106
107 // 1.3 부등식 제약조건 잔여물 (r_ineq = A_ineq*x + s - b_ineq)
108 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> r_ineq;
109 if constexpr (N_ineq > 0) r_ineq = (A_ineq * x) + s - b_ineq;
110
111 // 2. Duality Gap 및 수렴 조건 검사
112 double gap = 0.0; // 상보성 여유(Complementarity slackness) 갭 (s^T * z / N_ineq)
113 if constexpr (N_ineq > 0) {
114 for (size_t i = 0; i < N_ineq; ++i)
115 gap += s(static_cast<int>(i)) * z(static_cast<int>(i));
116 gap /= static_cast<double>(N_ineq); // 평균 갭 계산
117 }
118
119 // 모든 잔여물의 최대 절댓값(Infinity Norm)을 구함
120 double res_norm = 0.0;
121 for (size_t i = 0; i < N_vars; ++i)
122 res_norm = std::max(res_norm, std::abs(r_L(static_cast<int>(i))));
123 if constexpr (N_eq > 0)
124 for (size_t i = 0; i < N_eq; ++i)
125 res_norm = std::max(res_norm, std::abs(r_eq(static_cast<int>(i))));
126 if constexpr (N_ineq > 0)
127 for (size_t i = 0; i < N_ineq; ++i)
128 res_norm = std::max(res_norm, std::abs(r_ineq(static_cast<int>(i))));
129
130 // 잔여물과 Duality Gap이 모두 허용 오차 이내로 들어오면 수렴한 것으로 판단
131 if (res_norm < tol && gap < tol) {
132 return true;
133 }
134
135 // 3. 중심 경로 매개변수 설정 (Centering Parameter)
136 // Barrier parameter(mu)를 점진적으로 줄이기 위한 설정 (보통 Mehrotra's
137 // predictor-corrector를 쓰지만 여기서는 고정된 sigma 사용)
138 double sigma = 0.1;
139 double mu_target = sigma * gap;
140
141 // KKT 행렬 및 우변 초기화
142 K.set_zero();
143 rhs.set_zero();
144
145 // 4. Augmented KKT System (K * delta = rhs) 구성
146 // 4.1 행렬 K의 좌상단: P 행렬 복사 및 정규화
147 for (size_t i = 0; i < N_vars; ++i) {
148 for (size_t j = 0; j < N_vars; ++j)
149 K(static_cast<int>(i), static_cast<int>(j)) =
150 P(static_cast<int>(i), static_cast<int>(j));
151
152 // [Architect's Armor 1] 티호노프 정규화 (Tikhonov Regularization)
153 // P가 양의 정부호가 아닐 경우나 특이점(Singularity) 도달 시 역행렬 계산이 폭발하는
154 // 것을 방지
155 K(static_cast<int>(i), static_cast<int>(i)) += 1e-6;
156 }
157
158 // 4.2 행렬 K: 등식 제약조건 블록 할당 (A_eq, A_eq^T)
159 if constexpr (N_eq > 0) {
160 for (size_t i = 0; i < N_eq; ++i) {
161 for (size_t j = 0; j < N_vars; ++j) {
162 K(static_cast<int>(i + N_vars), static_cast<int>(j)) =
163 A_eq(static_cast<int>(i), static_cast<int>(j));
164 K(static_cast<int>(j), static_cast<int>(i + N_vars)) =
165 A_eq(static_cast<int>(i), static_cast<int>(j));
166 }
167 }
168 }
169
170 // 4.3 행렬 K: 부등식 제약조건 블록 할당 (A_ineq, A_ineq^T) 및 대각 블록 보정 (-s/z)
171 if constexpr (N_ineq > 0) {
172 for (size_t i = 0; i < N_ineq; ++i) {
173 for (size_t j = 0; j < N_vars; ++j) {
174 K(static_cast<int>(i + N_vars + N_eq), static_cast<int>(j)) =
175 A_ineq(static_cast<int>(i), static_cast<int>(j));
176 K(static_cast<int>(j), static_cast<int>(i + N_vars + N_eq)) =
177 A_ineq(static_cast<int>(i), static_cast<int>(j));
178 }
179 // Slack 변수의 역수 계산 시 z가 0에 가까워져 0으로 나누어지는 오류(Division by
180 // Zero) 방지
181 double z_safe = std::max(z(static_cast<int>(i)), 1e-12);
182 // 행렬의 우하단 블록에 부등식 정보(-S^-1 * Z 의 역방향 반영인 -S/Z 로 스케일링)
183 // 반영
184 K(static_cast<int>(i + N_vars + N_eq), static_cast<int>(i + N_vars + N_eq)) =
185 -s(static_cast<int>(i)) / z_safe;
186 }
187 }
188
189 // 4.4 우변(rhs) 벡터 구성
190 for (size_t i = 0; i < N_vars; ++i)
191 rhs(static_cast<int>(i)) = -r_L(static_cast<int>(i)); // Primal 업데이트 요구량
192 if constexpr (N_eq > 0)
193 for (size_t i = 0; i < N_eq; ++i)
194 rhs(static_cast<int>(i + N_vars)) =
195 -r_eq(static_cast<int>(i)); // 등식 제약 조건 충족 요구량
196 if constexpr (N_ineq > 0) {
197 for (size_t i = 0; i < N_ineq; ++i) {
198 // 상보성 조건 잔여물 (Complementarity condition): s * z - mu_target
199 double r_c = s(static_cast<int>(i)) * z(static_cast<int>(i)) - mu_target;
200 double z_safe = std::max(z(static_cast<int>(i)), 1e-12);
201 // 부등식 제약조건 및 상보성 보정 반영
202 rhs(static_cast<int>(i + N_vars + N_eq)) =
203 -r_ineq(static_cast<int>(i)) + r_c / z_safe;
204 }
205 }

207 // 5. 가우스 소거법(Gaussian Elimination)을 이용한 선형 시스템(K_aug * delta = rhs) 풀이
208 StaticMatrix<double, (KKT_size > 0 ? KKT_size : 1), (KKT_size > 0 ? KKT_size : 1)>
209 K_aug = K;
210 delta = rhs;
211 constexpr size_t N_kkt = (KKT_size > 0 ? KKT_size : 1);
212
213 // 전방 소거(Forward Elimination) 과정 (부분 피벗팅 적용)
214 for (size_t i = 0; i < N_kkt; ++i) {
215 size_t pivot = i;
216 double max_val = std::abs(K_aug(static_cast<int>(i), static_cast<int>(i)));
217 for (size_t j = i + 1; j < N_kkt; ++j) {
218 if (std::abs(K_aug(static_cast<int>(j), static_cast<int>(i))) > max_val) {
219 max_val = std::abs(K_aug(static_cast<int>(j), static_cast<int>(i)));
220 pivot = j;
221 }
222 }
223
224 // 수치적 불안정성 방지: 피벗 값이 너무 작으면 해당 행의 소거를 건너뜀 (정규화 덕에
225 // 거의 발생 안함)
226 if (max_val < 1e-12) {
227 continue;
228 }
229
230 // 행 교환(Row Swap) 수행
231 if (pivot != i) {
232 for (size_t j = i; j < N_kkt; ++j) {
233 double tmp = K_aug(static_cast<int>(i), static_cast<int>(j));
234 K_aug(static_cast<int>(i), static_cast<int>(j)) =
235 K_aug(static_cast<int>(pivot), static_cast<int>(j));
236 K_aug(static_cast<int>(pivot), static_cast<int>(j)) = tmp;
237 }
238 double tmp_d = delta(static_cast<int>(i));
239 delta(static_cast<int>(i)) = delta(static_cast<int>(pivot));
240 delta(static_cast<int>(pivot)) = tmp_d;
241 }
242
243 // 행 뺄셈 연산 수행
244 for (size_t j = i + 1; j < N_kkt; ++j) {
245 double factor = K_aug(static_cast<int>(j), static_cast<int>(i)) /
246 K_aug(static_cast<int>(i), static_cast<int>(i));
247 for (size_t k = i; k < N_kkt; ++k) {
248 K_aug(static_cast<int>(j), static_cast<int>(k)) -=
249 factor * K_aug(static_cast<int>(i), static_cast<int>(k));
250 }
251 delta(static_cast<int>(j)) -= factor * delta(static_cast<int>(i));
252 }
253 }
254
255 // 후방 대입(Backward Substitution)을 통한 해 도출
256 for (int i = static_cast<int>(N_kkt) - 1; i >= 0; --i) {
257 for (size_t j = i + 1; j < N_kkt; ++j) {
258 delta(i) -= K_aug(i, static_cast<int>(j)) * delta(static_cast<int>(j));
259 }
260 delta(i) /= (K_aug(i, i) != 0.0 ? K_aug(i, i) : 1e-12);
261 }
262
263 // 6. 계산된 델타(Delta) 분리 및 방향 벡터(Step Directions) 설정
264 StaticVector<double, N_vars> dx; // Primal 업데이트
265 StaticVector<double, (N_eq > 0 ? N_eq : 1)> dy; // 등식 Dual 업데이트
266 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> dz; // 부등식 Dual 업데이트
267 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> ds; // Slack 변수 업데이트
268
269 for (size_t i = 0; i < N_vars; ++i) {
270 // [안전 장치] IPM 내부 클램핑 (1차 방어)
271 // 해가 한 번의 스텝에서 비정상적으로 튀는 것(Overshooting)을 방지
272 double val = delta(static_cast<int>(i));
273 if (val > 10.0) val = 10.0;
274 if (val < -10.0) val = -10.0;
275 dx(static_cast<int>(i)) = val;
276 }
277 if constexpr (N_eq > 0)
278 for (size_t i = 0; i < N_eq; ++i)
279 dy(static_cast<int>(i)) = delta(static_cast<int>(i + N_vars));
280
281 // Slack 변수의 방향(ds) 계산: ds = -r_ineq - A_ineq * dx
282 if constexpr (N_ineq > 0) {
283 for (size_t i = 0; i < N_ineq; ++i) {
284 dz(static_cast<int>(i)) = delta(static_cast<int>(i + N_vars + N_eq));
285 double Adx = 0.0;
286 for (size_t j = 0; j < N_vars; ++j)
287 Adx += A_ineq(static_cast<int>(i), static_cast<int>(j)) *
288 dx(static_cast<int>(j));
289 ds(static_cast<int>(i)) = -r_ineq(static_cast<int>(i)) - Adx;
290 }
291 }
292
293 // 7. Step Size (보폭) 계산 (Fraction-to-the-boundary rule)
294 // Primal 변수와 Dual 변수가 항상 양의 제약(s > 0, z > 0)을 만족하도록 보폭을 제한
295 double alpha_prim = 1.0;
296 double alpha_dual = 1.0;
297 constexpr double tau = 0.995; // 경계에 너무 가까이 붙는 것을 막기 위한 백오프 파라미터
298
299 if constexpr (N_ineq > 0) {
300 for (size_t i = 0; i < N_ineq; ++i) {
301 // ds < 0 일 때, s + alpha*ds >= (1-tau)*s 를 만족하는 최대 alpha_prim 탐색
302 if (ds(static_cast<int>(i)) < 0.0)
303 alpha_prim = std::min(
304 alpha_prim, -tau * s(static_cast<int>(i)) / ds(static_cast<int>(i)));
305 // dz < 0 일 때, z + alpha*dz >= (1-tau)*z 를 만족하는 최대 alpha_dual 탐색
306 if (dz(static_cast<int>(i)) < 0.0)
307 alpha_dual = std::min(
308 alpha_dual, -tau * z(static_cast<int>(i)) / dz(static_cast<int>(i)));
309 }
310 }
311
312 // 8. 상태 변수 업데이트 (Update Variables)
313 x = x + (dx * alpha_prim);
314 if constexpr (N_eq > 0) y = y + (dy * alpha_dual);
315 if constexpr (N_ineq > 0) {
316 z = z + (dz * alpha_dual);
317 s = s + (ds * alpha_prim); // Slack 변수는 Primal 스텝 사이즈 적용
318 }
319 }
320
321 // [Architect's Safe Return]
322 // 타임아웃(최대 반복 횟수 도달)이 발생하더라도, 그동안 계산된 최선의 해(x)를 반환하여
323 // 이를 호출하는 상위 루프(예: SQP)가 완전히 무너지지 않도록 방어적인 설계를 취함
324 return true;
325 }
