32 bool solve(StaticVector<double, N_vars>& x, int max_iter = 50, double tol = 1e-3) {
33 StaticVector<double, (N_eq > 0 ? N_eq : 1)> y;
34 y.set_zero();
35 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> z;
36 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> s;
37
38 for (size_t i = 0; i < N_ineq; ++i) {
39 z(static_cast<int>(i)) = 1.0;
40 s(static_cast<int>(i)) = std::max(1.0, b_ineq(static_cast<int>(i)));
41 }

45 StaticVector<double, (KKT_size > 0 ? KKT_size : 1)> rhs;
46 StaticVector<double, (KKT_size > 0 ? KKT_size : 1)> delta;
47
48 for (int iter = 0; iter < max_iter; ++iter) {
49 StaticVector<double, N_vars> r_L = (P * x) + q;
50 if constexpr (N_eq > 0) r_L = r_L + (A_eq.transpose() * y);
51 if constexpr (N_ineq > 0) r_L = r_L + (A_ineq.transpose() * z);
52
53 StaticVector<double, (N_eq > 0 ? N_eq : 1)> r_eq;
54 if constexpr (N_eq > 0) r_eq = (A_eq * x) - b_eq;
55
56 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> r_ineq;
57 if constexpr (N_ineq > 0) r_ineq = (A_ineq * x) + s - b_ineq;
58
59 double gap = 0.0;
60 if constexpr (N_ineq > 0) {
61 for (size_t i = 0; i < N_ineq; ++i)
62 gap += s(static_cast<int>(i)) * z(static_cast<int>(i));
63 gap /= static_cast<double>(N_ineq);
64 }
65
66 double res_norm = 0.0;
67 for (size_t i = 0; i < N_vars; ++i)
68 res_norm = std::max(res_norm, std::abs(r_L(static_cast<int>(i))));
69 if constexpr (N_eq > 0)
70 for (size_t i = 0; i < N_eq; ++i)
71 res_norm = std::max(res_norm, std::abs(r_eq(static_cast<int>(i))));
72 if constexpr (N_ineq > 0)
73 for (size_t i = 0; i < N_ineq; ++i)
74 res_norm = std::max(res_norm, std::abs(r_ineq(static_cast<int>(i))));
75
76 if (res_norm < tol && gap < tol) {
77 return true;
78 }

81 double mu_target = sigma * gap;

83 K.set_zero();
84 rhs.set_zero();

86 for (size_t i = 0; i < N_vars; ++i) {
87 for (size_t j = 0; j < N_vars; ++j)
88 K(static_cast<int>(i), static_cast<int>(j)) =
89 P(static_cast<int>(i), static_cast<int>(j));
90 // [Architect's Armor 1] 티호노프 정규화 (Tikhonov Regularization)
91 // 역행렬 분해 시 0으로 나누어지는 특이점(Singularity) 폭발을 영구히 차단
92 K(static_cast<int>(i), static_cast<int>(i)) += 1e-6;
93 }
94
95 if constexpr (N_eq > 0) {
96 for (size_t i = 0; i < N_eq; ++i) {
97 for (size_t j = 0; j < N_vars; ++j) {
98 K(static_cast<int>(i + N_vars), static_cast<int>(j)) =
99 A_eq(static_cast<int>(i), static_cast<int>(j));
100 K(static_cast<int>(j), static_cast<int>(i + N_vars)) =
101 A_eq(static_cast<int>(i), static_cast<int>(j));
102 }
103 }
104 }

106 if constexpr (N_ineq > 0) {
107 for (size_t i = 0; i < N_ineq; ++i) {
108 for (size_t j = 0; j < N_vars; ++j) {
109 K(static_cast<int>(i + N_vars + N_eq), static_cast<int>(j)) =
110 A_ineq(static_cast<int>(i), static_cast<int>(j));
111 K(static_cast<int>(j), static_cast<int>(i + N_vars + N_eq)) =
112 A_ineq(static_cast<int>(i), static_cast<int>(j));
113 }
114 // Z가 0이 되어 폭발하는 것 방지
115 double z_safe = std::max(z(static_cast<int>(i)), 1e-12);
116 K(static_cast<int>(i + N_vars + N_eq), static_cast<int>(i + N_vars + N_eq)) =
117 -s(static_cast<int>(i)) / z_safe;
118 }

123 if constexpr (N_eq > 0)

125 rhs(static_cast<int>(i + N_vars)) = -r_eq(static_cast<int>(i));
126 if constexpr (N_ineq > 0) {
127 for (size_t i = 0; i < N_ineq; ++i) {
128 double r_c = s(static_cast<int>(i)) * z(static_cast<int>(i)) - mu_target;
129 double z_safe = std::max(z(static_cast<int>(i)), 1e-12);
130 rhs(static_cast<int>(i + N_vars + N_eq)) =

137 delta = rhs;
138 constexpr size_t N_kkt = (KKT_size > 0 ? KKT_size : 1);

143 for (size_t j = i + 1; j < N_kkt; ++j) {
144 if (std::abs(K_aug(static_cast<int>(j), static_cast<int>(i))) > max_val) {
145 max_val = std::abs(K_aug(static_cast<int>(j), static_cast<int>(i)));
146 pivot = j;
147 }
148 }

152 continue;
153 }

156 for (size_t j = i; j < N_kkt; ++j) {
157 double tmp = K_aug(static_cast<int>(i), static_cast<int>(j));
158 K_aug(static_cast<int>(i), static_cast<int>(j)) =
159 K_aug(static_cast<int>(pivot), static_cast<int>(j));
160 K_aug(static_cast<int>(pivot), static_cast<int>(j)) = tmp;
161 }
162 double tmp_d = delta(static_cast<int>(i));
163 delta(static_cast<int>(i)) = delta(static_cast<int>(pivot));
164 delta(static_cast<int>(pivot)) = tmp_d;
165 }

167 for (size_t j = i + 1; j < N_kkt; ++j) {
168 double factor = K_aug(static_cast<int>(j), static_cast<int>(i)) /
169 K_aug(static_cast<int>(i), static_cast<int>(i));
170 for (size_t k = i; k < N_kkt; ++k) {
171 K_aug(static_cast<int>(j), static_cast<int>(k)) -=
172 factor * K_aug(static_cast<int>(i), static_cast<int>(k));
173 }
174 delta(static_cast<int>(j)) -= factor * delta(static_cast<int>(i));
175 }
176 }

180 delta(i) -= K_aug(i, static_cast<int>(j)) * delta(static_cast<int>(j));
181 }
182 delta(i) /= (K_aug(i, i) != 0.0 ? K_aug(i, i) : 1e-12);
183 }

185 StaticVector<double, N_vars> dx;

188 StaticVector<double, (N_ineq > 0 ? N_ineq : 1)> ds;
189
190 for (size_t i = 0; i < N_vars; ++i) {
191 // IPM 내부 클램핑 (1차 방어)
192 double val = delta(static_cast<int>(i));
193 if (val > 10.0) val = 10.0;
194 if (val < -10.0) val = -10.0;
195 dx(static_cast<int>(i)) = val;
196 }
197 if constexpr (N_eq > 0)
198 for (size_t i = 0; i < N_eq; ++i)
199 dy(static_cast<int>(i)) = delta(static_cast<int>(i + N_vars));
200 if constexpr (N_ineq > 0) {
201 for (size_t i = 0; i < N_ineq; ++i) {
202 dz(static_cast<int>(i)) = delta(static_cast<int>(i + N_vars + N_eq));
203 double Adx = 0.0;
204 for (size_t j = 0; j < N_vars; ++j)
205 Adx += A_ineq(static_cast<int>(i), static_cast<int>(j)) *
206 dx(static_cast<int>(j));
207 ds(static_cast<int>(i)) = -r_ineq(static_cast<int>(i)) - Adx;
208 }
209 }

211 double alpha_prim = 1.0;
212 double alpha_dual = 1.0;
213 constexpr double tau = 0.995;
214
215 if constexpr (N_ineq > 0) {
216 for (size_t i = 0; i < N_ineq; ++i) {
217 if (ds(static_cast<int>(i)) < 0.0)
218 alpha_prim = std::min(
219 alpha_prim, -tau * s(static_cast<int>(i)) / ds(static_cast<int>(i)));
220 if (dz(static_cast<int>(i)) < 0.0)
221 alpha_dual = std::min(
222 alpha_dual, -tau * z(static_cast<int>(i)) / dz(static_cast<int>(i)));
223 }
224 }
225
226 x = x + (dx * alpha_prim);
227 if constexpr (N_eq > 0) y = y + (dy * alpha_dual);
228 if constexpr (N_ineq > 0) {
229 z = z + (dz * alpha_dual);
230 s = s + (ds * alpha_prim);
231 }
232 }
233
234 // [Architect's Safe Return] 타임아웃이 걸리더라도, 중간에 계산된 최선의 해(x)를 버리지 않고
235 // 반환하여 SQP를 살림
236 return true;
237 }
