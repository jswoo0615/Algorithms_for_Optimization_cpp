29 static KKT_Metrics evaluate_EQP(const StaticMatrix<double, N_vars, N_vars>& P,
30 const StaticVector<double, N_vars>& q,
31 const StaticMatrix<double, N_cons, N_vars>& A,
32 const StaticVector<double, N_cons>& b,
33 const StaticVector<double, N_vars>& u_opt,
34 const StaticVector<double, N_cons>& lambda_opt) {
35 KKT_Metrics metrics;
36
37 // 1. Stationarity : || ∇_u L ||_inf = || P*u + q + A^T*λ ||_inf <= TOL
38 StaticMatrix<double, N_vars, N_cons> A_T = A.transpose();
39 StaticVector<double, N_vars> grad_L = (P * u_opt) + q + (A_T * lambda_opt);
40
41 double stat_err = 0.0;
42 for (size_t i = 0; i < N_vars; ++i) {
43 stat_err = std::max(stat_err, std::abs(grad_L(i, 0)));
44 }
45 metrics.stationarity_error = stat_err;
46
47 // 2. Primal Feasibility: || Au - b ||_inf <= TOL
48 StaticVector<double, N_cons> eq_res = (A * u_opt) - b;
49 double prim_err = 0.0;
50 for (size_t i = 0; i < N_cons; ++i) {
51 prim_err = std::max(prim_err, std::abs(eq_res(i, 0)));
52 }
53 metrics.primal_feasibility_error = prim_err;
54
55 // 판정
56 metrics.is_optimal = (metrics.stationarity_error <= TOLERANCE) &&
57 (metrics.primal_feasibility_error <= TOLERANCE);
58
59 return metrics;
60 }
