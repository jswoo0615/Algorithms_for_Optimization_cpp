77 KKT_Metrics metrics;
78
79 // =========================================================================
80 // 1. Stationarity (정류성) 검사
81 // =========================================================================
82 // 라그랑지안 함수의 그래디언트(∇_u L)가 0벡터에 얼마나 가까운지(Infinity Norm) 측정합니다.
83 // 수식: || P*u + q + A^T*λ ||_inf <= TOLERANCE
84 StaticMatrix<double, N_vars, N_cons> A_T = A.transpose();
85 StaticVector<double, N_vars> grad_L = (P * u_opt) + q + (A_T * lambda_opt);
86
87 double stat_err = 0.0;
88 for (size_t i = 0; i < N_vars; ++i) {
89 // 잔차 벡터 원소들 중 가장 큰 절댓값을 찾음 (L-infinity Norm)
90 stat_err = std::max(stat_err, std::abs(grad_L(i, 0)));
91 }
92 metrics.stationarity_error = stat_err;
93
94 // =========================================================================
95 // 2. Primal Feasibility (원문제 실현가능성) 검사
96 // =========================================================================
97 // 도출된 해가 주어진 등식 제약 조건(A*u = b)을 얼마나 잘 만족하는지 측정합니다.
98 // 수식: || A*u - b ||_inf <= TOLERANCE
99 StaticVector<double, N_cons> eq_res = (A * u_opt) - b;
100 double prim_err = 0.0;
101 for (size_t i = 0; i < N_cons; ++i) {
102 // 위반 정도의 가장 큰 절댓값을 찾음 (L-infinity Norm)
103 prim_err = std::max(prim_err, std::abs(eq_res(i, 0)));
104 }
105 metrics.primal_feasibility_error = prim_err;
106
107 // =========================================================================
108 // 3. 최종 최적성(Optimality) 판정
109 // =========================================================================
110 // 두 가지 오차가 모두 지정된 허용 오차(TOLERANCE) 이하이면 올바른 KKT 포인트(최적해)로
111 // 인정합니다.
112 metrics.is_optimal = (metrics.stationarity_error <= TOLERANCE) &&
113 (metrics.primal_feasibility_error <= TOLERANCE);
114
115 return metrics;
116 }
