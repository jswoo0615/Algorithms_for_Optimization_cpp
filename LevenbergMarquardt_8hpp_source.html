16template <size_t N, size_t M, typename Functor>
17SolverStatus solve_LM(StaticVector<double, N>& x_opt, const Functor& calc_residuals,
18 int max_iter = 50, double tol = 1e-6, double initial_lambda = 1e-3) {
19 using ADVar = DualVec<double, N>;
20 double lambda = initial_lambda;
21
22 for (int iter = 0; iter < max_iter; ++iter) {
23 // 1. DualVec 변수 초기화 (시드 주입)
24 StaticVector<ADVar, N> x_dual;
25 for (size_t i = 0; i < N; ++i) {
26 x_dual(static_cast<int>(i)) = ADVar::make_variable(x_opt(static_cast<int>(i)), i);
27 }
28
29 // 2. 잔차 벡터 F(x) 및 자코비안 J 동시 평가
30 StaticVector<ADVar, M> residuals_dual = calc_residuals(x_dual);
31
32 // 3. H = J^T * J와 g = J^T * F 조립 (명시적 루프로 캐시 최적화)
33 StaticMatrix<double, N, N> H;
34 StaticVector<double, N> g;
35 double current_cost = 0.0;
36
37 // H, g 0으로 초기화
38 for (size_t i = 0; i < N; ++i) {
39 g(static_cast<int>(i)) = 0.0;
40 for (size_t j = 0; j < N; ++j) {
41 H(static_cast<int>(i), static_cast<int>(j)) = 0.0;
42 }
43 }
44
45 for (size_t k = 0; k < M; ++k) {
46 double r_val = residuals_dual(static_cast<int>(k)).v;
47 current_cost += 0.5 * r_val * r_val;
48
49 for (size_t i = 0; i < N; ++i) {
50 double J_ki = residuals_dual(static_cast<int>(k)).g[i];
51 g(static_cast<int>(i)) += J_ki * r_val; // g = J^T * F
52
53 for (size_t j = 0; j < N; ++j) {
54 double J_kj = residuals_dual(static_cast<int>(k)).g[j];
55 H(static_cast<int>(i), static_cast<int>(j)) += J_ki * J_kj; // H = J^T * J
56 }
57 }
58 }
59
60 // 4. 수렴 판정 (Gradient의 최대 크기가 허용 오차 이내인가?)
61 double max_grad = 0.0;
62 for (size_t i = 0; i < N; ++i) {
63 if (std::abs(g(static_cast<int>(i))) > max_grad) {
64 max_grad = std::abs(g(static_cast<int>(i)));
65 }
66 }
67 if (max_grad < tol) {
68 return SolverStatus::SUCCESS;
69 }
70
71 // 5. LM Damping 추가 : H_lm = H + lambda * I (또는 대각성분 비례)
72 StaticMatrix<double, N, N> H_lm = H;
73 for (size_t i = 0; i < N; ++i) {
74 // H_lm(i, i) += lambda; // 기본적인 Identity Damping
75 H_lm(static_cast<int>(i), static_cast<int>(i)) +=
76 lambda * (H(static_cast<int>(i), static_cast<int>(i)) + 1e-6);
77 }
78
79 // 6. 선형 시스템 해 찾기 : H_lm * dx = -g
80 // H_lm은 대칭 양의 정부호 (SPD)이므로 LDLT 솔버를 사용
81 for (size_t i = 0; i < N; ++i) {
82 g(static_cast<int>(i)) = -g(static_cast<int>(i)); // -g
83 }
84 MathStatus m_status = linalg::LDLT_decompose(H_lm);
85 if (m_status != MathStatus::SUCCESS) {
86 // LDLT가 실패하면 Damping을 늘려 행렬의 조건수 (Condition number)를 개선
87 lambda *= 10.0;
88 continue;
89 }
90
91 StaticVector<double, N> dx = linalg::LDLT_solve(H_lm, g);
92
93 // 7. 업데이트 평가 (스텝을 밟았을 때 Cost가 줄어드는가?)
94 StaticVector<double, N> x_new;
95 for (size_t i = 0; i < N; ++i) {
96 x_new(static_cast<int>(i)) = x_opt(static_cast<int>(i)) + dx(static_cast<int>(i));
97 }
98
99 // 새 위치에서의 Cost 계산 (Value만 필요하므로 double 오버로딩 또는 Dual 그대로 사용)
100 StaticVector<ADVar, N> x_new_dual;
101 for (size_t i = 0; i < N; ++i) {
102 x_new_dual(static_cast<int>(i)) = ADVar(x_new(static_cast<int>(i)));
103 }
104
105 StaticVector<ADVar, M> new_residuals = calc_residuals(x_new_dual);
106 double new_cost = 0.0;
107 for (size_t k = 0; k < M; ++k) {
108 new_cost +=
109 0.5 * new_residuals(static_cast<int>(k)).v * new_residuals(static_cast<int>(k)).v;
110 }
111
112 if (new_cost < current_cost) {
113 // 성공적인 스텝 : x 업데이트 및 Damping 감소 (가우스-뉴턴에 가깝게)
114 for (size_t i = 0; i < N; ++i) {
115 x_opt(static_cast<int>(i)) = x_new(static_cast<int>(i));
116 }
117 lambda = std::max(1e-7, lambda / 10.0);
118 } else {
119 // 실패한 스텝 : x 유지, Damping 증가 (경사하강법에 가깝게)
120 lambda *= 10.0;
121 }
122 }
123 return SolverStatus::MAX_ITERATION_REACHED;
124}
