22template <size_t N, size_t M, typename Functor>
23inline SolverStatus solve_LM(StaticVector<double, N>& x_opt, const Functor& calc_residuals,
24 int max_iter = 50, double tol = 1e-6, double initial_lambda = 1e-3) {
25 static_assert(M >= N, "Residuals M must be greater than or equal to variables N");
26
27 using ADVar = DualVec<double, N>;
28 double lambda = initial_lambda;
29
30 // [Architect's Update] 메모리 호이스팅 (루프 외부로 메모리 블록을 통째로 이전)
31 StaticVector<ADVar, N> x_dual;
32 StaticMatrix<double, M, N> J_mat;
33 StaticVector<double, M> F_vec;
34 StaticMatrix<double, N, N> H;
35 StaticVector<double, N> g;
36 StaticMatrix<double, N, N> H_lm;
37 StaticVector<double, N> neg_g;
38 StaticVector<double, N> dx;
39 StaticVector<double, N> x_new;
40 StaticVector<ADVar, N> x_new_dual;
41
42 for (int iter = 0; iter < max_iter; ++iter) {
43 // 1. DualVec 변수 초기화 (시드 주입)
44 for (size_t i = 0; i < N; ++i) {
45 x_dual(i) = ADVar::make_variable(x_opt(i), i);
46 }
47
48 // 2. 잔차 벡터 F(x) 및 자코비안 J 동시 평가 (가장 무거운 연산)
49 StaticVector<ADVar, M> residuals_dual = calc_residuals(x_dual);
50
51 // 3. F 벡터 및 J 행렬 추출, 현재 Cost 계산
52 double current_cost = 0.0;
53 for (size_t k = 0; k < M; ++k) {
54 double r_val = Optimization::get_value(residuals_dual(k));
55 F_vec(k) = r_val;
56 current_cost += 0.5 * r_val * r_val;
57
58 for (size_t i = 0; i < N; ++i) {
59 J_mat(k, i) = residuals_dual(k).g[i];
60 }
61 }
62
63 // 4. H = J^T * J 와 g = J^T * F 조립 (SIMD 가속 엔진 타격)
64 linalg::multiply_AT_B(J_mat, J_mat, H);
65 linalg::multiply_AT_B(J_mat, F_vec, g);
66
67 // 5. 수렴 판정 (Gradient의 최대 크기가 허용 오차 이내인가?)
68 double max_grad = 0.0;
69 for (size_t i = 0; i < N; ++i) {
70 double abs_g = MathTraits<double>::abs(g(i));
71 if (abs_g > max_grad) max_grad = abs_g;
72 }
73 if (max_grad < tol) {
74 return SolverStatus::SUCCESS; // 성공적 수렴
75 }
76
77 // 6. [Architect's Update] Inner Retry Loop (자코비안 재계산 방어)
78 bool step_accepted = false;
79 int inner_retry = 0;
80
81 while (!step_accepted && inner_retry < 10) {
82 // LM Damping 추가 : H_lm = H + lambda * diag(H)
83 H_lm = H; // 복사
84 for (size_t i = 0; i < N; ++i) {
85 H_lm(i, i) += lambda * (H(i, i) + 1e-6);
86 neg_g(i) = -g(i); // 우변 벡터 부호 반전
87 }
88
89 // H_lm은 대칭 양의 정부호(SPD) 보장 -> LDLT 분해
90 MathStatus m_status = linalg::LDLT_decompose(H_lm);
91 if (m_status != MathStatus::SUCCESS) {
92 // 수치적 불안정: Damping을 늘려 조건수(Condition Number) 개선 후 재시도
93 lambda *= 10.0;
94 inner_retry++;
95 continue;
96 }
97
98 // Zero-Allocation LDLT Solve
99 linalg::LDLT_solve(H_lm, neg_g, dx);
100
101 // Step Size(Stagnation) 검증
102 double max_dx = 0.0;
103 for (size_t i = 0; i < N; ++i) {
104 double abs_dx = MathTraits<double>::abs(dx(i));
105 if (abs_dx > max_dx) max_dx = abs_dx;
106 }
107 if (max_dx < std::numeric_limits<double>::epsilon() * 10.0) {
108 return SolverStatus::STEP_SIZE_TOO_SMALL; // 탐색 불가
109 }
110
111 // x_new = x_opt + dx (SIMD 가속)
112 x_new = x_opt;
113 x_new += dx;
114
115 // 7. 새로운 위치의 Cost 평가 (Gradient는 필요 없으므로 상수 듀얼 값 주입)
116 for (size_t i = 0; i < N; ++i) {
117 x_new_dual(i) = ADVar(x_new(i));
118 }
119
120 StaticVector<ADVar, M> new_residuals = calc_residuals(x_new_dual);
121 double new_cost = 0.0;
122 for (size_t k = 0; k < M; ++k) {
123 double r_val = Optimization::get_value(new_residuals(k));
124 new_cost += 0.5 * r_val * r_val;
125 }
126
127 // 8. 스텝 승인(Accept) 및 거절(Reject) 판단
128 if (new_cost < current_cost) {
129 // 성공적인 스텝 : x 업데이트 및 Damping 감소 (가우스-뉴턴에 근접)
130 x_opt = x_new;
131 lambda = MathTraits<double>::max(1e-7, lambda / 10.0);
132 step_accepted = true; // 내부 루프 탈출
133 } else {
134 // 실패한 스텝 : 위치는 유지하되 Damping 증가 (경사하강법에 근접)
135 lambda *= 10.0;
136 inner_retry++; // J를 다시 구하지 않고 while문 재진입
137 }
138 }
139
140 if (!step_accepted) {
141 // 댐핑을 10번이나 올렸는데도 Cost를 줄이는 스텝을 찾지 못함 -> 평탄 지형(Stagnation)
142 return SolverStatus::STEP_SIZE_TOO_SMALL;
143 }
144 }
145
146 return SolverStatus::MAX_ITERATION_REACHED;
147}
