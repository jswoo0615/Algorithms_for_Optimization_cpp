34 template <size_t N>
35 static StaticVector<double, N> ray_point(const StaticVector<double, N>& x,
36 const StaticVector<double, N>& d, double alpha) {
37 StaticVector<double, N> pt;
38 for (size_t i = 0; i < N; ++i) {
39 pt(static_cast<int>(i)) = x(static_cast<int>(i)) + alpha * d(static_cast<int>(i));
40 }
41 return pt;
42 }

54 template <size_t N, typename Func>
55 static double directional_derivative(Func f, const StaticVector<double, N>& x,
56 const StaticVector<double, N>& d, double alpha) {
57 auto pt = ray_point<N>(x, d, alpha);
58 auto grad = AutoDiff::gradient<N>(f, pt);
59 double dir_deriv = 0.0;
60 for (size_t i = 0; i < N; ++i) {
61 dir_deriv += grad(static_cast<int>(i)) * d(static_cast<int>(i));
62 }
63 return dir_deriv;
64 }

67 // ==================================================================
68 // 1. Bracket Minimum (Algorithm 3.1)
69 // 핵심 수학적 아이디어 : a < b < c 위치에 있는 세 점을 찾을 때,
70 // 함수값이 f(a) > f(b)이면서 동시에 f(b) < f(c)를 만족한다면
71 // 함수가 감소하다가 다시 증가한 것이므로, 구간 [a, c] 내부에 반드시 최솟값이 존재합니다.
72 // 이 알고리즘은 초기 스텝 s에서 시작해 조건을 만족할 때까지 스텝을 k배씩 늘려가며 탐색합니다.
73 // ==================================================================
84 template <size_t N, typename Func>
85 static std::pair<double, double> bracket_minimum(Func f, const StaticVector<double, N>& x,
86 const StaticVector<double, N>& d,
87 double s = 1e-2, double k = 2.0,
88 bool verbose = false) {
89 double a = 0.0;
90 double ya = AutoDiff::value<N>(f, ray_point(x, d, a));
91 double b = s;
92 double yb = AutoDiff::value<N>(f, ray_point<N>(x, d, b));
93
94 // 탐색 방향의 반대쪽으로 이동해야 함수값이 감소하는 경우
95 // 방향을 반대로 뒤집어서 탐색을 시작합니다.
96 if (yb > ya) {
97 std::swap(a, b);
98 std::swap(ya, yb);
99 s = -s;
100 }
101
102 size_t iter = 0;
103 while (true) {
104 iter++;
105 double c = b + s;
106 double yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
107
108 if (verbose) {
109 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] a: " << std::fixed
110 << std::setprecision(5) << a << " | b: " << b << " | c: " << c
111 << " | f(c): " << yc << "\n";
112 }
113
114 // f(c) > f(b)를 만족하면 f(a) > f(b) < f(c) 조건이 완성되어
115 // 구간 내에 지역 최솟값이 존재함이 보장됩니다.
116 if (yc > yb) {
117 return (a < c) ? std::make_pair(a, c) : std::make_pair(c, a);
118 }
119
120 // 아직 감소 추세라면 범위를 다음으로 이동하고 스텝(s)을 늘립니다.
121 a = b;
122 ya = yb;
123 b = c;
124 yb = yc;
125 s *= k; // 스텝 크기를 k배 증가 (가속 탐색)
126 }
127 }

130 // 2. Golden Section Search (Algorithm 3.3)
131 // 최솟값이 포함된 구간 [a, b]가 주어졌을 때, 황금비(약 0.618)를 이용하여
132 // 평가 횟수를 최소화하면서 구간을 지속적으로 축소해 나가는 탐색 기법입니다.
133 // 기존 구간을 황금비로 나누는 두 점 c, d 중 하나는 다음 스텝에서도 재사용되므로 연산 효율이 좋습니다.
134 // ================================================================================
146 template <size_t N, typename Func>
147 static double golden_section_search(Func f, const StaticVector<double, N>& x,
148 const StaticVector<double, N>& d, double a, double b,
149 double tol = 1e-5, bool verbose = false) {
150 const double phi = (3.0 - std::sqrt(5.0)) / 2.0; // 약 0.381966 (1 - 0.618034)
151 double d_step = b - a;
152 double c = a + phi * d_step;
153 double d_val = b - phi * d_step;
154
155 double yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
156 double yd = AutoDiff::value<N>(f, ray_point<N>(x, d, d_val));
157
158 size_t iter = 0;
159 // 구간의 길이(d_step)가 tol 이하로 좁혀질 때까지 반복
160 while (std::abs(d_step) > tol) {
161 iter++;
162 if (verbose) {
163 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Range: [" << std::fixed
164 << std::setprecision(6) << a << ", " << b
165 << "] | Gap: " << std::abs(b - a) << "\n";
166 }
167
168 // f(c) < f(d) 이면 최솟값은 [a, d_val] 사이에 존재
169 if (yc < yd) {
170 b = d_val; // 구간의 오른쪽 끝을 d_val로 축소
171 d_val = c; // 이전 c가 새로운 d_val이 됨 (함수값 재사용)
172 yd = yc;
173 d_step = b - a;
174 c = a + phi * d_step; // 새로운 c점만 평가
175 yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
176 }
177 // f(c) >= f(d) 이면 최솟값은 [c, b] 사이에 존재
178 else {
179 a = c; // 구간의 왼쪽 끝을 c로 축소
180 c = d_val; // 이전 d_val이 새로운 c가 됨 (함수값 재사용)
181 yc = yd;
182 d_step = b - a;
183 d_val = b - phi * d_step; // 새로운 d_val점만 평가
184 yd = AutoDiff::value<N>(f, ray_point<N>(x, d, d_val));
185 }
186 }
187 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
188 // 좁혀진 구간의 중앙값을 반환
189 return (a + b) / 2.0;
190 }

193 // 3. Quadratic Fit Search (Algorithm 3.4)
194 // 세 점 (a, f(a)), (b, f(b)), (c, f(c))를 지나는 2차 다항식(포물선)을 근사하여,
195 // 그 포물선의 최솟값을 가지는 꼭짓점 위치(x*)를 다음 탐색점으로 삼는 방법입니다.
196 // 함수가 최솟값 근처에서 2차 함수와 비슷한 형태를 띠는 경우 수렴 속도가 매우 빠릅니다.
197 // ======================================
211 template <size_t N, typename Func>
212 static double quadratic_fit_search(Func f, const StaticVector<double, N>& x,
213 const StaticVector<double, N>& d, double a, double b,
214 double c, size_t max_iter = 50, double tol = 1e-5,
215 bool verbose = false) {
216 auto eval = [&](double alpha) { return AutoDiff::value<N>(f, ray_point<N>(x, d, alpha)); };
217
218 double ya = eval(a), yb = eval(b), yc = eval(c);
219
220 for (size_t iter = 1; iter <= max_iter; ++iter) {
221 // 세 점을 지나는 2차 다항식의 꼭짓점 계산을 위한 분자와 분모
222 double num = (b - a) * (b - a) * (yb - yc) - (b - c) * (b - c) * (yb - ya);
223 double den = (b - a) * (yb - yc) - (b - c) * (yb - ya);
224
225 // 세 점이 거의 일직선상에 있어 2차 근사가 불가능한 경우 루프 탈출
226 if (std::abs(den) < 1e-16) {
227 break;
228 }
229
230 // 2차 근사 다항식의 최소점 위치 (x_star)
231 double x_star = b - 0.5 * num / den;
232 double y_star = eval(x_star);
233
234 if (verbose) {
235 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] x*: " << std::fixed
236 << std::setprecision(6) << x_star << " | f(x*): " << y_star
237 << " | Shift: " << std::abs(x_star - b) << "\n";
238 }
239
240 // 이동한 거리가 허용 오차보다 작으면 수렴한 것으로 판단
241 if (std::abs(x_star - b) < tol) {
242 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
243 return x_star;
244 }
245
246 // 새로 찾은 x_star와 b의 위치 관계, 그리고 각각의 함수값을 비교하여
247 // 다음 반복에서 사용할 a, b, c 세 점을 갱신합니다.
248 // 항상 b가 세 점 중 가운데 위치를 유지하도록 조정합니다.
249 if (x_star > b) {
250 if (y_star > yb) {
251 c = x_star;
252 yc = y_star;
253 } else {
254 a = b;
255 ya = yb;
256 b = x_star;
257 yb = y_star;
258 }
259 } else {
260 if (y_star > yb) {
261 a = x_star;
262 ya = y_star;
263 } else {
264 c = b;
265 yc = yb;
266 b = x_star;
267 yb = y_star;
268 }
269 }
270 }
271 return b;
272 }

275 // 4. Shubert-Piyavskii Algorithm (Algorithm 3.5)
276 // 립시츠 연속(Lipschitz Continuous) 함수에서, 립시츠 상수(L)를 알고 있을 때
277 // 함수의 전역 최솟값(Global Minimum)을 찾기 위해 사용되는 알고리즘입니다.
278 // 기존에 평가된 점들로부터 V자 형태의 하한선(Lower Bound)을 생성하여,
279 // 하한선의 교차점 중 가장 낮은 지점을 다음 탐색점으로 선택합니다.
280 // ==================================================================
293 template <size_t N, size_t MAX_NODES = 100, typename Func>
294 static double shubert_piyavskii(Func f, const StaticVector<double, N>& x,
295 const StaticVector<double, N>& d, double a, double b, double L,
296 double tol = 1e-4, bool verbose = false) {
297 auto eval = [&](double alpha) { return AutoDiff::value<N>(f, ray_point<N>(x, d, alpha)); };
298
299 // 탐색 구간의 하한선 교차점을 저장하는 구조체
300 struct SPNode {
301 double a, ya, b, yb, x_m, y_m;
302 bool active;
303 };
304 std::array<SPNode, MAX_NODES> pool;
305 size_t pool_size = 0;
306
307 double ya = eval(a), yb = eval(b);
308 double min_val = std::min(ya, yb);
309 double min_alpha = (ya < yb) ? a : b;
310
311 // 새로운 구간 [n_a, n_b]에 대한 하한선 교차점(x_m, y_m)을 계산하여 노드 풀에 추가
312 auto add_node = [&](double n_a, double n_ya, double n_b, double n_yb) {
313 if (pool_size >= MAX_NODES) {
314 return;
315 }
316 // 두 점을 지나는 기울기 ±L인 직선들의 교차점 x 좌표
317 double x_m = 0.5 * (n_a + n_b) + (n_ya - n_yb) / (2.0 * L);
318 // 교차점에서의 y 좌표 (하한값)
319 double y_m = 0.5 * (n_ya + n_yb) - L * (n_b - n_a) / 2.0;
320 pool[pool_size++] = {n_a, n_ya, n_b, n_yb, x_m, y_m, true};
321 };
322
323 add_node(a, ya, b, yb);
324
325 size_t iter = 0;
326 // 최대 반복 횟수는 노드 공간을 초과하지 않도록 MAX_NODES / 2 로 제한
327 for (iter = 1; iter <= MAX_NODES / 2; ++iter) {
328 double best_ym = 1e99;
329 size_t best_idx = 0;
330 bool found = false;
331
332 // 현재 활성화된 구간들 중 가장 낮은 하한값(y_m)을 가진 구간을 선택
333 for (size_t i = 0; i < pool_size; ++i) {
334 if (pool[i].active && pool[i].y_m < best_ym) {
335 best_ym = pool[i].y_m;
336 best_idx = i;
337 found = true;
338 }
339 }
340
341 if (!found) break; // 더 이상 탐색할 활성 구간이 없음
342
343 double x_m = pool[best_idx].x_m;
344 double y_eval = eval(x_m); // 교차점 위치에서의 실제 함수값 평가
345
346 // 현재까지 발견된 최소 함수값 갱신
347 if (y_eval < min_val) {
348 min_val = y_eval;
349 min_alpha = x_m;
350 }
351
352 if (verbose) {
353 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Probe x_m: " << std::fixed
354 << std::setprecision(5) << x_m
355 << " | LowerBound y_m: " << pool[best_idx].y_m
356 << " | Gap: " << std::abs(y_eval - pool[best_idx].y_m) << "\n";
357 }
358
359 // 실제 함수값(y_eval)과 예측 하한값(y_m)의 차이가 tol 이하라면 수렴한 것으로 간주
360 if (std::abs(y_eval - pool[best_idx].y_m) < tol) {
361 break;
362 }
363
364 // 탐색한 구간을 비활성화하고, x_m을 기준으로 두 개의 새로운 하위 구간을 생성
365 pool[best_idx].active = false;
366 add_node(pool[best_idx].a, pool[best_idx].ya, x_m, y_eval);
367 add_node(x_m, y_eval, pool[best_idx].b, pool[best_idx].yb);
368 }
369 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
370 return min_alpha;
371 }

373 // ==================================================================
374 // 5. Bracket Sign Change (Algorithm 3.6)
375 // 함수의 도함수(Derivative)가 연속이라고 가정할 때, 방향 도함수의 부호가
376 // 음수에서 양수로 바뀌는 구간을 찾는 알고리즘입니다. 도함수의 부호가 바뀌는
377 // 구간 내부에는 도함수가 0이 되는 지점(즉, 최솟값 후보인 임계점)이 존재하게 됩니다.
378 // ==================================================================
390 template <size_t N, typename Func>
391 static std::pair<double, double> bracket_sign_change(Func f, const StaticVector<double, N>& x,
392 const StaticVector<double, N>& d, double a,
393 double b, double k = 2.0,
394 bool verbose = false) {
395 if (a > b) {
396 std::swap(a, b);
397 }
398 double center = (a + b) / 2.0;
399 double half_width = (b - a) / 2.0;
400 auto eval_deriv = [&](double alpha) { return directional_derivative<N>(f, x, d, alpha); };
401
402 double fp_a = eval_deriv(a);
403 double fp_b = eval_deriv(b);
404
405 size_t iter = 0;
406 // 양 끝점의 도함수 부호가 같으면 (곱이 양수) 최솟값을 포함하지 않으므로 구간을 확장
407 while (fp_a * fp_b > 0.0) {
408 iter++;
409 half_width *= k; // 구간을 k배 만큼 중심을 기준으로 확장
410 a = center - half_width;
411 b = center + half_width;
412 fp_a = eval_deriv(a);
413 fp_b = eval_deriv(b);
414
415 if (verbose) {
416 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Expanding: [" << std::fixed
417 << std::setprecision(5) << a << ", " << b << "] | f'(a)=" << std::showpos
418 << fp_a << " | f'(b)=" << fp_b << std::noshowpos << "\n";
419 }
420 }
421 return std::make_pair(a, b);
422 }

425 // 6. Bisection Method (Algorithm 3.7)
426 // 도함수의 부호가 다른 두 점 a, b (f'(a) < 0, f'(b) > 0)가 주어졌을 때,
427 // 구간을 반씩 줄여가며 도함수가 0이 되는 지점(근)을 찾는 이분법 기반 라인 서치입니다.
428 // Bracket Sign Change 알고리즘과 연계하여 사용하기 좋습니다.
429 // ======================================
441 template <size_t N, typename Func>
442 static double bisection(Func f, const StaticVector<double, N>& x,
443 const StaticVector<double, N>& d, double a, double b, double tol = 1e-5,
444 bool verbose = false) {
445 if (a > b) std::swap(a, b); // a가 항상 작도록 정렬
446 auto eval_deriv = [&](double alpha) { return directional_derivative<N>(f, x, d, alpha); };
447 double ya = eval_deriv(a);
448 double yb = eval_deriv(b);
449
450 // 이미 양 끝점 중 하나가 최솟점(도함수가 0에 매우 가까움)인 경우 바로 반환
451 if (std::abs(ya) <= 1e-9) return a;
452 if (std::abs(yb) <= 1e-9) return b;
453
454 size_t iter = 0;
455 // 탐색 구간 크기가 허용 오차보다 클 때까지 반복
456 while ((b - a) > tol) {
457 iter++;
458 double mid = (a + b) / 2.0;
459 double y_mid = eval_deriv(mid);
460
461 if (verbose) {
462 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Root Search: ["
463 << std::fixed << std::setprecision(6) << a << ", " << b
464 << "] | Mid: " << mid << " | f'(mid): " << std::showpos << y_mid
465 << std::noshowpos << "\n";
466 }
467
468 // 중간점의 도함수 값이 0에 매우 가까우면 찾은 것으로 판단하고 종료
469 if (std::abs(y_mid) <= 1e-9) {
470 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
471 return mid;
472 }
473
474 // 중간점에서의 도함수 부호에 따라 탐색 구간 축소
475 // ya와 y_mid의 부호가 같으면(둘 다 양수거나 둘 다 음수), 근은 mid와 b 사이에 존재
476 if ((y_mid > 0.0 && ya > 0.0) || (y_mid < 0.0 && ya < 0.0)) {
477 a = mid;
478 ya = y_mid;
479 }
480 // 다르면 근은 a와 mid 사이에 존재
481 else {
482 b = mid;
483 yb = y_mid;
484 }
485 }
486 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
487 // 허용 오차 이내로 구간이 좁혀지면, 해당 구간의 중앙값을 결과로 반환
488 return (a + b) / 2.0;
489 }
