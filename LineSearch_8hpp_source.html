17 template <size_t N>
18 static StaticVector<double, N> ray_point(const StaticVector<double, N>& x,
19 const StaticVector<double, N>& d, double alpha) {
20 StaticVector<double, N> pt;
21 for (size_t i = 0; i < N; ++i) {
22 pt(static_cast<int>(i)) = x(static_cast<int>(i)) + alpha * d(static_cast<int>(i));
23 }
24 return pt;
25 }
26
27 template <size_t N, typename Func>
28 static double directional_derivative(Func f, const StaticVector<double, N>& x,
29 const StaticVector<double, N>& d, double alpha) {
30 auto pt = ray_point<N>(x, d, alpha);
31 auto grad = AutoDiff::gradient<N>(f, pt);
32 double dir_deriv = 0.0;
33 for (size_t i = 0; i < N; ++i) {
34 dir_deriv += grad(static_cast<int>(i)) * d(static_cast<int>(i));
35 }
36 return dir_deriv;
37 }

40 // ==================================================================
41 // 1. Bracket Minimum (Algorithm 3.1)
42 // 핵심 수학적 아이디어 : a < b < c 위치에 있는 세 점을 찾을 때,
43 // 함수값이 f(a) > f(b)이면서 동시에 f(b) < f(c)를 만족한다면
44 // 함수가 감소하다가 다시 증가한 것이므로, 구간 [a, c] 내부에 반드시 최솟값이 존재
45 // ==================================================================
46 template <size_t N, typename Func>
47 static std::pair<double, double> bracket_minimum(Func f, const StaticVector<double, N>& x,
48 const StaticVector<double, N>& d,
49 double s = 1e-2, double k = 2.0,
50 bool verbose = false) {
51 double a = 0.0;
52 double ya = AutoDiff::value<N>(f, ray_point(x, d, a));
53 double b = s;
54 double yb = AutoDiff::value<N>(f, ray_point<N>(x, d, b));
55
56 if (yb > ya) {
57 std::swap(a, b);
58 std::swap(ya, yb);
59 s = -s;
60 }
61
62 size_t iter = 0;
63 while (true) {
64 iter++;
65 double c = b + s;
66 double yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
67
68 if (verbose) {
69 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] a: " << std::fixed
70 << std::setprecision(5) << a << " | b: " << b << " | c: " << c
71 << " | f(c): " << yc << "\n";
72 }
73
74 if (yc > yb) {
75 return (a < c) ? std::make_pair(a, c) : std::make_pair(c, a);
76 }
77
78 a = b;
79 ya = yb;
80 b = c;
81 yb = yc;
82 s *= k;
83 }
84 }

87 // 2. Golden Section Search (Algorithm 3.3)
88 // ================================================================================
89 template <size_t N, typename Func>
90 static double golden_section_search(Func f, const StaticVector<double, N>& x,
91 const StaticVector<double, N>& d, double a, double b,
92 double tol = 1e-5, bool verbose = false) {
93 const double phi = (3.0 - std::sqrt(5.0)) / 2.0;
94 double d_step = b - a;
95 double c = a + phi * d_step;
96 double d_val = b - phi * d_step;
97
98 double yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
99 double yd = AutoDiff::value<N>(f, ray_point<N>(x, d, d_val));
100
101 size_t iter = 0;
102 while (std::abs(d_step) > tol) {
103 iter++;
104 if (verbose) {
105 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Range: [" << std::fixed
106 << std::setprecision(6) << a << ", " << b
107 << "] | Gap: " << std::abs(b - a) << "\n";
108 }
109
110 if (yc < yd) {
111 b = d_val;
112 d_val = c;
113 yd = yc;
114 d_step = b - a;
115 c = a + phi * d_step;
116 yc = AutoDiff::value<N>(f, ray_point<N>(x, d, c));
117 } else {
118 a = c;
119 c = d_val;
120 yc = yd;
121 d_step = b - a;
122 d_val = b - phi * d_step;
123 yd = AutoDiff::value<N>(f, ray_point<N>(x, d, d_val));
124 }
125 }
126 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
127 return (a + b) / 2.0;
128 }

130 // ======================================
131 // 3. Quadratic Fit Search (Algorithm 3.4)
132 // ======================================
133 template <size_t N, typename Func>
134 static double quadratic_fit_search(Func f, const StaticVector<double, N>& x,
135 const StaticVector<double, N>& d, double a, double b,
136 double c, size_t max_iter = 50, double tol = 1e-5,
137 bool verbose = false) {
138 auto eval = [&](double alpha) { return AutoDiff::value<N>(f, ray_point<N>(x, d, alpha)); };
139
140 double ya = eval(a), yb = eval(b), yc = eval(c);
141
142 for (size_t iter = 1; iter <= max_iter; ++iter) {
143 double num = (b - a) * (b - a) * (yb - yc) - (b - c) * (b - c) * (yb - ya);
144 double den = (b - a) * (yb - yc) - (b - c) * (yb - ya);
145 if (std::abs(den) < 1e-16) {
146 break;
147 }
148 double x_star = b - 0.5 * num / den;
149 double y_star = eval(x_star);
150
151 if (verbose) {
152 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] x*: " << std::fixed
153 << std::setprecision(6) << x_star << " | f(x*): " << y_star
154 << " | Shift: " << std::abs(x_star - b) << "\n";
155 }
156
157 if (std::abs(x_star - b) < tol) {
158 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
159 return x_star;
160 }
161
162 if (x_star > b) {
163 if (y_star > yb) {
164 c = x_star;
165 yc = y_star;
166 } else {
167 a = b;
168 ya = yb;
169 b = x_star;
170 yb = y_star;
171 }
172 } else {
173 if (y_star > yb) {
174 a = x_star;
175 ya = y_star;
176 } else {
177 c = b;
178 yc = yb;
179 b = x_star;
180 yb = y_star;
181 }
182 }
183 }
184 return b;
185 }

188 // 4. Shubert-Piyavskii Algorithm (Algorithm 3.5)
189 // ==================================================================
190 template <size_t N, size_t MAX_NODES = 100, typename Func>
191 static double shubert_piyavskii(Func f, const StaticVector<double, N>& x,
192 const StaticVector<double, N>& d, double a, double b, double L,
193 double tol = 1e-4, bool verbose = false) {
194 auto eval = [&](double alpha) { return AutoDiff::value<N>(f, ray_point<N>(x, d, alpha)); };
195 struct SPNode {
196 double a, ya, b, yb, x_m, y_m;
197 bool active;
198 };
199 std::array<SPNode, MAX_NODES> pool;
200 size_t pool_size = 0;
201 double ya = eval(a), yb = eval(b);
202 double min_val = std::min(ya, yb);
203 double min_alpha = (ya < yb) ? a : b;
204
205 auto add_node = [&](double n_a, double n_ya, double n_b, double n_yb) {
206 if (pool_size >= MAX_NODES) {
207 return;
208 }
209 double x_m = 0.5 * (n_a + n_b) + (n_ya - n_yb) / (2.0 * L);
210 double y_m = 0.5 * (n_ya + n_yb) - L * (n_b - n_a) / 2.0;
211 pool[pool_size++] = {n_a, n_ya, n_b, n_yb, x_m, y_m, true};
212 };
213 add_node(a, ya, b, yb);
214
215 size_t iter = 0;
216 for (iter = 1; iter <= MAX_NODES / 2; ++iter) {
217 double best_ym = 1e99;
218 size_t best_idx = 0;
219 bool found = false;
220 for (size_t i = 0; i < pool_size; ++i) {
221 if (pool[i].active && pool[i].y_m < best_ym) {
222 best_ym = pool[i].y_m;
223 best_idx = i;
224 found = true;
225 }
226 }
227
228 if (!found) break;
229 double x_m = pool[best_idx].x_m;
230 double y_eval = eval(x_m);
231
232 if (y_eval < min_val) {
233 min_val = y_eval;
234 min_alpha = x_m;
235 }
236
237 if (verbose) {
238 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Probe x_m: " << std::fixed
239 << std::setprecision(5) << x_m
240 << " | LowerBound y_m: " << pool[best_idx].y_m
241 << " | Gap: " << std::abs(y_eval - pool[best_idx].y_m) << "\n";
242 }
243
244 if (std::abs(y_eval - pool[best_idx].y_m) < tol) {
245 break;
246 }
247 pool[best_idx].active = false;
248 add_node(pool[best_idx].a, pool[best_idx].ya, x_m, y_eval);
249 add_node(x_m, y_eval, pool[best_idx].b, pool[best_idx].yb);
250 }
251 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
252 return min_alpha;
253 }

256 // 5. Bracket Sign Change (Algorithm 3.6)
257 // ==================================================================
258 template <size_t N, typename Func>
259 static std::pair<double, double> bracket_sign_change(Func f, const StaticVector<double, N>& x,
260 const StaticVector<double, N>& d, double a,
261 double b, double k = 2.0,
262 bool verbose = false) {
263 if (a > b) {
264 std::swap(a, b);
265 }
266 double center = (a + b) / 2.0;
267 double half_width = (b - a) / 2.0;
268 auto eval_deriv = [&](double alpha) { return directional_derivative<N>(f, x, d, alpha); };
269 double fp_a = eval_deriv(a);
270 double fp_b = eval_deriv(b);
271
272 size_t iter = 0;
273 while (fp_a * fp_b > 0.0) {
274 iter++;
275 half_width *= k;
276 a = center - half_width;
277 b = center + half_width;
278 fp_a = eval_deriv(a);
279 fp_b = eval_deriv(b);
280
281 if (verbose) {
282 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Expanding: [" << std::fixed
283 << std::setprecision(5) << a << ", " << b << "] | f'(a)=" << std::showpos
284 << fp_a << " | f'(b)=" << fp_b << std::noshowpos << "\n";
285 }
286 }
287 return std::make_pair(a, b);
288 }

290 // 6. Bisection Method (Algorithm 3.7)
291 // ======================================
292 template <size_t N, typename Func>
293 static double bisection(Func f, const StaticVector<double, N>& x,
294 const StaticVector<double, N>& d, double a, double b, double tol = 1e-5,
295 bool verbose = false) {
296 if (a > b) std::swap(a, b);
297 auto eval_deriv = [&](double alpha) { return directional_derivative<N>(f, x, d, alpha); };
298 double ya = eval_deriv(a);
299 double yb = eval_deriv(b);
300
301 if (std::abs(ya) <= 1e-9) return a;
302 if (std::abs(yb) <= 1e-9) return b;
303
304 size_t iter = 0;
305 while ((b - a) > tol) {
306 iter++;
307 double mid = (a + b) / 2.0;
308 double y_mid = eval_deriv(mid);
309
310 if (verbose) {
311 std::cout << " ↳ [Iter " << std::setw(2) << iter << "] Root Search: ["
312 << std::fixed << std::setprecision(6) << a << ", " << b
313 << "] | Mid: " << mid << " | f'(mid): " << std::showpos << y_mid
314 << std::noshowpos << "\n";
315 }
316
317 if (std::abs(y_mid) <= 1e-9) {
318 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
319 return mid;
320 }
321
322 if ((y_mid > 0.0 && ya > 0.0) || (y_mid < 0.0 && ya < 0.0)) {
323 a = mid;
324 ya = y_mid;
325 } else {
326 b = mid;
327 yb = y_mid;
328 }
329 }
330 if (verbose) std::cout << " ↳ [Converged] Total Iterations: " << iter << "\n";
331 return (a + b) / 2.0;
332 }
