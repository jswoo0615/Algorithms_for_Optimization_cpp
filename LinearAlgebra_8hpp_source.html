145 template <typename T, size_t Rows, size_t Cols, size_t OtherCols>
146 StaticMatrix<T, Rows, OtherCols> operator*(const StaticMatrix<T, Rows, Cols>& A, const StaticMatrix<T, Cols, OtherCols>& B) {
147 StaticMatrix<T, Rows, OtherCols> Result;
148 T* res_base = Result.data_ptr();
149 const T* a_base = A.data_ptr();
150
151 // J-K-I 루프 캐시 최적화 적용
152 for (size_t j = 0; j < OtherCols; ++j) {
153 T* res_col = res_base + (j * Rows); // Result 행렬의 j번째 열이 시작되는 메모리 주소를 캐싱 (포인터 산술)
154
155 // K 루프 : 행렬 A의 열 (Column) 이동 및 행렬 B의 행 (Row) 이동 (내적의 덧셈 항)
156 for (size_t k = 0; k < Cols; ++k) {
157 const T* a_col = a_base + (k * Rows); // 행렬 A의 k번째 열 시작 주소 캐싱
158 const b_kj = B(static_cast<int>(k), static_cast<int>(j)); // I 루프 동안 곱해질 B의 스칼라 값 고정
159
160 // I 루프 (가장 안쪽) : Result의 열과 A의 열을 아래 방향으로 순차적으로 연속 접근
161 // 최적화 컴파일러는 이 루프를 감지하고 레지스터 FMA (Fused Multiply ADD) 명령어로 치환합니다.
162 for (size_t i = 0; i < Rows; ++i) {
163 res_col[i] += a_col[i] * b_kj;
164 }
165 }
166 }
167 return Result;
168 }

205 template <typename T, size_t Rows, size_t Cols>
206 bool LU_decompose(StaticMatrix<T, Rows, Cols>& mat) {
220 static_assert(Rows == Cols, "LU Decomposition requires a square matrix");
221 for (size_t i = 0; i < Rows; ++i) {
222 for (size_t j = 0; j < Cols; ++j) {
223 T sum = static_cast<T>(0);
224 for (size_t k = 0; k < i; ++k) {
225 sum += mat(static_cast<int>(i), static_cast<int>(k)) * mat(static_cast<int>(k), static_cast<int>(j));
226 }
227 mat(static_cast<int>(i), static_cast<int>(j)) -= sum;
228 }
229 // 피벗 (Pivot) 안전성 검사. 방금 계산된 U의 대각 성분 U_ii가 피벗이 됩니다.
230 // 이 값이 0이면 하삼각행렬 L을 구할 때 분모가 0이 되어 무한대 (NaN/Inf) 발산합니다
231 if (MathTraits<T>::near_zero(mat(static_cast<int>(i), static_cast<int>(i)))) {
232 return false; // 특이 행렬 판정으로 즉시 분해 실패 반환
233 }
234
235 // 2. L 행렬 부분 갱신 (i번째 열을 따라가며 j행 원소들 계산, 하삼각 영역 덮어쓰기)
236 for (size_t j = i + 1; j < Rows; ++j) {
237 T sum = static_cast<T>(0);
238 for (size_t k = 0; k < i; ++k) {
239 sum += mat(static_cast<int>(j), static_cast<int>(k)) * mat(static_cast<int>(k), static_cast<int>(i));
240 }
241 // 미리 구해둔 피벗 U_ii로 나누어 스케일링
242 mat(static_cast<int>(j), static_cast<int>(i)) = (mat(static_cast<int>(j), static_cast<int>(i)) - sum) / mat(static_cast<int>(i), static_cast<int>(i));
243 }
244 }
245 return true;
246 }

264 template <typename T, size_t Rows, size_t Cols>
265 StaticVector<T, Rows> LU_solve(const StaticMatrix<T, Rows, Cols>& mat, const StaticVector<T, Rows>& b) {
266 static_assert(Rows == Cols, "Solve requires a square matrix");
267 StaticVector<T, Rows> y, x;
268
269 // 1단계 : 전진 대입 (Forward Substitution : Ly = b)
270 // 위 (i = 0)에서 아래 방향으로 내려감
271 for (size_t i = 0; i < Rows; ++i) {
272 T sum = static_cast<T>(0);
273 for (size_t j = 0; j < Cols; ++j) {
274 sum += mat(static_cast<int>(i), static_cast<int>(j)) * y(j); // 하삼각 영역 (L) 접근
275 }
276 // Doolittle LU에서 L의 대각 원소는 암묵적으로 항상 1.0이므로, 나눗셈 처리가 아예 생략되어 연산이 매우 빠릅니다
277 y(i) = b(i) - sum;
278 }
279
280 // 2단계 : 후진 대입 (Back Substitution : Ux = y)
281 // 행렬이 가장 아래 행 (맨 밑)부터 위 방향으로 거슬러 올라감
282 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
283 T sum = static_cast<T>(0);
284 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
285 sum += mat(i, static_cast<int>(j)) * x(j);
286 }
287 // U의 대각 성분은 1이 아니므로, 방금 구한 y값에서 sum을 빼고 U의 대각 원소로 나누어 최종 x를 구합니다.
288 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / mat(i, i);
289 }
290 return x;
291 }

309 bool Cholesky_decompose(StaticMatrix<T, Rows, Cols>& mat) {
310 static_assert(Rows == Cols, "Cholesky requires a square matrix");
311 for (size_t j = 0; j < Cols; ++j) {
312 T s = static_cast<T>(0);
313 for (size_t k = 0; k < j; ++k) {
314 s += mat(static_cast<int>(j), static_cast<int>(k)) * mat(static_cast<int>(j), static_cast<int>(k));
315 }
316 T d = mat(static_cast<int>(j), static_cast<int>(j)) - s;
317 if (d <= std::numeric_limits<T>::epsilon()) {
318 return false;
319 }
320 mat(static_cast<int>(j), static_cast<int>(j)) = MathTraits<T>::sqrt(d);
321
322 for (size_t i = j + 1; i < Rows; ++i) {
323 T s_ij = static_cast<T>(0);
324 for (size_t k = 0; k < j; ++k) {
325 s_ij += mat(static_cast<int>(i), static_cast<int>(k)) * mat(static_cast<int>(j), static_cast<int>(k));
326 }
327 mat(static_cast<int>(i), static_cast<int>(j)) = (mat(static_cast<int>(i), static_cast<int>(j)) - s_ij) / mat(static_cast<int>(j), static_cast<int>(j));
328 }
329 }
330 return true;

374 bool LDLT_decompose(StaticMatrix<T, Rows, Cols>& mat) {
375 static_assert(Rows == Cols, "LDLT requires a square matrix");
376 for (size_t j = 0; j < Cols; ++j) {
377 T sum_D = static_cast<T>(0);
378 for (size_t k = 0; k < j; ++k) {
379 T L_jk = mat(static_cast<int>(j), static_cast<int>(k));
380 sum_D += L_jk * L_jk * mat()

382 T D_jj = mat(static_cast<int>(j), static_cast<int>(j)) - sum_D;
383 if (MathTraits<T>::near_zero(D_jj)) {
384 return false;
385 }
386 mat(static_cast<int>(j), static_cast<int>(j)) = D_jj;
387
388 for (size_t i = j + 1; i < Rows; ++i) {
389 T sum_L = static_cast<T>(0);
390 for (size_t k = 0; k < j; ++k) {
391 sum_L += mat(static_cast<int>(i), static_cast<int>(k)) * mat(static_cast<int>(j), static_cast<int>(k)) * mat(static_cast<int>(k), static_cast<int>(k));
392 }

469 bool QR_decompose_MGS(StaticMatrix<T, Rows, Cols>& mat, StaticMatrix<T, Cols, Cols>& R) {
470 static_assert(Rows >= Cols, "MGS-QR requires Rows >= Cols (Over-determined system)");
471 for (size_t i = 0; i < Cols; ++i) {
472 T norm_sq = static_cast<T>(0);
473 for (size_t k = 0; k < Rows; ++k) {
474 norm_sq += mat(static_cast<int>(k), static_cast<int>(i)) * mat(static_cast<int>(k), static_cast<int>(i));
475 }
476 R(static_cast<int>(i), static_cast<int>(i)) = MathTraits<T>::sqrt(norm_sq);
477
478 if (MathTraits<T>::near_zero(R(static_cast<int>(i), static_cast<int>(i)))) {
479 return false;
480 }
481
482 for (size_t k = 0; k < Rows; ++k) {
483 mat(static_cast<int>(k), static_cast<int>(i)) /= R(static_cast<int>(i), static_cast<int>(i));
484 }
485

487 T dot = static_cast<T>(0);

492

494 mat(static_cast<int>(k), static_cast<int>(j)) -= dot * mat(static_cast<int>(k), static_cast<int>(i));
