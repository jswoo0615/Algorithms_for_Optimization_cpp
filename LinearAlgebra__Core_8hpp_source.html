57template <typename T, size_t Rows, size_t Cols, size_t OtherCols>
58StaticMatrix<T, Rows, OtherCols> operator*(const StaticMatrix<T, Rows, Cols>& A,
59 const StaticMatrix<T, Cols, OtherCols>& B) {
60 StaticMatrix<T, Rows, OtherCols> Result;
61 T* res_base = Result.data_ptr();
62 const T* a_base = A.data_ptr();
63
64 // J-K-I 루프 캐시 최적화 적용
65 for (size_t j = 0; j < OtherCols; ++j) {
66 T* res_col = res_base + (j * Rows);
67 for (size_t k = 0; k < Cols; ++k) {
68 const T* a_col = a_base + (k * Rows);
69 const T b_kj = B(static_cast<int>(k), static_cast<int>(j));
70 for (size_t i = 0; i < Rows; ++i) {
71 res_col[i] += a_col[i] * b_kj;
72 }
73 }
74 }
75 return Result;
