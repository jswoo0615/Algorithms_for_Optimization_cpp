23template <typename T, size_t N>
24inline MathStatus LU_decompose(StaticMatrix<T, N, N>& mat, StaticVector<int, N>& P) {
25 for (size_t i = 0; i < N; ++i) {
26 P(i) = static_cast<int>(i);
27 }
28
29 for (size_t i = 0; i < N; ++i) {
30 // 1. Pivot selection (Column 단위 순회이므로 캐시 친화적)
31 T max_val = static_cast<T>(0);
32 size_t pivot_row = i;
33 for (size_t j = i; j < N; ++j) {
34 T abs_val = MathTraits<T>::abs(mat(j, i));
35 // Dual 타입 호환성을 위해 get_value 사용
36 if (Optimization::get_value(abs_val) > Optimization::get_value(max_val)) {
37 max_val = abs_val;
38 pivot_row = j;
39 }
40 }
41
42 if (MathTraits<T>::near_zero(max_val)) {
43 return MathStatus::SINGULAR;
44 }
45
46 // 2. Row Swap (물리적 행 교환)
47 if (pivot_row != i) {
48 for (size_t k = 0; k < N; ++k) {
49 T temp = mat(i, k);
50 mat(i, k) = mat(pivot_row, k);
51 mat(pivot_row, k) = temp;
52 }
53 int temp_p = P(i);
54 P(i) = P(pivot_row);
55 P(pivot_row) = temp_p;
56 }
57
58 // 3. Scale column i (L 행렬 요소 생성)
59 T inv_pivot = static_cast<T>(1.0) / mat(i, i);
60 for (size_t j = i + 1; j < N; ++j) {
61 mat(j, i) *= inv_pivot;
62 }
63
64 // 4. Rank-1 Update (Schur Complement) -> Column-oriented SAXPY
65 for (size_t j = i + 1; j < N; ++j) {
66 T temp = mat(i, j);
67 size_t k = i + 1;
68
69 // --- SIMD 가속 구간: col_j[k...] -= temp * col_i[k...] ---
70 #if defined(__ARM_NEON) || defined(__ARM_NEON__)
71 if constexpr (std::is_same_v<T, float>) {
72 float32x4_t v_temp = vdupq_n_f32(-temp);
73 for (; k + 3 < N; k += 4) {
74 float32x4_t v_col_i = vld1q_f32(&mat(k, i));
75 float32x4_t v_col_j = vld1q_f32(&mat(k, j));
76 vst1q_f32(&mat(k, j), vfmaq_f32(v_col_j, v_col_i, v_temp));
77 }
78 } else if constexpr (std::is_same_v<T, double>) {
79 float64x2_t v_temp = vdupq_n_f64(-temp);
80 for (; k + 1 < N; k += 2) {
81 float64x2_t v_col_i = vld1q_f64(&mat(k, i));
82 float64x2_t v_col_j = vld1q_f64(&mat(k, j));
83 vst1q_f64(&mat(k, j), vfmaq_f64(v_col_j, v_col_i, v_temp));
84 }
85 }
86 #elif defined(__AVX2__)
87 if constexpr (std::is_same_v<T, float>) {
88 __m256 v_temp = _mm256_set1_ps(-temp);
89 for (; k + 7 < N; k += 8) {
90 __m256 v_col_i = _mm256_loadu_ps(&mat(k, i));
91 __m256 v_col_j = _mm256_loadu_ps(&mat(k, j));
92 _mm256_storeu_ps(&mat(k, j), _mm256_fmadd_ps(v_col_i, v_temp, v_col_j));
93 }
94 } else if constexpr (std::is_same_v<T, double>) {
95 __m256d v_temp = _mm256_set1_pd(-temp);
96 for (; k + 3 < N; k += 4) {
97 __m256d v_col_i = _mm256_loadu_pd(&mat(k, i));
98 __m256d v_col_j = _mm256_loadu_pd(&mat(k, j));
99 _mm256_storeu_pd(&mat(k, j), _mm256_fmadd_pd(v_col_i, v_temp, v_col_j));
100 }
101 }
102 #endif
103
104 // --- Scalar Fallback ---
105 for (; k < N; ++k) {
106 mat(k, j) -= mat(k, i) * temp;
107 }
108 }
109 }
110 return MathStatus::SUCCESS;
111}

117template <typename T, size_t N>
118inline void LU_solve(const StaticMatrix<T, N, N>& mat,
119 const StaticVector<int, N>& P,
120 const StaticVector<T, N>& b,
121 StaticVector<T, N>& x) {
122
123 // 1. Apply Permutation
124 for (size_t i = 0; i < N; ++i) {
125 x(i) = b(P(i));
126 }
127
128 // 2. Forward substitution (L * y = b) -> x에 in-place 수행
129 // L 행렬의 대각 원소는 1이라고 가정합니다.
130 for (size_t k = 0; k < N; ++k) {
131 T x_k = x(k);
132 // Column-oriented update
133 for (size_t i = k + 1; i < N; ++i) {
134 x(i) -= mat(i, k) * x_k;
135 }
136 }
137
138 // 3. Backward substitution (U * x = y) -> x에 in-place 수행
139 // [Architect's Note] 통상적인 Row-oriented 내적이 아닌, Column-oriented 소거법 적용
140 for (int k = static_cast<int>(N) - 1; k >= 0; --k) {
141 x(k) /= mat(k, k); // 대각 원소로 나누어 x(k) 확정
142 T x_k = x(k);
143 // 확정된 x(k)를 이용하여 그 위에 있는 모든 행의 값을 미리 빼줍니다 (Column 순회)
144 for (int i = 0; i < k; ++i) {
145 x(i) -= mat(i, k) * x_k;
146 }
147 }
148}

150// 편의성 래퍼 (RVO 지원)
151template <typename T, size_t N>
152inline StaticVector<T, N> LU_solve(const StaticMatrix<T, N, N>& mat,
153 const StaticVector<int, N>& P,
154 const StaticVector<T, N>& b) {
155 StaticVector<T, N> x;
156 LU_solve(mat, P, b, x);
157 return x;
158}
