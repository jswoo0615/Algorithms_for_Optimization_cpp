32class MeshAdaptiveDirectSearch {
33 private:
34 // O(1) 정적 배열 및 난수 참조 전달을 통한 지연 없는 랜덤 기저 (Basis) 생성
35 template <size_t N, typename Generator>
36 [[nodiscard]] static std::array<std::array<double, N>, N + 1> generate_positive_spanning_set(
37 double alpha, Generator& gen) {
38 alignas(64) std::array<std::array<double, N>, N + 1> D = {0.0};
39 alignas(64) std::array<std::array<double, N>, N> L = {0.0};
40
41 // 1. 델타 (Delta) 크기 결정
42 int delta = static_cast<int>(std::round(1.0 / std::sqrt(alpha)));
43 if (delta < 1) {
44 delta = 1;
45 }
46 std::uniform_int_distribution<int> sign_dist(0, 1);
47 std::uniform_int_distribution<int> lower_dist(-delta + 1, delta - 1);
48
49 // 2. 하삼각 행렬 (Lower Triangular Matrix) L 생성
50 for (size_t i = 0; i < N; ++i) {
51 L[i][i] = (sign_dist(gen) == 1) ? delta : -delta; // 대각 성분
52 for (size_t j = 0; j < i; ++j) {
53 L[i][j] = lower_dist(gen); // 하삼각 성분
54 }
55 }
56
57 // 3. 행과 열을 랜덤하게 섞기 (Permutation)
58 std::array<size_t, N> row_p, col_p;
59 std::iota(row_p.begin(), row_p.end(), 0); // 0, 1, 2, ... 할당
60 std::iota(col_p.begin(), col_p.end(), 0);
61 std::shuffle(row_p.begin(), row_p.end(), gen);
62 std::shuffle(col_p.begin(), col_p.end(), gen);
63
64 // 4. 방향 행렬 D 계산 및 N + 1 번째 방향 (Negative Sum) 추가
65 for (size_t d_idx = 0; d_idx < N; ++d_idx) {
66 for (size_t dim = 0; dim < N; ++dim) {
67 double val = L[row_p[dim]][col_p[d_idx]];
68 D[d_idx][dim] = val;
69 D[N][dim] -= val; // 나머지 벡터들의 역벡터 합 (Positive Spanning Set 조건 완수)
70 }
71 }
72 return D;
73 }
74
75 public:
76 MeshAdaptiveDirectSearch() = delete; // 인스턴스화 방지

78 template <size_t N, typename Func>
79 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
80 double tol = 1e-5,
81 size_t max_iter = 10000,
82 bool verbose = false) {
83 static_assert(N > 0, "Dimension N must be greater than 0");
84 auto start_clock = std::chrono::high_resolution_clock::now();
85
86 double alpha = 1.0;
87 alignas(64) std::array<double, N> x = x_init;
88 double y = f(x);
89
90 // [핵심] 정적 thread_local 난수 생성기로 지연 (Latency) 원천 차단
91 static thread_local std::mt19937 gen(std::random_device{}());
92
93 if (verbose) {
94 std::cout << "========================================================\n";
95 std::cout << " 🕸️ MADS (Mesh Adaptive Direct Search) Started \n";
96 std::cout << "========================================================\n";
97 }
98
99 size_t iter = 0;
100 while (alpha > tol && iter < max_iter) {
101 iter++;
102 bool improved = false;
103
104 // N + 1개의 무작위 탐색 방향 세트 획득
105 auto D = generate_positive_spanning_set<N>(alpha, gen);
106
107 // [기회주의적 탐색]
108 for (size_t i = 0; i <= N; ++i) {
109 alignas(64) std::array<double, N> x_prime = {0.0};
110#pragma omp simd
111 for (size_t j = 0; j < N; ++j) {
112 x_prime[j] = std::fma(alpha, D[i][j], x[j]);
113 }
114
115 double y_prime = f(x_prime);
116
117 if (y_prime < y) {
118 x = x_prime;
119 y = y_prime;
120 improved = true;
121
122 // [가속 스텝] 해당 방향으로 성과가 좋으면 3배 (3 * alpha) 반영
123 alignas(64) std::array<double, N> x_exp = {0.0};
124#pragma omp simd
125 for (size_t j = 0; j < N; ++j) {
126 x_exp[j] = std::fma(3.0 * alpha, D[i][j], x[j]);
127 }
128
129 double y_exp = f(x_exp);
130
131 if (y_exp < y) {
132 x = x_exp;
133 y = y_exp;
134 }
135 break; // 하나라도 개선되면 즉시 내부 루프 탈출
136 }
137 }
138
139 // 성과에 따른 Mesh 크기 및 보폭 조절
140 if (improved) {
141 alpha = std::min(4.0 * alpha, 1.0);
142 } else {
143 alpha /= 4.0;
144 }
145
146 if (verbose && (!improved || iter % 10 == 0)) {
147 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
148 << std::setprecision(6) << y << " | alpha: " << alpha << "\n";
149 }
150 }
151 auto end_clock = std::chrono::high_resolution_clock::now();
152 auto duration =
153 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
154
155 if (verbose) {
156 std::cout << "========================================================\n";
157 std::cout << " 🏁 Final Optimal Point: [" << x[0];
158 if constexpr (N > 1) std::cout << ", " << x[1];
159 if constexpr (N > 2) std::cout << ", ...";
160 std::cout << "]\n========================================================\n";
161 }
162
163 return {x, y, iter, duration.count()};
164 }
