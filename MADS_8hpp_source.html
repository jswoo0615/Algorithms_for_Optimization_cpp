58 template <size_t N, typename Generator>
59 [[nodiscard]] static std::array<std::array<double, N>, N + 1> generate_positive_spanning_set(
60 double alpha, Generator& gen) {
61
62 // 캐시 라인(64 byte) 정렬을 통해 SIMD 연산 및 메모리 접근 속도 최적화
63 alignas(64) std::array<std::array<double, N>, N + 1> D = {0.0};
64 alignas(64) std::array<std::array<double, N>, N> L = {0.0};
65
66 // 1. 델타 (Delta) 크기 결정
67 // alpha 값이 작아질수록 델타가 커져 더 넓은 범위의 기저 방향들을 생성하게 됨
68 int delta = static_cast<int>(std::round(1.0 / std::sqrt(alpha)));
69 if (delta < 1) {
70 delta = 1;
71 }
72
73 // 난수 분포 설정
74 std::uniform_int_distribution<int> sign_dist(0, 1); // 0 또는 1 반환 (부호 결정용)
75 std::uniform_int_distribution<int> lower_dist(-delta + 1, delta - 1); // 하삼각 요소 범위
76
77 // 2. 하삼각 행렬 (Lower Triangular Matrix) L 생성
78 for (size_t i = 0; i < N; ++i) {
79 // 대각 성분은 ±delta 로 강제하여 선형 독립성을 보장
80 L[i][i] = (sign_dist(gen) == 1) ? delta : -delta;
81 for (size_t j = 0; j < i; ++j) {
82 // 대각선 아래 값들은 범위 내의 임의의 정수 할당
83 L[i][j] = lower_dist(gen);
84 }
85 }
86
87 // 3. 행과 열을 랜덤하게 섞기 (Permutation)
88 // 이 과정을 통해 하삼각 행렬 구조에 편향되지 않은 균일한 방향 탐색을 수행함
89 std::array<size_t, N> row_p, col_p;
90 std::iota(row_p.begin(), row_p.end(), 0); // 0, 1, 2, ... N-1 할당
91 std::iota(col_p.begin(), col_p.end(), 0);
92
93 // 배열을 섞어서 무작위 순열 생성
94 std::shuffle(row_p.begin(), row_p.end(), gen);
95 std::shuffle(col_p.begin(), col_p.end(), gen);
96
97 // 4. 방향 행렬 D 계산 및 N + 1 번째 방향 (Negative Sum) 추가
98 for (size_t d_idx = 0; d_idx < N; ++d_idx) {
99 for (size_t dim = 0; dim < N; ++dim) {
100 // 섞인 순열 인덱스를 활용하여 L 행렬에서 값 복사
101 double val = L[row_p[dim]][col_p[d_idx]];
102 D[d_idx][dim] = val;
103
104 // N+1 번째 벡터 (인덱스 N)는 나머지 N개 벡터의 역방향 합산 (Negative Sum)
105 // 이렇게 해야 N+1개의 벡터가 공간의 모든 방향을 양의 조합(Positive Span)으로 표현할 수 있음
106 D[N][dim] -= val;
107 }
108 }
109 return D;
110 }

125 template <size_t N, typename Func>
126 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_init,
127 double tol = 1e-5,
128 size_t max_iter = 10000,
129 bool verbose = false) {
130 static_assert(N > 0, "Dimension N must be greater than 0"); // 컴파일 타임 차원 검사
131
132 // 연산 소요 시간 측정을 위한 타이머 시작
133 auto start_clock = std::chrono::high_resolution_clock::now();
134
135 double alpha = 1.0; // 초기 스텝 사이즈 (Mesh 파라미터)
136 alignas(64) std::array<double, N> x = x_init; // 캐시 정렬된 현재 탐색 위치 변수
137 double y = f(x); // 현재 위치의 목적 함수값
138
139 // [핵심] 정적 thread_local 난수 생성기로 지연 (Latency) 원천 차단
140 // 매 반복마다 생성기를 다시 인스턴스화하지 않아 난수 생성 오버헤드를 극도로 줄임
141 static thread_local std::mt19937 gen(std::random_device{}());
142
143 if (verbose) {
144 std::cout << "========================================================\n";
145 std::cout << " 🕸️ MADS (Mesh Adaptive Direct Search) Started \n";
146 std::cout << "========================================================\n";
147 }
148
149 size_t iter = 0;
150 // alpha 값이 허용 오차보다 크고, 최대 반복 횟수를 넘지 않을 동안 탐색 지속
151 while (alpha > tol && iter < max_iter) {
152 iter++;
153 bool improved = false; // 현재 이터레이션에서 해의 개선이 있었는지 여부
154
155 // alpha에 의존적인 N + 1개의 무작위 탐색 방향 세트(Positive Spanning Set) 획득
156 auto D = generate_positive_spanning_set<N>(alpha, gen);
157
158 // [기회주의적 탐색 (Opportunistic Polling)]
159 // 모든 방향을 다 평가하는 것이 아니라, 더 좋은 해를 찾는 즉시 방향 탐색을 멈추고 이동함
160 for (size_t i = 0; i <= N; ++i) {
161 alignas(64) std::array<double, N> x_prime = {0.0};
162
163// OpenMP SIMD 프라그마를 사용하여 벡터 연산 가속 (컴파일러 지원 시)
164#pragma omp simd
165 for (size_t j = 0; j < N; ++j) {
166 // x_prime = x + alpha * D[i]
167 // std::fma(Fused Multiply-Add)를 사용하여 하드웨어 레벨에서 단일 명령어로 고속/고정밀 계산
168 x_prime[j] = std::fma(alpha, D[i][j], x[j]);
169 }
170
171 double y_prime = f(x_prime); // 새로운 위치에서 함수 평가
172
173 // 만약 새로운 위치가 기존보다 더 낫다면 (최소화 문제)
174 if (y_prime < y) {
175 x = x_prime;
176 y = y_prime;
177 improved = true;
178
179 // [가속 스텝 (Accelerated Step)]
180 // 현재 방향으로의 성과가 좋으므로, 같은 방향으로 더 멀리(3배) 뻗어봄
181 alignas(64) std::array<double, N> x_exp = {0.0};
182#pragma omp simd
183 for (size_t j = 0; j < N; ++j) {
184 x_exp[j] = std::fma(3.0 * alpha, D[i][j], x[j]);
185 }
186
187 double y_exp = f(x_exp); // 가속 지점에서 함수 평가
188
189 // 더 멀리 갔더니 결과가 더 좋으면 가속 위치 채택
190 if (y_exp < y) {
191 x = x_exp;
192 y = y_exp;
193 }
194
195 // 더 나은 점을 하나라도 발견했으므로, 나머지 방향 평가는 건너뛰고 바로 루프 탈출
196 // (기회주의적 탐색 방식이 탐색 횟수를 획기적으로 줄여줌)
197 break;
198 }
199 }
200
201 // 성과(improved)에 따른 Mesh 크기 및 스텝 폭 조절
202 if (improved) {
203 // 탐색에 성공했다면 방향의 해상도(알파)를 키움
204 // 다만 최대 alpha 값을 1.0으로 제한하여 스텝이 무한정 커지는 것 방지
205 alpha = std::min(4.0 * alpha, 1.0);
206 } else {
207 // 어떤 방향으로도 개선하지 못했다면 현재 점이 골짜기 내부에 있을 수 있음
208 // 탐색 반경(스텝 크기)을 1/4로 줄여서 더 촘촘하게 국소적 탐색을 진행함
209 alpha /= 4.0;
210 }
211
212 // 진행 상태 출력: 개선에 실패했거나(구간이 축소될 때), 10회 주기마다 출력
213 if (verbose && (!improved || iter % 10 == 0)) {
214 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
215 << std::setprecision(6) << y << " | alpha: " << alpha << "\n";
216 }
217 }
218
219 // 탐색 완료 후 시간 측정 종료
220 auto end_clock = std::chrono::high_resolution_clock::now();
221 auto duration =
222 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
223
224 if (verbose) {
225 std::cout << "========================================================\n";
226 std::cout << " 🏁 Final Optimal Point: [" << x[0];
227 if constexpr (N > 1) std::cout << ", " << x[1];
228 if constexpr (N > 2) std::cout << ", ...";
229 std::cout << "]\n========================================================\n";
230 }
231
232 // 최적해와 부가 정보를 결과 구조체에 담아 반환
233 return {x, y, iter, duration.count()};
234 }
