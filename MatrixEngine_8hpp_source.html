38//
39// [배경 설명]
40// 최적화 솔버(Layer 4)가 Jacobian 행렬이나 Hessian 행렬을 추출하는 과정에서,
41// 행렬의 원소 자료형(Type) `T`는 단순 실수형(`double`)일 수도 있고,
42// 미분값(Gradient)을 함께 추적하는 `Optimization::Dual` 타입일 수도 있습니다.
43// C++ 제네릭 프로그래밍 환경에서는 이 두 가지 완전히 다른 타입에 대해
44// 단일화된 수학 연산 인터페이스를 제공해야 컴파일 오류가 발생하지 않습니다.
45// 이를 위해 `MathTraits`라는 템플릿 구조체를 도입하여 타입별 연산 방식을 분기 처리합니다.

104template <typename T, size_t Rows, size_t Cols>
105class StaticMatrix {
106 private:
107 // ============================================================
108 // [메모리 최적화: 정렬(Alignment) 및 공간 지역성(Spatial Locality)]
109 // alignas(64): 최신 CPU(x86_64, ARM 등)의 L1 캐시 라인(Cache Line) 크기인 64바이트 단위로 메모리 시작 주소를 정렬합니다.
110 // - 캐시 히트율(Cache Hit Rate) 극대화: 프로세서가 메모리를 읽을 때 64바이트 덩어리(청크) 단위로 가져오므로, 정렬이 어긋나서 캐시 라인을 두 번 읽는 패널티(False Sharing 등)를 방지합니다.
111 // - SIMD 가속 호환: AVX-512 등의 벡터화(Vectorization) 명령어는 메모리가 정렬되어 있지 않으면 심각한 성능 저하나 하드웨어 예외를 발생시킵니다.
112 // - 이 배열은 Column-major(열 우선) 방식이므로, `data` 배열 전체가 메모리 상에 완벽하게 일렬로 연속된(Contiguous) 하나의 1차원 블록을 형성합니다.
113 // ============================================================
114 alignas(64) T data[Rows * Cols]{};
115
116 public:
117 // ============================================================
118 // 메모리 접근자 (Accessors - Column-major 방식)
119 //
120 // [Column-major(열 우선) 레이아웃의 이유]
121 // Fortran, BLAS, LAPACK, Eigen 등 세계적인 표준 수치해석 라이브러리들은 모두 Column-major를 사용합니다.
122 // 메모리 주소 계산 공식: `data[col * Rows + row]`
123 // 즉, 같은 열(Column)에 있는 원소들이 메모리 상에 연속적으로 배치되어 있습니다.
124 // 열 단위의 연산(예: 행렬 곱셈, 블록 삽입/추출) 시 공간 지역성을 극대화하여 캐시 미스(Cache Miss)를 획기적으로 줄입니다.
125 // ============================================================
126 T& operator()(int r, int c) {

128 // 릴리즈 빌드(NDEBUG 정의 시)에서는 이 assert 구문이 완전히 컴파일에서 제외되므로
129 // 오버헤드(if문 분기) 없이 최고 속도의 메모리 직접 접근이 가능해집니다.
130 assert(r >= 0 && r < static_cast<int>(Rows) && c >= 0 && c < static_cast<int>(Cols));

172 StaticMatrix<T, Rows, Cols> operator+(const StaticMatrix<T, Rows, Cols>& Other) const {
173 StaticMatrix<T, Rows, Cols> Result;
174 const size_t total = Rows * Cols;
175 for (size_t i = 0; i < total; ++i) {
176 Result.data[i] = this->data[i] + Other.data[i];
177 }
178 return Result;
179 }

199 StaticMatrix<T, Rows, Cols> operator/(T scalar) const {
200 // AD(자동 미분) 호환 Epsilon 검사를 통해 나누려는 스칼라 값이 0에 너무 가깝지는 않은지 확인합니다.
201 // 0으로 나누기(Divide by Zero) 시도 시 프로그램이 즉각 종료(Crash)되는 것을 막고 예외 처리(Exception)로 유도.
202 if (MathTraits<T>::near_zero(scalar)) {
203 throw std::invalid_argument("Division by near-zero scalar in StaticMatrix::operator/");
204 }
205 StaticMatrix<T, Rows, Cols> Result;
206 const size_t total = Rows * Cols;
207
208 // [연산 최적화 팁]
209 // CPU 명령어 사이클 측면에서 부동소수점 나눗셈(FDIV)은 곱셈(FMUL)에 비해 약 10~20배 정도 압도적으로 느립니다.
210 // 루프 내에서 N*M 번 나누는 대신, 먼저 스칼라의 역수(Inverse)를 단 1번만 구하고,
211 // 루프 안에서는 역수를 곱해주는 방식으로 변환하여 엄청난 속도 이득을 얻습니다.
212 T inv_scalar = static_cast<T>(1.0) / scalar;
213 for (size_t i = 0; i < total; ++i) {
214 Result.data[i] = this->data[i] * inv_scalar;
215 }
216 return Result;
217 }

239 template <size_t OtherCols>
240 StaticMatrix<T, Rows, OtherCols> operator*(
241 const StaticMatrix<T, Cols, OtherCols>& Other) const {
242 StaticMatrix<T, Rows, OtherCols> Result;
243 T* res_base = Result.data_ptr();
244 const T* a_base = this->data;
245
246 // J 루프: 행렬 B(Other)의 열(Column)을 이동
247 for (size_t j = 0; j < OtherCols; ++j) {
248 T* res_col = res_base + (j * Rows); // Result 행렬의 j번째 열이 시작되는 메모리 주소를 캐싱 (포인터 산술)
249
250 // K 루프: 행렬 A의 열(Column) 이동 및 행렬 B의 행(Row) 이동 (내적의 덧셈 항)
251 for (size_t k = 0; k < Cols; ++k) {
252 const T* a_col = a_base + (k * Rows); // 행렬 A의 k번째 열 시작 주소 캐싱
253 const T b_kj = Other(static_cast<int>(k), static_cast<int>(j)); // I 루프 동안 곱해질 B의 스칼라 값 고정
254
255 // I 루프 (가장 안쪽): Result의 열과 A의 열을 아래 방향으로 순차적으로 연속 접근
256 // 최적화 컴파일러는 이 루프를 감지하고 레지스터 FMA(Fused Multiply-Add) 명령어로 치환합니다.
257 for (size_t i = 0; i < Rows; ++i) {
258 res_col[i] += a_col[i] * b_kj;
259 }
260 }
261 }
262 return Result;
263 }

268 StaticMatrix<T, Cols, Rows> transpose() const {
269 StaticMatrix<T, Cols, Rows> Result;
270 // 원본 행렬의 열(j)을 순차적(Sequential)으로 쭉 읽어서,
271 // 대상 행렬의 행(j) 위치에 여기저기 흩뿌려(Scatter) 할당하는 방식입니다.
272 // 읽는 쪽이라도 캐시 연속성을 확보하기 위해 외부 루프를 j로 잡습니다.
273 for (size_t j = 0; j < Cols; ++j) {
274 const T* src_col = this->data_ptr() + (j * Rows);
275 for (size_t i = 0; i < Rows; ++i) {
276 // 대상 Result 행렬은 (j, i) 순으로 접근되어 메모리 점프가 발생하지만, 전치의 특성상 불가피합니다.
277 Result(static_cast<int>(j), static_cast<int>(i)) = src_col[i];
278 }
279 }
280 return Result;
281 }

284 // 선형 시스템 분해 솔버 (Decompositions & Solvers)
285 //
286 // [솔버 설계의 필요성]
287 // 모델 예측 제어(MPC)의 핵심은 매 주기마다 거대한 KKT 형태의 선형 연립 방정식 시스템(Ax = b)을 푸는 것입니다.
288 // 수학적으로 x = A^{-1} * b 이지만, 컴퓨터로 직접 A의 역행렬(Inverse)을 통째로 구하는 것은
289 // O(N^3)의 막대한 연산량이 소모되며 무엇보다 부동소수점 오차가 기하급수적으로 누적되어 시스템이 불안정해집니다(수치적 불안정성).
290 // 따라서 현대 수치해석에서는 반드시 행렬 A를 다루기 쉬운 두세 개의 삼각 행렬 조각으로 "분해(Decomposition)"한 뒤,
291 // 전진 대입(Forward Substitution)과 후진 대입(Backward Substitution)이라는 아주 싼 연산(O(N^2))을 통해 빠르고 정확하게 해(x)를 구합니다.

305 bool LU_decompose() {
306 // 정방 행렬이 아니면 컴파일 자체를 거부합니다.
307 static_assert(Rows == Cols, "LU Decomposition requires a square matrix");
308
309 for (size_t i = 0; i < Rows; ++i) {
310 // 1. U 행렬 부분 갱신 (i번째 행을 따라가며 j열 원소들 계산, 상삼각 영역 덮어쓰기)
311 for (size_t j = i; j < Cols; ++j) {
312 T sum = static_cast<T>(0);
313 for (size_t k = 0; k < i; ++k) {
314 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) *
315 (*this)(static_cast<int>(k), static_cast<int>(j));
316 }
317 (*this)(static_cast<int>(i), static_cast<int>(j)) -= sum;
318 }
319
320 // 피벗(Pivot) 안전성 검사. 방금 계산된 U의 대각 성분 U_ii가 피벗이 됩니다.
321 // 이 값이 0이면 하삼각행렬 L을 구할 때 분모가 0이 되어 무한대(NaN/Inf) 폭발이 발생합니다.
322 if (MathTraits<T>::near_zero((*this)(static_cast<int>(i), static_cast<int>(i)))) {
323 return false; // 특이 행렬 판정으로 즉시 분해 실패 반환
324 }
325
326 // 2. L 행렬 부분 갱신 (i번째 열을 따라가며 j행 원소들 계산, 하삼각 영역 덮어쓰기)
327 for (size_t j = i + 1; j < Rows; ++j) {
328 T sum = static_cast<T>(0);
329 for (size_t k = 0; k < i; ++k) {
330 sum += (*this)(static_cast<int>(j), static_cast<int>(k)) *
331 (*this)(static_cast<int>(k), static_cast<int>(i));
332 }
333 // 미리 구해둔 피벗 U_ii 로 나누어 스케일링
334 (*this)(static_cast<int>(j), static_cast<int>(i)) =
335 ((*this)(static_cast<int>(j), static_cast<int>(i)) - sum) /
336 (*this)(static_cast<int>(i), static_cast<int>(i));
337 }
338 }
339 return true;
340 }

354 StaticVector<T, Rows> LU_solve(const StaticVector<T, Rows>& b) const {
355 static_assert(Rows == Cols, "Solve requires a square matrix");
356 StaticVector<T, Rows> y, x;
357
358 // 1단계: 전진 대입 (Forward substitution: Ly = b)
359 // 위(i=0)에서 아래 방향으로 내려감
360 for (size_t i = 0; i < Rows; ++i) {
361 T sum = static_cast<T>(0);
362 for (size_t j = 0; j < i; ++j) {
363 sum += (*this)(static_cast<int>(i), static_cast<int>(j)) * y(j); // 하삼각 영역(L) 접근
364 }
365 // Doolittle LU에서 L의 대각 원소는 암묵적으로 항상 1.0 이므로, 나눗셈 처리가 아예 생략되어 연산이 매우 빠릅니다.
366 y(i) = b(i) - sum;
367 }
368
369 // 2단계: 후진 대입 (Back substitution: Ux = y)
370 // 행렬의 가장 아래 행(맨 밑)부터 위 방향으로 거슬러 올라감
371 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
372 T sum = static_cast<T>(0);
373 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
374 sum += (*this)(i, static_cast<int>(j)) * x(j); // 상삼각 영역(U) 접근
375 }
376 // U의 대각 성분은 1이 아니므로, 방금 구한 y값에서 sum을 빼고 U의 대각 원소로 나누어 최종 x를 구합니다.
377 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
378 }
379 return x;
380 }

394 bool Cholesky_decompose() {
395 static_assert(Rows == Cols, "Cholesky requires a square matrix");
396
397 for (size_t j = 0; j < Cols; ++j) {
398 T s = static_cast<T>(0);
399 // 1. 현재 열 j의 대각 성분(Pivot)이 될 값의 후보를 계산하기 위한 누산
400 for (size_t k = 0; k < j; ++k) {
401 s += (*this)(static_cast<int>(j), static_cast<int>(k)) *
402 (*this)(static_cast<int>(j), static_cast<int>(k));
403 }
404 // 원래 대각 원소 값에서 누산된 제곱합을 뺌
405 T d = (*this)(static_cast<int>(j), static_cast<int>(j)) - s;
406
407 // [양정치성(Positive Definiteness) 엄격 검증]
408 // 행렬이 양의 정부호라면 이 d 값은 무조건 0보다 뚜렷하게 커야 합니다.
409 // 0 이하거나 입실론 이하의 수치적 찌꺼기라면, 행렬이 볼록성(Convexity)을 잃었거나 특이행렬이 된 것이므로
410 // 치명적인 허수(Imaginary number) 발생을 막기 위해 무조건 실패(false)를 반환하고 즉시 탈출합니다.
411 if (d <= std::numeric_limits<T>::epsilon()) {
412 return false;
413 }
414 // 양수임이 확인되었으므로 안심하고 루트(sqrt)를 씌워 L의 대각 성분 확정
415 (*this)(static_cast<int>(j), static_cast<int>(j)) = MathTraits<T>::sqrt(d);
416
417 // 2. L 행렬의 비대각 성분들(현재 대각 요소보다 아래에 있는 행들) 갱신
418 for (size_t i = j + 1; i < Rows; ++i) {
419 T s_ij = static_cast<T>(0);
420 for (size_t k = 0; k < j; ++k) {
421 s_ij += (*this)(static_cast<int>(i), static_cast<int>(k)) *
422 (*this)(static_cast<int>(j), static_cast<int>(k));
423 }
424 (*this)(static_cast<int>(i), static_cast<int>(j)) =
425 ((*this)(static_cast<int>(i), static_cast<int>(j)) - s_ij) /
426 (*this)(static_cast<int>(j), static_cast<int>(j)); // 확정된 L_jj로 나누기
427 }
428 }
429 return true;
430 }

437 StaticVector<T, Rows> Cholesky_solve(const StaticVector<T, Rows>& b) const {
438 static_assert(Rows == Cols, "Solve requires a square matrix");
439 StaticVector<T, Rows> x, y;
440
441 // 1단계: 전진 대입 (Forward substitution: Ly = b)
442 for (size_t i = 0; i < Rows; ++i) {
443 T sum = static_cast<T>(0);
444 for (size_t k = 0; k < i; ++k) {
445 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) * y(k);
446 }
447 // 촐레스키 L의 대각 성분은 1이 아니므로 L_ii로 직접 나눠줘야 합니다.
448 y(i) = (b(i) - sum) / (*this)(static_cast<int>(i), static_cast<int>(i));
449 }
450
451 // 2단계: 후진 대입 (Back substitution: L^T * x = y)
452 // 주의점: 물리적인 상삼각행렬 데이터가 존재하지 않습니다.
453 // 대신 하삼각행렬 L을 논리적으로 전치(Transpose)시켰다고 가정하고,
454 // 메모리에 접근할 때 원본 L 행렬의 인덱스 접근 순서(행과 열)를 뒤집어서 읽어오는( (*this)(k, i) ) 영리한 트릭을 사용합니다.
455 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
456 T sum = static_cast<T>(0);
457 for (size_t k = static_cast<size_t>(i) + 1; k < Rows; ++k) {
458 // L^T의 (i, k) 원소는 L의 (k, i) 원소와 같습니다.
459 sum += (*this)(static_cast<int>(k), i) * x(k);
460 }
461 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
462 }
463 return x;
464 }

480 bool LDLT_decompose() {
481 static_assert(Rows == Cols, "LDLT requires a square matrix");
482
483 for (size_t j = 0; j < Cols; ++j) {
484 T sum_D = static_cast<T>(0);
485 for (size_t k = 0; k < j; ++k) {
486 T L_jk = (*this)(static_cast<int>(j), static_cast<int>(k));
487 // L의 원소 제곱에 대각 원소 D_kk를 스케일링 팩터로 곱하여 누산 (루트 대신 사용되는 핵심 트릭)
488 sum_D += L_jk * L_jk * (*this)(static_cast<int>(k), static_cast<int>(k));
489 }
490
491 // D 행렬의 대각 성분 D_jj 결정
492 T D_jj = (*this)(static_cast<int>(j), static_cast<int>(j)) - sum_D;
493
494 // 비특이성(Non-singularity) 및 양정치 검사 (0에 근접하거나 음수면 거부)
495 if (MathTraits<T>::near_zero(D_jj)) {
496 return false;
497 }
498 (*this)(static_cast<int>(j), static_cast<int>(j)) = D_jj; // 대각 원소 자리에 D 성분 저장
499
500 // L 행렬 비대각 성분 갱신
501 for (size_t i = j + 1; i < Rows; ++i) {
502 T sum_L = static_cast<T>(0);
503 for (size_t k = 0; k < j; ++k) {
504 sum_L +=
505 (*this)(static_cast<int>(i), static_cast<int>(k)) *
506 (*this)(static_cast<int>(j), static_cast<int>(k)) *
507 (*this)(static_cast<int>(k), static_cast<int>(k)); // 공식 전개: L_ik * L_jk * D_kk
508 }
509 (*this)(static_cast<int>(i), static_cast<int>(j)) =
510 ((*this)(static_cast<int>(i), static_cast<int>(j)) - sum_L) /
511 (*this)(static_cast<int>(j), static_cast<int>(j)); // 확정된 D_jj로 나누어 마무리
512 }
513 }
514 return true;
515 }

522 StaticVector<T, Rows> LDLT_solve(const StaticVector<T, Rows>& b) const {
523 static_assert(Rows == Cols, "Solve requires a square matrix");
524 StaticVector<T, Rows> z, y, x;
525
526 // 1단계: 전진 대입 Lz = b
527 // 주의: LDLT에서 L의 대각 성분은 암묵적으로 항상 1.0으로 고정되어 있으므로,
528 // 이 부분에서는 나눗셈이 발생하지 않아 연산이 쾌적합니다.
529 for (size_t i = 0; i < Rows; ++i) {
530 T sum = static_cast<T>(0);
531 for (size_t k = 0; k < i; ++k) {
532 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) * z(k);
533 }
534 z(i) = b(i) - sum;
535 }
536
537 // 2단계: 중앙 대각 행렬(D) 스케일링: Dy = z
538 // 이 한 줄의 나눗셈 과정이 Cholesky 분해의 무거운 무리수 루트(sqrt) 연산을 훌륭하게 대체하는 수학적 역할을 합니다.
539 for (size_t i = 0; i < Rows; ++i) {
540 y(i) = z(i) / (*this)(static_cast<int>(i), static_cast<int>(i));
541 }
542
543 // 3단계: 후진 대입 L^T x = y
544 // 여기서도 마찬가지로 L^T의 대각 성분은 암묵적 1이므로,
545 // 뺄셈만 하고 L_ii 로 나누는 절차는 생략됩니다.
546 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
547 T sum = static_cast<T>(0);
548 for (size_t k = static_cast<size_t>(i) + 1; k < Rows; ++k) {
549 sum += (*this)(static_cast<int>(k), i) * x(k); // 인덱스 뒤집기 트릭
550 }
551 x(static_cast<size_t>(i)) = y(static_cast<size_t>(i)) - sum;
552 }
553 return x;
554 }

573 bool QR_decompose_MGS(StaticMatrix<T, Cols, Cols>& R) {
574 static_assert(Rows >= Cols, "MGS-QR requires Rows >= Cols (Over-determined system)");
575
576 for (size_t i = 0; i < Cols; ++i) {
577 // 1. 현재 열 벡터 a_i 의 Norm(크기, 길이)을 내적을 통해 도출
578 T norm_sq = static_cast<T>(0);
579 for (size_t k = 0; k < Rows; ++k) {
580 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
581 (*this)(static_cast<int>(k), static_cast<int>(i));
582 }
583 // R의 대각 원소는 현재 열 벡터의 Norm 길이가 할당됨
584 R(static_cast<int>(i), static_cast<int>(i)) = MathTraits<T>::sqrt(norm_sq);
585
586 // 열 벡터의 길이가 사실상 0 이라면, 선형 종속성(Linear Dependency)으로 인해 직교 기저(Orthogonal basis)를 형성할 수 없으므로 포기합니다.
587 if (MathTraits<T>::near_zero(R(static_cast<int>(i), static_cast<int>(i)))) {
588 return false;
589 }
590
591 // 2. 현재 열 벡터를 앞서 구한 길이 Norm으로 나누어 크기가 1인 단위 벡터(정규화) Q_i로 변환 (원본 행렬 공간 덮어쓰기)
592 for (size_t k = 0; k < Rows; ++k) {
593 (*this)(static_cast<int>(k), static_cast<int>(i)) /=
594 R(static_cast<int>(i), static_cast<int>(i));
595 }
596
597 // 3. 투영 소거 (MGS 알고리즘의 심장부)
598 // 이제 막 확정된 정규 단위 벡터 Q_i 를 기준으로, 그보다 오른쪽에 있는 모든 열(j)들에 대해
599 // Q_i 방향으로 겹쳐진 그림자 성분(성분 길이 = 내적 dot)을 계산하고, 그 그림자 성분을 완전히 빼버려 독립적으로 만듭니다.
600 for (size_t j = i + 1; j < Cols; ++j) {
601 T dot = static_cast<T>(0);
602 for (size_t k = 0; k < Rows; ++k) {
603 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
604 (*this)(static_cast<int>(k), static_cast<int>(j));
605 }
606 // 투영된 성분의 길이를 상삼각 R 행렬에 기록
607 R(static_cast<int>(i), static_cast<int>(j)) = dot;
608
609 // 나머지 열 벡터 j 에서 해당 투영 벡터 성분을 물리적으로 빼내어 직교화 완료
610 for (size_t k = 0; k < Rows; ++k) {
611 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
612 dot * (*this)(static_cast<int>(k), static_cast<int>(i));
613 }
614 }
615 }
616 return true;
617 }

625 StaticVector<T, Cols> QR_solve(const StaticMatrix<T, Cols, Cols>& R,
626 const StaticVector<T, Rows>& b) const {
627 StaticVector<T, Cols> y, x;
628
629 // 1단계: 행렬-벡터 곱셈 (Q^T * b 연산 수행)
630 // 메모리에 저장된 *this 자체가 직교 행렬 Q입니다.
631 for (size_t i = 0; i < Cols; ++i) {
632 T dot = static_cast<T>(0);
633 for (size_t k = 0; k < Rows; ++k) {
634 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) * b(k);
635 }
636 y(i) = dot;
637 }
638
639 // 2단계: 후진 대입 (Back substitution: R * x = y)
640 // R 행렬은 외부에서 전달받은 상삼각 형태이므로 밑에서부터 채워 올라옵니다.
641 for (int i = static_cast<int>(Cols) - 1; i >= 0; --i) {
642 T sum = static_cast<T>(0);
643 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
644 sum += R(i, static_cast<int>(j)) * x(j);
645 }
646 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / R(i, i);
647 }
648 return x;
649 }

668 bool QR_decompose_Householder(StaticVector<T, Cols>& tau) {
669 static_assert(Rows >= Cols, "Householder-QR requires Rows >= Cols");
670
671 for (size_t i = 0; i < Cols; ++i) {
672 // 현재 타겟이 되는 부분 열(서브벡터 i부터 최하단 Rows까지)의 Norm 크기를 누산.
673 T norm_sq = static_cast<T>(0);
674 for (size_t k = i; k < Rows; ++k) {
675 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
676 (*this)(static_cast<int>(k), static_cast<int>(i));
677 }
678 T norm_x = MathTraits<T>::sqrt(norm_sq); // 누산 후 무거운 루트 연산은 단 1회만 실행
679
680 // 부분 벡터 전체가 완벽하게 0벡터에 가깝다면, 반사(Reflection) 액션을 수행할 필요 없이 스킵.
681 if (MathTraits<T>::near_zero(norm_x)) {
682 tau(i) = static_cast<T>(0);
683 continue;
684 }
685
686 // 부호 처리 (Sign Handling)
687 // 컴퓨터 수학의 고질적 문제인 수치 소거 오류(Cancellation Error, 비슷한 크기의 부동소수점을 뺄 때 유효 숫자가 뭉텅이로 날아가는 치명적 현상)를 방지하기 위한 안전장치입니다.
688 // 기존 대각 원소의 부호와 완전히 반대 부호를 취하여 뺄셈이 덧셈처럼 작용하도록 만듭니다.
689 T sign = ((*this)(static_cast<int>(i), static_cast<int>(i)) >= static_cast<T>(0))
690 ? static_cast<T>(1.0)
691 : static_cast<T>(-1.0);
692 T v0 = (*this)(static_cast<int>(i), static_cast<int>(i)) + sign * norm_x;
693
694 // 반사 벡터 v 구축 및 Compact하게 쑤셔넣기
695 // v의 첫 원소인 v0는 기준점이 되므로 스케일 정규화 용도로 나누고,
696 // 나머지 성분들은 행렬의 사용되지 않는 하삼각 영역에 차곡차곡 은밀하게 보관합니다.
697 for (size_t k = i + 1; k < Rows; ++k) {
698 (*this)(static_cast<int>(k), static_cast<int>(i)) /= v0;
699 }
700
701 // Tau (반사 크기 조절용 스케일 팩터) 계산
702 T v_sq_norm = static_cast<T>(1.0); // 방금 나눈 v0 부분은 정규화되어 1의 제곱으로 치급
703 for (size_t k = i + 1; k < Rows; ++k) {
704 v_sq_norm += (*this)(static_cast<int>(k), static_cast<int>(i)) *
705 (*this)(static_cast<int>(k), static_cast<int>(i));
706 }
707 tau(i) = static_cast<T>(2.0) / v_sq_norm; // 거울 반사 방정식 계수 (2.0)
708
709 // 분해된 상삼각 R 행렬의 대각 성분이 되는 부분을 최종 확정 지음
710 (*this)(static_cast<int>(i), static_cast<int>(i)) = -sign * norm_x;
711
712 // [Rank-1 Update 투영]
713 // 방금 만들어낸 거울(Householder Reflection Vector)을 이용해
714 // 현재 열보다 우측에 위치한 모든 열 벡터(A 행렬의 뒷부분)들을 전부 이 거울에 통과시켜 각도를 반사시켜 접어버립니다. (A <- (I - tau*v*v^T) A)
715 for (size_t j = i + 1; j < Cols; ++j) {
716 T dot = (*this)(static_cast<int>(i), static_cast<int>(j)); // v_0 = 1 기준 성분 초기화
717 for (size_t k = i + 1; k < Rows; ++k) {
718 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
719 (*this)(static_cast<int>(k), static_cast<int>(j));
720 }
721 T tau_dot = tau(i) * dot;
722
723 // 거울에 반사된 만큼의 각도를 원본 행렬에서 즉각 갱신(수정)
724 (*this)(static_cast<int>(i), static_cast<int>(j)) -= tau_dot;
725 for (size_t k = i + 1; k < Rows; ++k) {
726 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
727 tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
728 }
729 }
730 }
731 return true;
732 }

742 StaticVector<T, Cols> QR_solve_Householder(const StaticVector<T, Cols>& tau,
743 const StaticVector<T, Rows>& b) const {
744 StaticVector<T, Rows> y = b;
745
746 // 1단계: b 벡터에 대해 순차적 거울 반사(Sequential Reflection) 시뮬레이션 적용 (Q^T b 계산 효과)
747 for (size_t i = 0; i < Cols; ++i) {
748 // 반사가 무효화(tau가 0)된 열은 패스
749 if (MathTraits<T>::near_zero(tau(i))) {
750 continue;
751 }
752 T dot = y(i);
753 // 하삼각에 숨겨둔 벡터 v 성분을 재조립
754 for (size_t k = i + 1; k < Rows; ++k) {
755 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) * y(k);
756 }
757 T tau_dot = tau(i) * dot;
758
759 // 벡터 b(현재 임시변수 y)를 거울에 통과시켜 방향을 접음
760 y(i) -= tau_dot;
761 for (size_t k = i + 1; k < Rows; ++k) {
762 y(k) -= tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
763 }
764 }
765
766 // 2단계: 후진 대입 (Back substitution: R * x = y)
767 // R 행렬의 데이터는 원본 배열의 상삼각 영역(Upper Triangular)에 고스란히 남아있으므로 이를 활용.
768 StaticVector<T, Cols> x;
769 for (int i = static_cast<int>(Cols) - 1; i >= 0; --i) {
770 T sum = static_cast<T>(0);
771 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
772 sum += (*this)(i, static_cast<int>(j)) * x(j);
773 }
774 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
775 }
776 return x;
777 }

804 template <size_t SubRows, size_t SubCols>
805 void insert_block(size_t start_row, size_t start_col,
806 const StaticMatrix<T, SubRows, SubCols>& block) {
807 // [컴파일 타임 에러 검출] 템플릿 크기 자체가 타겟 행렬의 허용 크기를 초과하면 애초에 빌드를 차단
808 static_assert(SubRows <= Rows, "SubMatrix rows exceed target");
809 static_assert(SubCols <= Cols, "SubMatrix cols exceed target");

811 // [런타임 인덱스 경계 에러 검출] 삽입 시작 좌표가 밀려서 꼬리가 삐져나가는 현상 방지
812 assert(start_row + SubRows <= Rows && "Row index out of bounds");
813 assert(start_col + SubCols <= Cols && "Col index out of bounds");

849 block(static_cast<int>(i), static_cast<int>(j));

862 template <size_t SubRows, size_t SubCols>
863 StaticMatrix<T, SubRows, SubCols> extract_block(size_t start_row, size_t start_col) const {
864 assert(start_row + SubRows <= Rows && "Row index out of bounds");
865 assert(start_col + SubCols <= Cols && "Col index out of bounds");
866
867 StaticMatrix<T, SubRows, SubCols> result;
868 // 행은 건너뛰고 열 단위 덩어리째 접근
869 for (size_t j = 0; j < SubCols; ++j) {
870 // 거대 원본 행렬에서 도려낼 좌표 메모리 주소(포인터) 계산
871 const T* src = this->data_ptr() + ((start_col + j) * Rows) + start_row;
872 // 새롭게 뽑아낼 결과 행렬 공간의 j번째 열 시작 메모리 주소
873 T* dest = result.data_ptr() + (j * SubRows);
874 // 덩어리 단위 통째 초고속 복사
875 std::copy(src, src + SubRows, dest);
876 }
877 return result;
