164 template <size_t OtherCols>
165 StaticMatrix<T, Rows, OtherCols> operator*(
166 const StaticMatrix<T, Cols, OtherCols>& Other) const {
167 StaticMatrix<T, Rows, OtherCols> Result;
168 T* res_base = Result.data_ptr();
169 const T* a_base = this->data;
170
171 for (size_t j = 0; j < OtherCols; ++j) {
172 T* res_col = res_base + (j * Rows); // Result의 j열 시작
173 for (size_t k = 0; k < Cols; ++k) {
174 const T* a_col = a_base + (k * Rows); // A의 k열 시작
175 const T b_kj = Other(static_cast<int>(k), static_cast<int>(j));
176 for (size_t i = 0; i < Rows; ++i) {
177 res_col[i] += a_col[i] * b_kj;
178 }
179 }
180 }
181 return Result;
182 }

211 bool LU_decompose() {
212 static_assert(Rows == Cols, "LU Decomposition requires a square matrix");
213
214 for (size_t i = 0; i < Rows; ++i) {
215 // U 행 갱신
216 for (size_t j = i; j < Cols; ++j) {
217 T sum = static_cast<T>(0);
218 for (size_t k = 0; k < i; ++k) {
219 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) *
220 (*this)(static_cast<int>(k), static_cast<int>(j));
221 }
222 (*this)(static_cast<int>(i), static_cast<int>(j)) -= sum;
223 }
224
225 if (MathTraits<T>::near_zero((*this)(static_cast<int>(i), static_cast<int>(i)))) {
226 return false; // Singular Matrix
227 }
228
229 // L 열 갱신
230 for (size_t j = i + 1; j < Rows; ++j) {
231 T sum = static_cast<T>(0);
232 for (size_t k = 0; k < i; ++k) {
233 sum += (*this)(static_cast<int>(j), static_cast<int>(k)) *
234 (*this)(static_cast<int>(k), static_cast<int>(i));
235 }
236 (*this)(static_cast<int>(j), static_cast<int>(i)) =
237 ((*this)(static_cast<int>(j), static_cast<int>(i)) - sum) /
238 (*this)(static_cast<int>(i), static_cast<int>(i));
239 }
240 }
241 return true;
242 }

251 StaticVector<T, Rows> LU_solve(const StaticVector<T, Rows>& b) const {
252 static_assert(Rows == Cols, "Solve requires a square matrix");
253 StaticVector<T, Rows> y, x;
254
255 // Forward substitution: Ly = b
256 for (size_t i = 0; i < Rows; ++i) {
257 T sum = static_cast<T>(0);
258 for (size_t j = 0; j < i; ++j) {
259 sum += (*this)(static_cast<int>(i), static_cast<int>(j)) * y(j);
260 }
261 y(i) = b(i) - sum;
262 }
263
264 // Back substitution: Ux = y
265 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
266 T sum = static_cast<T>(0);
267 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
268 sum += (*this)(i, static_cast<int>(j)) * x(j);
269 }
270 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
271 }
272 return x;
273 }

284 bool Cholesky_decompose() {
285 static_assert(Rows == Cols, "Cholesky requires a square matrix");
286
287 for (size_t j = 0; j < Cols; ++j) {
288 T s = static_cast<T>(0);
289 for (size_t k = 0; k < j; ++k) {
290 s += (*this)(static_cast<int>(j), static_cast<int>(k)) *
291 (*this)(static_cast<int>(j), static_cast<int>(k));
292 }
293 T d = (*this)(static_cast<int>(j), static_cast<int>(j)) - s;
294
295 // 양정치성 검사: d > epsilon 이어야 함
296 if (d <= std::numeric_limits<T>::epsilon()) {
297 return false;
298 }
299 (*this)(static_cast<int>(j), static_cast<int>(j)) = MathTraits<T>::sqrt(d);
300
301 for (size_t i = j + 1; i < Rows; ++i) {
302 T s_ij = static_cast<T>(0);
303 for (size_t k = 0; k < j; ++k) {
304 s_ij += (*this)(static_cast<int>(i), static_cast<int>(k)) *
305 (*this)(static_cast<int>(j), static_cast<int>(k));
306 }
307 (*this)(static_cast<int>(i), static_cast<int>(j)) =
308 ((*this)(static_cast<int>(i), static_cast<int>(j)) - s_ij) /
309 (*this)(static_cast<int>(j), static_cast<int>(j));
310 }
311 }
312 return true;

359 bool LDLT_decompose() {
360 static_assert(Rows == Cols, "LDLT requires a square matrix");
361
362 for (size_t j = 0; j < Cols; ++j) {
363 T sum_D = static_cast<T>(0);
364 for (size_t k = 0; k < j; ++k) {
365 T L_jk = (*this)(static_cast<int>(j), static_cast<int>(k));
366 sum_D += L_jk * L_jk * (*this)(static_cast<int>(k), static_cast<int>(k));
367 }
368
369 T D_jj = (*this)(static_cast<int>(j), static_cast<int>(j)) - sum_D;
370
371 // 비특이성 검사 (음수 허용, 0 근접만 거부)
372 if (MathTraits<T>::near_zero(D_jj)) {
373 return false;
374 }
375 (*this)(static_cast<int>(j), static_cast<int>(j)) = D_jj;
376
377 for (size_t i = j + 1; i < Rows; ++i) {
378 T sum_L = static_cast<T>(0);
379 for (size_t k = 0; k < j; ++k) {
380 sum_L += (*this)(static_cast<int>(i), static_cast<int>(k)) *
381 (*this)(static_cast<int>(j), static_cast<int>(k)) *
382 (*this)(static_cast<int>(k), static_cast<int>(k));
383 }
384 (*this)(static_cast<int>(i), static_cast<int>(j)) =
385 ((*this)(static_cast<int>(i), static_cast<int>(j)) - sum_L) /
386 (*this)(static_cast<int>(j), static_cast<int>(j));
387 }
388 }
389 return true;
390 }

401 static_assert(Rows == Cols, "Solve requires a square matrix");

444 bool QR_decompose_MGS(StaticMatrix<T, Cols, Cols>& R) {
445 static_assert(Rows >= Cols, "MGS-QR requires Rows >= Cols");
446
447 for (size_t i = 0; i < Cols; ++i) {
448 T norm_sq = static_cast<T>(0);
449 for (size_t k = 0; k < Rows; ++k) {
450 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
451 (*this)(static_cast<int>(k), static_cast<int>(i));
452 }
453 R(static_cast<int>(i), static_cast<int>(i)) = MathTraits<T>::sqrt(norm_sq);
454
455 if (MathTraits<T>::near_zero(R(static_cast<int>(i), static_cast<int>(i)))) {
456 return false; // 선형종속 열
457 }
458
459 for (size_t k = 0; k < Rows; ++k) {
460 (*this)(static_cast<int>(k), static_cast<int>(i)) /=
461 R(static_cast<int>(i), static_cast<int>(i));
462 }
463
464 for (size_t j = i + 1; j < Cols; ++j) {
465 T dot = static_cast<T>(0);
466 for (size_t k = 0; k < Rows; ++k) {
467 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
468 (*this)(static_cast<int>(k), static_cast<int>(j));
469 }
470 R(static_cast<int>(i), static_cast<int>(j)) = dot;
471
472 for (size_t k = 0; k < Rows; ++k) {
473 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
474 dot * (*this)(static_cast<int>(k), static_cast<int>(i));
475 }
476 }
477 }
478 return true;
479 }

529 bool QR_decompose_Householder(StaticVector<T, Cols>& tau) {
530 static_assert(Rows >= Cols, "Householder-QR requires Rows >= Cols");
531
532 for (size_t i = 0; i < Cols; ++i) {
533 // norm 누산 완료 후 sqrt — 루프 분리가 핵심
534 T norm_sq = static_cast<T>(0);
535 for (size_t k = i; k < Rows; ++k) {
536 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
537 (*this)(static_cast<int>(k), static_cast<int>(i));
538 }
539 T norm_x = MathTraits<T>::sqrt(norm_sq);
540
541 if (MathTraits<T>::near_zero(norm_x)) {
542 tau(i) = static_cast<T>(0);
543 continue;
544 }
545
546 T sign = ((*this)(static_cast<int>(i), static_cast<int>(i)) >= static_cast<T>(0))
547 ? static_cast<T>(1.0)
548 : static_cast<T>(-1.0);
549 T v0 = (*this)(static_cast<int>(i), static_cast<int>(i)) + sign * norm_x;
550
551 // Householder 벡터 compact 저장 (하삼각 영역)
552 for (size_t k = i + 1; k < Rows; ++k) {
553 (*this)(static_cast<int>(k), static_cast<int>(i)) /= v0;
554 }
555
556 T v_sq_norm = static_cast<T>(1.0);
557 for (size_t k = i + 1; k < Rows; ++k) {
558 v_sq_norm += (*this)(static_cast<int>(k), static_cast<int>(i)) *
559 (*this)(static_cast<int>(k), static_cast<int>(i));
560 }
561 tau(i) = static_cast<T>(2.0) / v_sq_norm;
562
563 (*this)(static_cast<int>(i), static_cast<int>(i)) = -sign * norm_x;
564
565 // (I - tau*v*v^T) A 적용 (j > i 열에 대해 rank-1 update)
566 for (size_t j = i + 1; j < Cols; ++j) {
567 T dot = (*this)(static_cast<int>(i), static_cast<int>(j));
568 for (size_t k = i + 1; k < Rows; ++k) {
569 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
570 (*this)(static_cast<int>(k), static_cast<int>(j));
571 }
572 T tau_dot = tau(i) * dot;
573 (*this)(static_cast<int>(i), static_cast<int>(j)) -= tau_dot;
574 for (size_t k = i + 1; k < Rows; ++k) {
575 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
576 tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
577 }
578 }
579 }
580 return true;
581 }

592 StaticVector<T, Cols> QR_solve_Householder(const StaticVector<T, Cols>& tau,
593 const StaticVector<T, Rows>& b) const {
594 StaticVector<T, Rows> y = b;
595
596 // Q^T b 적용 (Householder reflections 순차 적용)
597 for (size_t i = 0; i < Cols; ++i) {
598 if (MathTraits<T>::near_zero(tau(i))) {
599 continue;
600 }
601 T dot = y(i);
602 for (size_t k = i + 1; k < Rows; ++k) {
603 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) * y(k);
604 }
605 T tau_dot = tau(i) * dot;
606 y(i) -= tau_dot;
607 for (size_t k = i + 1; k < Rows; ++k) {
608 y(k) -= tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
609 }
610 }
611
612 // Back substitution: Rx = Q^T b
613 StaticVector<T, Cols> x;
614 for (int i = static_cast<int>(Cols) - 1; i >= 0; --i) {
615 T sum = static_cast<T>(0);
616 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
617 sum += (*this)(i, static_cast<int>(j)) * x(j);
618 }
619 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
620 }
621 return x;
622 }

696 template <size_t SubRows, size_t SubCols>
697 StaticMatrix<T, SubRows, SubCols> extract_block(size_t start_row, size_t start_col) const {
698 assert(start_row + SubRows <= Rows && "Row index out of bounds");
699 assert(start_col + SubCols <= Cols && "Col index out of bounds");
700
701 StaticMatrix<T, SubRows, SubCols> result;
702 for (size_t j = 0; j < SubCols; ++j) {
