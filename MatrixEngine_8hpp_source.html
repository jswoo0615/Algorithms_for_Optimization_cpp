37// ============================================================
38// AD(Auto Differentiation, 자동 미분) 호환 Traits 구조체
39//
40// [배경 설명]
41// 최적화 솔버(Layer 4)가 Jacobian 행렬이나 Hessian 행렬을 추출하는 과정에서,
42// 행렬의 원소 자료형(Type) `T`는 단순 실수형(`double`)일 수도 있고,
43// 미분값(Gradient)을 함께 추적하는 `Optimization::Dual` 타입일 수도 있습니다.
44// C++ 제네릭 프로그래밍 환경에서는 이 두 가지 완전히 다른 타입에 대해
45// 단일화된 수학 연산 인터페이스를 제공해야 컴파일 오류가 발생하지 않습니다.
46// 이를 위해 `MathTraits`라는 템플릿 구조체를 도입하여 타입별 연산 방식을 분기 처리합니다.
47// ============================================================
48template <typename T>
49struct MathTraits {
50 // 일반 스칼라 타입(예: double, float)에 대한 수학 함수 매핑
51 static T abs(const T& x) { return std::abs(x); }
52 static T sqrt(const T& x) { return std::sqrt(x); }
53
64 static bool near_zero(const T& x) { return std::abs(x) <= std::numeric_limits<T>::epsilon(); }
65};

67// Optimization::Dual 타입에 대한 템플릿 특수화(Specialization)
68template <typename T>
69struct MathTraits<Optimization::Dual<T>> {
70 // Dual 타입은 내부에 실수부(v)와 미분부(d)를 모두 가지므로,
71 // Dual 헤더에 오버로딩된 특수 수학 함수를 명시적으로 호출합니다.
72 static Optimization::Dual<T> abs(const Optimization::Dual<T>& x) {
73 return Optimization::ad::abs(x); // Dual 내부의 오버로딩된 abs 호출
74 }
75 static Optimization::Dual<T> sqrt(const Optimization::Dual<T>& x) {
76 return Optimization::ad::sqrt(x); // Dual 내부의 오버로딩된 sqrt 호출
77 }
78
88 static bool near_zero(const Optimization::Dual<T>& x) {
89 return std::abs(x.v) <= std::numeric_limits<T>::epsilon();
90 }

108template <typename T, size_t Rows, size_t Cols>
109class StaticMatrix {
110 private:
111 // ============================================================
112 // [메모리 최적화: 정렬(Alignment) 및 공간 지역성(Spatial Locality)]
113 // alignas(64): 최신 CPU(x86_64, ARM 등)의 L1 캐시 라인(Cache Line) 크기인 64바이트 단위로
114 // 메모리 시작 주소를 정렬합니다.
115 // - 캐시 히트율(Cache Hit Rate) 극대화: 프로세서가 메모리를 읽을 때 64바이트 덩어리(청크)
116 // 단위로 가져오므로, 정렬이 어긋나서 캐시 라인을 두 번 읽는 패널티(False Sharing 등)를
117 // 방지합니다.
118 // - SIMD 가속 호환: AVX-512 등의 벡터화(Vectorization) 명령어는 메모리가 정렬되어 있지 않으면
119 // 심각한 성능 저하나 하드웨어 예외를 발생시킵니다.
120 // - 이 배열은 Column-major(열 우선) 방식이므로, `data` 배열 전체가 메모리 상에 완벽하게
121 // 일렬로 연속된(Contiguous) 하나의 1차원 블록을 형성합니다.
122 // ============================================================
123 alignas(64) T data[Rows * Cols]{};
124
125 public:
126 // ============================================================
127 // 메모리 접근자 (Accessors - Column-major 방식)
128 //
129 // [Column-major(열 우선) 레이아웃의 이유]
130 // Fortran, BLAS, LAPACK, Eigen 등 세계적인 표준 수치해석 라이브러리들은 모두 Column-major를
131 // 사용합니다. 메모리 주소 계산 공식: `data[col * Rows + row]` 즉, 같은 열(Column)에 있는
132 // 원소들이 메모리 상에 연속적으로 배치되어 있습니다. 열 단위의 연산(예: 행렬 곱셈, 블록
133 // 삽입/추출) 시 공간 지역성을 극대화하여 캐시 미스(Cache Miss)를 획기적으로 줄입니다.
134 // ============================================================
135 T& operator()(int r, int c) {
136 // 인덱스 안전성 검증.
137 // 릴리즈 빌드(NDEBUG 정의 시)에서는 이 assert 구문이 완전히 컴파일에서 제외되므로
138 // 오버헤드(if문 분기) 없이 최고 속도의 메모리 직접 접근이 가능해집니다.
139 assert(r >= 0 && r < static_cast<int>(Rows) && c >= 0 && c < static_cast<int>(Cols));
140 return data[c * Rows + r];
141 }
142 const T& operator()(int r, int c) const {
143 assert(r >= 0 && r < static_cast<int>(Rows) && c >= 0 && c < static_cast<int>(Cols));
144 return data[c * Rows + r];
145 }

174 // ============================================================
175 // 기본 선형 대수 연산 (Algebraic Operators)
176 //
177 // 두 행렬의 덧셈(+), 뺄셈(-) 및 스칼라배(*, /)는 각 성분끼리 매칭되는
178 // 독립적인 요소별(Element-wise) 연산입니다.
179 // 따라서 2차원(행, 열) 인덱스를 계산하며 순회할 필요 없이,
180 // 1차원 선형 배열(data)을 0번 인덱스부터 끝까지 한 번에 쭉 순회(Linear Scan)하여 루프
181 // 오버헤드를 최소화합니다.
182 // ============================================================
183 StaticMatrix<T, Rows, Cols> operator+(const StaticMatrix<T, Rows, Cols>& Other) const {
184 StaticMatrix<T, Rows, Cols> Result;
185 const size_t total = Rows * Cols;
186 for (size_t i = 0; i < total; ++i) {
187 Result.data[i] = this->data[i] + Other.data[i];
188 }
189 return Result;
190 }

195 for (size_t i = 0; i < total; ++i) {
196 Result.data[i] = this->data[i] - Other.data[i];
197 }
198 return Result;

210 StaticMatrix<T, Rows, Cols> operator/(T scalar) const {
211 // AD(자동 미분) 호환 Epsilon 검사를 통해 나누려는 스칼라 값이 0에 너무 가깝지는 않은지
212 // 확인합니다. 0으로 나누기(Divide by Zero) 시도 시 프로그램이 즉각 종료(Crash)되는 것을
213 // 막고 예외 처리(Exception)로 유도.
214 if (MathTraits<T>::near_zero(scalar)) {
215 throw std::invalid_argument("Division by near-zero scalar in StaticMatrix::operator/");
216 }
217 StaticMatrix<T, Rows, Cols> Result;
218 const size_t total = Rows * Cols;
219
220 // [연산 최적화 팁]
221 // CPU 명령어 사이클 측면에서 부동소수점 나눗셈(FDIV)은 곱셈(FMUL)에 비해 약 10~20배 정도
222 // 압도적으로 느립니다. 루프 내에서 N*M 번 나누는 대신, 먼저 스칼라의 역수(Inverse)를 단
223 // 1번만 구하고, 루프 안에서는 역수를 곱해주는 방식으로 변환하여 엄청난 속도 이득을
224 // 얻습니다.
225 T inv_scalar = static_cast<T>(1.0) / scalar;
226 for (size_t i = 0; i < total; ++i) {
227 Result.data[i] = this->data[i] * inv_scalar;
228 }
229 return Result;
230 }

255 template <size_t OtherCols>
256 StaticMatrix<T, Rows, OtherCols> operator*(
257 const StaticMatrix<T, Cols, OtherCols>& Other) const {
258 StaticMatrix<T, Rows, OtherCols> Result;
259 T* res_base = Result.data_ptr();
260 const T* a_base = this->data;
261
262 // J 루프: 행렬 B(Other)의 열(Column)을 이동
263 for (size_t j = 0; j < OtherCols; ++j) {
264 T* res_col =
265 res_base +
266 (j * Rows); // Result 행렬의 j번째 열이 시작되는 메모리 주소를 캐싱 (포인터 산술)
267
268 // K 루프: 행렬 A의 열(Column) 이동 및 행렬 B의 행(Row) 이동 (내적의 덧셈 항)
269 for (size_t k = 0; k < Cols; ++k) {
270 const T* a_col = a_base + (k * Rows); // 행렬 A의 k번째 열 시작 주소 캐싱
271 const T b_kj = Other(static_cast<int>(k),
272 static_cast<int>(j)); // I 루프 동안 곱해질 B의 스칼라 값 고정
273
274 // I 루프 (가장 안쪽): Result의 열과 A의 열을 아래 방향으로 순차적으로 연속 접근
275 // 최적화 컴파일러는 이 루프를 감지하고 레지스터 FMA(Fused Multiply-Add) 명령어로
276 // 치환합니다.
277 for (size_t i = 0; i < Rows; ++i) {
278 res_col[i] += a_col[i] * b_kj;
279 }
280 }
281 }
282 return Result;
283 }

286 // 행렬 전치 (Matrix Transpose)

288 StaticMatrix<T, Cols, Rows> transpose() const {
289 StaticMatrix<T, Cols, Rows> Result;
290 // 원본 행렬의 열(j)을 순차적(Sequential)으로 쭉 읽어서,
291 // 대상 행렬의 행(j) 위치에 여기저기 흩뿌려(Scatter) 할당하는 방식입니다.
292 // 읽는 쪽이라도 캐시 연속성을 확보하기 위해 외부 루프를 j로 잡습니다.
293 for (size_t j = 0; j < Cols; ++j) {
294 const T* src_col = this->data_ptr() + (j * Rows);
295 for (size_t i = 0; i < Rows; ++i) {
296 // 대상 Result 행렬은 (j, i) 순으로 접근되어 메모리 점프가 발생하지만, 전치의 특성상
297 // 불가피합니다.
298 Result(static_cast<int>(j), static_cast<int>(i)) = src_col[i];
299 }
300 }
301 return Result;
302 }

305 // 선형 시스템 분해 솔버 (Decompositions & Solvers)
306 //
307 // [솔버 설계의 필요성]
308 // 모델 예측 제어(MPC)의 핵심은 매 주기마다 거대한 KKT 형태의 선형 연립 방정식 시스템(Ax = b)을
309 // 푸는 것입니다. 수학적으로 x = A^{-1} * b 이지만, 컴퓨터로 직접 A의 역행렬(Inverse)을 통째로
310 // 구하는 것은 O(N^3)의 막대한 연산량이 소모되며 무엇보다 부동소수점 오차가 기하급수적으로
311 // 누적되어 시스템이 불안정해집니다(수치적 불안정성). 따라서 현대 수치해석에서는 반드시 행렬 A를
312 // 다루기 쉬운 두세 개의 삼각 행렬 조각으로 "분해(Decomposition)"한 뒤, 전진 대입(Forward
313 // Substitution)과 후진 대입(Backward Substitution)이라는 아주 싼 연산(O(N^2))을 통해 빠르고
314 // 정확하게 해(x)를 구합니다.
315 // ============================================================

330 bool LU_decompose() {
331 // 정방 행렬이 아니면 컴파일 자체를 거부합니다.
332 static_assert(Rows == Cols, "LU Decomposition requires a square matrix");
333
334 for (size_t i = 0; i < Rows; ++i) {
335 // 1. U 행렬 부분 갱신 (i번째 행을 따라가며 j열 원소들 계산, 상삼각 영역 덮어쓰기)
336 for (size_t j = i; j < Cols; ++j) {
337 T sum = static_cast<T>(0);
338 for (size_t k = 0; k < i; ++k) {
339 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) *
340 (*this)(static_cast<int>(k), static_cast<int>(j));
341 }
342 (*this)(static_cast<int>(i), static_cast<int>(j)) -= sum;
343 }
344
345 // 피벗(Pivot) 안전성 검사. 방금 계산된 U의 대각 성분 U_ii가 피벗이 됩니다.
346 // 이 값이 0이면 하삼각행렬 L을 구할 때 분모가 0이 되어 무한대(NaN/Inf) 폭발이
347 // 발생합니다.
348 if (MathTraits<T>::near_zero((*this)(static_cast<int>(i), static_cast<int>(i)))) {
349 return false; // 특이 행렬 판정으로 즉시 분해 실패 반환
350 }
351
352 // 2. L 행렬 부분 갱신 (i번째 열을 따라가며 j행 원소들 계산, 하삼각 영역 덮어쓰기)
353 for (size_t j = i + 1; j < Rows; ++j) {
354 T sum = static_cast<T>(0);
355 for (size_t k = 0; k < i; ++k) {
356 sum += (*this)(static_cast<int>(j), static_cast<int>(k)) *
357 (*this)(static_cast<int>(k), static_cast<int>(i));
358 }
359 // 미리 구해둔 피벗 U_ii 로 나누어 스케일링
360 (*this)(static_cast<int>(j), static_cast<int>(i)) =
361 ((*this)(static_cast<int>(j), static_cast<int>(i)) - sum) /
362 (*this)(static_cast<int>(i), static_cast<int>(i));
363 }
364 }
365 return true;
366 }

383 StaticVector<T, Rows> LU_solve(const StaticVector<T, Rows>& b) const {
384 static_assert(Rows == Cols, "Solve requires a square matrix");
385 StaticVector<T, Rows> y, x;
386
387 // 1단계: 전진 대입 (Forward substitution: Ly = b)
388 // 위(i=0)에서 아래 방향으로 내려감
389 for (size_t i = 0; i < Rows; ++i) {
390 T sum = static_cast<T>(0);
391 for (size_t j = 0; j < i; ++j) {
392 sum += (*this)(static_cast<int>(i), static_cast<int>(j)) *
393 y(j); // 하삼각 영역(L) 접근
394 }
395 // Doolittle LU에서 L의 대각 원소는 암묵적으로 항상 1.0 이므로, 나눗셈 처리가 아예
396 // 생략되어 연산이 매우 빠릅니다.
397 y(i) = b(i) - sum;
398 }
399
400 // 2단계: 후진 대입 (Back substitution: Ux = y)
401 // 행렬의 가장 아래 행(맨 밑)부터 위 방향으로 거슬러 올라감
402 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
403 T sum = static_cast<T>(0);
404 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
405 sum += (*this)(i, static_cast<int>(j)) * x(j); // 상삼각 영역(U) 접근
406 }
407 // U의 대각 성분은 1이 아니므로, 방금 구한 y값에서 sum을 빼고 U의 대각 원소로 나누어
408 // 최종 x를 구합니다.
409 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
410 }
411 return x;
412 }

430 bool Cholesky_decompose() {
431 static_assert(Rows == Cols, "Cholesky requires a square matrix");
432
433 for (size_t j = 0; j < Cols; ++j) {
434 T s = static_cast<T>(0);
435 // 1. 현재 열 j의 대각 성분(Pivot)이 될 값의 후보를 계산하기 위한 누산
436 for (size_t k = 0; k < j; ++k) {
437 s += (*this)(static_cast<int>(j), static_cast<int>(k)) *
438 (*this)(static_cast<int>(j), static_cast<int>(k));
439 }
440 // 원래 대각 원소 값에서 누산된 제곱합을 뺌
441 T d = (*this)(static_cast<int>(j), static_cast<int>(j)) - s;
442
443 // [양정치성(Positive Definiteness) 엄격 검증]
444 // 행렬이 양의 정부호라면 이 d 값은 무조건 0보다 뚜렷하게 커야 합니다.
445 // 0 이하거나 입실론 이하의 수치적 찌꺼기라면, 행렬이 볼록성(Convexity)을 잃었거나
446 // 특이행렬이 된 것이므로 치명적인 허수(Imaginary number) 발생을 막기 위해 무조건
447 // 실패(false)를 반환하고 즉시 탈출합니다.
448 if (d <= std::numeric_limits<T>::epsilon()) {
449 return false;
450 }
451 // 양수임이 확인되었으므로 안심하고 루트(sqrt)를 씌워 L의 대각 성분 확정
452 (*this)(static_cast<int>(j), static_cast<int>(j)) = MathTraits<T>::sqrt(d);
453
454 // 2. L 행렬의 비대각 성분들(현재 대각 요소보다 아래에 있는 행들) 갱신
455 for (size_t i = j + 1; i < Rows; ++i) {
456 T s_ij = static_cast<T>(0);
457 for (size_t k = 0; k < j; ++k) {
458 s_ij += (*this)(static_cast<int>(i), static_cast<int>(k)) *
459 (*this)(static_cast<int>(j), static_cast<int>(k));
460 }
461 (*this)(static_cast<int>(i), static_cast<int>(j)) =
462 ((*this)(static_cast<int>(i), static_cast<int>(j)) - s_ij) /
463 (*this)(static_cast<int>(j), static_cast<int>(j)); // 확정된 L_jj로 나누기
464 }
465 }
466 return true;
467 }

475 StaticVector<T, Rows> Cholesky_solve(const StaticVector<T, Rows>& b) const {
476 static_assert(Rows == Cols, "Solve requires a square matrix");
477 StaticVector<T, Rows> x, y;
478
479 // 1단계: 전진 대입 (Forward substitution: Ly = b)
480 for (size_t i = 0; i < Rows; ++i) {
481 T sum = static_cast<T>(0);
482 for (size_t k = 0; k < i; ++k) {
483 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) * y(k);
484 }
485 // 촐레스키 L의 대각 성분은 1이 아니므로 L_ii로 직접 나눠줘야 합니다.
486 y(i) = (b(i) - sum) / (*this)(static_cast<int>(i), static_cast<int>(i));
487 }
488
489 // 2단계: 후진 대입 (Back substitution: L^T * x = y)
490 // 주의점: 물리적인 상삼각행렬 데이터가 존재하지 않습니다.
491 // 대신 하삼각행렬 L을 논리적으로 전치(Transpose)시켰다고 가정하고,
492 // 메모리에 접근할 때 원본 L 행렬의 인덱스 접근 순서(행과 열)를 뒤집어서 읽어오는(
493 // (*this)(k, i) ) 영리한 트릭을 사용합니다.
494 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
495 T sum = static_cast<T>(0);
496 for (size_t k = static_cast<size_t>(i) + 1; k < Rows; ++k) {
497 // L^T의 (i, k) 원소는 L의 (k, i) 원소와 같습니다.
498 sum += (*this)(static_cast<int>(k), i) * x(k);
499 }
500 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
501 }
502 return x;
503 }

522 bool LDLT_decompose() {
523 static_assert(Rows == Cols, "LDLT requires a square matrix");
524
525 for (size_t j = 0; j < Cols; ++j) {
526 T sum_D = static_cast<T>(0);
527 for (size_t k = 0; k < j; ++k) {
528 T L_jk = (*this)(static_cast<int>(j), static_cast<int>(k));
529 // L의 원소 제곱에 대각 원소 D_kk를 스케일링 팩터로 곱하여 누산 (루트 대신 사용되는
530 // 핵심 트릭)
531 sum_D += L_jk * L_jk * (*this)(static_cast<int>(k), static_cast<int>(k));
532 }
533
534 // D 행렬의 대각 성분 D_jj 결정
535 T D_jj = (*this)(static_cast<int>(j), static_cast<int>(j)) - sum_D;
536
537 // 비특이성(Non-singularity) 및 양정치 검사 (0에 근접하거나 음수면 거부)
538 if (MathTraits<T>::near_zero(D_jj)) {
539 return false;
540 }
541 (*this)(static_cast<int>(j), static_cast<int>(j)) =
542 D_jj; // 대각 원소 자리에 D 성분 저장
543
544 // L 행렬 비대각 성분 갱신
545 for (size_t i = j + 1; i < Rows; ++i) {
546 T sum_L = static_cast<T>(0);
547 for (size_t k = 0; k < j; ++k) {
548 sum_L += (*this)(static_cast<int>(i), static_cast<int>(k)) *
549 (*this)(static_cast<int>(j), static_cast<int>(k)) *
550 (*this)(static_cast<int>(k),
551 static_cast<int>(k)); // 공식 전개: L_ik * L_jk * D_kk
552 }
553 (*this)(static_cast<int>(i), static_cast<int>(j)) =
554 ((*this)(static_cast<int>(i), static_cast<int>(j)) - sum_L) /
555 (*this)(static_cast<int>(j),
556 static_cast<int>(j)); // 확정된 D_jj로 나누어 마무리
557 }
558 }
559 return true;
560 }

567 StaticVector<T, Rows> LDLT_solve(const StaticVector<T, Rows>& b) const {
568 static_assert(Rows == Cols, "Solve requires a square matrix");
569 StaticVector<T, Rows> z, y, x;
570
571 // 1단계: 전진 대입 Lz = b
572 // 주의: LDLT에서 L의 대각 성분은 암묵적으로 항상 1.0으로 고정되어 있으므로,
573 // 이 부분에서는 나눗셈이 발생하지 않아 연산이 쾌적합니다.
574 for (size_t i = 0; i < Rows; ++i) {
575 T sum = static_cast<T>(0);
576 for (size_t k = 0; k < i; ++k) {
577 sum += (*this)(static_cast<int>(i), static_cast<int>(k)) * z(k);
578 }
579 z(i) = b(i) - sum;
580 }
581
582 // 2단계: 중앙 대각 행렬(D) 스케일링: Dy = z
583 // 이 한 줄의 나눗셈 과정이 Cholesky 분해의 무거운 무리수 루트(sqrt) 연산을 훌륭하게
584 // 대체하는 수학적 역할을 합니다.
585 for (size_t i = 0; i < Rows; ++i) {
586 y(i) = z(i) / (*this)(static_cast<int>(i), static_cast<int>(i));
587 }
588
589 // 3단계: 후진 대입 L^T x = y
590 // 여기서도 마찬가지로 L^T의 대각 성분은 암묵적 1이므로,
591 // 뺄셈만 하고 L_ii 로 나누는 절차는 생략됩니다.
592 for (int i = static_cast<int>(Rows) - 1; i >= 0; --i) {
593 T sum = static_cast<T>(0);
594 for (size_t k = static_cast<size_t>(i) + 1; k < Rows; ++k) {
595 sum += (*this)(static_cast<int>(k), i) * x(k); // 인덱스 뒤집기 트릭
596 }
597 x(static_cast<size_t>(i)) = y(static_cast<size_t>(i)) - sum;
598 }
599 return x;
600 }

622 bool QR_decompose_MGS(StaticMatrix<T, Cols, Cols>& R) {
623 static_assert(Rows >= Cols, "MGS-QR requires Rows >= Cols (Over-determined system)");
624
625 for (size_t i = 0; i < Cols; ++i) {
626 // 1. 현재 열 벡터 a_i 의 Norm(크기, 길이)을 내적을 통해 도출
627 T norm_sq = static_cast<T>(0);
628 for (size_t k = 0; k < Rows; ++k) {
629 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
630 (*this)(static_cast<int>(k), static_cast<int>(i));
631 }
632 // R의 대각 원소는 현재 열 벡터의 Norm 길이가 할당됨
633 R(static_cast<int>(i), static_cast<int>(i)) = MathTraits<T>::sqrt(norm_sq);
634
635 // 열 벡터의 길이가 사실상 0 이라면, 선형 종속성(Linear Dependency)으로 인해 직교
636 // 기저(Orthogonal basis)를 형성할 수 없으므로 포기합니다.
637 if (MathTraits<T>::near_zero(R(static_cast<int>(i), static_cast<int>(i)))) {
638 return false;
639 }
640
641 // 2. 현재 열 벡터를 앞서 구한 길이 Norm으로 나누어 크기가 1인 단위 벡터(정규화) Q_i로
642 // 변환 (원본 행렬 공간 덮어쓰기)
643 for (size_t k = 0; k < Rows; ++k) {
644 (*this)(static_cast<int>(k), static_cast<int>(i)) /=
645 R(static_cast<int>(i), static_cast<int>(i));
646 }
647
648 // 3. 투영 소거 (MGS 알고리즘의 심장부)
649 // 이제 막 확정된 정규 단위 벡터 Q_i 를 기준으로, 그보다 오른쪽에 있는 모든 열(j)들에
650 // 대해 Q_i 방향으로 겹쳐진 그림자 성분(성분 길이 = 내적 dot)을 계산하고, 그 그림자
651 // 성분을 완전히 빼버려 독립적으로 만듭니다.
652 for (size_t j = i + 1; j < Cols; ++j) {
653 T dot = static_cast<T>(0);
654 for (size_t k = 0; k < Rows; ++k) {
655 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
656 (*this)(static_cast<int>(k), static_cast<int>(j));
657 }
658 // 투영된 성분의 길이를 상삼각 R 행렬에 기록
659 R(static_cast<int>(i), static_cast<int>(j)) = dot;
660
661 // 나머지 열 벡터 j 에서 해당 투영 벡터 성분을 물리적으로 빼내어 직교화 완료
662 for (size_t k = 0; k < Rows; ++k) {
663 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
664 dot * (*this)(static_cast<int>(k), static_cast<int>(i));
665 }
666 }
667 }
668 return true;
669 }

678 StaticVector<T, Cols> QR_solve(const StaticMatrix<T, Cols, Cols>& R,
679 const StaticVector<T, Rows>& b) const {
680 StaticVector<T, Cols> y, x;
681
682 // 1단계: 행렬-벡터 곱셈 (Q^T * b 연산 수행)
683 // 메모리에 저장된 *this 자체가 직교 행렬 Q입니다.
684 for (size_t i = 0; i < Cols; ++i) {
685 T dot = static_cast<T>(0);
686 for (size_t k = 0; k < Rows; ++k) {
687 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) * b(k);
688 }
689 y(i) = dot;
690 }
691
692 // 2단계: 후진 대입 (Back substitution: R * x = y)
693 // R 행렬은 외부에서 전달받은 상삼각 형태이므로 밑에서부터 채워 올라옵니다.
694 for (int i = static_cast<int>(Cols) - 1; i >= 0; --i) {
695 T sum = static_cast<T>(0);
696 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
697 sum += R(i, static_cast<int>(j)) * x(j);
698 }
699 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / R(i, i);
700 }
701 return x;
702 }

725 bool QR_decompose_Householder(StaticVector<T, Cols>& tau) {
726 static_assert(Rows >= Cols, "Householder-QR requires Rows >= Cols");
727
728 for (size_t i = 0; i < Cols; ++i) {
729 // 현재 타겟이 되는 부분 열(서브벡터 i부터 최하단 Rows까지)의 Norm 크기를 누산.
730 T norm_sq = static_cast<T>(0);
731 for (size_t k = i; k < Rows; ++k) {
732 norm_sq += (*this)(static_cast<int>(k), static_cast<int>(i)) *
733 (*this)(static_cast<int>(k), static_cast<int>(i));
734 }
735 T norm_x = MathTraits<T>::sqrt(norm_sq); // 누산 후 무거운 루트 연산은 단 1회만 실행
736
737 // 부분 벡터 전체가 완벽하게 0벡터에 가깝다면, 반사(Reflection) 액션을 수행할 필요 없이
738 // 스킵.
739 if (MathTraits<T>::near_zero(norm_x)) {
740 tau(i) = static_cast<T>(0);
741 continue;
742 }
743
744 // 부호 처리 (Sign Handling)
745 // 컴퓨터 수학의 고질적 문제인 수치 소거 오류(Cancellation Error, 비슷한 크기의
746 // 부동소수점을 뺄 때 유효 숫자가 뭉텅이로 날아가는 치명적 현상)를 방지하기 위한
747 // 안전장치입니다. 기존 대각 원소의 부호와 완전히 반대 부호를 취하여 뺄셈이 덧셈처럼
748 // 작용하도록 만듭니다.
749 T sign = ((*this)(static_cast<int>(i), static_cast<int>(i)) >= static_cast<T>(0))
750 ? static_cast<T>(1.0)
751 : static_cast<T>(-1.0);
752 T v0 = (*this)(static_cast<int>(i), static_cast<int>(i)) + sign * norm_x;
753
754 // 반사 벡터 v 구축 및 Compact하게 쑤셔넣기
755 // v의 첫 원소인 v0는 기준점이 되므로 스케일 정규화 용도로 나누고,
756 // 나머지 성분들은 행렬의 사용되지 않는 하삼각 영역에 차곡차곡 은밀하게 보관합니다.
757 for (size_t k = i + 1; k < Rows; ++k) {
758 (*this)(static_cast<int>(k), static_cast<int>(i)) /= v0;
759 }
760
761 // Tau (반사 크기 조절용 스케일 팩터) 계산
762 T v_sq_norm = static_cast<T>(1.0); // 방금 나눈 v0 부분은 정규화되어 1의 제곱으로 치급
763 for (size_t k = i + 1; k < Rows; ++k) {
764 v_sq_norm += (*this)(static_cast<int>(k), static_cast<int>(i)) *
765 (*this)(static_cast<int>(k), static_cast<int>(i));
766 }
767 tau(i) = static_cast<T>(2.0) / v_sq_norm; // 거울 반사 방정식 계수 (2.0)
768
769 // 분해된 상삼각 R 행렬의 대각 성분이 되는 부분을 최종 확정 지음
770 (*this)(static_cast<int>(i), static_cast<int>(i)) = -sign * norm_x;
771
772 // [Rank-1 Update 투영]
773 // 방금 만들어낸 거울(Householder Reflection Vector)을 이용해
774 // 현재 열보다 우측에 위치한 모든 열 벡터(A 행렬의 뒷부분)들을 전부 이 거울에 통과시켜
775 // 각도를 반사시켜 접어버립니다. (A <- (I - tau*v*v^T) A)
776 for (size_t j = i + 1; j < Cols; ++j) {
777 T dot =
778 (*this)(static_cast<int>(i), static_cast<int>(j)); // v_0 = 1 기준 성분 초기화
779 for (size_t k = i + 1; k < Rows; ++k) {
780 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) *
781 (*this)(static_cast<int>(k), static_cast<int>(j));
782 }
783 T tau_dot = tau(i) * dot;
784
785 // 거울에 반사된 만큼의 각도를 원본 행렬에서 즉각 갱신(수정)
786 (*this)(static_cast<int>(i), static_cast<int>(j)) -= tau_dot;
787 for (size_t k = i + 1; k < Rows; ++k) {
788 (*this)(static_cast<int>(k), static_cast<int>(j)) -=
789 tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
790 }
791 }
792 }
793 return true;
794 }

805 StaticVector<T, Cols> QR_solve_Householder(const StaticVector<T, Cols>& tau,
806 const StaticVector<T, Rows>& b) const {
807 StaticVector<T, Rows> y = b;
808
809 // 1단계: b 벡터에 대해 순차적 거울 반사(Sequential Reflection) 시뮬레이션 적용 (Q^T b 계산
810 // 효과)
811 for (size_t i = 0; i < Cols; ++i) {
812 // 반사가 무효화(tau가 0)된 열은 패스
813 if (MathTraits<T>::near_zero(tau(i))) {
814 continue;
815 }
816 T dot = y(i);
817 // 하삼각에 숨겨둔 벡터 v 성분을 재조립
818 for (size_t k = i + 1; k < Rows; ++k) {
819 dot += (*this)(static_cast<int>(k), static_cast<int>(i)) * y(k);
820 }
821 T tau_dot = tau(i) * dot;
822
823 // 벡터 b(현재 임시변수 y)를 거울에 통과시켜 방향을 접음
824 y(i) -= tau_dot;
825 for (size_t k = i + 1; k < Rows; ++k) {
826 y(k) -= tau_dot * (*this)(static_cast<int>(k), static_cast<int>(i));
827 }
828 }
829
830 // 2단계: 후진 대입 (Back substitution: R * x = y)
831 // R 행렬의 데이터는 원본 배열의 상삼각 영역(Upper Triangular)에 고스란히 남아있으므로 이를
832 // 활용.
833 StaticVector<T, Cols> x;
834 for (int i = static_cast<int>(Cols) - 1; i >= 0; --i) {
835 T sum = static_cast<T>(0);
836 for (size_t j = static_cast<size_t>(i) + 1; j < Cols; ++j) {
837 sum += (*this)(i, static_cast<int>(j)) * x(j);
838 }
839 x(static_cast<size_t>(i)) = (y(static_cast<size_t>(i)) - sum) / (*this)(i, i);
840 }
841 return x;
842 }

870 template <size_t SubRows, size_t SubCols>
871 void insert_block(size_t start_row, size_t start_col,
872 const StaticMatrix<T, SubRows, SubCols>& block) {
873 // [컴파일 타임 에러 검출] 템플릿 크기 자체가 타겟 행렬의 허용 크기를 초과하면 애초에 빌드를
874 // 차단
875 static_assert(SubRows <= Rows, "SubMatrix rows exceed target");
876 static_assert(SubCols <= Cols, "SubMatrix cols exceed target");
877
878 // [런타임 인덱스 경계 에러 검출] 삽입 시작 좌표가 밀려서 꼬리가 삐져나가는 현상 방지
879 assert(start_row + SubRows <= Rows && "Row index out of bounds");
880 assert(start_col + SubCols <= Cols && "Col index out of bounds");
881
882 // 루프는 '열(Column)' 단위로만 돕니다. 행 단위 접근은 통째로 생략됨.
883 for (size_t j = 0; j < SubCols; ++j) {
884 // 원본 블록 행렬의 j번째 열 데이터 덩어리가 시작되는 첫 번째 메모리 주소(포인터) 계산
885 const T* src = block.data_ptr() + (j * SubRows);
886 // 삽입 당할 거대 타겟 행렬 내에서, 정확히 박힐 시작 메모리 주소(포인터) 계산

888
889 // src 포인터부터 1열 크기(SubRows 개수)만큼 통째로 긁어서 dest 에 한 번에 복사
890 std::copy(src, src + SubRows, dest);
891 }

943 T* dest = result.data_ptr() + (j * SubRows);

958 StaticMatrix<T, Cols, Cols> quadratic_multiply(const StaticMatrix<T, P_Dim, P_Dim>& P) const {

960 StaticMatrix<T, P_Dim, Cols> Temp = P * (*this);
961 // 2. Result = A^T * Temp (Cols x P_Dim * P_Dim x Cols)
