18 template <size_t N, typename Func>
19 static std::array<double, N> optimize(Func f, std::array<double, N> x,
20 double alpha=0.001, double beta=0.9,
21 size_t max_iter=15000, double tol=1e-4,
22 bool verbose=false) {
23 std::array<double, N> v = {0.0}; // 초기 속도 (Velocity)는 0으로 시작
24 if (verbose) {
25 std::cout << "========================================================\n";
26 std::cout << " 🌪️ Momentum Optimizer Started (alpha=" << alpha << ", beta=" << beta << ")\n";
27 std::cout << "========================================================\n";
28 }
29 for (size_t iter = 1; iter <= max_iter; ++iter) {
30 double f_x;
31 std::array<double, N> g;
32
33 // 1. 현재 위치의 함수값 및 기울기 평가
34 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
35
36 // 2. 기울기 노름 (Norm) 계산 및 종료 조건 확인
37 double g_norm = 0.0;
38 for (size_t i = 0; i < N; ++i)
39 g_norm += g[i] * g[i];
40 g_norm = std::sqrt(g_norm);
41
42 if (g_norm < tol) {
43 if (verbose)
44 std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
45 break;
46 }
47
48
49 // 3. 모멘텀 업데이트
50 for (size_t i = 0; i < N; ++i) {
51 // 과거의 속도 (v)를 beta만큼 살리고, 현재 기울기 (g)를 alpha만큼 빼줌
52 v[i] = beta * v[i] - alpha * g[i];
53
54 // 새로운 위치로 이동
55 x[i] = x[i] + v[i];
56 }
57 if (verbose && iter % 1000 == 0) {
58 std::cout << "[Iter " << std::setw(5) << iter << "] f(x): " << std::fixed << std::setprecision(6) << f_x
59 << " | ||g||: " << g_norm << "\n";
60 }
61 }
62 if (verbose) {
63 std::cout << "========================================================\n";
64 std::cout << " 🏁 Final Optimal Point: [" << x[0] << ", " << x[1] << "]\n";
65 std::cout << "========================================================\n";
66 }
67 return x;
68 }
