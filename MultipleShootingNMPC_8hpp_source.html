17template <size_t H, size_t Nx = 6, size_t Nu = 2>
18struct NMPCResiduals {
19 // Z 벡터 차원: x가 H+1개, u가 H개
20 static constexpr size_t Nz = Nx * (H + 1) + Nu * H;
21
22 // 잔차 벡터 차원 M:
23 // 1. 초기 상태 구속 (Nx)
24 // 2. 동역학 제약 (H * Nx)
25 // 3. 상태 추종 오차 ((H+1) * Nx)
26 // 4. 입력 최소화/추종 (H * Nu)
27 static constexpr size_t M = Nx + (H * Nx) + ((H + 1) * Nx) + (H * Nu);
28
29 StaticVector<double, Nx> x_current; // 현재 차량의 실제 상태 (Initial Condition)
30 StaticVector<double, Nx> x_reference; // 목표 상태 (일단 고정 궤적으로 단순화)
31 double dt;
32
33 // 가중치 (Penalty Weights)
34 double W_init = 10000.0; // 초기 상태는 무조건 지켜야 함
35 double W_dyn = 5000.0; // 물리 법칙을 어기면 심각한 페널티
36 double W_state = 10.0; // 목표 궤적 추종 가중치 (Q)
37 double W_control = 1.0; // 입력 최소화 가중치 (R)
38
39 template <typename T>
40 StaticVector<T, M> operator()(const StaticVector<T, Nz>& Z) const {
41 StaticVector<T, M> residuals;
42 size_t res_idx = 0;
43
44 vehicle::DynamicBicycleModel model;
45
46 // --- 헬퍼 람다: Z 벡터에서 x_k 와 u_k 를 0-allocation으로 추출 ---
47 auto get_x = [&](size_t k) {
48 StaticVector<T, Nx> x;
49 for (size_t i = 0; i < Nx; ++i)
50 x(static_cast<int>(i)) = Z(static_cast<int>(k * (Nx + Nu) + i));
51 return x;
52 };
53 auto get_u = [&](size_t k) {
54 StaticVector<T, Nu> u;
55 for (size_t i = 0; i < Nu; ++i)
56 u(static_cast<int>(i)) = Z(static_cast<int>(k * (Nx + Nu) + Nx + i));
57 return u;
58 };
59
60 // 1. 초기 상태 구속 (Initial Condition: x_0 == x_current)
61 StaticVector<T, Nx> x_0 = get_x(0);
62 for (size_t i = 0; i < Nx; ++i) {
63 residuals(static_cast<int>(res_idx++)) =
64 T(W_init) * (x_0(static_cast<int>(i)) - T(x_current(static_cast<int>(i))));
65 }
66
67 // 2. 동역학 제약 (Dynamic Constraints: x_{k+1} == RK4(x_k, u_k))
68 for (size_t k = 0; k < H; ++k) {
69 StaticVector<T, Nx> x_k = get_x(k);
70 StaticVector<T, Nu> u_k = get_u(k);
71 StaticVector<T, Nx> x_next = get_x(k + 1);
72
73 // RK4 적분기를 통해 예측된 물리적 다음 상태
74 StaticVector<T, Nx> x_pred = integrator::step_rk4<Nx, Nu>(model, x_k, u_k, dt);
75
76 for (size_t i = 0; i < Nx; ++i) {
77 // 예측된 상태와 최적화 변수 x_{k+1}의 차이가 0이 되어야 함
78 residuals(static_cast<int>(res_idx++)) =
79 T(W_dyn) * (x_next(static_cast<int>(i)) - x_pred(static_cast<int>(i)));
80 }
81 }
82
83 // 3. 상태 추종 오차 (State Tracking Cost)
84 for (size_t k = 0; k <= H; ++k) {
85 StaticVector<T, Nx> x_k = get_x(k);
86 for (size_t i = 0; i < Nx; ++i) {
87 residuals(static_cast<int>(res_idx++)) =
88 T(W_state) * (x_k(static_cast<int>(i)) - T(x_reference(static_cast<int>(i))));
89 }
90 }
91
92 // 4. 제어 입력 최소화 (Control Minimization)
93 for (size_t k = 0; k < H; ++k) {
94 StaticVector<T, Nu> u_k = get_u(k);
95 for (size_t i = 0; i < Nu; ++i) {
96 residuals(static_cast<int>(res_idx++)) = T(W_control) * u_k(static_cast<int>(i));
97 }
98 }
99
100 return residuals;
101 }
102};
