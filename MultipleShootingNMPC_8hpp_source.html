34 u_last.set_zero();
35 dt = 0.1;
36 for (size_t k = 0; k < H; ++k) U_guess[k].set_zero();
37 for (size_t k = 0; k <= H; ++k) X_guess[k].set_zero();

46 inline void shift_sequence() {
47 for (size_t k = 0; k < H - 1; ++k) {
48 U_guess[k] = U_guess[k + 1];
49 X_guess[k] = X_guess[k + 1];
50 }
51 X_guess[H - 1] = X_guess[H];
52 U_guess[H - 1](0) *= 0.5;
53 U_guess[H - 1](1) *= 0.5;

71 NMPCResult solve_rt_qp(const StaticVector<double, Nx>& x_curr_frenet,
72 const NMPCTuningConfig& config) {
73 NMPCResult result;
74 result.sqp_iterations = 1;
75
76 vehicle::DynamicBicycleModel model;
77 model.kappa = config.kappa;
78
79 // 1. 초기 상태 구속
80 X_guess[0] = x_curr_frenet;
81 X_guess[0](3) = MathTraits<double>::max(0.1, x_curr_frenet(3));
82
83 // =========================================================================
84 // [Architect's Shield: Cold Start Ignition]
85 // 예측 궤적의 끝단(X_guess[H]) 속도가 0이라는 것은 메모리가 방금 할당되었음을 의미합니다.
86 // 거대한 결함(Defect) 오차로 인한 폭발을 막기 위해 1회 명목 궤적을 쏴줍니다.
87 // =========================================================================
88 if (MathTraits<double>::abs(X_guess[H](3)) < 1e-3) {
89 for (size_t k = 0; k < H; ++k) {
90 X_guess[k + 1] = integrator::step_rk4<Nx, Nu>(model, X_guess[k], U_guess[k], dt);
91 }
92 }
93
94 SparseNMPC<H, Nx, Nu> cost_evaluator;
95 cost_evaluator.dt = this->dt;
96 cost_evaluator.obstacles = this->obstacles;
97
98 using ADVar = DualVec<double, Nx + Nu>;
99
100 for (size_t k = 0; k < H; ++k) {
101 StaticVector<double, Nu> u_prev = (k == 0) ? u_last : U_guess[k - 1];
102
103 StaticVector<ADVar, Nx> x_dual;
104 StaticVector<ADVar, Nu> u_dual;
105 StaticVector<ADVar, Nu> u_prev_dual;
106
107 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(X_guess[k](i), i);
108 for (size_t i = 0; i < Nu; ++i) {
109 u_dual(i) = ADVar::make_variable(U_guess[k](i), Nx + i);
110 u_prev_dual(i) = ADVar(u_prev(i));
111 }
112
113 StaticVector<ADVar, Nx> x_next_dual = integrator::step_rk4<Nx, Nu>(model, x_dual, u_dual, dt);
114
115 StaticVector<double, Nx> x_pred_val;
116 for (size_t i = 0; i < Nx; ++i) {
117 x_pred_val(i) = x_next_dual(i).v;
118 for (size_t j = 0; j < Nx; ++j) riccati.A[k](i, j) = x_next_dual(i).g[j];
119 for (size_t j = 0; j < Nu; ++j) riccati.B[k](i, j) = x_next_dual(i).g[Nx + j];
120 }
121
122 // Defect Constraint (마법의 열쇠)
123 riccati.d[k] = x_pred_val - X_guess[k + 1];
124
125 StaticVector<ADVar, NUM_RESIDUALS> res_dual =
126 cost_evaluator.eval_node_residuals(x_dual, u_dual, u_prev_dual, config, k);
127
128 riccati.Q[k].set_zero();
129 riccati.R[k].set_zero();
130 riccati.q[k].set_zero();
131 riccati.r[k].set_zero();
132
133 for (size_t res_idx = 0; res_idx < NUM_RESIDUALS; ++res_idx) {
134 double r_val = res_dual(res_idx).v;
135 for (size_t i = 0; i < Nx; ++i) {
136 double J_xi = res_dual(res_idx).g[i];
137 riccati.q[k](i) += J_xi * r_val;
138 for (size_t j = 0; j < Nx; ++j) riccati.Q[k](i, j) += J_xi * res_dual(res_idx).g[j];
139 }
140 for (size_t i = 0; i < Nu; ++i) {
141 double J_ui = res_dual(res_idx).g[Nx + i];
142 riccati.r[k](i) += J_ui * r_val;
143 for (size_t j = 0; j < Nu; ++j) riccati.R[k](i, j) += J_ui * res_dual(res_idx).g[Nx + j];
144 }
145 }
146
147 for (size_t i = 0; i < Nu; ++i) {
148 double u_val = U_guess[k](i);
149 double s_upper = MathTraits<double>::max(config.u_max[i] - u_val, 1e-3);
150 double s_lower = MathTraits<double>::max(u_val - config.u_min[i], 1e-3);
151
152 double grad_barrier = config.barrier_mu * (1.0 / s_upper - 1.0 / s_lower);
153 double hess_barrier = config.barrier_mu * (1.0 / (s_upper * s_upper) + 1.0 / (s_lower * s_lower));
154
155 riccati.r[k](i) += std::clamp(grad_barrier, -100.0, 100.0);
156 riccati.R[k](i, i) += MathTraits<double>::min(hess_barrier, 500.0);
157 }
158
159 for (size_t i = 0; i < Nx; ++i) riccati.Q[k](i, i) += config.damping_Q;
160 for (size_t i = 0; i < Nu; ++i) riccati.R[k](i, i) += config.damping_R;
161 }
162
163 riccati.Q[H].set_zero();
164 riccati.q[H].set_zero();
165
166 for (size_t i = 0; i < Nx; ++i) {
167 if (i == 1 || i == 2 || i == 3) {
168 double err = (i == 3) ? (X_guess[H](i) - config.target_vx) : X_guess[H](i);
169 double qf_i = (i == 1) ? config.Q_D * 5.0 :
170 (i == 2) ? config.Q_mu * 5.0 :
171 config.Q_Vx * 5.0;
172 riccati.Q[H](i, i) += qf_i;
173 riccati.q[H](i) += qf_i * err;
174 }
175 }
176
177 if (riccati.solve() != SolverStatus::SUCCESS) {
178 return execute_fallback(result, "Factorization Failed (MS Riccati)");
179 }
180
181 double alpha = config.step_alpha;
182 for (size_t k = 0; k < H; ++k) {
183 for (size_t i = 0; i < Nu; ++i) {
184 double du = riccati.du[k](i);
185 if (du > 1e-4) {
186 double max_alpha = (config.u_max[i] - U_guess[k](i)) / du;
187 if (max_alpha > 0.0) alpha = MathTraits<double>::min(alpha, config.safety_fraction * max_alpha);
188 } else if (du < -1e-4) {
189 double max_alpha = (config.u_min[i] - U_guess[k](i)) / du;
190 if (max_alpha > 0.0) alpha = MathTraits<double>::min(alpha, config.safety_fraction * max_alpha);
191 }
192 }
193 }
194 alpha = MathTraits<double>::max(alpha, 0.05);
195
196 double max_kkt = 0.0;
197 for (size_t k = 0; k < H; ++k) {
198 for (size_t i = 0; i < Nu; ++i) {
199 U_guess[k](i) += alpha * riccati.du[k](i);
200 U_guess[k](i) = std::clamp(U_guess[k](i), config.u_min[i] + 1e-3, config.u_max[i] - 1e-3);
201 max_kkt = MathTraits<double>::max(max_kkt, MathTraits<double>::abs(riccati.du[k](i)));
202 }
203 for (size_t i = 0; i < Nx; ++i) {
204 X_guess[k + 1](i) += alpha * riccati.dx[k + 1](i);
205 }
206 }
207
208 result.max_kkt_error = max_kkt;
209 if (std::isnan(max_kkt) || max_kkt > 20.0) {
210 return execute_fallback(result, "Divergence Detected (MS KKT Exploded)");
211 }
212
213 u_last = U_guess[0];
214 result.success = true;
215 return result;
216 }
