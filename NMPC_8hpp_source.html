61 struct CostFunc {
62 StaticVector<double, Nx> x0;
63 StaticVector<double, Nx> x_ref;
64 StaticVector<double, Nx> Q, Qf;
65 StaticVector<double, Nu> R, R_rate;
66 StaticVector<double, Nu> u_last;
67 DynamicBicycleModel model;
68 double dt;
69 std::array<Obstacle, 10> obs;
70
71 template <typename T>
72 T operator()(const StaticVector<T, N_vars>& U) const {
73 T cost(0.0);
74 StaticVector<T, Nx> x;
75 for(size_t i = 0; i < Nx; ++i) x(static_cast<int>(i)) = T(x0(static_cast<int>(i)));
76
77 auto squash = [](T val, double limit) -> T {
78 if constexpr (std::is_same_v<T, double>) {
79 return limit * std::tanh(val / limit);
80 } else {
81 return T(limit) * ad::tanh(val / T(limit));
82 }
83 };
84
85 // [Architect's Add] 이전 스텝의 입력을 추적하기 위한 변수 (초기값은 u_last)
86 StaticVector<T, Nu> u_prev;
87 u_prev(0) = T(u_last(0));
88 u_prev(1) = T(u_last(1));
89
90 for (size_t k = 0; k < Np; ++k) {
91 StaticVector<T, Nu> u;
92 u(0) = squash(U(static_cast<int>(k * Nu + 0)), 3.0); // 가속도
93 u(1) = squash(U(static_cast<int>(k * Nu + 1)), 0.5); // 조향
94
95 // 1. 상태 오차 페널티 (정차 로직 포함)
96 for(size_t i = 0; i < Nx; ++i) {
97 T err = x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i)));
98 cost += T(Q(static_cast<int>(i))) * err * err;
99 }
100
101 // 2. 제어 입력 크기(Magnitude) 페널티
102 cost += T(R(0)) * U(static_cast<int>(k * Nu + 0)) * U(static_cast<int>(k * Nu + 0)) +
103 T(R(1)) * U(static_cast<int>(k * Nu + 1)) * U(static_cast<int>(k * Nu + 1));
104
105 // 3. [핵심] 제어 입력 변화율(Slew Rate/Jerk) 페널티 주입
106 T delta_a = u(0) - u_prev(0);
107 T delta_steer = u(1) - u_prev(1);
108 cost += T(R_rate(0)) * delta_a * delta_a + T(R_rate(1)) * delta_steer * delta_steer;
109 u_prev = u; // 다음 스텝(k+1)의 변화율 계산을 위해 갱신
110
111 // 4. Soft Obstacle Penalty (장애물 회피망 구축)
112 for (size_t obs_i = 0; obs_i < 10; ++obs_i) {
113 T dx = x(0) - T(obs[obs_i].x);
114 T dy = x(1) - T(obs[obs_i].y);
115 T dist_sq = dx * dx + dy * dy;
116
117 // 장애물 반경 + 차량 여유 반경(0.8m)
118 T r_safe = T(obs[obs_i].r + 0.8);
119 T violation = r_safe * r_safe - dist_sq;
120
121 // 안전 거리 안으로 침범할 경우 막대한 2차 페널티 부과
122 if (Optimization::get_value(violation) > 0.0) {
123 cost += T(8000.0) * violation * violation;
124 }
125 }
126
127 // 5. 플랜트 롤아웃
128 x = Integrator::rk4<Nx, Nu, DynamicBicycleModel, T>(model, x, u, dt);
129 }
130
131 // 6. Terminal Cost
132 for(size_t i = 0; i < Nx; ++i) {
133 T err = x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i)));
134 cost += T(Qf(static_cast<int>(i))) * err * err;
135 }
136
137 return cost;
138 }
