64 struct CostFunc {
65 StaticVector<double, Nx> x0;
66 StaticVector<double, Nx> x_ref;
67 StaticVector<double, Nx> Q, Qf;
68 StaticVector<double, Nu> R, R_rate;
69 StaticVector<double, Nu> u_last;
70 DynamicBicycleModel model;
71 double dt;
72 std::array<Obstacle, 10> obs;
73
74 template <typename T>
75 T operator()(const StaticVector<T, N_vars>& U) const {
76 T cost(0.0);
77 StaticVector<T, Nx> x;
78 for (size_t i = 0; i < Nx; ++i) x(static_cast<int>(i)) = T(x0(static_cast<int>(i)));
79
80 StaticVector<T, Nu> u_prev;
81 u_prev(0) = T(u_last(0));
82 u_prev(1) = T(u_last(1));
83
84 for (size_t k = 0; k < Np; ++k) {
85 StaticVector<T, Nu> u;
86 // [Architect's Fix] 임시방편이었던 Tanh 스쿼싱 완전 삭제
87 u(0) = U(static_cast<int>(k * Nu + 0));
88 u(1) = U(static_cast<int>(k * Nu + 1));
89
90 // 1. 상태 오차 페널티
91 for (size_t i = 0; i < Nx; ++i) {
92 T err = x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i)));
93 cost += T(Q(static_cast<int>(i))) * err * err;
94 }
95
96 // 2. 크기 및 변화율(Slew Rate) 페널티
97 cost += T(R(0)) * u(0) * u(0) + T(R(1)) * u(1) * u(1);
98 T delta_a = u(0) - u_prev(0);
99 T delta_steer = u(1) - u_prev(1);
100 cost += T(R_rate(0)) * delta_a * delta_a + T(R_rate(1)) * delta_steer * delta_steer;
101 u_prev = u;
102
103 // 3. 장애물 회피망 (장애물은 볼록성이 보장되지 않아 Soft Penalty 유지)
104 for (size_t obs_i = 0; obs_i < 10; ++obs_i) {
105 T dx = x(0) - T(obs[obs_i].x);
106 T dy = x(1) - T(obs[obs_i].y);
107 T dist_sq = dx * dx + dy * dy;
108 T r_safe = T(obs[obs_i].r + 0.8);
109 T violation = r_safe * r_safe - dist_sq;
110 if (Optimization::get_value(violation) > 0.0) {
111 cost += T(8000.0) * violation * violation;
112 }
113 }
114
115 // 4. 플랜트 롤아웃
116 x = Integrator::rk4<Nx, Nu, DynamicBicycleModel, T>(model, x, u, dt);
117 }
118
119 // 5. Terminal Cost
120 for (size_t i = 0; i < Nx; ++i) {
121 T err = x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i)));
122 cost += T(Qf(static_cast<int>(i))) * err * err;
123 }
124 return cost;
125 }
126 };

128 struct DummyEq {
129 template <typename T>
130 StaticVector<T, 0> operator()(const StaticVector<T, N_vars>& U) const {
131 (void)U;

136 // [Architect's Add] 하드 제약 조건 정의 (A_ineq * x - b_ineq <= 0)
137 struct BoundIneq {
138 template <typename T>
139 StaticVector<T, N_ineq> operator()(const StaticVector<T, N_vars>& U) const {
140 StaticVector<T, N_ineq> ineq;
141 for (size_t k = 0; k < Np; ++k) {
142 T a = U(static_cast<int>(k * Nu + 0));
143 T delta = U(static_cast<int>(k * Nu + 1));
144
145 // 가속도 한계 [-3.0, 3.0]
146 ineq(static_cast<int>(k * 4 + 0)) = a - T(3.0);
147 ineq(static_cast<int>(k * 4 + 1)) = -a - T(3.0);
148
149 // 조향각 한계 [-0.5, 0.5] rad
150 ineq(static_cast<int>(k * 4 + 2)) = delta - T(0.5);
151 ineq(static_cast<int>(k * 4 + 3)) = -delta - T(0.5);
152 }
153 return ineq;
154 }
155 };
