57 struct Vertex {
58 alignas(64) std::array<double, N> x;

79 template <size_t N, typename Func>
80 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_start,
81 double step = 1.0, double tol = 1e-6,
82 size_t max_iter = 2000,
83 bool verbose = false) noexcept {
84 static_assert(N > 0, "Dimension N must be greater than 0"); // 컴파일 타임 차원 검사
85 auto start_clock = std::chrono::high_resolution_clock::now(); // 소요 시간 측정 시작
86
87 // Nelder-Mead 형태 변환 파라미터 (표준 설정값)
88 const double alpha = 1.0; // 반사 계수 (Reflection)
89 const double gamma = 2.0; // 확장 계수 (Expansion)
90 const double rho = 0.5; // 수축 계수 (Contraction)
91 const double sigma = 0.5; // 축소 계수 (Shrinkage)
92
93 // 수렴 조건 검사 시 불필요한 제곱근(sqrt) 연산을 피하기 위해 오차의 제곱을 미리 계산
94 const double tol_sq = tol * tol;
95
96 // 1. 초기 심플렉스 (Simplex) 생성
97 // N차원 공간에서는 N+1개의 정점이 필요합니다. (예: 2차원이면 3개의 정점인 삼각형 구성)
98 std::array<Vertex<N>, N + 1> simplex = {};
99
100 // 첫 번째 정점은 주어진 초기 위치로 설정
101 simplex[0].x = x_start;
102 simplex[0].f_val = f(simplex[0].x);
103
104 // 나머지 N개의 정점은 초기 위치에서 각 축 방향으로 step만큼 이동하여 생성 (직교 정점)
105 for (size_t i = 0; i < N; ++i) {
106 simplex[i + 1].x = x_start;
107 simplex[i + 1].x[i] += step;
108 simplex[i + 1].f_val = f(simplex[i + 1].x);
109 }
110
111 // 반복문 시작 전 최초 1회 정렬 (0: Best, N: Worst)
112 std::sort(simplex.begin(), simplex.end());
113
114 if (verbose) {
115 std::cout << "========================================================\n";
116 std::cout << " 🦠 Nelder-Mead Simplex (The Amoeba) Started\n";
117 std::cout << "========================================================\n";
118 }
119
120 size_t iter = 0;
121 for (iter = 1; iter <= max_iter; ++iter) {
122
123 // 2. 정점 정렬 (0번 인덱스가 최고(Best), N번 인덱스가 최악(Worst)의 점이 됨)
124 // 매 이터레이션의 끝에서 정점이 갱신되므로 항상 정렬 상태를 유지해야 함
125 std::sort(simplex.begin(), simplex.end());
126
127 // 3. 수렴 조건 검사 (Variance Check)
128 // N+1개의 함수값들의 분산을 계산하여, 이 분산이 허용 오차의 제곱(tol_sq) 미만이면 수렴한 것으로 판단
129 double mean_y = 0.0;
130#pragma omp simd
131 for (size_t i = 0; i <= N; ++i) {
132 mean_y += simplex[i].f_val;
133 }
134 mean_y /= (N + 1);
135
136 double var_y = 0.0;
137#pragma omp simd
138 for (size_t i = 0; i <= N; ++i) {
139 double diff = simplex[i].f_val - mean_y;
140 // var_y = diff * diff + var_y 를 fma로 단일 명령어 처리
141 var_y = std::fma(diff, diff, var_y);
142 }
143 var_y /= (N + 1);
144
145 if (var_y < tol_sq) {
146 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
147 break;
148 }
149
150 // 4. Centroid (무게중심) 계산
151 // 가장 함수값이 나쁜 점(simplex[N])을 제외한 나머지 N개 정점들의 평균 위치를 구함
152 alignas(64) std::array<double, N> x_o = {0.0};
153 for (size_t i = 0; i < N; ++i) { // i < N (최악점 인덱스 N은 제외됨)
154#pragma omp simd
155 for (size_t j = 0; j < N; ++j) {
156 x_o[j] += simplex[i].x[j];
157 }
158 }
159#pragma omp simd
160 for (size_t j = 0; j < N; ++j) {
161 x_o[j] /= N;
162 }
163
164 // 5. 반사 (Reflection)
165 // 최악의 점을 중심점(x_o)을 기준으로 반대편으로 넘김
166 alignas(64) std::array<double, N> x_r;
167#pragma omp simd
168 for (size_t j = 0; j < N; ++j) {
169 // x_r = x_o + alpha * (x_o - x_worst)
170 x_r[j] = std::fma(alpha, x_o[j] - simplex[N].x[j], x_o[j]);
171 }
172 double f_r = f(x_r);
173
174 // 반사된 점(x_r)이 최고점(Best)보다는 나쁘지만 차악점(Second Worst, N-1)보다는 좋을 경우
175 // 평범하게 반사된 점을 수용함
176 if (f_r >= simplex[0].f_val && f_r < simplex[N - 1].f_val) {
177 simplex[N].x = x_r;
178 simplex[N].f_val = f_r;
179 }
180 // 반사된 점이 현재 최고점(Best)보다도 더 좋은 경우 (올바른 방향을 찾았음)
181 else if (f_r < simplex[0].f_val) {
182 // 6. 확장 (Expansion)
183 // 방향이 좋으므로 반사된 방향으로 더 멀리(gamma) 확장하여 탐색함
184 alignas(64) std::array<double, N> x_e;
185#pragma omp simd
186 for (size_t j = 0; j < N; ++j) {
187 // x_e = x_o + gamma * (x_r - x_o)
188 x_e[j] = std::fma(gamma, x_r[j] - x_o[j], x_o[j]);
189 }
190 double f_e = f(x_e);
191
192 // 확장한 점이 반사한 점보다 더 좋으면 확장을 채택, 아니면 그냥 반사를 채택
193 if (f_e < f_r) {
194 simplex[N].x = x_e;
195 simplex[N].f_val = f_e;
196 } else {
197 simplex[N].x = x_r;
198 simplex[N].f_val = f_r;
199 }
200 }
201 // 반사된 점이 차악점(Second Worst)보다도 나쁜 경우 (해당 방향이 아니거나 계곡임)
202 else {
203 // 7. 수축 (Contraction)
204 bool do_shrink = false; // 축소(Shrinkage) 연산 수행 플래그
205
206 // 그래도 최악의 점(Worst)보다는 반사된 점이 나은 경우 -> 외부 수축 (Outside Contraction)
207 if (f_r < simplex[N].f_val) {
208 alignas(64) std::array<double, N> x_c;
209#pragma omp simd
210 for (size_t j = 0; j < N; ++j) {
211 // x_c = x_o + rho * (x_r - x_o) -> 반사점과 중심점의 사이
212 x_c[j] = std::fma(rho, x_r[j] - x_o[j], x_o[j]);
213 }
214 double f_c = f(x_c);
215
216 // 수축한 점이 반사점보다 좋거나 같으면 채택
217 if (f_c <= f_r) {
218 simplex[N].x = x_c;
219 simplex[N].f_val = f_c;
220 } else {
221 do_shrink = true; // 수축도 실패했으므로 전체 축소
222 }
223 }
224 // 반사된 점이 최악의 점보다도 더 나빠진 경우 -> 내부 수축 (Inside Contraction)
225 else {
226 alignas(64) std::array<double, N> x_c;
227#pragma omp simd
228 for (size_t j = 0; j < N; ++j) {
229 // x_c = x_o + rho * (x_worst - x_o) -> 중심점과 최악점의 사이
230 x_c[j] = std::fma(rho, simplex[N].x[j] - x_o[j], x_o[j]);
231 }
232 double f_c = f(x_c);
233
234 // 수축한 점이 기존 최악점보다 좋아졌으면 채택
235 if (f_c < simplex[N].f_val) {
236 simplex[N].x = x_c;
237 simplex[N].f_val = f_c;
238 } else {
239 do_shrink = true; // 수축도 실패했으므로 전체 축소
240 }
241 }
242
243 // 8. 축소 (Shrinkage)
244 // 수축(Contraction)으로도 나은 점을 찾지 못한 경우, 심플렉스가 골짜기에 꼈다고 판단하고
245 // 최우수 점(simplex[0])을 제외한 모든 점들을 최우수 점 방향으로 일정 비율(sigma) 당김
246 if (do_shrink) {
247 for (size_t i = 1; i <= N; ++i) { // 0번(Best)은 놔두고 나머지 모두
248#pragma omp simd
249 for (size_t j = 0; j < N; ++j) {
250 // x_new = x_best + sigma * (x_old - x_best)
251 simplex[i].x[j] =
252 std::fma(sigma, simplex[i].x[j] - simplex[0].x[j], simplex[0].x[j]);
253 }
254 // 위치가 바뀌었으므로 함수값 재평가
255 simplex[i].f_val = f(simplex[i].x);
256 }
257 }
258 }
259
260 // 진행 상태 출력: 첫 이터레이션이거나 매 50번마다 출력
261 if (verbose && (iter % 50 == 0 || iter == 1)) {
262 std::cout << "[Iter " << std::setw(3) << iter << "] Best f(x): " << std::fixed
263 << std::setprecision(8) << simplex[0].f_val << " | Variance: " << var_y
264 << "\n";
265 }
266 }
267
268 // 최적화 종료 및 소요 시간 계산
269 auto end_clock = std::chrono::high_resolution_clock::now();
270 auto duration =
271 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
272
273 if (verbose) {
274 std::cout << "========================================================\n";
275 std::cout << " 🏁 Final Optimal Point: [" << simplex[0].x[0];
276 if constexpr (N > 1) std::cout << ", " << simplex[0].x[1];
277 if constexpr (N > 2) std::cout << ", ...";
278 std::cout << "]\n========================================================\n";
279 }
280
281 // 가장 함수값이 작은 최우수 정점의 위치(simplex[0].x) 반환
282 return {simplex[0].x, simplex[0].f_val, iter, duration.count()};
283 }
