16 // 미래의 위치 (Look-ahead)를 먼저 예측하고 그곳의 기울기를 이용해 업데이트합니다
17 // ==============================================================
18 template <size_t N, typename Func>
19 static std::array<double, N> optimize(Func f, std::array<double, N> x, double alpha = 0.001,
20 double beta = 0.9, size_t max_iter = 15000,
21 double tol = 1e-4, bool verbose = false) {
22 std::array<double, N> v = {0.0};
23 if (verbose) {
24 std::cout << "========================================================\n";
25 std::cout << " 🔮 Nesterov Momentum Started (alpha=" << alpha << ", beta=" << beta
26 << ")\n";
27 std::cout << "========================================================\n";
28 }
29 for (size_t iter = 1; iter <= max_iter; ++iter) {
30 // 1. 미래의 위치 (Look-ahead point) 계산 : 먼저 관성대로 가본다
31 std::array<double, N> x_lookahead;
32 for (size_t i = 0; i < N; ++i) {
33 x_lookahead[i] = x[i] + beta * v[i];
34 }
35
36 // 2. 미래 위치에서의 기울기 (g_lookahead) 획득
37 double f_lookahead;
38 std::array<double, N> g_lookahead;
39 AutoDiff::value_and_gradient<N>(f, x_lookahead, f_lookahead, g_lookahead);
40
41 // 종료 조건 검사 (미래 위치의 기울기 기준)
42 double g_norm = 0.0;
43 for (size_t i = 0; i < N; ++i) {
44 g_norm += g_lookahead[i] * g_lookahead[i];
45 }
46 g_norm = std::sqrt(g_norm);
47
48 if (g_norm < tol) {
49 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
50 break;
51 }
52
53 // 3. 네스테로프 업데이트 (미래의 기울기를 반영하여 속도와 위치 갱신)
54 for (size_t i = 0; i < N; ++i) {
55 v[i] = beta * v[i] - alpha * g_lookahead[i];
56 x[i] = x[i] + v[i]; // 실제 위치 업데이트
57 }
58 }
59
60 if (verbose) {
61 std::cout << "========================================================\n";
62 // 10차원이니까 다 출력하면 기니, 앞의 두 개만 확인합시다!
63 std::cout << " 🏁 Final Optimal Point: [" << x[0] << ", " << x[1] << "]\n";
64 std::cout << "========================================================\n";
65 }
66 return x;
67 }
