48 template <size_t N, typename Func>
49 static std::array<double, N> optimize(Func f, std::array<double, N> x, double alpha = 0.001,
50 double beta = 0.9, size_t max_iter = 15000,
51 double tol = 1e-4, bool verbose = false) {
52 std::array<double, N> v = {0.0}; // 속도(Velocity) 누적 변수 초기화
53
54 if (verbose) {
55 std::cout << "========================================================\n";
56 std::cout << " 🔮 Nesterov Momentum Started (alpha=" << alpha << ", beta=" << beta
57 << ")\n";
58 std::cout << "========================================================\n";
59 }
60
61 for (size_t iter = 1; iter <= max_iter; ++iter) {
62 // -----------------------------------------------------------
63 // 1. 미래의 위치 (Look-ahead point) 계산
64 // 일반 모멘텀과 달리, 현재 위치(x)에서 현재 관성(beta * v)만큼 먼저 이동해봅니다.
65 // -----------------------------------------------------------
66 std::array<double, N> x_lookahead;
67 for (size_t i = 0; i < N; ++i) {
68 x_lookahead[i] = x[i] + beta * v[i];
69 }
70
71 // -----------------------------------------------------------
72 // 2. 미래 위치에서의 함수값 및 기울기(g_lookahead) 획득
73 // AutoDiff를 통해 미래 위치(x_lookahead)에서의 정확한 기울기를 계산합니다.
74 // 이 기울기는 "앞으로 내리막이 계속될지, 오르막으로 변할지"에 대한 정보를 담고
75 // 있습니다.
76 // -----------------------------------------------------------
77 double f_lookahead;
78 std::array<double, N> g_lookahead;
79 AutoDiff::value_and_gradient<N>(f, x_lookahead, f_lookahead, g_lookahead);
80
81 // 종료 조건 검사: "미래 위치"의 기울기 노름(Norm) 계산
82 double g_norm = 0.0;
83 for (size_t i = 0; i < N; ++i) {
84 g_norm += g_lookahead[i] * g_lookahead[i];
85 }
86 g_norm = std::sqrt(g_norm);
87
88 // 기울기 노름이 허용 오차보다 작으면 최적점(임계점)에 도달한 것으로 판단하고 종료
89 if (g_norm < tol) {
90 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
91 break;
92 }
93
94 // -----------------------------------------------------------
95 // 3. 네스테로프 업데이트 (파라미터 갱신)
96 // 미래 위치에서 구한 기울기를 현재 속도 업데이트에 반영하여 실제 이동할 위치를
97 // 계산합니다.
98 // -----------------------------------------------------------
99 for (size_t i = 0; i < N; ++i) {
100 // 속도 업데이트: 기존 관성 유지 + (미래 위치에서의 기울기 기반 하강)
101 v[i] = beta * v[i] - alpha * g_lookahead[i];
102
103 // 위치 업데이트: 계산된 새로운 속도를 이용해 현재 위치를 이동시킵니다.
104 x[i] = x[i] + v[i];
105 }
106
107 // (선택) 진행 과정 출력 로직을 추가하려면 이곳에 배치할 수 있습니다.
108 }
109
110 if (verbose) {
111 std::cout << "========================================================\n";
112 // 차원 N과 상관없이 유연하게 출력할 수 있도록 루프 처리
113 std::cout << " 🏁 Final Optimal Point: [" << x[0];
114 for (size_t i = 1; i < N; ++i) {
115 std::cout << ", " << x[i];
116 }
117 std::cout << "]\n========================================================\n";
118 }
119
120 return x;
121 }
