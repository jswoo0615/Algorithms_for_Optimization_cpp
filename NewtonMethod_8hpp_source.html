54 template <size_t N>
55 [[nodiscard]] static constexpr std::optional<std::array<double, N>> solve_newton_system(
56 std::array<std::array<double, N>, N> H, std::array<double, N> g) noexcept {
57
58 if constexpr (N == 2) {
59 // [2차원 최적화] 역행렬 공식을 Cramer's rule 기반으로 풀이
60 // 판별식(Determinant) 계산: H[0][0]*H[1][1] - H[0][1]*H[1][0]
61 // std::fma(Fused Multiply-Add)를 사용하여 하드웨어 단일 명령어로 빠르고 정밀하게 처리
62 const double det = std::fma(H[0][0], H[1][1], -H[0][1] * H[1][0]);
63
64 // 판별식이 0에 매우 가까우면 역행렬이 존재하지 않음 (지형이 완전 평탄하거나 안장점(Saddle Point))
65 if (std::abs(det) < 1e-15) {
66 return std::nullopt; // Singular Matrix 대처
67 }
68
69 const double inv_det = 1.0 / det;
70
71 // 2x2 역행렬 공식:
72 // H^{-1} = inv_det * [ H[1][1] -H[0][1] ]
73 // [ -H[1][0] H[0][0] ]
74 // p = -H^{-1} * g 계산 후 바로 반환
75 return std::array<double, 2>{-inv_det * std::fma(H[1][1], g[0], -H[0][1] * g[1]),
76 -inv_det * std::fma(H[0][0], g[1], -H[1][0] * g[0])};
77 } else {
78 // [N차원 최적화] 정적 가우스 소거법 (Static Gaussian Elimination)
79 // 동적 배열(std::vector) 할당(new/malloc)을 없애 속도 최적화
80 for (size_t i = 0; i < N; ++i) {
81 // Partial Pivoting: 수치적 안정성을 위해 절댓값이 가장 큰 원소를 피벗으로 선택
82 size_t pivot = i;
83 for (size_t j = i + 1; j < N; ++j) {
84 if (std::abs(H[j][i]) > std::abs(H[pivot][i])) pivot = j;
85 }
86 // 행 교환
87 std::swap(H[i], H[pivot]);
88 std::swap(g[i], g[pivot]);
89
90 // 피벗 값이 0에 너무 가까우면 해를 구할 수 없음 (Singular)
91 if (std::abs(H[i][i]) < 1e-15) return std::nullopt;
92
93 // 전진 소거 (Forward Elimination)
94 for (size_t j = i + 1; j < N; ++j) {
95 const double factor = H[j][i] / H[i][i];
96 for (size_t k = i; k < N; ++k) H[j][k] -= factor * H[i][k];
97 g[j] -= factor * g[i];
98 }
99 }
100
101 // 후진 대입 (Backward Substitution)
102 std::array<double, N> p = {0.0};
103 for (size_t i = N; i-- > 0;) {
104 double sum = 0.0;
105 for (size_t j = i + 1; j < N; ++j) {
106 sum = std::fma(H[i][j], p[j], sum);
107 }
108 // 원래 풀어야 하는 방정식은 H * p_raw = g 였으나,
109 // 업데이트 공식을 위해 방향 반전 (-H^-1 * g)이 필요하므로 부호를 뒤집음
110 p[i] = -(g[i] - sum) / H[i][i];
111 }
112 return p;
113 }
114 }

128 template <size_t N, typename Func>
129 [[nodiscard]] static OptimizationResult<N> optimize(Func f, std::array<double, N> x_init,
130 double tol = 1e-6,
131 size_t max_iter = 50) noexcept {
132 auto start_clock = std::chrono::high_resolution_clock::now(); // 타이머 시작
133
134 std::array<double, N> x = x_init;
135
136 // 분산/크기 비교를 위해 매번 제곱근(sqrt) 연산을 하는 비용을 없애기 위해 제곱 형태로 비교
137 const double tol_sq = tol * tol;
138 size_t iter = 0;
139
140 for (iter = 1; iter <= max_iter; ++iter) {
141 double f_val = 0.0;
142 std::array<double, N> g = {0.0};
143
144 // 1. 현재 위치에서의 목적 함수값과 1차 미분(Gradient) 획득
145 // 리포지토리에 자체 구현된 AutoDiff 엔진을 활용하여 O(1) 정적 할당만으로 기울기를 구합니다.
146 AutoDiff::value_and_gradient<N>(f, x, f_val, g);
147
148 // 2. 기울기 노름의 제곱(Gradient Norm Squared) 계산
149 double g_norm_sq = 0.0;
150// OpenMP SIMD 지시어를 통해 벡터 누적 연산을 가속 (컴파일러 지원 시)
151#pragma omp simd
152 for (size_t i = 0; i < N; ++i) {
153 g_norm_sq = std::fma(g[i], g[i], g_norm_sq); // g[i]*g[i] + g_norm_sq
154 }
155
156 // 기울기가 허용 오차보다 작으면 최적점(임계점)에 도달한 것으로 판단하고 조기 종료
157 if (g_norm_sq < tol_sq) {
158 break;
159 }
160
161 // 3. 현재 위치에서의 2차 미분(Hessian Matrix) 획득
162 // 리포지토리의 AutoDiff.hpp에서 제공하는 Central Difference 기반의 헤시안 행렬을 구합니다.
163 auto H = AutoDiff::hessian<N>(f, x);
164
165 // 4. 뉴턴 스텝 계산 (선형 시스템 H * p = -g 풀이)
166 auto p = solve_newton_system<N>(H, g);
167
168 // 헤시안이 특이 행렬이어서 방향을 계산할 수 없다면 (지형 평탄) 최적화를 멈춥니다.
169 if (!p) {
170 break; // 역행렬(선형 방정식) 실패 시 즉시 중단
171 }
172
173 // 5. 위치 업데이트
174 // p 내부에는 이미 방향 반전(-H^-1 * g)이 반영되어 있으므로 단순 덧셈(+) 처리
175#pragma omp simd
176 for (size_t i = 0; i < N; ++i) {
177 x[i] += (*p)[i];
178 }
179 }
180
181 auto end_clock = std::chrono::high_resolution_clock::now();
182 auto duration =
183 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
184
185 // 최종 위치(x)에서의 목적 함수값을 다시 평가하여 반환 구조체에 담아 리턴합니다.
186 return {x, AutoDiff::value<N>(f, x), iter, duration.count()};
187 }
