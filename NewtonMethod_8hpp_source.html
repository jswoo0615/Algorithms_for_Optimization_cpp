30class NewtonMethod {
31 private:
37 template <size_t N, typename Func>
38 static void compute_grad_and_hessian(const Func& f, const StaticVector<double, N>& x,
39 double& f_val, StaticVector<double, N>& g,
40 StaticMatrix<double, N, N>& H) noexcept {
41 using ADVar = DualVec<double, N>;
42 constexpr double h = 1e-5; // 차분 스텝
43 constexpr double inv_2h = 1.0 / (2.0 * h);
44
45 // 1. 현재 위치의 함수값 및 Gradient (AD 1회 평가)
46 StaticVector<ADVar, N> x_dual;
47 for (size_t i = 0; i < N; ++i) x_dual(i) = ADVar::make_variable(x(i), i);
48
49 ADVar f_dual = f(x_dual);
50 f_val = Optimization::get_value(f_dual);
51 for (size_t i = 0; i < N; ++i) g(i) = f_dual.g[i];
52
53 // 2. Hessian 계산 (AD + Central Difference)
54 StaticVector<ADVar, N> x_plus_dual, x_minus_dual;
55
56 for (size_t i = 0; i < N; ++i) {
57 // x + h*e_i
58 for (size_t k = 0; k < N; ++k)
59 x_plus_dual(k) = ADVar::make_variable(x(k) + (k == i ? h : 0.0), k);
60 ADVar f_plus = f(x_plus_dual);
61
62 // x - h*e_i
63 for (size_t k = 0; k < N; ++k)
64 x_minus_dual(k) = ADVar::make_variable(x(k) - (k == i ? h : 0.0), k);
65 ADVar f_minus = f(x_minus_dual);
66
67 for (size_t j = 0; j < N; ++j) {
68 H(j, i) = (f_plus.g[j] - f_minus.g[j]) * inv_2h;
69 }
70 }
71
72 // 수치적 오차 방지를 위한 헤시안 대칭화 (Symmetrization)
73 for (size_t i = 0; i < N; ++i) {
74 for (size_t j = 0; j < i; ++j) {
75 double sym = 0.5 * (H(i, j) + H(j, i));
76 H(i, j) = sym;
77 H(j, i) = sym;
78 }
79 }
80 }

85 template <size_t N, typename Func>
86 [[nodiscard]] static OptimizationResult<N> optimize(const Func& f,
87 const StaticVector<double, N>& x_init,
88 double tol = 1e-6,
89 size_t max_iter = 50) noexcept {
90 auto start_clock = std::chrono::high_resolution_clock::now();
91
92 StaticVector<double, N> x = x_init;
93 const double tol_sq = tol * tol;
94 size_t iter = 0;
95
96 // 메모리 호이스팅 (Zero-Allocation)
97 StaticVector<double, N> g, neg_g, p;
98 StaticMatrix<double, N, N> H;
99 double f_val = 0.0;
100 SolverStatus status = SolverStatus::MAX_ITERATION_REACHED;
101
102 for (iter = 1; iter <= max_iter; ++iter) {
103 // 1. 값, Gradient, Hessian 동시 도출
104 compute_grad_and_hessian(f, x, f_val, g, H);
105
106 // 2. Gradient Norm 검사 (수렴 판정)
107 double g_norm_sq = 0.0;
108 for (size_t i = 0; i < N; ++i) {
109 g_norm_sq += g(i) * g(i);
110 neg_g(i) = -g(i); // LDLT_solve를 위한 부호 반전
111 }
112
113 if (g_norm_sq < tol_sq) {
114 status = SolverStatus::SUCCESS;
115 break;
116 }
117
118 // 3. 선형 시스템 풀이 (H * p = -g)
119 // 헤시안(H)은 대칭행렬이므로 LDLT 분해가 가장 빠르고 적합합니다.
120 if (linalg::LDLT_decompose(H) != MathStatus::SUCCESS) {
121 // 헤시안이 특이(Singular)하거나 음의 곡률(Negative Curvature)을 만나면 즉시 탈출
122 status = SolverStatus::MATH_ERROR;
123 break;
124 }
125
126 linalg::LDLT_solve(H, neg_g, p);
127
128 // 4. Step Size (Stagnation) 검증
129 double max_p = 0.0;
130 for (size_t i = 0; i < N; ++i) {
131 double abs_p = MathTraits<double>::abs(p(i));
132 if (abs_p > max_p) max_p = abs_p;
133 }
134 if (max_p < 1e-12) {
135 status = SolverStatus::STEP_SIZE_TOO_SMALL;
136 break;
137 }
138
139 // 5. 위치 업데이트 (x = x + p) SIMD 타격
140 x += p;
141 }
142
143 auto end_clock = std::chrono::high_resolution_clock::now();
144 auto duration =
145 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
146
147 return {x, f_val, iter, duration.count(), status};
148 }
149};
