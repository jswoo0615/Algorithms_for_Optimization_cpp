14template <size_t N, typename Functor>
15SolverStatus solve_newton(StaticVector<double, N>& x_opt, const Functor& f, int max_iter = 50,
16 double tol = 1e-6) {
17 using ADVar = DualVec<double, N>;
18 for (int iter = 0; iter < max_iter; ++iter) {
19 // 1. 현재 x 값을 DualVec 독립 변수로 변환 (시드 주입)
20 StaticVector<ADVar, N> x_dual;
21 for (size_t i = 0; i < N; ++i) {
22 x_dual(static_cast<int>(i)) = ADVar::make_variable(x_opt(static_cast<int>(i)), i);
23 }
24
25 // 2. 비선형 함수 평가 (Value와 Jacobian이 동시에 계산됨)
26 StaticVector<ADVar, N> y_dual = f(x_dual);
27
28 // 3. 선형 시스템 조립 : J * dx = -F
29 StaticMatrix<double, N, N> J;
30 StaticVector<double, N> neg_F;
31 double max_residual = 0.0;
32
33 for (size_t i = 0; i < N; ++i) {
34 double f_val = y_dual(static_cast<int>(i)).v;
35 neg_F(static_cast<int>(i)) = -f_val; // -F 벡터 구성
36
37 // 수렴 판정용 최대 잔차 (Residual) 업데이트
38 double abs_f = std::abs(f_val);
39 if (abs_f > max_residual) {
40 max_residual = abs_f;
41 }
42
43 // 자코비안 행렬 J 추출
44 for (size_t j = 0; j < N; ++j) {
45 J(static_cast<int>(i), static_cast<int>(j)) = y_dual(static_cast<int>(i)).g[j];
46 }
47 }
48
49 // 4. 수렴 조건 검사 (허용 오차 이내 진입 시 즉시 종료)
50 if (max_residual < tol) {
51 return SolverStatus::SUCCESS;
52 }
53
54 // 5. 탐색 방향 (Search Direction) 계산 : LU 분해 활용 (비대칭 행렬 가능성이 높으므로 LU
55 // 사용)
56 StaticVector<int, N> P;
57 MathStatus m_status = linalg::LU_decompose(J, P);
58
59 // 수학적으로 발산하면 즉시 실패 선언
60 if (m_status != MathStatus::SUCCESS) {
61 return SolverStatus::MATH_ERROR;
62 }
63
64 StaticVector<double, N> dx = linalg::LU_solve(J, P, neg_F);
65
66 // 6. 상태 업데이트 : x_new = x_old + dx
67 for (size_t i = 0; i < N; ++i) {
68 x_opt(static_cast<int>(i)) += dx(static_cast<int>(i));
69 }
70 }
71 // 최대 반복 횟수를 채웠으나, 수렴하지 못한 경우
72 return SolverStatus::MAX_ITERATION_REACHED;
73}
