34class NoisyDescent {
35 public:
36 NoisyDescent() = delete; // 인스턴스화 방지
37 // SigmaFunc : 반복 횟수 (iter)를 받아 현재의 표준 편차 sigma를 반환하는 스케줄링 콜백 함수
38 template <size_t N, typename Func, typename SigmaFunc>
39 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_start,
40 SigmaFunc sigma_func, double alpha = 0.01,
41 double tol = 1e-5,
42 size_t max_iter = 10000,
43 bool verbose = false) noexcept {
44 static_assert(N > 0, "Dimension N must be greater than 0.");
45 auto start_clock = std::chrono::high_resolution_clock::now();
46
47 const double tol_sq = tol * tol;
48 alignas(64) std::array<double, N> x = x_start;
49
50 // [핵심 최적화] 하드웨어 엔트로피 획득 및 난수 엔진 초기화는 스레드당 1회만 수행
51 // 매 iteration 또는 함수 호출 시 발생하는 병목 완전 제거
52 static thread_local std::mt19937 gen(std::random_device{}());
53 std::normal_distribution<double> dist(0.0, 1.0);
54
55 if (verbose) {
56 std::cout << "========================================================\n";
57 std::cout << " 🎲 Noisy Descent Started (alpha=" << alpha << ")\n";
58 std::cout << "========================================================\n";
59 }
60
61 double f_x = 0.0;
62 size_t iter = 0;
63
64 for (iter = 1; iter <= max_iter; ++iter) {
65 alignas(64) std::array<double, N> g{0.0}; // Gradient 벡터 (캐시 라인 정렬)
66
67 // 1. Auto Diff를 통한 정확한 목적 함수 값 및 1차 미분 (Gradient) 획득
68 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
69
70 double g_norm_sq = 0.0;
71#pragma omp simd
72 for (size_t i = 0; i < N; ++i) {
73 g_norm_sq = std::fma(g[i], g[i], g_norm_sq);
74 }
75
76 // 2. 외부 주입된 노이즈 스케줄링 콜백을 통해 현재 sigma 계산
77 double sigma = sigma_func(iter);
78
79 // 3. 종료 조건 검사 : 기울기가 0에 수렴하고, 노이즈가 허용치 이하로 소멸했을 때
80 if (g_norm_sq < tol_sq && sigma < tol) {
81 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
82 break;
83 }
84
85// 4. 파라미터 업데이트 (경사 하강 + 가우시안 확률론적)
86#pragma omp simd
87 for (size_t i = 0; i < N; ++i) {
88 double noise_term = sigma * dist(gen);
89 // x[i] = x[i] - alpha * g[i] + noise_term; // 기존 업데이트
90 x[i] =
91 std::fma(-alpha, g[i], x[i]) + noise_term; // FMA 적용하여 정확도 및 성능 향상
92 }
93
94 if (verbose && (iter % 100 == 0 || iter == 1)) {
95 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
96 << std::setprecision(6) << f_x << " | ||g||: " << std::sqrt(g_norm_sq)
97 << " | sigma: " << sigma << "\n";
98 }
99 }
100 auto end_clock = std::chrono::high_resolution_clock::now();
101 auto duration =
102 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
103
104 if (verbose) {
105 std::cout << "========================================================\n";
106 std::cout << " 🏁 Final Optimal Point: [" << x[0];
107 if constexpr (N > 1) std::cout << ", " << x[1];
108 if constexpr (N > 2) std::cout << ", ...";
109 std::cout << "]\n========================================================\n";
110 }
111
112 return {x, f_x, iter, duration.count()};
113 }
114};
