79 [[nodiscard]] static OptimizationResultND<N> optimize(Func f, std::array<double, N> x_start,
80 SigmaFunc sigma_func, double alpha = 0.01,
81 double tol = 1e-5,
82 size_t max_iter = 10000,
83 bool verbose = false) noexcept {
84 static_assert(N > 0, "Dimension N must be greater than 0."); // 컴파일 타임 차원 검사
85 auto start_clock = std::chrono::high_resolution_clock::now(); // 소요 시간 측정 시작
86
87 // 반복문 내부에서 불필요한 제곱근 연산을 피하기 위해 오차 허용치의 제곱을 미리 계산
88 const double tol_sq = tol * tol;
89
90 // 캐시 라인(64 byte) 정렬을 통해 메모리 접근 속도 최적화
91 alignas(64) std::array<double, N> x = x_start;
92
93 // [핵심 최적화] 하드웨어 엔트로피 획득 및 난수 엔진 초기화는 스레드당 1회만 수행
94 // 매 이터레이션(반복) 혹은 함수 호출 시마다 엔진을 새로 생성하면 병목(Bottleneck)이
95 // 발생하므로 이를 완전 제거
96 static thread_local std::mt19937 gen(std::random_device{}());
97 std::normal_distribution<double> dist(0.0, 1.0); // 평균 0, 표준 편차 1인 표준 정규 분포
98
99 if (verbose) {
100 std::cout << "========================================================\n";
101 std::cout << " 🎲 Noisy Descent Started (alpha=" << alpha << ")\n";
102 std::cout << "========================================================\n";
103 }
104
105 double f_x = 0.0;
106 size_t iter = 0;
107
108 for (iter = 1; iter <= max_iter; ++iter) {
109 // 캐시 정렬된 기울기 벡터 초기화
110 alignas(64) std::array<double, N> g{0.0};
111
112 // -------------------------------------------------------------
113 // 1. 목적 함수값 및 기울기 평가
114 // AutoDiff(자동 미분) 엔진을 호출하여 수치적 오차 없는 정확한 함수값과 기울기(g)를
115 // 동시에 획득
116 // -------------------------------------------------------------
117 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
118
119 // 기울기 노름의 제곱(Gradient Norm Squared) 계산
120 double g_norm_sq = 0.0;
121// OpenMP SIMD 지시어를 사용하여 벡터 누적 연산을 가속 (컴파일러 지원 시)
122#pragma omp simd
123 for (size_t i = 0; i < N; ++i) {
124 // g_norm_sq += g[i] * g[i] 와 동일한 FMA 연산
125 g_norm_sq = std::fma(g[i], g[i], g_norm_sq);
126 }
127
128 // -------------------------------------------------------------
129 // 2. 노이즈 스케줄링(Noise Scheduling) 적용
130 // 외부에서 주입된 콜백 함수(sigma_func)를 호출하여 현재 반복 횟수(iter)에 맞는
131 // 가우시안 노이즈의 표준 편차(sigma) 크기를 결정합니다. (일반적으로 점진적 감소)
132 // -------------------------------------------------------------
133 double sigma = sigma_func(iter);
134
135 // -------------------------------------------------------------
136 // 3. 조기 종료(Convergence) 조건 검사
137 // 단순 경사 하강법과 달리, 1) 기울기가 0에 수렴해야 하고 (g_norm_sq < tol_sq)
138 // 2) 주입되는 노이즈 또한 허용치(tol) 이하로 소멸되어야만 최종 수렴한 것으로
139 // 판단합니다.
140 // -------------------------------------------------------------
141 if (g_norm_sq < tol_sq && sigma < tol) {
142 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
143 break;
144 }
145
146 // -------------------------------------------------------------
147 // 4. 파라미터 업데이트 (경사 하강 방향 이동 + 가우시안 노이즈 주입)
148 // -------------------------------------------------------------
149#pragma omp simd
150 for (size_t i = 0; i < N; ++i) {
151 // 표준 정규 분포에서 뽑은 난수에 sigma를 곱해 N(0, sigma^2) 노이즈 생성
152 double noise_term = sigma * dist(gen);
153
154 // 기존 수식: x[i] = x[i] - alpha * g[i] + noise_term;
155 // FMA(Fused Multiply-Add)를 적용하여 곱셈과 덧셈을 한 번에 정확하게 처리
156 // std::fma(-alpha, g[i], x[i]) 가 (x[i] - alpha * g[i]) 와 동일함
157 x[i] = std::fma(-alpha, g[i], x[i]) + noise_term;
158 }
159
160 // 진행 상태 출력: 첫 세대이거나 매 100세대마다 한 번씩 출력
161 if (verbose && (iter % 100 == 0 || iter == 1)) {
162 std::cout << "[Iter " << std::setw(4) << iter << "] f(x): " << std::fixed
163 << std::setprecision(6) << f_x << " | ||g||: " << std::sqrt(g_norm_sq)
164 << " | sigma: " << sigma << "\n";
165 }
166 }
167
168 // 탐색 종료 후 시간 측정 완료
169 auto end_clock = std::chrono::high_resolution_clock::now();
170 auto duration =
171 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
172
173 if (verbose) {
174 std::cout << "========================================================\n";
175 std::cout << " 🏁 Final Optimal Point: [" << x[0];
176 if constexpr (N > 1) std::cout << ", " << x[1];
177 if constexpr (N > 2) std::cout << ", ...";
178 std::cout << "]\n========================================================\n";
179 }
180
181 // 최적해와 부가 정보를 결과 구조체에 담아 반환
182 return {x, f_x, iter, duration.count()};
183 }
