72 alignas(64) std::array<double, N> x = x_init;
73

75 const double tol_sq = tol * tol;
76
77 // 탐색 방향 (Search directions)을 저장할 캐시 정렬된 정적 배열 U
78 // 초기에는 기본 좌표축 방향(단위 행렬)으로 설정
79 alignas(64) std::array<std::array<double, N>, N> U = {0.0};
80 reset_basis<N>(U);
81
82 size_t iter = 0;
83
84 if (verbose) {
85 std::cout << "========================================================\n";
86 std::cout << " 🌪️ Powell's Method Started \n";
87 std::cout << "========================================================\n";
88 }

92 // 이번 이터레이션 시작점 백업 (각 방향별 선탐색 시작점)
93 alignas(64) std::array<double, N> x_prime = x;
94
95 // 1. 현재 유지 중인 방향 세트 U의 모든 방향(d)에 대해 순차적으로 1차원 선탐색 수행

98 // 각 방향으로 선탐색하여 x_prime 위치 업데이트
99 x_prime = line_search(f, x_prime, d);
100 }
101
102 // --- Powell 방법의 핵심: 새로운 켤레 방향(Conjugate Direction) 학습 ---
103
104 // 2. 가장 오래된 방향(맨 앞, U[0])을 버리고 배열을 한 칸씩 앞으로 이동 (Shift)
105 for (size_t i = 0; i < N - 1; ++i) {
106 U[i] = U[i + 1];
107 }
108
109 // 3. 1 사이클(N번의 선탐색) 전체를 통해 이동한 "알짜 이동 방향" (d_new) 계산
110 alignas(64) std::array<double, N> d_new = {0.0};
111#pragma omp simd // 벡터화(SIMD) 가속 지시어
112 for (size_t i = 0; i < N; ++i) {
113 d_new[i] = x_prime[i] - x[i]; // (사이클 후 위치) - (사이클 전 위치)
114 }
115 // 이 알짜 이동 방향을 새로운 마지막 탐색 방향으로 등록
116 U[N - 1] = d_new;

118 // 4. 새롭게 학습한 방향(d_new)을 따라 원래 시작점 (x)에서부터 다시 한 번 선탐색 수행

120
121 // 5. 이번 이터레이션 동안의 총 이동 거리의 제곱(Norm squared) 계산
122 // (수렴 여부 판별 용도)
123 double delta_sq = 0.0;
124#pragma omp simd
125 for (size_t i = 0; i < N; ++i) {
126 const double diff = x_next[i] - x[i];
127 // Fused Multiply-Add 연산 (delta_sq = diff * diff + delta_sq) 최적화
128 delta_sq = std::fma(diff, diff, delta_sq);
129 }

136 }

139 x = x_next;
140

142 if (delta_sq < tol_sq) {
143 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
144 break;
145 }

148 // Powell 방법은 반복하다 보면 새 방향들이 선형 종속적으로 변하여 탐색 공간이 축소될 수

150 // 초기화시킴.
151 if (iter % (N + 1) == 0) {
152 reset_basis<N>(U);
153 }
154 }
155

157 auto end_clock = std::chrono::high_resolution_clock::now();
158 auto duration =
159 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
160
161 // 최종 결과 로그 출력
162 if (verbose) {
163 std::cout << "========================================================\n";
164 std::cout << " 🏁 Final Optimal Point: [" << x[0];
165 if constexpr (N > 1) std::cout << ", " << x[1];
166 if constexpr (N > 2) std::cout << ", ...";
167 std::cout << "]\n========================================================\n";
168 }

172 }
