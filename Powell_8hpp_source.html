20 // 기저 벡터 집합 U를 단위 행렬로 초기화
21 template <size_t N>
22 static constexpr void reset_basis(std::array<std::array<double, N>, N>& U) noexcept {
23 for (size_t i = 0; i < N; ++i) {
24 for (size_t j = 0; j < N; ++j) {
25 U[i][j] = (i == j) ? 1.0 : 0.0;
26 }
27 }
28 }

31 PowellMethod() = delete; // 인스턴스화 방지
32 template <size_t N, typename Func, typename LineSearchFunc = GoldenSectionStrategy>
33 [[nodiscard]] static OptimizationResultND<N> optimize(
34 Func f, std::array<double, N> x_init, LineSearchFunc line_search = GoldenSectionStrategy{},
35 double tol = 1e-5, size_t max_iter = 1000, bool verbose = false) noexcept {
36 static_assert(N > 0, "Dimension N must be greater than 0");
37 auto start_clock = std::chrono::high_resolution_clock::now();
38
39 alignas(64) std::array<double, N> x = x_init;
40 const double tol_sq = tol * tol;
41
42 // 탐색 방향 (Search directions)을 저장할 캐시 정렬된 정적 배열 U
43 alignas(64) std::array<std::array<double, N>, N> U = {0.0};
44 reset_basis<N>(U);
45
46 size_t iter = 0;
47
48 if (verbose) {
49 std::cout << "========================================================\n";
50 std::cout << " 🌪️ Powell's Method Started \n";
51 std::cout << "========================================================\n";
52 }
53
54 for (iter = 1; iter <= max_iter; ++iter) {
55 alignas(64) std::array<double, N> x_prime = x;
56
57 // 1. 현재 방향 세트 U의 모든 방향에 대해 순차적으로 선탐색 수행
58 for (size_t i = 0; i < N; ++i) {
59 alignas(64) std::array<double, N> d = U[i];
60 x_prime = line_search(f, x_prime, d);
61 }
62
63 // 2. 가장 오래된 방향 U를 버리고 배열을 한 칸씩 앞으로 당김 (Shift)
64 for (size_t i = 0; i < N - 1; ++i) {
65 U[i] = U[i + 1];
66 }
67
68 // 3. 1 사이클 전체의 알짜 이동 방향 (d_new) 도출
69 alignas(64) std::array<double, N> d_new = {0.0};
70#pragma omp simd
71 for (size_t i = 0; i < N; ++i) {
72 d_new[i] = x_prime[i] - x[i];
73 }
74 U[N - 1] = d_new; // 새로운 이동 방향을 마지막 탐색 방향으로 등록
75
76 // 4. 새로운 방향을 따라 시작점 (x)에서부터 다시 한 번 선탐색 수행
77 alignas(64) std::array<double, N> x_next = line_search(f, x, d_new);
78
79 // 이동 거리 (Norm) 제곱 계산 및 FMA 적용
80 double delta_sq = 0.0;
81#pragma omp simd
82 for (size_t i = 0; i < N; ++i) {
83 const double diff = x_next[i] - x[i];
84 delta_sq = std::fma(diff, diff, delta_sq);
85 }
86
87 if (verbose && (iter % 10 == 0 || iter == 1)) {
88 std::cout << "[Iter " << std::setw(3) << iter << "] f(x): " << std::fixed
89 << std::setprecision(6) << f(x_next)
90 << " | ||Δ||: " << std::sqrt(delta_sq) << "\n";
91 }
92
93 x = x_next; // 위치 갱신
94
95 // 종료 조건 확인
96 if (delta_sq < tol_sq) {
97 if (verbose) std::cout << " ↳ ✅ Converged at Iteration: " << iter << "\n";
98 break;
99 }
100
101 // [안전성 보장] 방향 벡터의 선형 종속(Linear dependence) 방지
102 if (iter % (N + 1) == 0) {
103 reset_basis<N>(U);
104 }
105 }
106
107 auto end_clock = std::chrono::high_resolution_clock::now();
108 auto duration =
109 std::chrono::duration_cast<std::chrono::nanoseconds>(end_clock - start_clock);
110
111 if (verbose) {
112 std::cout << "========================================================\n";
113 // [버그 수정] std::cout << x 와 x[5] 하드코딩 오류 제거
114 std::cout << " 🏁 Final Optimal Point: [" << x[0];
115 if constexpr (N > 1) std::cout << ", " << x[1];
116 if constexpr (N > 2) std::cout << ", ...";
117 std::cout << "]\n========================================================\n";
118 }
119 return {x, f(x), iter, duration.count()};
120 }
