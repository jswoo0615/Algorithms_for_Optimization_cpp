24template <size_t Nx, size_t Nc>
25class QPSolver_IPM {
26 public:
27 static constexpr double SIGMA = 0.5; // Centering parameter
28 static constexpr double TAU = 0.995; // Fraction-to-boundary (경계 접근 한계율)
29 static constexpr double MIN_SLACK = 1e-8; // 수치 붕괴 방지용 최소 슬랙값
30
34 static SolverStatus solve(const StaticMatrix<double, Nx, Nx>& H,
35 const StaticVector<double, Nx>& g,
36 const StaticMatrix<double, Nc, Nx>& C,
37 const StaticVector<double, Nc>& d,
38 StaticVector<double, Nx>& x_opt,
39 int max_iter = 20, double tol = 1e-6) {
40
41 // 메모리 호이스팅 (Zero-Allocation)
42 StaticVector<double, Nc> s, z; // Slack(s) > 0, Dual(z) > 0
43 StaticVector<double, Nc> ds, dz; // Search directions
44 StaticVector<double, Nx> dx;
45
46 // 초기화 (Warm-start가 없다면 충분히 큰 양수로 초기화하여 내부에서 시작)
47 for (size_t i = 0; i < Nc; ++i) {
48 s(i) = 1.0;
49 z(i) = 1.0;
50 }
51
52 StaticMatrix<double, Nx, Nx> H_sys;
53 StaticVector<double, Nx> g_sys;
54 StaticVector<double, Nc> r_c, r_inq; // Complementarity & Inequality residuals
55 StaticVector<double, Nx> r_d; // Dual residual
56
57 for (int iter = 0; iter < max_iter; ++iter) {
58 // 1. KKT 잔차(Residuals) 계산
59 // r_inq = C*x - d + s (= 0)
60 linalg::multiply(C, x_opt, r_inq);
61 r_inq.saxpy(-1.0, d);
62 r_inq += s;
63
64 // r_d = H*x + g + C^T*z (= 0)
65 linalg::multiply(H, x_opt, r_d);
66 r_d += g;
67 StaticVector<double, Nx> CT_z;
68 linalg::multiply_AT_B(C, z, CT_z);
69 r_d += CT_z;
70
71 // Duality Gap 및 Barrier parameter (mu) 계산
72 double duality_gap = 0.0;
73 for (size_t i = 0; i < Nc; ++i) {
74 duality_gap += s(i) * z(i);
75 }
76 double mu = duality_gap / static_cast<double>(Nc);
77
78 // 종료 조건 검사
79 double max_res = 0.0;
80 for (size_t i = 0; i < Nx; ++i) max_res = MathTraits<double>::max(max_res, MathTraits<double>::abs(r_d(i)));
81 for (size_t i = 0; i < Nc; ++i) max_res = MathTraits<double>::max(max_res, MathTraits<double>::abs(r_inq(i)));
82
83 if (max_res < tol && duality_gap < tol) {
84 return SolverStatus::SUCCESS;
85 }
86
87 // Target barrier parameter
88 double target_mu = SIGMA * mu;
89
90 // r_c = s * z - target_mu (= 0)
91 for (size_t i = 0; i < Nc; ++i) {
92 r_c(i) = s(i) * z(i) - target_mu;
93 }
94
95 // 2. Schur Complement를 통한 KKT 시스템 압축
96 // H_sys = H + C^T * (S^{-1} * Z) * C
97 H_sys = H;
98 StaticMatrix<double, Nc, Nx> Sigma_C = C; // Sigma * C
99 for (size_t i = 0; i < Nc; ++i) {
100 double sigma_i = z(i) / MathTraits<double>::max(s(i), MIN_SLACK); // 0으로 나누기 방지
101 for (size_t j = 0; j < Nx; ++j) {
102 Sigma_C(i, j) *= sigma_i;
103 }
104 }
105
106 StaticMatrix<double, Nx, Nx> CTSigmaC;
107 linalg::multiply_AT_B(C, Sigma_C, CTSigmaC); // SIMD 가상 전치 내적
108 H_sys += CTSigmaC;
109
110 // g_sys = -r_d - C^T * S^{-1} * (z * r_inq - r_c)
111 g_sys = r_d;
112 for (size_t i = 0; i < Nx; ++i) g_sys(i) = -g_sys(i); // g_sys = -r_d
113
114 StaticVector<double, Nc> right_term;
115 for (size_t i = 0; i < Nc; ++i) {
116 right_term(i) = (z(i) * r_inq(i) - r_c(i)) / MathTraits<double>::max(s(i), MIN_SLACK);
117 }
118
119 StaticVector<double, Nx> CT_right;
120 linalg::multiply_AT_B(C, right_term, CT_right);
121
122 // [Architect's Fix] operator-= 대신 하드웨어 가속 FMA(saxpy) 강제 적용
123 g_sys.saxpy(-1.0, CT_right);
124
125 // 3. 탐색 방향(Step Direction) 도출 (LDLT 가속)
126 if (linalg::LDLT_decompose(H_sys) != MathStatus::SUCCESS) {
127 // 수치적 붕괴 발생: 10ms 실시간 제어에서 뻗는 것을 방지
128 return SolverStatus::MATH_ERROR;
129 }
130 linalg::LDLT_solve(H_sys, g_sys, dx); // Zero-Allocation
131
132 // ds = -r_inq - C*dx
133 linalg::multiply(C, dx, ds);
134 for (size_t i = 0; i < Nc; ++i) {
135 ds(i) = -r_inq(i) - ds(i);
136 }
137
138 // dz = -S^{-1} * (r_c + Z * ds)
139 for (size_t i = 0; i < Nc; ++i) {
140 dz(i) = -(r_c(i) + z(i) * ds(i)) / MathTraits<double>::max(s(i), MIN_SLACK);
141 }
142
143 // 4. Fraction-to-Boundary Rule (스텝 사이즈 결정)
144 double alpha_primal = 1.0;
145 double alpha_dual = 1.0;
146
147 for (size_t i = 0; i < Nc; ++i) {
148 if (ds(i) < 0.0) {
149 alpha_primal = MathTraits<double>::min(alpha_primal, -TAU * s(i) / ds(i));
150 }
151 if (dz(i) < 0.0) {
152 alpha_dual = MathTraits<double>::min(alpha_dual, -TAU * z(i) / dz(i));
153 }
154 }
155
156 // [안전장치] Stagnation 방어
157 if (alpha_primal < 1e-6 || alpha_dual < 1e-6) {
158 return SolverStatus::STEP_SIZE_TOO_SMALL;
159 }
160
161 // 5. 변수 업데이트
162 for (size_t i = 0; i < Nx; ++i) x_opt(i) += alpha_primal * dx(i);
163 for (size_t i = 0; i < Nc; ++i) {
164 s(i) += alpha_primal * ds(i);
165 z(i) += alpha_dual * dz(i);
166 }
167 }
168
169 // RTI 기법 등에서 최대 반복 횟수에 도달하더라도 부분 수렴해(Suboptimal)를 제어기에 넘김
170 return SolverStatus::SUBOPTIMAL;
171 }
172};
