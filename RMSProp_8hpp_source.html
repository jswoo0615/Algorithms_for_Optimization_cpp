33 const double tol_sq = tol * tol;
34

36 std::cout << "========================================================\n";

38 << ")\n";
39 std::cout << "========================================================\n";
40 }
41
42 for (size_t iter = 1; iter <= max_iter; ++iter) {
43 double f_x = 0.0;
44 std::array<double, N> g = {0.0};
45
46 // 1. O(1) 할당 Auto Diff 호출
47 AutoDiff::value_and_gradient<N>(f, x, f_x, g);
48
49 // 2. 기울기 L2-Norm 제곱 계산
50 double g_norm_sq = 0.0;
51 for (size_t i = 0; i < N; ++i) {

59 }

61 }

63 // 3. RMSProp 파라미터 업데이트
64 for (size_t i = 0; i < N; ++i) {
65 // 과거 기억 (G)은 decay만큼 남기고, 새로운 기울기 제곱을 더함 (EMA)
66 G[i] = decay * G[i] + (1.0 - decay) * g[i] * g[i];
67
68 x[i] -= (alpha / (std::sqrt(G[i]) + epsilon)) * g[i];
69 }

75 << "\n"; // 출력 시에만 sqrt 연산 수행
76 }
77 }

80 std::cout << " 🏁 Final Optimal Point: [" << x[0];
81 if constexpr (N > 1) std::cout << ", " << x[1];
82 if constexpr (N > 2) std::cout << ", ...";
83 std::cout << "]\n========================================================\n";
