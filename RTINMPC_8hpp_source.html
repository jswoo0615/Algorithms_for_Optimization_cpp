52 StaticVector<T, Nx> x; for(size_t i=0; i<Nx; ++i) x(i) = T(x0(i));
53 StaticVector<T, Nu> u_prev; u_prev(0)=T(u_last(0)); u_prev(1)=T(u_last(1));
54
55 for (size_t k = 0; k < Np; ++k) {
56 StaticVector<T, Nu> u; u(0) = U(static_cast<int>(k*2)); u(1) = U(static_cast<int>(k*2+1));
57 for(size_t i=0; i<Nx; ++i) r(idx++) = T(std::sqrt(Q(static_cast<int>(i)))) * (x(static_cast<int>(i)) - T(ref_horizon[k](static_cast<int>(i))));
58 r(idx++) = T(std::sqrt(R(0))) * u(0); r(idx++) = T(std::sqrt(R(1))) * u(1);
59 r(idx++) = T(std::sqrt(R_rate(0))) * (u(0)-u_prev(0)); r(idx++) = T(std::sqrt(R_rate(1))) * (u(1)-u_prev(1));
60 u_prev = u;
61 for (size_t i=0; i<10; ++i) {
62 T dx = x(0)-T(obs[i].x), dy = x(1)-T(obs[i].y);
63 // 좁은 통로 통과를 위해 안전 마진을 0.25m로 더 타이트하게 조정
64 T viol = T(obs[i].r + 0.25) * T(obs[i].r + 0.25) - (dx*dx+dy*dy);
65 r(idx++) = (Optimization::get_value(viol) > 0.0) ? T(std::sqrt(4000.0)) * viol : T(0.0);
66 }
67 x = Integrator::rk4<Nx, Nu, DynamicBicycleModel, T>(model, x, u, dt);
68 }
