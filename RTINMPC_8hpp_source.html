51 struct ResidualFunc {
52 StaticVector<double, Nx> x0;
53 StaticVector<double, Nx> ref_horizon[Np];
54 StaticVector<double, Nx> Q, Qf;
55 StaticVector<double, Nu> R, R_rate;
56 StaticVector<double, Nu> u_last;
57 DynamicBicycleModel model;
58 double dt;

61 template <typename T>
62 StaticVector<T, N_res> operator()(const StaticVector<T, N_vars>& U) const {
63 StaticVector<T, N_res> r;
64 r.set_zero();
65 int idx = 0;
66 StaticVector<T, Nx> x;
67 for (size_t i = 0; i < Nx; ++i) x(i) = T(x0(i));
68 StaticVector<T, Nu> u_prev;
69 u_prev(0) = T(u_last(0));
70 u_prev(1) = T(u_last(1));
71
72 for (size_t k = 0; k < Np; ++k) {
73 StaticVector<T, Nu> u;
74 u(0) = U(static_cast<int>(k * 2));
75 u(1) = U(static_cast<int>(k * 2 + 1));
76 for (size_t i = 0; i < Nx; ++i)
77 r(idx++) = T(std::sqrt(Q(static_cast<int>(i)))) *
78 (x(static_cast<int>(i)) - T(ref_horizon[k](static_cast<int>(i))));
79 r(idx++) = T(std::sqrt(R(0))) * u(0);
80 r(idx++) = T(std::sqrt(R(1))) * u(1);
81 r(idx++) = T(std::sqrt(R_rate(0))) * (u(0) - u_prev(0));
82 r(idx++) = T(std::sqrt(R_rate(1))) * (u(1) - u_prev(1));
83 u_prev = u;
84 for (size_t i = 0; i < 10; ++i) {
85 T dx = x(0) - T(obs[i].x), dy = x(1) - T(obs[i].y);
86 // 좁은 통로 통과를 위해 안전 마진을 0.25m로 더 타이트하게 조정
87 T viol = T(obs[i].r + 0.25) * T(obs[i].r + 0.25) - (dx * dx + dy * dy);
88 r(idx++) = (Optimization::get_value(viol) > 0.0) ? T(std::sqrt(4000.0)) * viol
89 : T(0.0);
90 }
91 x = Integrator::rk4<Nx, Nu, DynamicBicycleModel, T>(model, x, u, dt);
92 }
93 for (size_t i = 0; i < Nx; ++i)
94 r(idx++) = T(std::sqrt(Qf(static_cast<int>(i)))) *
95 (x(static_cast<int>(i)) - T(ref_horizon[Np - 1](static_cast<int>(i))));
96 return r;
97 }

100 struct BoundIneq {
101 StaticVector<double, Nx> x0;
102 DynamicBicycleModel model;
103 double dt;
104 template <typename T>
105 StaticVector<T, N_ineq> operator()(const StaticVector<T, N_vars>& U) const {
106 StaticVector<T, N_ineq> ineq;
107 StaticVector<T, Nx> x;
108 for (int i = 0; i < 6; ++i) x(i) = T(x0(i));
109 for (size_t k = 0; k < Np; ++k) {
110 T a = U(static_cast<int>(k * 2)), d = U(static_cast<int>(k * 2 + 1));
111 ineq(static_cast<int>(k * 8 + 0)) = a - T(3.0);
112 ineq(static_cast<int>(k * 8 + 1)) = -a - T(3.0);
113 ineq(static_cast<int>(k * 8 + 2)) = d - T(0.5);
114 ineq(static_cast<int>(k * 8 + 3)) = -d - T(0.5);
115 StaticVector<T, 2> ut;
116 ut(0) = a;
117 ut(1) = d;
118 x = Integrator::rk4<6, 2, DynamicBicycleModel, T>(model, x, ut, dt);
119 ineq(static_cast<int>(k * 8 + 4)) = x(0) - T(11.0);
120 ineq(static_cast<int>(k * 8 + 5)) = T(-1.0) - x(0);
121 ineq(static_cast<int>(k * 8 + 6)) = x(1) - T(11.0);
122 ineq(static_cast<int>(k * 8 + 7)) = T(-1.0) - x(1);
123 }
124 return ineq;
125 }

132 for (size_t k = 0; k < Np; ++k) res_f.ref_horizon[k] = ref_horizon[k];
133 BoundIneq ineq_f{current_x, model, dt};
134 bool success = rti.solve(
135 U_guess, res_f, [](const auto&) { return StaticVector<double, 0>(); }, ineq_f);
136 u_last(0) = U_guess(0);
137 u_last(1) = U_guess(1);
138 return success;
