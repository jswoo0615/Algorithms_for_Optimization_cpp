75 StaticVector<double, Nx> x0;
76 StaticVector<double, Nx> x_ref;

84 template <typename T>
85 StaticVector<T, N_res> operator()(const StaticVector<T, N_vars>& U) const {
86 StaticVector<T, N_res> r;
87 r.set_zero();
88 int idx = 0;
89
90 StaticVector<T, Nx> x;
91 for (size_t i = 0; i < Nx; ++i) x(static_cast<int>(i)) = T(x0(static_cast<int>(i)));
92
93 StaticVector<T, Nu> u_prev;
94 u_prev(0) = T(u_last(0));
95 u_prev(1) = T(u_last(1));
96
97 for (size_t k = 0; k < Np; ++k) {
98 StaticVector<T, Nu> u;
99 u(0) = U(static_cast<int>(k * Nu + 0));
100 u(1) = U(static_cast<int>(k * Nu + 1));
101
102 // 1. 상태 잔차: J^T J 과정에서 제곱되므로 sqrt(Q)를 곱함
103 for(size_t i = 0; i < Nx; ++i) {
104 r(idx++) = T(std::sqrt(Q(static_cast<int>(i)))) * (x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i))));
105 }
106
107 // 2. 제어 잔차
108 r(idx++) = T(std::sqrt(R(0))) * u(0);
109 r(idx++) = T(std::sqrt(R(1))) * u(1);
110
111 // 3. 변화율 잔차
112 r(idx++) = T(std::sqrt(R_rate(0))) * (u(0) - u_prev(0));
113 r(idx++) = T(std::sqrt(R_rate(1))) * (u(1) - u_prev(1));
114 u_prev = u;
115
116 // 4. 장애물 회피 잔차
117 for (size_t obs_i = 0; obs_i < 10; ++obs_i) {
118 T dx = x(0) - T(obs[obs_i].x);
119 T dy = x(1) - T(obs[obs_i].y);
120 T dist_sq = dx * dx + dy * dy;
121 T r_safe = T(obs[obs_i].r + 0.8);
122 T violation = r_safe * r_safe - dist_sq;
123
124 if (Optimization::get_value(violation) > 0.0) {
125 r(idx++) = T(std::sqrt(8000.0)) * violation;
126 } else {
127 r(idx++) = T(0.0);
128 }
129 }
130
131 x = Integrator::rk4<Nx, Nu, DynamicBicycleModel, T>(model, x, u, dt);
132 }
133
134 // 5. 종단 잔차
135 for(size_t i = 0; i < Nx; ++i) {
136 r(idx++) = T(std::sqrt(Qf(static_cast<int>(i)))) * (x(static_cast<int>(i)) - T(x_ref(static_cast<int>(i))));
137 }
138
139 return r;
140 }
141 };

151 struct BoundIneq {
152 template <typename T>
153 StaticVector<T, N_ineq> operator()(const StaticVector<T, N_vars>& U) const {
154 StaticVector<T, N_ineq> ineq;
155 for (size_t k = 0; k < Np; ++k) {
156 T a = U(static_cast<int>(k * Nu + 0));
157 T delta = U(static_cast<int>(k * Nu + 1));
158
159 ineq(static_cast<int>(k * 4 + 0)) = a - T(3.0);
160 ineq(static_cast<int>(k * 4 + 1)) = -a - T(3.0);
161 ineq(static_cast<int>(k * 4 + 2)) = delta - T(0.5);
162 ineq(static_cast<int>(k * 4 + 3)) = -delta - T(0.5);
163 }
164 return ineq;
165 }
