35 // 1. Dense Solver (가우스-뉴턴 기반 조밀 행렬 해법)
36 // ====================================================================
37 template <typename ResidualFunc, typename EqFunc, typename IneqFunc>
38 bool solve_dense(StaticVector<double, N_vars>& u, ResidualFunc res_f, EqFunc eq_f, IneqFunc ineq_f) {
39 auto start = std::chrono::high_resolution_clock::now();
40 profile.current_engine = "Dense (Gaussian IPM)";
41
42 StaticVector<double, N_res> r_val = res_f(u);
43 StaticMatrix<double, N_res, N_vars> J_res = AutoDiff::jacobian<N_res, N_vars>(res_f, u);
44 StaticMatrix<double, N_vars, N_vars> H_GN;
45 StaticVector<double, N_vars> grad_f;
46
47 H_GN.set_zero();
48 grad_f.set_zero();
49
50 for (size_t i = 0; i < N_vars; ++i) {
51 for (size_t j = 0; j < N_vars; ++j) {
52 double sum_h = 0.0;
53 for (size_t k = 0; k < N_res; ++k) {
54 sum_h += J_res(k, i) * J_res(k, j);
55 }
56 H_GN(i, j) = sum_h;
57 }
58 double sum_g = 0.0;
59 for (size_t k = 0; k < N_res; ++k) {
60 sum_g += J_res(k, i) * r_val(k);
61 }
62 grad_f(i) = sum_g;
63 H_GN(i, i) += 5.0; // Levenberg-Marquardt damping
64 }
65
66 StaticVector<double, N_ineq> ineq_val = ineq_f(u);
67 StaticMatrix<double, N_ineq, N_vars> J_ineq = AutoDiff::jacobian<N_ineq, N_vars>(ineq_f, u);
68
69 for (size_t i = 0; i < N_ineq; ++i) {
70 qp_solver_dense.b_ineq(i) = -ineq_val(i);
71 for (size_t j = 0; j < N_vars; ++j) {
72 qp_solver_dense.A_ineq(i, j) = J_ineq(i, j);
73 }
74 }
75
76 qp_solver_dense.P = H_GN;
77 qp_solver_dense.q = grad_f;
78
79 StaticVector<double, N_vars> p;
80 p.set_zero();
81 bool success = qp_solver_dense.solve(p, 15, 1e-3);
82
83 if (success) {
84 for (size_t i = 0; i < N_vars; ++i) {
85 u(i) += p(i);
86 }
87 }
88
89 auto end = std::chrono::high_resolution_clock::now();
90 std::chrono::duration<double, std::milli> elapsed = end - start;
91 profile.last_exec_time_ms = elapsed.count();
92 if (profile.last_exec_time_ms > profile.wcet_ms) {
93 profile.wcet_ms = profile.last_exec_time_ms;
94 }
95
96 return success;
97 }

100 // 2. Sparse Solver (구조적 필터링이 적용된 희소 행렬 해법)
101 // ====================================================================
102 template <typename ResidualFunc, typename EqFunc, typename IneqFunc>
103 bool solve_sparse(StaticVector<double, N_vars>& u, ResidualFunc res_f, EqFunc eq_f, IneqFunc ineq_f) {
104 auto start = std::chrono::high_resolution_clock::now();
105 profile.current_engine = "Sparse (Matrix-Free IPM)";
106
107 // 1. Hessian 행렬과 Gradient 벡터 계산
108 StaticVector<double, N_res> r_val = res_f(u);
109 StaticMatrix<double, N_res, N_vars> J_res = AutoDiff::jacobian<N_res, N_vars>(res_f, u);
110
111 qp_solver_sparse.P.nnz_count = 0;
112 qp_solver_sparse.P.row_ptr.set_zero();
113
114 for (size_t i = 0; i < N_vars; ++i) {
115 for (size_t j = 0; j < N_vars; ++j) {
116 double val = 0.0;
117 for (size_t k = 0; k < N_res; ++k) {
118 val += J_res(k, i) * J_res(k, j);
119 }
120 if (i == j) {
121 val += 5.0; // Levenberg-Marquardt damping
122 }
123 if (std::abs(val) > 1e-9) {
124 qp_solver_sparse.P.add_value(static_cast<int>(i), static_cast<int>(j), val);
125 }
126 } // j 루프 종료
127
128 double sum_g = 0.0;
129 for (size_t k = 0; k < N_res; ++k) {
130 sum_g += J_res(k, i) * r_val(k);
131 }
132 qp_solver_sparse.q(i) = sum_g;
133 } // i 루프 종료
134
135 qp_solver_sparse.P.finalize();
136
137 // 2. Constraint 조립
138 StaticVector<double, N_ineq> ineq_val = ineq_f(u);
139 StaticMatrix<double, N_ineq, N_vars> J_ineq = AutoDiff::jacobian<N_ineq, N_vars>(ineq_f, u);
140
141 qp_solver_sparse.A_ineq.nnz_count = 0;
142 qp_solver_sparse.A_ineq.row_ptr.set_zero();
143
144 for (size_t i = 0; i < N_ineq; ++i) {
145 qp_solver_sparse.b_ineq(i) = -ineq_val(i);
146 for (size_t j = 0; j < N_vars; ++j) {
147 if (std::abs(J_ineq(i, j)) > 1e-9) {
148 qp_solver_sparse.A_ineq.add_value(static_cast<int>(i), static_cast<int>(j), J_ineq(i, j));
149 }
150 }
151 }
152 qp_solver_sparse.A_ineq.finalize();
153
154 // 3. Sparse IPM Solve (Warm-start u)
155 StaticVector<double, N_vars> p;
156 p.set_zero();
157 bool success = qp_solver_sparse.solve(p, 10, 1e-3);
158
159 if (success) {
160 for (size_t i = 0; i < N_vars; ++i) {
161 u(i) += p(i);
162 }
163 }
164
165 // 4. 타이밍 프로파일링
166 auto end = std::chrono::high_resolution_clock::now();
167 std::chrono::duration<double, std::milli> elapsed = end - start;
168 profile.last_exec_time_ms = elapsed.count();
169 if (profile.last_exec_time_ms > profile.wcet_ms) {
170 profile.wcet_ms = profile.last_exec_time_ms;
171 }
172
173 return success;
174 }
