19template <size_t H, size_t Nx, size_t Nu>
20struct RiccatiSolver {
21 std::array<StaticMatrix<double, Nx, Nx>, H> A;
22 std::array<StaticMatrix<double, Nx, Nu>, H> B;
23
24 std::array<StaticMatrix<double, Nx, Nx>, H + 1> Q;
25 std::array<StaticMatrix<double, Nu, Nu>, H> R;
26
27 std::array<StaticVector<double, Nx>, H + 1> q;
28 std::array<StaticVector<double, Nu>, H> r;
29
30 std::array<StaticVector<double, Nx>, H> d;
31
32 // [출력 데이터]
33 std::array<StaticVector<double, Nx>, H + 1> dx;
34 std::array<StaticVector<double, Nu>, H> du;
35
36 SolverStatus solve() {
37 std::array<StaticMatrix<double, Nx, Nx>, H + 1> P;
38 std::array<StaticVector<double, Nx>, H + 1> p;
39
40 std::array<StaticMatrix<double, Nu, Nx>, H> K;
41 std::array<StaticVector<double, Nu>, H> k_ff;
42
43 // --- 1. Backward Pass (꼬리에서 머리로) ---
44 P[H] = Q[H];
45 p[H] = q[H];
46
47 // 루프 내부의 스택 생성/소멸 오버헤드를 막기 위한 캐시 변수 사전 선언
48 StaticMatrix<double, Nx, Nu> PB;
49 StaticMatrix<double, Nx, Nx> PA;
50 StaticMatrix<double, Nu, Nu> Bt_PB;
51 StaticMatrix<double, Nu, Nx> Qux;
52 StaticMatrix<double, Nx, Nx> At_PA;
53 StaticVector<double, Nx> Pd;
54 StaticVector<double, Nx> p_next_mod;
55 StaticVector<double, Nu> Bt_p;
56 StaticVector<double, Nx> At_p;
57
58 for (int k = H - 1; k >= 0; --k) {
59 // [결합 법칙 최적화] P_{k+1} * B_k 와 P_{k+1} * A_k 를 미리 계산 (재사용)
60 linalg::multiply(P[k + 1], B[k], PB);
61 linalg::multiply(P[k + 1], A[k], PA);
62
63 // Quu = R_k + B_k^T * (P_{k+1} * B_k)
64 StaticMatrix<double, Nu, Nu> Quu = R[k];
65 linalg::multiply_AT_B(B[k], PB, Bt_PB); // 가상 전치(Virtual Transpose) 타격
66 Quu += Bt_PB;
67
68 // Qux = B_k^T * (P_{k+1} * A_k)
69 linalg::multiply_AT_B(B[k], PA, Qux);
70
71 // Qxx = Q_k + A_k^T * (P_{k+1} * A_k)
72 StaticMatrix<double, Nx, Nx> Qxx = Q[k];
73 linalg::multiply_AT_B(A[k], PA, At_PA);
74 Qxx += At_PA;
75
76 // p_next_mod = p_{k+1} + P_{k+1} * d_k
77 linalg::multiply(P[k + 1], d[k], Pd);
78 p_next_mod = p[k + 1] + Pd;
79
80 // qu = r_k + B_k^T * p_next_mod
81 StaticVector<double, Nu> qu = r[k];
82 linalg::multiply_AT_B(B[k], p_next_mod, Bt_p);
83 qu += Bt_p;
84
85 // qx = q_k + A_k^T * p_next_mod
86 StaticVector<double, Nx> qx = q[k];
87 linalg::multiply_AT_B(A[k], p_next_mod, At_p);
88 qx += At_p;
89
90 // --- 고속 LDLT 분해 (Quu는 항상 SPD) ---
91 StaticMatrix<double, Nu, Nu> Quu_factored = Quu;
92 if (linalg::LDLT_decompose(Quu_factored) != MathStatus::SUCCESS) {
93 return SolverStatus::MATH_ERROR; // 제어 불가능 지점
94 }
95
96 // K = -Quu^{-1} * Qux 계산 (Column by Column in-place solve)
97 for (size_t j = 0; j < Nx; ++j) {
98 StaticVector<double, Nu> neg_Qux_col;
99 for (size_t i = 0; i < Nu; ++i) neg_Qux_col(i) = -Qux(i, j); // 부호 반전
100
101 StaticVector<double, Nu> K_col;
102 linalg::LDLT_solve(Quu_factored, neg_Qux_col, K_col); // Zero-Allocation Solve
103 for (size_t i = 0; i < Nu; ++i) K[k](i, j) = K_col(i);
104 }
105
106 // k_ff = -Quu^{-1} * qu 계산
107 StaticVector<double, Nu> neg_qu;
108 for (size_t i = 0; i < Nu; ++i) neg_qu(i) = -qu(i);
109 linalg::LDLT_solve(Quu_factored, neg_qu, k_ff[k]);
110
111 // P_k = Qxx + Qux^T * K_k
112 P[k] = Qxx;
113 StaticMatrix<double, Nx, Nx> QuxT_K;
114 linalg::multiply_AT_B(Qux, K[k], QuxT_K);
115 P[k] += QuxT_K;
116
117 // p_k = qx + Qux^T * k_ff_k
118 p[k] = qx;
119 StaticVector<double, Nx> QuxT_kff;
120 linalg::multiply_AT_B(Qux, k_ff[k], QuxT_kff);
121 p[k] += QuxT_kff;
122 }
123
124 // --- 2. Forward Pass (머리에서 꼬리로) ---
125 dx[0].set_zero(); // 초기 상태 섭동은 0
126
127 for (size_t k = 0; k < H; ++k) {
128 // du_k = K_k * dx_k + k_ff_k
129 StaticVector<double, Nu> K_dx;
130 linalg::multiply(K[k], dx[k], K_dx);
131 du[k] = K_dx + k_ff[k];
132
133 // dx_{k+1} = A_k * dx_k + B_k * du_k + d_k
134 StaticVector<double, Nx> A_dx, B_du;
135 linalg::multiply(A[k], dx[k], A_dx);
136 linalg::multiply(B[k], du[k], B_du);
137 dx[k + 1] = A_dx + B_du + d[k];
138 }
139
140 return SolverStatus::SUCCESS;
141 }
142};
