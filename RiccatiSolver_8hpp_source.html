19template <size_t H, size_t Nx, size_t Nu>
20struct RiccatiSolver {
21 // [입력 데이터] 각 스텝 k에서의 선형화된 행렬들
22 std::array<StaticMatrix<double, Nx, Nx>, H> A; // 상태 자코비안
23 std::array<StaticMatrix<double, Nx, Nu>, H> B; // 입력 자코비안
24
25 std::array<StaticMatrix<double, Nx, Nx>, H + 1> Q; // 상태 비용 (Hessian)
26 std::array<StaticMatrix<double, Nu, Nu>, H> R; // 입력 비용 (Hessian)
27
28 std::array<StaticVector<double, Nx>, H + 1> q; // 상태 그래디언트 (Residual)
29 std::array<StaticVector<double, Nu>, H> r; // 입력 그래디언트 (Residual)
30
31 std::array<StaticVector<double, Nx>, H> d; // 동역학 오차 (Dynamics Gap: x_{next} - f(x,u))
32
33 // [출력 데이터] 계산된 최적 스텝 (탐색 방향)
34 std::array<StaticVector<double, Nx>, H + 1> dx;
35 std::array<StaticVector<double, Nu>, H> du;
36
40 SolverStatus solve() {
41 // Value Function P (Cost-to-go Hessian) 및 p (Cost-to-go Gradient)
42 std::array<StaticMatrix<double, Nx, Nx>, H + 1> P;
43 std::array<StaticVector<double, Nx>, H + 1> p;
44
45 // Feedback Gain K 및 Feedforward term k
46 std::array<StaticMatrix<double, Nu, Nx>, H> K;
47 std::array<StaticVector<double, Nu>, H> k_ff;
48
49 // --- 1. Backward Pass (꼬리에서 머리로) ---
50 // Terminal 조건 초기화
51 P[H] = Q[H];
52 p[H] = q[H];
53
54 for (int k = H - 1; k >= 0; --k) {
55 // Quu = R_k + B_k^T * P_{k+1} * B_k
56 StaticMatrix<double, Nu, Nu> Quu = R[k] + B[k].transpose() * P[k + 1] * B[k];
57 // Qux = B_k^T * P_{k+1} * A_k
58 StaticMatrix<double, Nu, Nx> Qux = B[k].transpose() * P[k + 1] * A[k];
59 // Qxx = Q_k + A_k^T * P_{k+1} * A_k
60 StaticMatrix<double, Nx, Nx> Qxx = Q[k] + A[k].transpose() * P[k + 1] * A[k];
61
62 // qu = r_k + B_k^T * p_{k+1} + B_k^T * P_{k+1} * d_k
63 StaticVector<double, Nu> qu =
64 r[k] + B[k].transpose() * p[k + 1] + B[k].transpose() * (P[k + 1] * d[k]);
65 // qx = q_k + A_k^T * p_{k+1} + A_k^T * P_{k+1} * d_k
66 StaticVector<double, Nx> qx =
67 q[k] + A[k].transpose() * p[k + 1] + A[k].transpose() * (P[k + 1] * d[k]);
68
69 // Quu의 역행렬을 구하여 Gain 계산 (Nu가 2이므로 초고속 연산 가능)
70 // -Quu * K = Qux => K = -Quu^{-1} * Qux
71 // -Quu * k_ff = qu => k_ff = -Quu^{-1} * qu
72 StaticMatrix<double, Nu, Nu> neg_Quu;
73 for (size_t i = 0; i < Nu; ++i)
74 for (size_t j = 0; j < Nu; ++j) neg_Quu(i, j) = -Quu(i, j);
75
76 // Nu 크기(2x2)의 LU 분해 (혹은 LDLT)
77 StaticVector<int, Nu> pivot;
78 if (linalg::LU_decompose(neg_Quu, pivot) != MathStatus::SUCCESS) {
79 return SolverStatus::MATH_ERROR; // 제어 불가능한 특이 지점
80 }
81
82 // 열벡터 단위로 해를 구하여 K 행렬 조립
83 for (size_t j = 0; j < Nx; ++j) {
84 StaticVector<double, Nu> Qux_col;
85 for (size_t i = 0; i < Nu; ++i) Qux_col(i) = Qux(i, j);
86 StaticVector<double, Nu> K_col = linalg::LU_solve(neg_Quu, pivot, Qux_col);
87 for (size_t i = 0; i < Nu; ++i) K[k](i, j) = K_col(i);
88 }
89
90 // k_ff 계산
91 k_ff[k] = linalg::LU_solve(neg_Quu, pivot, qu);
92
93 // P_k = Qxx + Qux^T * K
94 P[k] = Qxx + Qux.transpose() * K[k];
95 // p_k = qx + Qux^T * k_ff
96 p[k] = qx + Qux.transpose() * k_ff[k];
97 }
98
99 // --- 2. Forward Pass (머리에서 꼬리로) ---
100 // 초기 상태는 고정되어 있으므로 변화량(dx_0)은 0
101 for (size_t i = 0; i < Nx; ++i) dx[0](i) = 0.0;
102
103 for (size_t k = 0; k < H; ++k) {
104 // du_k = K_k * dx_k + k_ff_k
105 du[k] = K[k] * dx[k] + k_ff[k];
106
107 // dx_{k+1} = A_k * dx_k + B_k * du_k + d_k
108 dx[k + 1] = A[k] * dx[k] + B[k] * du[k] + d[k];
109 }
110
111 return SolverStatus::SUCCESS;
112 }
113};
