29template <size_t N_vars, size_t N_eq, size_t N_ineq>
30class SQPSolver {
31 public:
32 // 하부 구조: 정적 메모리 기반 Primal-Dual IPM(Interior Point Method) 엔진
33 // 매 SQP 반복(iteration)마다 발생하는 QP 하위 문제를 풀기 위해 사용됩니다.
34 IPMQPSolver<N_vars, N_eq, N_ineq> qp_solver;

36 // 헤시안 근사 행렬 (Hessian Approximation Matrix)
37 // 목적 함수의 라그랑지안에 대한 2차 미분(헤시안) 정보를 담고 있으며,
38 // BFGS(Broyden-Fletcher-Goldfarb-Shanno) 알고리즘을 통해 매 반복마다 점진적으로 업데이트됩니다.
39 StaticMatrix<double, N_vars, N_vars> H;
40
47 SQPSolver() {

68 template <typename CostFunc, typename EqFunc, typename IneqFunc>
69 bool solve(StaticVector<double, N_vars>& u, CostFunc cost_f, EqFunc eq_f, IneqFunc ineq_f,
70 int max_iter = 50) {
71 for (int iter = 0; iter < max_iter; ++iter) {
72 // 1. 현재 상태에서 목적 함수의 값과 기울기(Gradient) 계산
73 double cost_val = 0.0;
74 StaticVector<double, N_vars> grad_f;
75 // AutoDiff(자동 미분)를 사용하여 목적 함수의 값과 기울기를 평가합니다.
76 AutoDiff::value_and_gradient<N_vars>(cost_f, u, cost_val, grad_f);
77
78 // 2. 등식 제약 조건 선형화 (Linearization of Equality Constraints)
79 // 테일러 1차 전개를 통해 등식 제약을 선형 제약으로 변환합니다.
80 // c(u + p) ~ c(u) + J_eq * p = 0 -> J_eq * p = -c(u)
81 if constexpr (N_eq > 0) {
82 StaticVector<double, N_eq> eq_val =
83 eq_f(u); // 현재 위치에서의 등식 제약 조건 값 평가
84 StaticMatrix<double, N_eq, N_vars> J_eq =
85 AutoDiff::jacobian<N_eq, N_vars>(eq_f, u); // 야코비안 계산
86 for (size_t i = 0; i < N_eq; ++i) {
87 qp_solver.b_eq(static_cast<int>(i)) =
88 -eq_val(static_cast<int>(i)); // 우변: -c(u)
89 for (size_t j = 0; j < N_vars; ++j) {
90 qp_solver.A_eq(static_cast<int>(i), static_cast<int>(j)) =
91 J_eq(static_cast<int>(i), static_cast<int>(j)); // 좌변: Jacobian 행렬
92 }
93 }
94 }
95
96 // 3. 부등식 제약 조건 선형화 (Linearization of Inequality Constraints)
97 // 테일러 1차 전개를 통해 부등식 제약을 선형 제약으로 변환합니다.
98 // g(u + p) ~ g(u) + J_ineq * p <= 0 -> J_ineq * p <= -g(u)
99 if constexpr (N_ineq > 0) {
100 StaticVector<double, N_ineq> ineq_val =
101 ineq_f(u); // 현재 위치에서의 부등식 제약 조건 값 평가
102 StaticMatrix<double, N_ineq, N_vars> J_ineq =
103 AutoDiff::jacobian<N_ineq, N_vars>(ineq_f, u); // 야코비안 계산
104 for (size_t i = 0; i < N_ineq; ++i) {
105 qp_solver.b_ineq(static_cast<int>(i)) =
106 -ineq_val(static_cast<int>(i)); // 우변: -g(u)
107 for (size_t j = 0; j < N_vars; ++j) {
108 qp_solver.A_ineq(static_cast<int>(i), static_cast<int>(j)) = J_ineq(
109 static_cast<int>(i), static_cast<int>(j)); // 좌변: Jacobian 행렬
110 }
111 }
112 }
113
114 // 4. QP 하위 문제 설정 및 풀이
115 // QP 목적 함수: 1/2 p^T * H * p + grad_f^T * p
116 qp_solver.P = H; // 이차항 계수 (헤시안 근사)
117 qp_solver.q = grad_f; // 일차항 계수 (목적 함수의 기울기)
118
119 StaticVector<double, N_vars> p;
120 p.set_zero();
121
122 // IPM 솔버 가동 (톨러런스를 1e-4로 설정하여 실시간성 확보)
123 // QP 솔버가 성공적으로 탐색 방향 p를 찾지 못한 경우에 대한 Fallback 메커니즘
124 if (!qp_solver.solve(p, 50, 1e-4)) {
125 // QP 풀이에 실패하면, 단순 경사 하강법(Gradient Descent) 방향으로 약간
126 // 이동시킵니다.
127 p = grad_f * -0.05;
128 }
129
130 // 구한 탐색 방향(p)에 대한 안정성 및 제약 처리
131 double p_norm = 0.0;
132 for (size_t i = 0; i < N_vars; ++i) {
133 // NaN 또는 Inf 값이 발생한 경우 0으로 초기화하여 발산 방지
134 if (std::isnan(p(static_cast<int>(i))) || std::isinf(p(static_cast<int>(i)))) {
135 p(static_cast<int>(i)) = 0.0;
136 }
137 // 한 제어 주기당 변화할 수 있는 물리적 최대 한계치 클리핑 (예: 가속도 +-3.0, 조향
138 // +-3.0 등) 너무 큰 step이 발생하여 시스템이 불안정해지는 것을 막습니다.
139 if (p(static_cast<int>(i)) > 3.0) p(static_cast<int>(i)) = 3.0;
140 if (p(static_cast<int>(i)) < -3.0) p(static_cast<int>(i)) = -3.0;
141
142 // 탐색 방향 벡터 p의 크기(L-infinity norm) 계산
143 p_norm = std::max(p_norm, std::abs(p(static_cast<int>(i))));
144 }
145
146 // 5. 수렴 조건 검사 (Convergence Check)
147 // 스텝 크기가 1e-6보다 작으면 더 이상 유의미한 변화가 없다고 판단하고 최적해 도달로
148 // 간주합니다.
149 if (p_norm < 1e-6) return true;
150
151 // 6. 라인 서치 (Line Search) 기반 스텝 크기 결정
152 // L1 Merit Function을 사용하여 목적 함수 값의 감소와 제약 조건 위반도 감소의 균형을
153 // 맞춥니다.
154 double alpha = 1.0; // 초기 스텝 사이즈 (1.0 = Newton step)
155 const double rho = 10.0; // 패널티 파라미터 (제약 조건 위반에 대한 벌점 가중치)
156
157 // 현재 위치에서의 Merit 함수 값 계산
158 double current_merit = cost_val;
159 if constexpr (N_eq > 0) {
160 StaticVector<double, N_eq> v = eq_f(u);
161 for (size_t i = 0; i < N_eq; ++i)
162 current_merit += rho * std::abs(v(static_cast<int>(i))); // 등식 제약 위반: |v|
163 }
164 if constexpr (N_ineq > 0) {
165 StaticVector<double, N_ineq> v = ineq_f(u);
166 for (size_t i = 0; i < N_ineq; ++i)
167 current_merit +=
168 rho * std::max(0.0, v(static_cast<int>(i))); // 부등식 제약 위반: max(0, v)
169 }
170
171 // 백트래킹 라인 서치 (Backtracking Line Search) 수행
172 StaticVector<double, N_vars> u_next;
173 for (int ls_iter = 0; ls_iter < 10; ++ls_iter) {
174 // 다음 시도 위치: u_next = u + alpha * p
175 for (size_t i = 0; i < N_vars; ++i)
176 u_next(static_cast<int>(i)) =
177 u(static_cast<int>(i)) + alpha * p(static_cast<int>(i));
178
179 // 시도 위치에서의 Merit 함수 값 평가
180 double next_cost = cost_f(u_next);
181 double next_merit = next_cost;
182
183 if constexpr (N_eq > 0) {
184 StaticVector<double, N_eq> v = eq_f(u_next);
185 for (size_t i = 0; i < N_eq; ++i)
186 next_merit += rho * std::abs(v(static_cast<int>(i)));
187 }
188 if constexpr (N_ineq > 0) {
189 StaticVector<double, N_ineq> v = ineq_f(u_next);
190 for (size_t i = 0; i < N_ineq; ++i)
191 next_merit += rho * std::max(0.0, v(static_cast<int>(i)));
192 }
193
194 // Armijo 조건과 유사하게, Merit 함수 값이 감소했으면 적절한 스텝 사이즈로 간주하고
195 // 루프 종료
196 if (next_merit < current_merit) break;
197
198 // 감소하지 않았다면 스텝 사이즈를 절반으로 줄이고 다시 시도
199 alpha *= 0.5;
200 }
201
202 // 7. Damped BFGS Update (헤시안 근사 행렬 업데이트)
203 // 새로운 위치(u_next)에서의 목적 함수 기울기 계산
204 StaticVector<double, N_vars> next_grad = AutoDiff::gradient<N_vars>(cost_f, u_next);
205 StaticVector<double, N_vars> s; // 변위 벡터: s = u_next - u
206 StaticVector<double, N_vars> y; // 기울기 변화 벡터: y = grad(u_next) - grad(u)
207 double s_norm_sq = 0.0;
208
209 for (size_t i = 0; i < N_vars; ++i) {
210 int idx = static_cast<int>(i);
211 s(idx) = u_next(idx) - u(idx);
212 y(idx) = next_grad(idx) - grad_f(idx);
213 s_norm_sq += s(idx) * s(idx); // ||s||^2
214 }
215
216 // 변위가 유의미할 때만(1e-12보다 클 때) 헤시안을 업데이트하여 수치적 불안정성 회피
217 if (s_norm_sq > 1e-12) {
218 StaticVector<double, N_vars> Hs;
219 Hs.set_zero();
220 double sHs = 0.0; // s^T * H * s
221 double ys = 0.0; // y^T * s
222
223 for (size_t i = 0; i < N_vars; ++i) {
224 int r = static_cast<int>(i);
225 for (size_t j = 0; j < N_vars; ++j) {
226 Hs(r) += H(r, static_cast<int>(j)) * s(static_cast<int>(j)); // H * s 계산
227 }
228 sHs += s(r) * Hs(r);
229 ys += y(r) * s(r);
230 }
231
232 // Damping factor 계산 (Powell's modification)
233 // 곡률 조건(Curvature condition, y^T * s > 0)을 보장하고 헤시안 행렬 H의
234 // 양의 정부호(Positive Definite) 성질을 유지하기 위해 사용하는 기법
235 double theta = 1.0;
236 if (ys < 0.2 * sHs) {
237 theta = (0.8 * sHs) / (sHs - ys + 1e-16);
238 }
239
240 // 수정된 y 벡터 (r_vec) 계산: r = theta * y + (1 - theta) * H * s
241 StaticVector<double, N_vars> r_vec;
242 double rs = 0.0; // r_vec^T * s
243 for (size_t i = 0; i < N_vars; ++i) {
244 int idx = static_cast<int>(i);
245 r_vec(idx) = theta * y(idx) + (1.0 - theta) * Hs(idx);
246 rs += r_vec(idx) * s(idx);
247 }
248
249 // BFGS 행렬 업데이트 공식 적용
250 // 수치적 에러를 방지하기 위해 분모(rs, sHs)가 매우 작지 않은지 검사
251 if (rs > 1e-12 && sHs > 1e-12) {
252 for (size_t i = 0; i < N_vars; ++i) {
253 for (size_t j = 0; j < N_vars; ++j) {
254 int row = static_cast<int>(i);
255 int col = static_cast<int>(j);
256 // H_new = H - (H*s * s^T*H) / (s^T*H*s) + (r * r^T) / (r^T*s)
257 H(row, col) = H(row, col) - (Hs(row) * Hs(col)) / sHs +
258 (r_vec(row) * r_vec(col)) / rs;
259 }
260 }
261 }
262 }
263
264 // 8. 변수 업데이트: 다음 반복(iteration)을 위해 u를 갱신
265 u = u_next;
266 }
267
268 // 최대 반복 횟수(max_iter)에 도달했지만 수렴 조건을 만족하지 못한 경우
269 return false;
270 }
