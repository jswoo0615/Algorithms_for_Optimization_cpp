32 template <typename CostFunc, typename EqFunc, typename IneqFunc>
33 bool solve(StaticVector<double, N_vars>& u, CostFunc cost_f, EqFunc eq_f, IneqFunc ineq_f, int max_iter = 50) {
34 for (int iter = 0; iter < max_iter; ++iter) {
35 // 1. 현재 지점의 함수값 및 Gradient 추출
36 double cost_val = 0.0;
37 StaticVector<double, N_vars> grad_f;
38 AutoDiff::value_and_gradient<N_vars>(cost_f, u, cost_val, grad_f);
39
40 // 2. 등식 제약 조건 선형화 (StaticMatrix 규격 적용)
41 if constexpr (N_eq > 0) {
42 StaticVector<double, N_eq> eq_val = eq_f(u);
43 StaticMatrix<double, N_eq, N_vars> J_eq = AutoDiff::jacobian<N_eq, N_vars>(eq_f, u);
44 qp_solver.A_eq = J_eq;
45 qp_solver.b_eq = eq_val * -1.0;
46 }
47
48 // 3. 부등식 제약 조건 선형화 및 Active Set 초기화
49 if constexpr (N_ineq > 0) {
50 StaticVector<double, N_ineq> ineq_val = ineq_f(u);
51 StaticMatrix<double, N_ineq, N_vars> J_ineq = AutoDiff::jacobian<N_ineq, N_vars>(ineq_f, u);
52 qp_solver.A_ineq = J_ineq;
53 qp_solver.b_ineq = ineq_val * -1.0;
54
55 // Infeasible Start 방지를 위한 제약 위반 감시
56 for (size_t i = 0; i < N_ineq; ++i) {
57 qp_solver.working_set[i] = (qp_solver.b_ineq(static_cast<int>(i)) < -1e-6);
58 }
59 }
60
61 // 4. QP 하위 문제 해결
62 qp_solver.P = H;
63 qp_solver.q = grad_f;
64 StaticVector<double, N_vars> p;
65 p.set_zero();
66
67 if (!qp_solver.solve(p, 100)) {
68 // QP 실패 시 Fallback : Steepest Descent 방향으로 비상 탈출
69 p = grad_f * -0.05;
70 }
71
72 // 5. 수렴 판정
73 double p_norm = 0.0;
74 for (size_t i = 0; i < N_vars; ++i) {
75 p_norm = std::max(p_norm, std::abs(p(static_cast<int>(i))));
76 }
77 if (p_norm < 1e-6) {
78 return true;
79 }
80
81 // 6. Merit Function 기반 Backtracking Line Search
82 // LineSearch::ray_point 개념을 활용한 스텝 제어
83 double alpha = 1.0;
84 const double rho = 10.0; // 제약 조건 패널티 가중치
85
86 // 현재 Merit 계산
87 double current_merit = cost_val;
88 if constexpr (N_eq > 0) {
89 StaticVector<double, N_eq> v = eq_f(u);
90 for (size_t i = 0; i < N_eq; ++i) {
91 current_merit += rho * std::abs(v(static_cast<int>(i)));;
92 }
93 }
94 if constexpr (N_ineq > 0) {
95 StaticVector<double, N_ineq> v = ineq_f(u);
96 for (size_t i = 0; i < N_ineq; ++i) {
97 current_merit += rho * std::max(0.0, v(static_cast<int>(i)) + alpha * p(static_cast<int>(i)));
98 }
99 }
100 StaticVector<double, N_vars> u_next;
101 for (int ls_iter = 0; ls_iter < 10; ++ls_iter) {
102 // u_next = u + alpha * p (LineSearch의 ray_point 로직과 동일)
103 for (size_t i = 0; i < N_vars; ++i) {
104 u_next(static_cast<int>(i)) = u(static_cast<int>(i)) + alpha * p(static_cast<int>(i));
105 }
106 double next_cost = cost_f(u_next);
107 double next_merit = next_cost;
108
109 if constexpr (N_eq > 0) {
110 StaticVector<double, N_eq> v = eq_f(u_next);
111 for (size_t i = 0; i < N_eq; ++i) {
112 next_merit += rho * std::abs(v(static_cast<int>(i)));
113 }
114 }
115 if constexpr (N_ineq > 0) {
116 StaticVector<double, N_ineq> v = ineq_f(u_next);
117 for (size_t i = 0; i < N_ineq; ++i) {
118 next_merit += rho * std::max(0.0, v(static_cast<int>(i)));
119 }
120 }
121 if (next_merit < current_merit) {
122 break;
123 }
124 alpha *= 0.5; // 보폭 축소
125 }
126
127 // 7. Damped BFGS Hessian Update (Powell's Trick)
128 StaticVector<double, N_vars> next_grad = AutoDiff::gradient<N_vars>(cost_f, u_next);
129 StaticVector<double, N_vars> s;
130 StaticVector<double, N_vars> y;
131 double s_norm_sq = 0.0;
132
133 for (size_t i = 0; i < N_vars; ++i) {
134 int idx = static_cast<int>(i);
135 s(idx) = u_next(idx) - u(idx);
136 // SQP는 라그랑지안 미분을 써야 하나, 안정성 확보를 위해 우선 목적함수 미분으로 근사
137 y(idx) = next_grad(idx) - grad_f(idx);
138 s_norm_sq += s(idx) * s(idx);
139 }
140
141 if (s_norm_sq > 1e-12) {
142 StaticVector<double, N_vars> Hs;
143 Hs.set_zero();
144 double sHs = 0.0;
145 double ys = 0.0;
146
147 // H * s 연산
148 for (size_t i = 0; i < N_vars; ++i) {
149 int r = static_cast<int>(i);
150 for (size_t j = 0; j < N_vars; ++j) {
151 Hs(r) += H(r, static_cast<int>(j)) * s(static_cast<int>(j));
152 }
153 sHs += s(r) * Hs(r);
154 ys += y(r) * s(r);
155 }
156
157 // Powell's Damping : 제약 조건으로 인해 H가 양의 정부호 (Positivie Definite)를 잃는 것을 방지
158 double theta = 1.0;
159 if (ys < 0.2 * sHs) {
160 theta = (0.8 * sHs) / (sHs - ys + 1e-16);
161 }
162
163 StaticVector<double, N_vars> r_vec;
164 double rs = 0.0;
165 for (size_t i = 0; i < N_vars; ++i) {
166 int idx = static_cast<int>(i);
167 r_vec(idx) = theta * y(idx) + (1.0 - theta) * Hs(idx);
168 rs += r_vec(idx) * s(idx);
169 }
170
171 // H_new = H - (Hs * HS^T)/sHs + (r * r^T)/rs
172 if (rs > 1e-12 && sHs > 1e-12) {
173 for (size_t i = 0; i < N_vars; ++i) {
174 for (size_t j = 0; j < N_vars; ++j) {
175 int row = static_cast<int>(i);
176 int col = static_cast<int>(j);
177 H(row, col) = H(row, col) - (Hs(row) * Hs(col)) / sHs + (r_vec(row) * r_vec(col)) / rs;
178 }
179 }
180 }
181 }
182 u = u_next;
183 }
184 return false;
185 }
