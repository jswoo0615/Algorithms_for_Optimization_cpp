24class SQPSolver {

29 static void compute_cost_derivatives(const CostFunctor& f, const StaticVector<double, Nx>& x,
30 StaticVector<double, Nx>& g,
31 StaticMatrix<double, Nx, Nx>& H) {
32 using ADVar = DualVec<double, Nx>;
33 constexpr double h = 1e-5;
34 constexpr double inv_2h = 1.0 / (2.0 * h);
35
36 StaticVector<ADVar, Nx> x_dual;
37 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(x(i), i);
38
39 ADVar f_dual = f(x_dual);
40 for (size_t i = 0; i < Nx; ++i) g(i) = f_dual.g[i];
41
42 // Central Difference for Exact Hessian
43 StaticVector<ADVar, Nx> x_plus, x_minus;
44 for (size_t i = 0; i < Nx; ++i) {
45 for (size_t k = 0; k < Nx; ++k)
46 x_plus(k) = ADVar::make_variable(x(k) + (k == i ? h : 0.0), k);
47 ADVar f_plus = f(x_plus);
48
49 for (size_t k = 0; k < Nx; ++k)
50 x_minus(k) = ADVar::make_variable(x(k) - (k == i ? h : 0.0), k);
51 ADVar f_minus = f(x_minus);
52
53 for (size_t j = 0; j < Nx; ++j) {
54 H(j, i) = (f_plus.g[j] - f_minus.g[j]) * inv_2h;
55 }
56 }
57
58 // [Architect's Update] 대칭화 및 SPD 강제 (Regularization)
59 // QP 솔버인 IPM이 무조건 풀 수 있도록 대각 성분에 미세한 양수를 더해줍니다.
60 for (size_t i = 0; i < Nx; ++i) {
61 for (size_t j = 0; j < i; ++j) {
62 double sym = 0.5 * (H(i, j) + H(j, i));
63 H(i, j) = sym;
64 H(j, i) = sym;
65 }
66 H(i, i) += 1e-4; // Regularization Shield
67 }
68 }

70 public:
71 static SolverStatus solve(const CostFunctor& cost_func, const IneqFunctor& ineq_func,
72 StaticVector<double, Nx>& x_opt, int sqp_max_iter = 15,
73 double tol = 1e-5) {
74 using ADVar = DualVec<double, Nx>;
75 StaticVector<ADVar, Nx> x_dual;
76 StaticMatrix<double, Nx, Nx> H;
77 StaticVector<double, Nx> g;
78 StaticMatrix<double, Nc, Nx> C;
79 StaticVector<double, Nc> d;
80 StaticVector<double, Nx> dx;
81
82 for (int sqp_iter = 0; sqp_iter < sqp_max_iter; ++sqp_iter) {
83 // 1. 비선형 목적 함수 -> QP용 H, g 도출
84 compute_cost_derivatives(cost_func, x_opt, g, H);
85
86 // 2. 비선형 제약조건 -> QP용 선형 제약 도출 (C * dx <= d)
87 // h(x + dx) \approx h(x) + J_h * dx <= 0 ==> J_h * dx <= -h(x)
88 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(x_opt(i), i);
89 StaticVector<ADVar, Nc> h_dual = ineq_func(x_dual);
90
91 for (size_t i = 0; i < Nc; ++i) {
92 d(i) = -Optimization::get_value(h_dual(i)); // d = -h(x)
93 for (size_t j = 0; j < Nx; ++j) {
94 C(i, j) = h_dual(i).g[j]; // C = J_h
95 }
96 }
97
98 // 3. QP 서브프라블럼 하청 (IPM 호출)
99 dx.set_zero();
100 SolverStatus qp_status = QPSolver_IPM<Nx, Nc>::solve(H, g, C, d, dx, 40, 1e-5);
101
102 // 수학적 붕괴 시 즉시 제어권 반환
103 if (qp_status == SolverStatus::MATH_ERROR || qp_status == SolverStatus::INFEASIBLE) {
104 return qp_status;
105 }
106
107 // 4. 프라이멀 변수 업데이트 (x_opt += dx)
108 double max_dx = 0.0;
109 for (size_t i = 0; i < Nx; ++i) {
110 x_opt(i) += dx(i);
111 double abs_dx = MathTraits<double>::abs(dx(i));
112 if (abs_dx > max_dx) max_dx = abs_dx;
113 }
114
115 // 5. SQP 수렴 판정 (더 이상 이동하지 않으면 최적해 도달)
116 if (max_dx < tol) {
117 return SolverStatus::SUCCESS;
118 }
119 }
120
121 return SolverStatus::MAX_ITERATION_REACHED;
122 }
