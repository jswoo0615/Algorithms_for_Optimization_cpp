74 void apply_H_sys(const StaticVector<double, N_vars>& p, const StaticVector<double, N_ineq>& W, StaticVector<double, N_vars>& y) const {
75 // 1. y = P * p (목적 함수 곡률 방향 투영)
76 P.multiply(p, y);
77
78 // 2. temp1 = A * p (제약 조건 방향 투영)
79 StaticVector<double, N_ineq> Ap;
80 A_ineq.multiply(p, Ap);
81
82 // 3. temp2 = W * (A * p) (가중치 대각 행렬 W 적용)
83 for (size_t i = 0; i < N_ineq; ++i) {
84 Ap(i) *= W(i);
85 }
86
87 // 4. y += A^T * temp2 (다시 원래 변수 공간으로 역투영하여 누적 합산)
88 StaticVector<double, N_vars> AT_Ap;
89 A_ineq.multiply_transpose(Ap, AT_Ap);
90 for (size_t i = 0; i < N_vars; ++i) {
91 y(i) += AT_Ap(i);
92 }
93 }

95 // [Conjugate Gradient] 축소된 KKT 시스템 해결
96 bool solve_implicit_cg(const StaticVector<double, N_vars>& rhs, StaticVector<double, N_ineq>& W, StaticVector<double, N_vars>& dx) const {
97 StaticVector<double, N_vars> r = rhs; // 초기 추정치 dx = 0
98 StaticVector<double, N_vars> p_cg = r;
99 StaticVector<double, N_vars> Ap_cg;
100
101 double rsold = 0.0;
102 for (size_t i = 0; i < N_vars; ++i) {
103 rsold += r(i) * r(i);
104 }
105 if (rsold < 1e-8) {
106 return true;
107 }
108
109 for (int iter = 0; iter < 30; ++iter) { // 고정 반복 (WCET 보장)
110 apply_H_sys(p_cg, Ap_cg);
111 double pAp = 0.0;
112 for (size_t i = 0; i < N_vars; ++i) {
113 pAp += p_cg(i) * Ap_cg(i);
114 }
115 if (std::abs(pAp) < 1e-12) {
116 break;
117 }
118 double alpha = rsold / pAp;
119 double rsnew = 0.0;
120 for (size_t i = 0; i < N_vars; ++i) {
121 dx(i) += alpha * p_cg(i);
122 r(i) -= alpha * Ap_cg(i);
123 rsnew += r(i) * r(i);
124 }
125 if (rsnew < 1e-6) {
126 return true;
127 }
128 double beta = rsnew / rsold;
129 for (size_t i = 0; i < N_vars; ++i) {
130 p_cg(i) = r(i) + beta * p_cg(i);
131 }
132 rsold = rsnew;
133 }
134 return false;
135 }

137 bool solve(StaticVector<double, N_vars>& out_x, int max_iter = 10, double tol = 1e-3) {
138 // Warm-start
139 for (size_t i = 0; i < N_vars; ++i) {
140 x(i) = out_x(i);
141 }
142
143 for (int iter = 0; iter < max_iter; ++iter) {
144 // 1. Residual 계산 (KKT 시스템의 잔차)
145 StaticVector<double, N_vars> Px;
146 P.multiply(x, Px);
147 StaticVector<double, N_vars> AT_lambda;
148 A_ineq.multiply_transpose(lambda, AT_lambda);
149 StaticVector<double, N_vars> r_d; // r_d = Px + q + A^T lambda
150 for (size_t i = 0; i < N_vars; ++i) {
151 r_d(i) = Px(i) + q(i) + AT_lambda(i);
152 }
153
154 StaticVector<double, N_ineq> Ax;
155 A_ineq.multiply(x, Ax);
156 StaticVector<double, N_ineq> r_p; // r_p = Ax - b + s
157 for (size_t i = 0; i < N_ineq; ++i) {
158 r_p(i) = Ax(i) - b_ineq(i); + s(i);
159 }
160
161 double gap = 0.0;
162 for (size_t i = 0; i < N_ineq; ++i) {
163 gap += lambda(i) * s(i);
164 }
165 double mu = gap / N_ineq;
166
167 if (mu < tol) { // 수렴 시
168 break;
169 }
170
171 double sigma = 0.1; // Centering 파라미터
172 double target_mu = sigma * mu;
173
174 // 2. Weight Matrix (W = lambda * S^{-1}) 및 RHS 조립
175 StaticVector<double, N_ineq> W;
176 StaticVector<double, N_ineq> rhs = r_d; // rhs = -r_d - A^T (S^{-1} (Lambda r_p - r_c))
177 StaticVector<double, N_ineq> temp_vec;
178 for (size_t i = 0; i < N_ineq; ++i) {
179 W(i) = lambda(i) / s(i);
180 double r_c = lambda(i) * s(i) - target_mu;
181 temp_vec(i) = (lambda(i) * r_p(i) - r_c) / s(i);
182 }
183
184 StaticVector<double, N_vars> AT_temp;
185 A_ineq.multiply_transpose(temp_vec, AT_temp);
186 for (size_t i = 0; i < N_vars; ++i) {
187 rhs(i) = -r_d(i) - AT_temp(i);
188 }
189
190 // 3. Newton Step 계산 (Implicit CG)
191 StaticVector<double, N_vars> dx;
192 dx.set_zero();
193 solve_implicit_cg(rhs, W, dx);
194
195 // 4. 슬랙 및 듀얼 스텝 복원
196 StaticVector<double, N_ineq> A_dx;
197 A_ineq.multiply(dx, A_dx);
198 StaticVector<double, N_ineq> ds, dlambda;
199 for (size_t i = 0; i < N_ineq; ++i) {
200 ds(i) = -r_p(i) - A_dx(i);
201 double r_c = lambda(i) * s(i) - target_mu;
202 dlambda(i) = -(r_c + lambda(i) * ds(i)) / s(i);
203 }
204
205 // 5. Fraction-to-the-boundary step size
206 double alpha_p = 1.0, alpha_d = 1.0;
207 for (size_t i = 0; i < N_ineq; ++i) {
208 if (ds(i) < 0.0) {
209 alpha_p = std::min(alpha_p, -0.99 * s(i) / ds(i));
210 }
211 if (dlambda(i) < 0.0) {
212 alpha_d = std::min(alpha_d, -0.99 * lambda(i) / dlambda(i));
213 }
214 }
215
216 // 6. 업데이트
217 for (size_t i = 0; i < N_vars; ++i) {
218 x(i) += alpha_p * dx(i);
219 }
220 for (size_t i = 0; i < N_ineq; ++i) {
221 s(i) += alpha_p * ds(i);
222 lambda(i) += alpha_d * dlambda(i);
223 }
224 }
225
226 // 결과 반환
227 for (size_t i = 0; i < N_vars; ++i) {
228 out_x(i) = x(i);
229 }
230 return true;
231 }
