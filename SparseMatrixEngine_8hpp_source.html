9// [Architect's True Sparse] 동적 할당이 완벽히 배제된 정적 CSR 행렬 구조
10template <typename T, size_t Rows, size_t Cols, size_t MaxNNZ>
11class StaticSparseMatrix {
12public:
13 StaticVector<T, MaxNNZ> values; // 0이 아닌 실제 값들
14 StaticVector<int, MaxNNZ> col_index; // 해당 값의 열(Column) 인덱스
15 StaticVector<int, Rows + 1> row_ptr; // 각 행(Row)이 시작되는 인덱스 포인터
16 size_t nnz_count; // 현재 채워진 Non-zero 개수
17
18 StaticSparseMatrix() : nnz_count(0) {
19 values.set_zero();
20 col_index.set_zero();
21 row_ptr.set_zero();
22 }
23
24 // 0이 아닌 값만 밀어 넣음 (메모리 폭발 방지)
25 void add_value(int row, int col, T val) {
26 if (nnz_count >= MaxNNZ) return; // 오버플로우 방탄조끼
27 values(nnz_count) = val;
28 col_index(nnz_count) = col;
29 row_ptr(row + 1)++;
30 nnz_count++;
31 }
32
33 void finalize() {
34 for (size_t i = 0; i < Rows; ++i) {
35 row_ptr(i + 1) += row_ptr(i);
36 }
37 }
38
39 // 핵심: O(N^2)이 아닌 O(NNZ) 속도의 초고속 행렬-벡터 곱셈 (SpMV)
40 void multiply(const StaticVector<T, Cols>& x, StaticVector<T, Rows>& y) const {
41 y.set_zero();
42 for (size_t i = 0; i < Rows; ++i) {
43 T sum = 0.0;
44 for (int j = row_ptr(i); j < row_ptr(i + 1); ++j) {
45 sum += values(j) * x(col_index(j));
46 }
47 y(static_cast<int>(i)) = sum;
48 }
49 }
50};
