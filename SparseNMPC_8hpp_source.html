93 template <typename T>
94 StaticVector<T, 25> eval_node_residuals(const StaticVector<T, Nx>& x,
95 const StaticVector<T, Nu>& u,
96 const StaticVector<T, Nu>& u_prev,
97 const StaticVector<double, Nx>& x_ref_local,
98 const StaticVector<double, Nx>& x_curr_global,
99 const StaticVector<double, Nx>& x_ref_global,
100 const NMPCTuningConfig& config,
101 int k)
102 {
103 using std::abs; using std::sqrt;
104
105 StaticVector<T, 25> res; res.set_zero();
106 int idx = 0;
107
108 T theta_c = T(x_curr_global(2));
109 T sin_t = T(std::sin(x_curr_global(2)));
110 T cos_t = T(std::cos(x_curr_global(2)));
111 T y_c = T(x_curr_global(1));
112 T y_target = T(x_ref_global(1));
113
114 // 글로벌 좌표계 기반의 투영 오차
115 T y_err = x(0) * sin_t + x(1) * cos_t + y_c - y_target;
116
117 // 1. 상태 페널티
118 res(idx++) = T(0.0); // X 무시
119 res(idx++) = T(sqrt(config.Q_Y)) * y_err;
120 res(idx++) = T(sqrt(config.Q_Yaw)) * (x(2) - T(x_ref_local(2)));
121 res(idx++) = T(sqrt(config.Q_Vx)) * (x(3) - T(10.0));
122
123 // 2. 제어 및 제어 변화량 페널티 (복원력)
124 res(idx++) = T(sqrt(config.R_Accel)) * u(0);
125 res(idx++) = T(sqrt(config.R_Steer)) * u(1);
126 res(idx++) = T(sqrt(config.R_Accel_Rate)) * (u(0) - u_prev(0));
127 res(idx++) = T(sqrt(config.R_Steer_Rate)) * (u(1) - u_prev(1));
128
129 // 3. 동적 장애물 예측 및 회피 페널티
130 double time_future = k * dt;
131 for (size_t i = 0; i < 10; ++i) {
132 // 시간 흐름에 따른 로컬 장애물 위치 투영
133 T obs_pred_x = T(local_obstacles[i].x + local_obstacles[i].vx * time_future);
134 T obs_pred_y = T(local_obstacles[i].y + local_obstacles[i].vy * time_future);
135
136 T dx = x(0) - obs_pred_x;
137 T dy = x(1) - obs_pred_y;
138 T dist_sq = dx * dx + dy * dy;
139 T r_safe = T(local_obstacles[i].r + config.Obstacle_Margin);
140 T viol = r_safe * r_safe - dist_sq;
141 res(idx++) = (Optimization::get_value(viol) > 0.0) ? T(sqrt(config.Obstacle_Penalty)) * viol : T(0.0);
142 }
143
144 // 4. 역주행 방지 가드 (Soft Constraint)
145 T v_min_viol = T(5.0) - x(3);
146 res(idx++) = (Optimization::get_value(v_min_viol) > 0.0) ? T(1000.0) * v_min_viol : T(0.0);
147
148 return res;
149 }

169 NMPCResult solve_rt_qp(const StaticVector<double, Nx>& x_curr_global,
170 const StaticVector<double, Nx>& x_ref_global,
171 const NMPCTuningConfig& config)
172 {
173 NMPCResult result;
174 result.sqp_iterations = 1;
175
176 // [Phase 1: 글로벌 -> 로컬 좌표계 변환 및 장애물 속도 회전]
177 auto T_g2l = utils::SE2Transform::get_global_to_local(x_curr_global(0), x_curr_global(1), x_curr_global(2));
178
179 double cos_yaw = std::cos(-x_curr_global(2));
180 double sin_yaw = std::sin(-x_curr_global(2));
181
182 for (size_t i = 0; i < 10; ++i) {
183 local_obstacles[i].r = global_obstacles[i].r;
184
185 // 위치 변환
186 utils::SE2Transform::transform_point(T_g2l, global_obstacles[i].x, global_obstacles[i].y, local_obstacles[i].x, local_obstacles[i].y);
187
188 // 속도 벡터 회전 변환
189 local_obstacles[i].vx = global_obstacles[i].vx * cos_yaw - global_obstacles[i].vy * sin_yaw;
190 local_obstacles[i].vy = global_obstacles[i].vx * sin_yaw + global_obstacles[i].vy * cos_yaw;
191 }
192
193 StaticVector<double, Nx> x_curr_local; x_curr_local.set_zero();
194 x_curr_local(3) = std::max(0.1, x_curr_global(3));
195 x_curr_local(4) = x_curr_global(4);
196 x_curr_local(5) = x_curr_global(5);
197
198 StaticVector<double, Nx> x_ref_local;
199 utils::SE2Transform::transform_point(T_g2l, x_ref_global(0), x_ref_global(1), x_ref_local(0), x_ref_local(1));
200 x_ref_local(2) = normalize_angle(x_ref_global(2) - x_curr_global(2));
201 x_ref_local(3) = x_ref_global(3);
202 x_ref_local(4) = x_ref_global(4);
203 x_ref_local(5) = x_ref_global(5);
204
205 vehicle::DynamicBicycleModel model;
206
207 // [Phase 2: 선형화 및 QP 문제 구성]
208 X_pred[0] = x_curr_local;
209 for (size_t k = 0; k < H; ++k) X_pred[k+1] = integrator::step_rk4<Nx, Nu>(model, X_pred[k], U_guess[k], dt);
210
211 for (size_t k = 0; k < H; ++k) {
212 StaticVector<double, Nu> u_prev = (k == 0) ? u_last : U_guess[k - 1];
213 using ADVar = DualVec<double, Nx + Nu>;
214 StaticVector<ADVar, Nx> x_dual;
215 StaticVector<ADVar, Nu> u_dual;
216 StaticVector<ADVar, Nu> u_prev_dual;
217
218 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(X_pred[k](i), i);
219 for (size_t i = 0; i < Nu; ++i) {
220 u_dual(i) = ADVar::make_variable(U_guess[k](i), Nx + i);
221 u_prev_dual(i) = ADVar(u_prev(i));
222 }
223
224 StaticVector<ADVar, Nx> x_next_dual = integrator::step_rk4<Nx, Nu>(model, x_dual, u_dual, dt);
225
226 for (size_t i = 0; i < Nx; ++i) {
227 for (size_t j = 0; j < Nx; ++j) riccati.A[k](i, j) = x_next_dual(i).g[j];
228 for (size_t j = 0; j < Nu; ++j) riccati.B[k](i, j) = x_next_dual(i).g[Nx + j];
229 riccati.d[k](i) = 0.0;
230 }
231
232 StaticVector<ADVar, 25> res_dual = eval_node_residuals(x_dual, u_dual, u_prev_dual, x_ref_local, x_curr_global, x_ref_global, config, k);
233
234 riccati.Q[k].set_zero();
235 riccati.R[k].set_zero();
236 riccati.q[k].set_zero();
237 riccati.r[k].set_zero();
238
239 for (size_t res_idx = 0; res_idx < 25; ++res_idx) {
240 double r_val = res_dual(res_idx).v;
241 for (size_t i = 0; i < Nx; ++i) {
242 double J_xi = res_dual(res_idx).g[i];
243 riccati.q[k](i) += J_xi * r_val;
244 for (size_t j = 0; j < Nx; ++j) riccati.Q[k](i, j) += J_xi * res_dual(res_idx).g[j];
245 }
246 for (size_t i = 0; i < Nu; ++i) {
247 double J_ui = res_dual(res_idx).g[Nx + i];
248 riccati.r[k](i) += J_ui * r_val;
249 for (size_t j = 0; j < Nu; ++j) riccati.R[k](i, j) += J_ui * res_dual(res_idx).g[Nx + j];
250 }
251 }
252
253 // [Phase 3: 완화된 Interior-Point Method (Hessian/Gradient 클램핑)]
254 for (size_t i = 0; i < Nu; ++i) {
255 double u_val = U_guess[k](i);
256 double s_upper = config.u_max[i] - u_val;
257 double s_lower = u_val - config.u_min[i];
258
259 s_upper = std::max(s_upper, 1e-3);
260 s_lower = std::max(s_lower, 1e-3);
261
262 double grad_barrier = config.barrier_mu * (1.0 / s_upper - 1.0 / s_lower);
263 double hess_barrier = config.barrier_mu * (1.0 / (s_upper * s_upper) + 1.0 / (s_lower * s_lower));
264
265 grad_barrier = std::clamp(grad_barrier, -100.0, 100.0);
266 hess_barrier = std::min(hess_barrier, 500.0);
267
268 riccati.r[k](i) += grad_barrier;
269 riccati.R[k](i, i) += hess_barrier;
270 }
271
272 // 동적 스케줄링 댐핑
273 for (size_t i = 0; i < Nx; ++i) riccati.Q[k](i, i) += config.damping_Q;
274 for (size_t i = 0; i < Nu; ++i) riccati.R[k](i, i) += config.damping_R;
275 }
276
277 // Terminal Cost 구성
278 riccati.Q[H].set_zero();
279 riccati.q[H].set_zero();
280
281 double sin_t = std::sin(x_curr_global(2));
282 double cos_t = std::cos(x_curr_global(2));
283 for(size_t i = 0; i < Nx; ++i) {
284 if (i == 1) {
285 double err_y = X_pred[H](0) * sin_t + X_pred[H](1) * cos_t + x_curr_global(1) - x_ref_global(1);
286 double qf_y = config.Q_Y * 5.0;
287 riccati.Q[H](1, 1) += qf_y * cos_t * cos_t;
288 riccati.Q[H](0, 0) += qf_y * sin_t * sin_t;
289 riccati.Q[H](0, 1) += qf_y * sin_t * cos_t;
290 riccati.Q[H](1, 0) += qf_y * sin_t * cos_t;
291 riccati.q[H](1) += qf_y * err_y * cos_t;
292 riccati.q[H](0) += qf_y * err_y * sin_t;
293 } else if (i != 0) {
294 double err = X_pred[H](i) - x_ref_local(i);
295 double qf_i = (i == 2) ? config.Q_Yaw * 5.0 : ((i == 3) ? config.Q_Vx * 5.0 : 0.0);
296 riccati.Q[H](i, i) += qf_i;
297 riccati.q[H](i) += qf_i * err;
298 }
299 }
300
301 // [Phase 4: Riccati Solve & Step Update]
302 if (riccati.solve() != SolverStatus::SUCCESS) return execute_fallback(result, "Factorization Failed");
303
304 // IPM 경계 준수를 위한 보폭(Alpha) 계산
305 double alpha = config.step_alpha;
306 for (size_t k = 0; k < H; ++k) {
307 for (size_t i = 0; i < Nu; ++i) {
308 double du = riccati.du[k](i);
309 if (du > 1e-4) {
310 double max_alpha = (config.u_max[i] - U_guess[k](i)) / du;
311 if (max_alpha > 0.0) alpha = std::min(alpha, config.safety_fraction * max_alpha);
312 } else if (du < -1e-4) {
313 double max_alpha = (config.u_min[i] - U_guess[k](i)) / du;
314 if (max_alpha > 0.0) alpha = std::min(alpha, config.safety_fraction * max_alpha);
315 }
316 }
317 }
318
319 // Stall 현상(보폭 0) 방지
320 alpha = std::max(alpha, 0.05);
321
322 double max_kkt = 0.0;
323 for (size_t k = 0; k < H; ++k) {
324 for (size_t i = 0; i < Nu; ++i) {
325 U_guess[k](i) += alpha * riccati.du[k](i);
326
327 // 수치 오류 방어막 (Strict Interior Projection)
328 U_guess[k](i) = std::clamp(U_guess[k](i), config.u_min[i] + 1e-3, config.u_max[i] - 1e-3);
329
330 max_kkt = std::max(max_kkt, std::abs(riccati.du[k](i)));
331 }
332 }
333
334 result.max_kkt_error = max_kkt;
335 if (std::isnan(max_kkt) || max_kkt > 20.0) return execute_fallback(result, "Divergence Detected");
336
337 u_last = U_guess[0];
338 result.success = true;
339 return result;
340 }
