37 SparseNMPC() {
38 // [Architect's Fix] X좌표 추종 페널티(Q(0)) 제거.
39 // 차량이 앞으로 나아가는 것을 처벌하면 차량은 스핀아웃을 선택합니다.
40 Q.set_zero(); Q(1) = 15.0; Q(2) = 5.0; Q(3) = 10.0;
41 Qf.set_zero(); for (size_t i=0; i<Nx; ++i) Qf(i) = Q(i) * 10.0;
42 R.set_zero(); R(0) = 0.5; R(1) = 10.0;
43 R_rate.set_zero(); R_rate(0) = 50.0; R_rate(1) = 250.0;
44 u_last.set_zero();
45 dt = 0.1;
46 for (size_t k=0; k<H; ++k) U_guess[k].set_zero();

48 // =========================================================
49 // [Architect's Fix] 유령 장애물 치우기
50 // 쓰지 않는 장애물 배열 초기값을 차량이 닿을 수 없는 곳으로 보냅니다.

59 // =========================================================================
60 // [Architect's Tactics] 각 노드(Node)별 비선형 잔차 함수 (M_res = 26)
61 // 1. 상태 추종 (6) + 2. 제어 크기 (2) + 3. Slew Rate (2)
62 // 4. 장애물 (10) + 5. Log-Barrier 기반 하드 제약 (4+2)
63 // =========================================================================
64 // 잔차의 개수를 정확히 24개로 맞춤 (상태 6 + 제어 2 + Slew 2 + 장애물 10 + 제약 4)
65 template <typename T>
66 StaticVector<T, 24> eval_node_residuals(const StaticVector<T, Nx>& x,
67 const StaticVector<T, Nu>& u,
68 const StaticVector<T, Nu>& u_prev,
69 const StaticVector<double, Nx>& x_ref)
70 {
71 // 이제 math_log, math_sqrt 라우터는 필요 없습니다. 완벽히 삭제하십시오.
72
73 StaticVector<T, 24> res; res.set_zero();
74 int idx = 0;
75
76 // 1. 상태 오차 잔차 (6)
77 for (size_t i = 0; i < Nx; ++i) res(idx++) = T(std::sqrt(Q(i))) * (x(i) - T(x_ref(i)));
78
79 // 2. 제어 입력 크기 잔차 (2)
80 for (size_t i = 0; i < Nu; ++i) res(idx++) = T(std::sqrt(R(i))) * u(i);
81
82 // 3. Slew Rate 잔차 (2)
83 for (size_t i = 0; i < Nu; ++i) res(idx++) = T(std::sqrt(R_rate(i))) * (u(i) - u_prev(i));
84
85 // 4. 장애물 회피망 (10)
86 for (size_t i = 0; i < 10; ++i) {
87 T dx = x(0) - T(obstacles[i].x);
88 T dy = x(1) - T(obstacles[i].y);
89 T dist_sq = dx * dx + dy * dy;
90 T r_safe = T(obstacles[i].r + 0.8);
91 T viol = r_safe * r_safe - dist_sq;
92 res(idx++) = (Optimization::get_value(viol) > 0.0) ? T(std::sqrt(8000.0)) * viol : T(0.0);
93 }
94
95 // =========================================================
96 // [Architect's Tactics] Inverse Barrier Function (Gauss-Newton IPM)
97 // NaN의 위험이 없고, 연산이 압도적으로 빠르며 완벽한 하드 제약을 형성합니다.
98 // =========================================================
99 double mu = 0.5; // Barrier Parameter
100
101 // 가속도 한계 [-3.0, 3.0]
102 T a_viol_upper = u(0) - T(3.0);
103 T a_viol_lower = T(-3.0) - u(0);
104 res(idx++) = (Optimization::get_value(a_viol_upper) > 0.0) ? T(std::sqrt(100.0)) * a_viol_upper : T(0.0);
105 res(idx++) = (Optimization::get_value(a_viol_lower) > 0.0) ? T(std::sqrt(100.0)) * a_viol_lower : T(0.0);
106
107 // 조향 한계 [-0.5, 0.5] rad
108 T steer_viol_upper = u(1) - T(0.5);
109 T steer_viol_lower = T(-0.5) - u(1);
110 res(idx++) = (Optimization::get_value(steer_viol_upper) > 0.0) ? T(std::sqrt(100.0)) * steer_viol_upper : T(0.0);
111 res(idx++) = (Optimization::get_value(steer_viol_lower) > 0.0) ? T(std::sqrt(100.0)) * steer_viol_lower : T(0.0);
112
113 return res;
114 }

116 void update_dynamic_weights(const StaticVector<double, Nx>& x_curr) {
117 double vx = x_curr(3); // 현재 종방향 속도 (m/s)
118
119 // 1. 기본 가중치 (저속 상태의 Base 값)
120 double base_R_steer = 10.0;
121 double base_R_rate = 250.0;
122 double base_Q_psi = 5.0;
123
124 // 2. 동적 스케줄링 (속도에 따른 페널티 증가)
125 // 고속일수록 조향을 무겁게 만들고(폭주 방지), 헤딩 추종을 날카롭게 만듭니다.
126 R(1) = base_R_steer + 0.5 * (vx * vx);
127 R_rate(1) = base_R_rate + 2.0 * (vx * vx);
128 Q(2) = base_Q_psi + 0.5 * std::abs(vx);
129 }

131 bool solve(const StaticVector<double, Nx>& x_curr, const StaticVector<double, Nx>& x_ref) {
132 vehicle::DynamicBicycleModel model;
133
134 // 1. Forward Rollout (Nominal Trajectory)
135 X_pred[0] = x_curr;
136 for (size_t k = 0; k < H; ++k) {
137 X_pred[k + 1] = integrator::step_rk4<Nx, Nu>(model, X_pred[k], U_guess[k], dt);
138 }
139
140 // 2. Linearization & Gauss-Newton Cost Formulation
141 for (size_t k = 0; k < H; ++k) {
142 // 이전 제어 입력 (k=0 이면 u_last, 아니면 이전 스텝의 guess)
143 StaticVector<double, Nu> u_prev = (k == 0) ? u_last : U_guess[k - 1];
144
145 // --- A. 동역학 선형화 (A_k, B_k) ---
146 using ADVar = DualVec<double, Nx + Nu>;
147 StaticVector<ADVar, Nx> x_dual;
148 StaticVector<ADVar, Nu> u_dual;
149 StaticVector<ADVar, Nu> u_prev_dual; // 잔차 평가용 상수로 취급
150
151 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(X_pred[k](i), i);
152 for (size_t i = 0; i < Nu; ++i) {
153 u_dual(i) = ADVar::make_variable(U_guess[k](i), Nx + i);
154 u_prev_dual(i) = ADVar(u_prev(i)); // Gradient 전파 안함
155 }
156
157 StaticVector<ADVar, Nx> x_next_dual = integrator::step_rk4<Nx, Nu>(model, x_dual, u_dual, dt);
158 for (size_t i = 0; i < Nx; ++i) {
159 for (size_t j = 0; j < Nx; ++j) riccati.A[k](i, j) = x_next_dual(i).g[j];
160 for (size_t j = 0; j < Nu; ++j) riccati.B[k](i, j) = x_next_dual(i).g[Nx + j];
161 }
162 for (size_t i = 0; i < Nx; ++i) riccati.d[k](i) = 0.0; // Nominal trajectory 일치
163
164 // --- B. 비용 함수 Gauss-Newton 선형화 (Q_k, R_k, q_k, r_k) ---
165 StaticVector<ADVar, 24> res_dual = eval_node_residuals(x_dual, u_dual, u_prev_dual, x_ref);
166
167 riccati.Q[k].set_zero(); riccati.R[k].set_zero();
168 riccati.q[k].set_zero(); riccati.r[k].set_zero();
169
170 // J^T * J 및 J^T * res 조립
171 for (size_t res_idx = 0; res_idx < 24; ++res_idx) {
172 double r_val = res_dual(res_idx).v;
173
174 // State 방향 (Q, q)
175 for (size_t i = 0; i < Nx; ++i) {
176 double J_xi = res_dual(res_idx).g[i];
177 riccati.q[k](i) += J_xi * r_val;
178 for (size_t j = 0; j < Nx; ++j) {
179 riccati.Q[k](i, j) += J_xi * res_dual(res_idx).g[j];
180 }
181 }
182 // Control 방향 (R, r)
183 for (size_t i = 0; i < Nu; ++i) {
184 double J_ui = res_dual(res_idx).g[Nx + i];
185 riccati.r[k](i) += J_ui * r_val;
186 for (size_t j = 0; j < Nu; ++j) {
187 riccati.R[k](i, j) += J_ui * res_dual(res_idx).g[Nx + j];
188 }
189 }
190 }
191
192 // =========================================================
193 // [Architect's Fix] Levenberg-Marquardt Damping 주입
194 // 순수 Gauss-Newton의 오버슛(급조향)을 억제하는 국소적 Trust Region.
195 // =========================================================
196 double lambda_q = 2.0;
197 // [Architect's Tuning] 조향이 미친듯이 튀는 것을 막는 최강의 물리적 제동 장치
198 double lambda_r = 500.0;
199 for (size_t i = 0; i < Nx; ++i) riccati.Q[k](i, i) += lambda_q;
200 for (size_t i = 0; i < Nu; ++i) riccati.R[k](i, i) += lambda_r;
201 }
202
203 // Terminal Node (H)
204 riccati.Q[H].set_zero(); riccati.q[H].set_zero();
205 for(size_t i=0; i<Nx; ++i) {
206 double err = X_pred[H](i) - x_ref(i);
207 riccati.Q[H](i, i) = Qf(i);
208 riccati.q[H](i) = Qf(i) * err;
209 }
210
211 // 3. Solve & Update
212 SolverStatus status = riccati.solve();
213 if (status != SolverStatus::SUCCESS) return false;
214
215 // =========================================================
216 // [Architect's Tactics] Strict Interior Projection
217 // 해가 장벽(Barrier)의 점근선을 뛰어넘어 기울기가 0이 되는 것을 방지합니다.
218 // 무조건 허용 구역 내부([-0.49, 0.49])에 머물도록 투영시킵니다.
219 // =========================================================
220 for (size_t k = 0; k < H; ++k) {
221 for (size_t i = 0; i < Nu; ++i) U_guess[k](i) += riccati.du[k](i);
222 }
223
224 // 다음 제어 주기를 위한 Warm Start
225 u_last = U_guess[0];
226
227 return true;
228 }
