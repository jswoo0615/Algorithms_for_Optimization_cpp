62 template <typename T>
63 StaticVector<T, 25> eval_node_residuals(const StaticVector<T, Nx>& x,
64 const StaticVector<T, Nu>& u,
65 const StaticVector<T, Nu>& u_prev,
66 const StaticVector<double, Nx>& x_ref_local,
67 const StaticVector<double, Nx>& x_curr_global,
68 const StaticVector<double, Nx>& x_ref_global)
69 {
70 using std::abs; using std::sqrt;
71
72 StaticVector<T, 25> res; res.set_zero();
73 int idx = 0;
74
75 T theta_c = T(x_curr_global(2));
76 T sin_t = T(std::sin(x_curr_global(2)));
77 T cos_t = T(std::cos(x_curr_global(2)));
78 T y_c = T(x_curr_global(1));
79 T y_target = T(x_ref_global(1));
80
81 // True global Y error projected approximately to local Y scale
82 // Y_global = x(0)*sin_t + x(1)*cos_t + y_c
83 // error = Y_global - y_target
84 T y_err = x(0) * sin_t + x(1) * cos_t + y_c - y_target;
85
86 // [Architect's New Law] X는 무시한다. 오직 차선 유지와 속도뿐.
87 res(idx++) = T(0.0); // X 오차 무시
88 res(idx++) = T(sqrt(200.0)) * y_err; // Y (차선) 엄격히 유지
89 res(idx++) = T(sqrt(500.0)) * (x(2) - T(x_ref_local(2))); // Yaw 엄격히 정렬
90 res(idx++) = T(sqrt(50.0)) * (x(3) - T(10.0)); // 속도 10m/s 유지
91
92 // 조향 자체를 죄악시한다 (R 가중치 극대화)
93 res(idx++) = T(sqrt(10.0)) * u(0); // Accel
94 res(idx++) = T(sqrt(5000.0)) * u(1); // Steering (매우 무겁게)
95
96 // 장애물 회피 (Margin 1.5m)
97 for (size_t i = 0; i < 10; ++i) {
98 T dx = x(0) - T(local_obstacles[i].x);
99 T dy = x(1) - T(local_obstacles[i].y);
100 T dist_sq = dx * dx + dy * dy;
101 T r_safe = T(local_obstacles[i].r + 1.5);
102 T viol = r_safe * r_safe - dist_sq;
103 res(idx++) = (Optimization::get_value(viol) > 0.0) ? T(sqrt(20000.0)) * viol : T(0.0);
104 }
105
106 // [Hard Defense] Vx가 5m/s 아래로 떨어지면 기하급수적 처벌 (역주행 방지)
107 T v_min_viol = T(5.0) - x(3);
108 res(idx++) = (Optimization::get_value(v_min_viol) > 0.0) ? T(1000.0) * v_min_viol : T(0.0);
109
110 return res;
111 }

130 NMPCResult solve(const StaticVector<double, Nx>& x_curr_global, const StaticVector<double, Nx>& x_ref_global, int max_sqp_iter = 1) {
131 NMPCResult result;
132 auto T_g2l = utils::SE2Transform::get_global_to_local(x_curr_global(0), x_curr_global(1), x_curr_global(2));
133 for (size_t i = 0; i < 10; ++i) {
134 local_obstacles[i].r = global_obstacles[i].r;
135 utils::SE2Transform::transform_point(T_g2l, global_obstacles[i].x, global_obstacles[i].y, local_obstacles[i].x, local_obstacles[i].y);
136 }
137
138 StaticVector<double, Nx> x_curr_local; x_curr_local.set_zero();
139 x_curr_local(3) = std::max(0.1, x_curr_global(3));
140 x_curr_local(4) = x_curr_global(4); x_curr_local(5) = x_curr_global(5);
141
142 StaticVector<double, Nx> x_ref_local;
143 utils::SE2Transform::transform_point(T_g2l, x_ref_global(0), x_ref_global(1), x_ref_local(0), x_ref_local(1));
144 x_ref_local(2) = normalize_angle(x_ref_global(2) - x_curr_global(2));
145 x_ref_local(3) = x_ref_global(3); x_ref_local(4) = x_ref_global(4); x_ref_local(5) = x_ref_global(5);
146
147 update_dynamic_weights(x_curr_local);
148 vehicle::DynamicBicycleModel model;
149
150 for (int sqp = 0; sqp < max_sqp_iter; ++sqp) {
151 result.sqp_iterations = sqp + 1;
152 X_pred[0] = x_curr_local;
153 for (size_t k = 0; k < H; ++k) X_pred[k+1] = integrator::step_rk4<Nx, Nu>(model, X_pred[k], U_guess[k], dt);
154
155 for (size_t k = 0; k < H; ++k) {
156 StaticVector<double, Nu> u_prev = (k == 0) ? u_last : U_guess[k - 1];
157 using ADVar = DualVec<double, Nx + Nu>;
158 StaticVector<ADVar, Nx> x_dual; StaticVector<ADVar, Nu> u_dual; StaticVector<ADVar, Nu> u_prev_dual;
159 for (size_t i = 0; i < Nx; ++i) x_dual(i) = ADVar::make_variable(X_pred[k](i), i);
160 for (size_t i = 0; i < Nu; ++i) {
161 u_dual(i) = ADVar::make_variable(U_guess[k](i), Nx + i);
162 u_prev_dual(i) = ADVar(u_prev(i));
163 }
164
165 StaticVector<ADVar, Nx> x_next_dual = integrator::step_rk4<Nx, Nu>(model, x_dual, u_dual, dt);
166 for (size_t i = 0; i < Nx; ++i) {
167 for (size_t j = 0; j < Nx; ++j) riccati.A[k](i, j) = x_next_dual(i).g[j];
168 for (size_t j = 0; j < Nu; ++j) riccati.B[k](i, j) = x_next_dual(i).g[Nx + j];
169 riccati.d[k](i) = 0.0;
170 }
171 StaticVector<ADVar, 25> res_dual = eval_node_residuals(x_dual, u_dual, u_prev_dual, x_ref_local, x_curr_global, x_ref_global);
172 riccati.Q[k].set_zero(); riccati.R[k].set_zero(); riccati.q[k].set_zero(); riccati.r[k].set_zero();
173 for (size_t res_idx = 0; res_idx < 25; ++res_idx) {
174 double r_val = res_dual(res_idx).v;
175 for (size_t i = 0; i < Nx; ++i) {
176 double J_xi = res_dual(res_idx).g[i]; riccati.q[k](i) += J_xi * r_val;
177 for (size_t j = 0; j < Nx; ++j) riccati.Q[k](i, j) += J_xi * res_dual(res_idx).g[j];
178 }
179 for (size_t i = 0; i < Nu; ++i) {
180 double J_ui = res_dual(res_idx).g[Nx + i]; riccati.r[k](i) += J_ui * r_val;
181 for (size_t j = 0; j < Nu; ++j) riccati.R[k](i, j) += J_ui * res_dual(res_idx).g[Nx + j];
182 }
183 }
184 // 강력한 정규화(Damping) 추가
185 for (size_t i = 0; i < Nx; ++i) riccati.Q[k](i, i) += 5.0;
186 for (size_t i = 0; i < Nu; ++i) riccati.R[k](i, i) += 500.0;
187 }
188 riccati.Q[H].set_zero(); riccati.q[H].set_zero();
189 double sin_t = std::sin(x_curr_global(2));
190 double cos_t = std::cos(x_curr_global(2));
191 for(size_t i=0; i<Nx; ++i) {
192 if (i == 1) {
193 double err_y = X_pred[H](0) * sin_t + X_pred[H](1) * cos_t + x_curr_global(1) - x_ref_global(1);
194 riccati.Q[H](1, 1) += Qf(1) * cos_t * cos_t;
195 riccati.Q[H](0, 0) += Qf(1) * sin_t * sin_t;
196 riccati.Q[H](0, 1) += Qf(1) * sin_t * cos_t;
197 riccati.Q[H](1, 0) += Qf(1) * sin_t * cos_t;
198 riccati.q[H](1) += Qf(1) * err_y * cos_t;
199 riccati.q[H](0) += Qf(1) * err_y * sin_t;
200 } else if (i != 0) { // i=0 is handled above, but original Q(0)=0 anyway
201 double err = X_pred[H](i) - x_ref_local(i);
202 riccati.Q[H](i, i) += Qf(i);
203 riccati.q[H](i) += Qf(i) * err;
204 }
205 }
206
207 if (riccati.solve() != SolverStatus::SUCCESS) return execute_fallback(result, "Factorization Failed");
208
209 double max_step = 0.0;
210 for (size_t k = 0; k < H; ++k) {
211 for (size_t i = 0; i < Nu; ++i) {
212 // Trust Region: 더 보수적인 스텝 클램핑
213 riccati.du[k](i) = std::clamp(riccati.du[k](i), -0.5, 0.5);
214 max_step = std::max(max_step, std::abs(riccati.du[k](i)));
215 }
216 }
217 result.max_kkt_error = max_step;
218 if (std::isnan(max_step) || max_step > 5.0) return execute_fallback(result, "Divergence Detected");
219
220 double alpha = 0.3; // 극단적으로 보수적인 보폭
221 for (size_t k = 0; k < H; ++k) {
222 for (size_t i = 0; i < Nu; ++i) U_guess[k](i) += alpha * riccati.du[k](i);
223 }
224 if (max_step < 1e-4) break;
225 }
226 u_last = U_guess[0]; result.success = true; return result;
227 }
228 };
