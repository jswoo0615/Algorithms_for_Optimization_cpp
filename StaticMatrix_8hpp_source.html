43 // =======================================================================================
44 // [메모리 최적화 : 정렬 (Alignment) 및 공간 지역성 (Spartial Locallity)]
45 // alignas(64) : 최신 CPU (x86_64, ARM 등)의 L1 캐시 라인 (Cache Line) 크기인 64바이트 단위로
46 // 메모리 시작 주소를 정렬합니다.
47 // - 캐시 히트율 (Cache Hit Rate) 극대화 : 프로세서가 메모리를 읽을 때 64바이트 덩어리
48 // (정크) 단위로 가져오므로, 정렬이 어긋나서 캐시 라인을 두 번 읽는 패널티 (False Sharing
49 // 등)를 방지합니다.
50 // - SIMD 가속 호환 : AVX-512 등의 벡터화 (Vectorization) 명령어는 메모리가 정렬되어 있지
51 // 않으면 심각한 성능 저하나 하드웨어 예외를 발생시킵니다.
52 // - 이 배열은 Column-major (열 우선) 방식으로, `data` 배열 전체가 메모리 상에 완벽하게
53 // 일렬로 연속된 (Contiguous) 하나의 1차원 블록을 형성합니다.
54 // =======================================================================================

64 // 원소들이 메모리 상에 연속적으로 배치되어 있습니다. 열 단위 연산 (예 : 행렬 곱셈, 블록
65 // 삽입/추출) 시 공간 지역성을 극대화하여 캐시 미스 (Cache Miss)를 획기적으로 줄입니다.

103 template <size_t SubRows, size_t SubCols>
104 void insert_block(size_t start_row, size_t start_col,
105 const StaticMatrix<T, SubRows, SubCols>& block) {
106 static_assert(SubRows <= Rows, "SubMatrix rows exceed target");
107 static_assert(SubCols <= Cols, "SubMatrix cols exceed target");
108 assert(start_row + SubRows <= Rows && "Row index out of bounds");
109 assert(start_col + SubCols <= Cols && "Col index out of bounds");
110
111 for (size_t j = 0; j < SubCols; ++j) {
112 const T* src = block.data_ptr() + (j * SubRows);
113 T* dest = this->data_ptr() + ((start_col + j) * Rows) + start_row;
114 std::copy(src, src + SubRows, dest);
115 }
116 }

119 void insert_transposed_block(size_t start_row, size_t start_col,
120 const StaticMatrix<T, SubRows, SubCols>& block) {
121 static_assert(SubCols <= Rows, "Trnasposed block rows exceed target");
122 static_assert(SubRows <= Cols, "Transposed block cols exceed target");
123 assert(start_col + SubRows <= Cols && "Col index out of bounds");
124 assert(start_row + SubCols <= Rows && "Row index out of bounds");
125
126 for (size_t j = 0; j < SubCols; ++j) {
127 for (size_t i = 0; i < SubRows; ++i) {
128 (*this)(static_cast<int>(start_row + j), static_cast<int>(start_col + i)) =
129 block(static_cast<int>(i), static_cast<int>(j));
130 }
131 }
