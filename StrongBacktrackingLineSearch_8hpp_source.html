30 template <size_t N, typename Func>
31 static double search(Func f, const std::array<double, N>& x, const std::array<double, N>& d,
32 double alpha_init = 1.0, double c1 = 1e-4, double c2 = 0.9,
33 bool verbose = false) {
34 // 1. 출발점 정보 획득
35 double f_x;
36 std::array<double, N> grad_x;
37 AutoDiff::value_and_gradient<N>(f, x, f_x, grad_x);
38
39 double dir_deriv = 0.0;
40 for (size_t i = 0; i < N; ++i) {
41 dir_deriv += grad_x[i] * d[i];
42 }
43 if (dir_deriv >= 0.0 && verbose) {
44 std::cout << " [Warning] Not a descent direction!\n";
45 }
46 // 2. 탐색 구간 초기화
47 double alpha_lo = 0.0;
48 double alpha_hi = 1e9; // 초기 상한선은 무한대로 설정
49 double alpha = alpha_init;
50
51 size_t iter = 0;
52 while (true) {
53 iter++;
54 auto x_new = ray_point<N>(x, d, alpha);
55 double f_new;
56 std::array<double, N> grad_new;
57 AutoDiff::value_and_gradient<N>(f, x_new, f_new, grad_new);
58
59 double dir_deriv_new = 0.0;
60 for (size_t i = 0; i < N; ++i) {
61 dir_deriv_new += grad_new[i] * d[i];
62 }
63 if (verbose) {
64 std::cout << " ↳ [Wolfe Iter " << std::setw(2) << iter << "] alpha: " << std::fixed
65 << std::setprecision(6) << alpha << " | Range: [" << alpha_lo << ", "
66 << (alpha_hi > 1e8 ? "INF" : std::to_string(alpha_hi)) << "]\n";
67 }
68
69 // [조건 1] Armijo 검사 : 너무 멀리 갔는가? (함수값이 충분히 안 떨어짐)
70 if (f_new > f_x + c1 * alpha * dir_deriv || dir_deriv_new > 0.0) {
71 if (verbose)
72 std::cout << " -> Armijo failed or went uphill. Shrinking upper bound.\n";
73 alpha_hi = alpha; // 상한선을 현재 위치로 당김
74 alpha = (alpha_lo + alpha_hi) / 2.0; // 구간의 절반으로 후퇴
75 }
76
77 // [조건 2] Strong Curvature 검사 : 너무 조금 갔는가? (기울기가 아직도 너무 가파름)
78 else if (std::abs(dir_deriv_new) > c2 * std::abs(dir_deriv)) {
79 if (verbose)
80 std::cout << " -> Curvature failed (Too steep). Expanding lower bound.\n";
81 alpha_lo = alpha; // 하한선을 현재 위치로 올림
82 if (alpha_hi >= 1e8) {
83 alpha *= 2.0; // 상한선을 아직 못 찾았다면 보폭을 2배로 늘림
84 } else {
85 alpha = (alpha_lo + alpha_hi) / 2.0; // 상한선이 있다면 그 사이의 절반으로 전진
86 }
87 }
88
89 // [조건 3] 완벽한 타격 (두 조건 모두 통과)
90 else {
91 if (verbose) std::cout << " ↳ [Accepted] Strong Wolfe conditions satisfied!\n";
92 return alpha;
93 }
94
95 // 무한 루프 방지 (정밀도 한계)
96 if (std::abs(alpha_hi - alpha_lo) < 1e-10) {
97 if (verbose)
98 std::cout << " ↳ [Failsafe] Bracket too small, returning best alpha.\n";
99 return alpha;
100 }
101 }
102 }
