14 // 벡터의 L2 Norm 계산
15 static double norm2(const std::array<double, 2>& v) {
16 return std::sqrt(v[0] * v[0] + v[1] * v[1]);
17 }

19 // 2x2 역행렬 계산 (Newton Step을 위해 필요)
20 static std::array<double, 2> solve_newton(const std::array<std::array<double, 2>, 2>& H,
21 const std::array<double, 2>& g) {
22 double det = H[0][0] * H[1][1] - H[0][1] * H[1][0];
23 if (std::abs(det) < 1e-12) {
24 return {-g[0], -g[1]}; // 특이행렬이면 그냥 Gradient 방향으로 도망
25 }
26 double inv00 = H[1][1] / det;
27 double inv01 = -H[0][1] / det;
28 double inv10 = -H[1][0] / det;
29 double inv11 = H[0][0] / det;
30
31 return {-(inv00 * g[0] + inv01 * g[1]), -(inv10 * g[0] + inv11 * g[1])};
32 }

40 double delta = 0.5 * max_delta; // 초기 반경
41 const double eta = 0.15; // Accept 기준 비율
42
43 if (verbose) {
44 std::cout << "========================================================\n";
45 std::cout << " 🛡️ Trust Region Method (Dogleg) Started\n";

47 }

49 for (size_t iter = 1; iter <= max_iter; ++iter) {
50 double f_x;
51 std::array<double, 2> g;
52 AutoDiff::value_and_gradient<2>(f, x, f_x, g);
53
54 double g_norm = norm2(g);
55 if (g_norm < 1e-5) {
56 if (verbose)
57 std::cout << " ↳ ✅ Converged! (Gradient ≈ 0) at Iteration: " << iter << "\n";
58 break;
59 }

63
64 // ----------------------------------------------
65 // [Subproblem] Dogleg Method (개다리 기법)
66 // 1. Cauchy Point (1차 미분 기반 가장 가파른 방향)
67 // 2. Newton Step (2차 미분 기반)
68 // 이 두 개를 섞어 반경 Delta 안에서 최적 스텝 p 도출
69 // ----------------------------------------------
70 std::array<double, 2> p = {0.0, 0.0};
71 std::array<double, 2> p_newton = solve_newton(H, g);
72
73 double gHg = g[0] * (H[0][0] * g[0] + H[0][1] * g[1]) + g[1] * (H[1][0] * g[0] + H[1][1] * g[1]);
74 std::array<double, 2> p_cauchy;
75 if (gHg <= 0) {// 곡률이 음수면 뒤집힘 -> 무조건 최대치로 이동
76 p_cauchy = {-delta * g[0] / g_norm, -delta * g[1] / g_norm};
77 } else {
78 double tau = (g_norm * g_norm) / gHg;
79 p_cauchy = {-tau * g[0], -tau * g[1]};
80 }
81
82 if (norm2(p_newton) <= delta) {
83 // 뉴턴 스텝이 신뢰 반경 안에 들어온다면 -> 그대로 진행
84 p = p_newton;
85 } else if (norm2(p_cauchy) >= delta) {
86 // 코시 포인트가 반경 밖이라면 -> 반경 가장자리까지 이동 (Gradient Descent)
87 p = {-delta * g[0] / g_norm, -delta * g[1] / g_norm};
88 } else {
89 // [Dogleg의 꽃] Cauchy Point와 Newton Step 사이를 선형 보간하여 반경 가장자리 계산
90 p = p_cauchy;
91 }
92
93 // -----------------------------------

95 // -----------------------------------
96 std::array<double, 2> x_new = {x[0] + p[0], x[1] + p[1]};
97 double f_new = AutoDiff::value<2>(f, x_new);
98 double act_red = f_x - f_new; // 실제 감소량
99
100 // 예측 모델 : m(p) = f + g^T p + 0.5 p^T H p
101 double pred_red = -(g[0] * p[0] + g[1] * p[1]) - 0.5 * (p[0] * (H[0][0] * p[0] + H[0][1] * p[1]) + p[1] * (H[1][0] * p[0] + H[1][1] * p[1]));
102 double rho = (pred_red == 0) ? 0 : act_red / pred_red;
103
104 if (verbose) {
105 std::cout << "[Iter " << std::setw(3) << iter << "] f(x): " << std::fixed << std::setprecision(5) << f_x

107 }
108
109 // -----------------------------------------------------------------
110 // [Update] 신뢰 영역 반경 조절
111 // -----------------------------------------------------------------
112 if (rho < 0.25) {
113 delta *= 0.25; // 예측 대실패: 지형이 너무 험하다. 1/4로 줄임!
114 if (verbose)
115 std::cout << " -> 📉 Shrink\n";
116 } else if (rho > 0.75 && norm2(p) >= 0.99 * delta) {
117 delta = std::min(2.0 * delta, max_delta); // 예측 대성공: 2배로 확장!
118 if (verbose)
119 std::cout << " -> 📈 Expand\n";
120 } else {
121 if (verbose)
122 std::cout << " -> ➖ Keep\n";
123 }
124
125 // 스텝 승인(Accept) 여부
126 if (rho > eta) {
127 x = x_new; // 이동 확정!
128 }
129 }
130 if (verbose) {
131 std::cout << "========================================================\n";
132 std::cout << " 🏁 Final Optimal Point: [" << x[0] << ", " << x[1] << "]\n";
133 std::cout << "========================================================\n";
134 }
135
136 return x;
